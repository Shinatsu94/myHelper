```java
class OddMultiplicationTable { // 修改類別名稱使其更具意義
    public static void main(String args[]) {
        // 設定起始值、結束值、間隔，提高程式碼可讀性和彈性
        int startOdd = 1;
        int endOdd = 9;
        int oddIncrement = 2;
        int startEven = 2;
        int endEven = 8;
        int evenIncrement = 2;

        // 外層迴圈控制奇數 (1, 3, 5, 7, 9)
        for (int i = startOdd; i <= endOdd; i += oddIncrement) {
            // 內層迴圈控制偶數 (2, 4, 6, 8)
            for (int j = startEven; j <= endEven; j += evenIncrement) {
                System.out.print(i + "*" + j + "=" + i * j);
                if (j != endEven) {
                    System.out.print("\t");
                }
            }
            System.out.println(); // 換行
        }
    }
}
```
