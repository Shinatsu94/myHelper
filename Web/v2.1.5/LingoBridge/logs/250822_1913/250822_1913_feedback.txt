```java
class hw4_2

      // 外迴圈控制奇數的乘數 (1, 3, 5, 7, 9)
      for(int i=1; i<=maxOddNumber; i+=2)
      {
         // 內迴圈控制偶數的被乘數 (2, 4, 6, 8)
         for(int j=2; j<=maxEvenNumber; j+=2)
         {
            // 印出乘法運算式和結果
            System.out.print(i + "*" + j + "=" + i*j);

            // 如果不是該行的最後一個元素，則印出 tab 分隔符
            if(j != maxEvenNumber) {
                System.out.print("\t");
            }
         }
         // 每完成一行後換行
         System.out.println();
      }
   }
}
```
