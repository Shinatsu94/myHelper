        final int START_EVEN = 2; // 偶數起始值
        final int MAX_EVEN = 8; // 最大偶數

        for (int i = 1; i <= MAX_ODD; i += 2) {

                System.out.print(i + "*" + j + "=" + i * j);
                if (j != MAX_EVEN) {

            }

        }
    }
}
