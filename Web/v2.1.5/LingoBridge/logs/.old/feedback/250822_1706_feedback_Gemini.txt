```java
import java.util.Scanner;
class hw4_2 {
   public static void main(String args[]) {
       // 新增Scanner物件，讀取使用者輸入
       Scanner scanner = new Scanner(System.in);
       System.out.print("請輸入一個1~9的奇數：");
       int n = scanner.nextInt();

       // 確保輸入的是奇數且在範圍內，如果不是則重新輸入
       while (n < 1 || n > 9 || n % 2 == 0) {
           System.out.println("輸入錯誤，請輸入1~9的奇數：");
           n = scanner.nextInt();
       }

       for (int i = 1; i <= n; i += 2) {
           for (int j = 2; j <= 8; j += 2) {
               System.out.print(i + "*" + j + "=" + i * j);
               if (j != 8) {
                   System.out.print("\t");
               }
           }
           System.out.println();
       }

       // 關閉Scanner物件
       scanner.close();
   }
}
```
