class hw4_2 {
    public static void main(String args[]) {
        // 印出一個 n x 9 乘法表，n 為 1 ~ 9 的奇數

        // 第一行為 n x 2，第二行為 n x 4，以此類推至 n x 8

        for (int i = 1; i <= 9; i += 2) { // 迴圈遍歷 1, 3, 5, 7, 9 (奇數)
            for (int j = 2; j <= 8; j += 2) { // 迴圈遍歷 2, 4, 6, 8 (偶數)
                System.out.print(i + "*" + j + "=" + i * j); // 輸出乘法算式
                if (j != 8) {
                    System.out.print("\t"); // 除了最後一個，加上 tab 分隔
                }
            }
            System.out.println(); // 換行
        }
    }
}
