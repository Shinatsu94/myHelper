        final int MAX_EVEN_NUMBER = 8;

            for (int j = START_EVEN_NUMBER; j <= MAX_EVEN_NUMBER; j += 2) {
                // 輸出乘法表的每一項

                // 除了最後一項，都加上 tab 分隔
                if (j != MAX_EVEN_NUMBER) {
                    System.out.print("\t");
                }
            }
            // 每一行輸出後換行

    }
}
