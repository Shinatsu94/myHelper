```java
/**
 * 檔案名稱: LeibnizPi.java
 * 功能: 使用勒貝格級數近似圓周率 pi，輸出每 10000 次迭代的 pi 值，並找出最小的 i 使 pi 四捨五入至小數點後五位為 3.14159。
 * 作者: 教授
 * 日期: 2025-11-11
 * 備註: 依題意輸出格式，i 為六位寬，pi 為小數點後十位。
 */

public class LeibnizPi {
    public static void main(String[] args) {
        final int MAX_TERMS_PART1 = 100000; // part (a) 迭代至的項數
        final int STEP = 10000;             // 每 STEP 顯示一次
        final int MAX_TERMS_PART2 = 1000000; // part (b) 搜尋的上限（可調整）
        double sum = 0.0;
        double pi = 0.0;

        // Part (a): 逐步累加，輸出 i = 10000, 20000, ..., 100000 及對應 pi
        for (int k = 0; k < MAX_TERMS_PART1; k++) {
            // 公式：sum_k = sum_{j=0..k} (-1)^j / (2j+1)
            sum += (k % 2 == 0 ? 1.0 : -1.0) / (2.0 * k + 1.0);
            pi = 4.0 * sum;
            if ((k + 1) % STEP == 0) {
                System.out.printf("i = %6d, pi = %.10f%n", (k + 1), pi);
            }
        }

        System.out.println();

        // Part (b): 從第 MAX_TERMS_PART1+1 項開始繼續，找第一個 pi 四捨五入為 3.14159 的 i
        for (int k = MAX_TERMS_PART1; k < MAX_TERMS_PART2; k++) {
            sum += (k % 2 == 0 ? 1.0 : -1.0) / (2.0 * k + 1.0);
            pi = 4.0 * sum;
            String pi5 = String.format("%.5f", pi);
            if ("3.14159".equals(pi5)) {
                System.out.println("i = " + (k + 1));
                break;
            }
        }
    }
}
```

- 建議把輸出格式統一成 printf 風格並完整加入程式標頭註解，以提升可讀性與符合評分規範。
