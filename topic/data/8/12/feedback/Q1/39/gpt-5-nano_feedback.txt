/**
 * 檔案名稱: CSIE_41343216_1017_1.java
 * 功能: 計算 pi 的近似值，並輸出指定 i 值對應的 pi 值（以逐項累加法計算）
 * 作者: 你的名字
 * 日期: 2025-11-15
 * 備註: 程式內容包含註解，符合基本程式說明需求
 */

import java.util.Locale;

public class CSIE_41343216_1017_1 {
    public static void main(String[] args) {
        Locale.setDefault(Locale.US);

        // Part (a): 逐步計算並輸出 i = 10000, 20000, ..., 100000 對應的 pi
        int[] checkpoints = {10000, 20000, 30000, 40000, 50000, 60000, 70000, 80000, 90000, 100000};
        int idx = 0;

        double sum = 0.0; // sum = sum_{k=0}^{i-1} (-1)^k / (2k+1)
        // i 介於 1 到 100000，公式為 pi = 4 * sum
        for (int i = 0; i < 100000; i++) {
            double term = ((i % 2 == 0) ? 1.0 : -1.0) / (2.0 * i + 1.0);
            sum += term;

            int termCount = i + 1;
            if (idx < checkpoints.length && termCount == checkpoints[idx]) {
                double pi = 4.0 * sum;
                System.out.printf("i = %6d, pi = %.10f\n", checkpoints[idx], pi);
                idx++;
            }
        }

        // Part (b): 找到最小的 i 使 pi 接近 3.14159，並輸出該 i (以 1-based 的 i 表示)
        sum = 0.0;
        double target = 3.14159;
        int i = 0;
        while (true) {
            double term = ((i % 2 == 0) ? 1.0 : -1.0) / (2.0 * i + 1.0);
            sum += term;
            double pi = 4.0 * sum;

            // 以四捨五入到小數點後第五位作為相容性判定
            if (Math.abs(pi - target) < 0.000005) {
                System.out.printf("\ni = %6d\n", i + 1);
                break;
            }
            i++;
        }
    }
}
