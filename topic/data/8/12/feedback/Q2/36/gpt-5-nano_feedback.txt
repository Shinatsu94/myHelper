```java
/**
 * 檔案名稱: LeibnizPi.java
 * 功能: 根據萊布尼茨級數計算 π 的逐步近似，輸出 i 為 10000, 20000, ..., 100000 的 pi 值，並格式化輸出
 * 作者: 請填寫
 * 日期: 2025-11-11
 * 版本: 1.0
 * 說明: 程式內含註解以解釋關鍵步驟
 */

public class LeibnizPi {
    public static void main(String[] args) {
        // 依題意，輸出 i 為 10000 到 100000，間隔為 10000
        for (int i = 10000; i <= 100000; i += 10000) {
            double sum = 0.0;
            for (int k = 0; k < i; k++) {
                // 第 k 項的符號依次為 +, -, +, -, ...
                double term = (k % 2 == 0) ? (1.0 / (2 * k + 1)) : (-1.0 / (2 * k + 1));
                sum += term;
            }
            double pi = 4.0 * sum;
            // i 顯示為六位寬度，pi 保留小數點後 10 位
            System.out.printf("i = %6d, pi = %.10f%n", i, pi);
        }
    }
}
```
