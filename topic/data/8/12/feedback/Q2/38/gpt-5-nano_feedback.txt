/*
File: CSIE_41343215_1017_2.java
Description: 判斷輸入的數字各位是否嚴格遞增（從左至右）。
Author: Student
Date: 2025-11-15
*/

import java.util.Scanner;

public class CSIE_41343215_1017_2 {

    // 檢查數字字串的每一位是否嚴格遞增
    public static boolean isWellOrdered(String numberStr) {
        for (int i = 0; i < numberStr.length() - 1; i++) {
            if (numberStr.charAt(i) >= numberStr.charAt(i + 1)) {
                return false;
            }
        }
        return true;
    }

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        // 讀取單行輸入並去除前後空白
        String userInput = scanner.nextLine().trim();

        if (isWellOrdered(userInput)) {
            System.out.println(userInput + " is well-ordered number.");
        } else {
            System.out.println(userInput + " is not well-ordered number.");
        }
        scanner.close();
    }
}
