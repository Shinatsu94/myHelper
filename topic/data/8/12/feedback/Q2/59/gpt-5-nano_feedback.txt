// 檔案名稱: LeibnizPi.java
// 功能: 計算 pi 的勒貝日茲級數近似，輸出指定 i 值的 pi 值與最小 i 使近似值顯示為 3.14159 的 i
// 作者: 你/教授
// 日期: 2025-11-12

public class LeibnizPi {
    public static void main(String[] args) {
        // Part (a): 針對 i = 10000, 20000, ..., 100000 計算並輸出 pi 值
        int[] targets = {10000, 20000, 30000, 40000, 50000,
                         60000, 70000, 80000, 90000, 100000};

        double sum = 0.0; // sum = Σ (-1)^k / (2k+1)，k 從 0 累加到 i-1
        int k = 0;        // 已累加的項數

        // 逐步累加，遇到目標時輸出結果
        for (int target : targets) {
            while (k < target) {
                double term = (k % 2 == 0) ? 1.0 : -1.0;
                sum += term / (2.0 * k + 1.0);
                k++;
            }
            double piValue = sum * 4.0;
            System.out.printf("i = %6d, pi = %.10f%n", target, piValue);
        }

        // Part (b): 找最小的 i 使近似值顯示為 3.14159（以 5 位小數四捨五入）
        // 以目前的累加狀態繼續累加，直到符合條件為止
        while (true) {
            double term = (k % 2 == 0) ? 1.0 : -1.0;
            sum += term / (2.0 * k + 1.0);
            k++;
            double piValue = sum * 4.0;

            // 以 5 位小數做四捨五入判斷
            double rounded5 = Math.floor(piValue * 100000.0 + 0.5) / 100000.0;
            if (rounded5 == 3.14159) {
                System.out.println("i = " + k);
                break;
            }
        }
    }
}
