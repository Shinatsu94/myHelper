import java.util.Scanner;

/*
 * File: WellOrderedNumber.java
 * Purpose: Determine whether a given positive integer's digits are strictly increasing
 *          from left to right.
 * Author: (Your Name)
 * Date: 2025-11-15
 * Notes: This program reads one number from standard input and prints whether it is a
 *        well-ordered number according to the definition:
 *        digits strictly increasing from left to right.
 */
public class WellOrderedNumber {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        String s = sc.next();  // 直接以字串處理，方便逐位比較
        sc.close();

        boolean wellOrdered = true;
        for (int i = 0; i < s.length() - 1; i++) {
            char a = s.charAt(i);
            char b = s.charAt(i + 1);
            // 若任意相鄰兩位不嚴格遞增，則不是 well-ordered
            if (a >= b) {
                wellOrdered = false;
                break;
            }
        }

        if (wellOrdered) {
            System.out.println(s + " is well-ordered number.");
        } else {
            System.out.println(s + " is not well-ordered number.");
        }
    }
}
