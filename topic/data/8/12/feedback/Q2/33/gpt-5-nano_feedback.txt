注：以下提供修改後的完整程式碼，使用字串方式檢查每一位是否嚴格遞增，並包含基本註解與更清晰的命名。

/**
 * File: WellOrderedNumber.java
 * Description: 判斷輸入數字的各位是否從左到右嚴格遞增
 * Author: 你自己
 * Date: 2025-11-15
 */
public class WellOrderedNumber {
    public static void main(String[] args) {
        java.util.Scanner sc = new java.util.Scanner(System.in);
        String s = sc.next(); // 直接取得輸入內容，保留原始表示以便輸出
        sc.close();

        boolean isWellOrdered = true;

        for (int i = 0; i < s.length() - 1; i++) {
            char c1 = s.charAt(i);
            char c2 = s.charAt(i + 1);
            if (c1 >= c2) {
                isWellOrdered = false;
                break;
            }
        }

        if (isWellOrdered) {
            System.out.printf("%s is well-ordered number.\n", s);
        } else {
            System.out.printf("%s is not well-ordered number.\n", s);
        }
    }
}
