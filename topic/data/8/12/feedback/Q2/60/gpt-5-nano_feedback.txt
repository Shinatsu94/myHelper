```java
/*
檔案名稱: LeibnizPiCalculator.java
功能: 計算 pi 的近似值，輸出 i 為 10000, 20000, ..., 100000 的 pi 值，並輸出能使 pi 為 3.14159 的最小 i
作者: 你的姓名
日期: 2025-11-12
說明: 使用 Leibniz 公式 pi = 4 * sum_{k=0}^{i-1} (-1)^k / (2k+1)
      輸出格式符合題目要求：i 顯示到6位數，pi 顯示到小數點後10位
*/
public class LeibnizPiCalculator {
    public static void main(String[] args) {
        // (a) 依序計算 i = 10000, 20000, ..., 100000 的 pi 值
        double sum = 0.0;
        for (int i = 1; i <= 100000; i++) {
            // 第 i 項的分母為 (2*i - 1)，符號為 (-1)^(i-1)
            double term = ((i % 2 == 1) ? 1.0 : -1.0) / (2.0 * i - 1.0);
            sum += term;

            if (i % 10000 == 0) {
                double pi = 4.0 * sum;
                System.out.printf("i = %6d, pi = %.10f%n", i, pi);
            }
        }

        // (b) 找到能使 pi 四捨五入到小數第5位等於 3.14159 的最小 i
        double sum2 = 0.0;
        int minI = -1;
        for (int i = 1; i <= 200000; i++) {
            double term = ((i % 2 == 1) ? 1.0 : -1.0) / (2.0 * i - 1.0);
            sum2 += term;
            double pi = 4.0 * sum2;
            double rounded = Math.round(pi * 100000.0) / 100000.0;
            if (rounded == 3.14159) {
                minI = i;
                break;
            }
        }
        System.out.println();
        System.out.println("i = " + minI);
    }
}
```
