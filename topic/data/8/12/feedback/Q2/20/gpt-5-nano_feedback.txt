/*
 * File: CSIE_41343112_1017_2.java
 * Description: Determine if a given positive integer's digits are strictly increasing from left to right.
 * Author: CSIE_41343112_1017_2
 * Date: 2025-11-15
 */

import java.util.Scanner;

public class CSIE_41343112_1017_2 {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int number = scanner.nextInt();
        scanner.close();

        boolean isWellOrdered = true;
        int rightDigit = number % 10;   // 最右邊的位數
        int leftPart = number / 10;      // 右邊位數以外的部分

        // 從右往左檢查每一位，左位必須比右位小（才能是左到右嚴格遞增）
        while (leftPart > 0) {
            int leftDigit = leftPart % 10;
            if (leftDigit >= rightDigit) {
                isWellOrdered = false;
                break;
            }
            rightDigit = leftDigit;
            leftPart /= 10;
        }

        if (isWellOrdered) {
            System.out.println(number + " is well-ordered number.");
        } else {
            System.out.println(number + " is not well-ordered number.");
        }
    }
}
