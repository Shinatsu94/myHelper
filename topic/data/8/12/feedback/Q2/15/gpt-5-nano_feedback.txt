修改說明

  - 對 i = 10000, 20000, ..., 100000 逐步輸出 pi，格式與題目給定相符（i 以六位寬度印出，pi 以小數點後十位印出）。
  - 再往後延伸計算，尋找最小的 i，使 pi 四捨五入到小數點五位等於 3.14159，輸出格式為單行 "i = ..."。

/*
File: LeibnizPiPrinter.java
功能:
  - 依勒讓得級數輸出 i = 10000, 20000, ..., 100000 時的 pi 值，格式符合要求。
  - 找出最小的 i，使 pi 四捨五入到五位小數等於 3.14159，並輸出該 i。
作者: 請填寫
日期: 2025-11-11
說明: 使用累積和的方式計算 pi，避免每次重新計算整個序列。
*/
import java.util.*;

public class LeibnizPiPrinter {
    public static void main(String[] args) {
        // 設定最大 i 值與輸出條件
        final int MAX_I = 100000;
        double sum = 0.0;

        // 逐步累加勒讓得級數的項，輸出當 i 為 10000 的整倍數時的 pi 值
        for (int i = 1; i <= MAX_I; i++) {
            int k = i - 1;
            double term = ( (k % 2 == 0) ? 1.0 : -1.0 ) / (2.0 * k + 1.0);
            sum += term;
            if (i % 10000 == 0) {
                double pi = 4.0 * sum;
                // i 以六位寬度對齊，pi 保留十位小數
                System.out.printf("i = %6d, pi = %.10f%n", i, pi);
            }
        }

        // 顯示空行以區隔兩段輸出
        System.out.println();

        // 從 100000 開始繼續往後找最小的 i，使 pi 四捨五入到五位小數等於 3.14159
        // 直接用累積和從 MAX_I 開始累加，避免重新計算前面的項
        double currentSum = sum; // sum 目前為 100000 項的總和
        int iMin = MAX_I;
        while (true) {
            iMin++;
            int k = iMin - 1;
            double term = ( (k % 2 == 0) ? 1.0 : -1.0 ) / (2.0 * k + 1.0);
            currentSum += term;
            double pi = 4.0 * currentSum;
            long roundedFive = Math.round(pi * 1e5); // 四捨五入到 5 位小數
            if (roundedFive == 314159) {
                System.out.println("i = " + iMin);
                break;
            }
            // 若需要，可加入上限保護以避免無限迴圈
        }
    }
}
