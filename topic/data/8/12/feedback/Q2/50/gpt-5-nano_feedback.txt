/*
檔案名稱: WellOrderedNumber.java
功能說明: 判斷一個整數的每一位數字是否由左至右遞增（strictly increasing）。
作者: 你的名字
日期: 2025-11-15
*/
import java.io.BufferedReader;
import java.io.InputStreamReader;

public class WellOrderedNumber {
    public static void main(String[] args) {
        String input;
        try (BufferedReader br = new BufferedReader(new InputStreamReader(System.in))) {
            input = br.readLine();
        } catch (Exception e) {
            // 若讀取失敗，直接結束
            return;
        }

        if (input == null) return;
        input = input.trim();
        if (input.length() == 0) return;

        boolean isWellOrdered = true;
        for (int i = 0; i < input.length() - 1; i++) {
            char c1 = input.charAt(i);
            char c2 = input.charAt(i + 1);
            // 確保都是位數字
            if (!Character.isDigit(c1) || !Character.isDigit(c2)) {
                isWellOrdered = false;
                break;
            }
            // 相鄰兩位需嚴格遞增
            if (c1 >= c2) {
                isWellOrdered = false;
                break;
            }
        }

        if (isWellOrdered) {
            System.out.println(input + " is well-ordered number.");
        } else {
            System.out.println(input + " is not well-ordered number.");
        }
    }
}
