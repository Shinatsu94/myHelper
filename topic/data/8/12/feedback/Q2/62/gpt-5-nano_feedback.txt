 File: LeibnizPiProgram.java
 功能: 計算 Leibniz 公式近似的 pi 值，輸出指定的 i 值與對應的 pi，並找出使 pi 至 3.14159 的最小 i。
 作者: 輸入者
 日期: 2025-11-11

public class LeibnizPiProgram {
    public static void main(String[] args) {
        // (a) 計算 pi 此公式的近似值，對 i = 10000, 20000, ..., 100000
        final int MAX_I = 100000;
        double sum = 0.0; // 1 - 1/3 + 1/5 - ... 的部分和
        for (int i = 1; i <= MAX_I; i++) {
            double term = 1.0 / (2 * i - 1);
            if (i % 2 == 1) sum += term;
            else sum -= term;
            if (i % 10000 == 0) {
                double pi = 4.0 * sum;
                System.out.printf("i = %6d, pi = %.10f%n", i, pi);
            }
        }

        // 空行分隔
        System.out.println();

        // (b) 找出最小的 i 使 pi ~ 3.14159
        // 以 pi 的近似值乘以 100000 後取整，若等於 314159 表示小數後五位為 3.14159
        double sumForMin = sum;
        int iMin = MAX_I;
        while (true) {
            iMin++;
            double term = 1.0 / (2 * iMin - 1);
            if (iMin % 2 == 1) sumForMin += term;
            else sumForMin -= term;
            double pi = 4.0 * sumForMin;
            if ((int)(pi * 100000.0) == 314159) {
                System.out.println("i = " + iMin);
                break;
            }
            // 為避免無限迴圈，實務上可加上上限，但此題預計於較大 i 即可達成
        }
    }
}
