/*
檔案: ReverseDigits.java
功能: 讀取整數，輸出其位元數字反轉的結果。負數亦會保持符號並反轉數字。
作者: 匿名
日期: 2025-11-15
*/
import java.util.Scanner;

public class ReverseDigits {
    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);

        long n = in.nextLong();          // 原始數字
        boolean negative = n < 0;          // 判斷是否為負數
        long m = Math.abs(n);              // 取絕對值以便逐位處理

        long reversed = 0;
        while (m != 0) {
            reversed = reversed * 10 + (m % 10);
            m /= 10;
        }

        if (negative) {
            reversed = -reversed;
        }

        System.out.println(reversed);
        in.close();
    }
}
