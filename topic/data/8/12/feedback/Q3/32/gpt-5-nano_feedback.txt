/*
 * 檔案名稱: PiLeibniz.java
 * 功能說明: 計算圓周率的近似值，使用 Leibniz 線性級數
 *          (a) 輸出 i = 10000, 20000, ..., 100000 時的 pi 值，格式為
 *              i =  %6d, pi = %.10f
 *          (b) 找出使 pi 五位小數為 3.14159 的最小 i，輸出
 *              i = %6d
 * 作者: 你的姓名
 * 日期: 2025-11-11
 */

public class PiLeibniz {
    // 計算條件為 terms 個項，返回 pi 的近似值
    public static double approximatePiLeibniz(int terms) {
        double sum = 0.0;
        int sign = 1;
        for (int k = 0; k < terms; k++) {
            sum += sign * 1.0 / (2 * k + 1);
            sign = -sign;
        }
        return 4.0 * sum;
    }

    // 找到使 pi 四捨五入到五位小數等於 target 的最小項數
    // 目標使用 3.14159 對應的整數 314159（乘以 1e5 後四捨五入）
    public static int minTermsForPiToFiveDecimal(double target) {
        double sum = 0.0;
        int sign = 1;
        int k = 0;
        long targetRounded = Math.round(target * 100000); // 314159
        while (true) {
            sum += sign * 1.0 / (2 * k + 1);
            double approx = 4.0 * sum;
            long rounded5 = Math.round(approx * 100000);
            if (rounded5 == targetRounded) {
                return k + 1; // 由於 k 從 0 開始，實際項數為 k+1
            }
            sign = -sign;
            k++;
            // 防護機制，避免無窮迴圈（極端情況）
            if (k > 100000000) {
                break;
            }
        }
        return -1;
    }

    public static void main(String[] args) {
        // (a) 輸出 i = 10000, 20000, ..., 100000 的 pi 值
        for (int i = 10000; i <= 100000; i += 10000) {
            double pi = approximatePiLeibniz(i);
            System.out.printf("i = %6d, pi = %.10f%n", i, pi);
        }

        // 空行分隔兩部分輸出（符合題目示例中的格式）
        System.out.println();

        // (b) 找出使 pi 四捨五入到五位小數等於 3.14159 的最小 i
        int minI = minTermsForPiToFiveDecimal(3.14159);
        if (minI != -1) {
            System.out.printf("i = %6d%n", minI);
        } else {
            System.out.println("未找到符合條件的 i");
        }
    }
}
