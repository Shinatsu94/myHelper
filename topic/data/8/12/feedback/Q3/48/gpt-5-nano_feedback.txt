/*
 * 檔案名稱: ReverseDigits.java
 * 功能: 將輸入整數的各位數字反轉並輸出，僅執行一次（單次輸入）
 * 作者: 教師
 * 日期: 2025-11-15
 * 說明: 支援正負數輸入，若輸入為 0 會輸出 0。
 */

import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        long n = scanner.nextLong();
        boolean negative = n < 0;
        long num = Math.abs(n);

        long rev = 0;
        while (num != 0) {
            rev = rev * 10 + (num % 10);
            num /= 10;
        }

        if (negative) rev = -rev;
        System.out.println(rev);
    }
}
