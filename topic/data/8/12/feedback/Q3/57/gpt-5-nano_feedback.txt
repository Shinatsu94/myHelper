/*
 * File: ReverseDigits.java
 * 功能: 讀取使用者輸入的整數，將其數字位元反轉輸出。
 * 內容說明: 支援負數，負號保留，數字部分解析後再重新組合成結果。
 * 作者: 你的名字
 * 日期: 2025-11-15
 */

import java.util.Scanner;

public class ReverseDigits {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 以 long 提供較大的數值範圍，並考慮負數情況
        long n = sc.nextLong();
        int sign = n < 0 ? -1 : 1;
        long m = Math.abs(n);

        long reversed = 0;

        // 反轉數字部分
        while (m > 0) {
            long digit = m % 10;
            reversed = reversed * 10 + digit;
            m /= 10;
        }

        long result = sign * reversed;
        System.out.println(result);

        sc.close();
    }
}
