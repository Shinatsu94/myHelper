修改後完整程式碼（Java）：
/*
 * File: ReverseDigits.java
 * Description: 讀入一個整數，輸出其數字反轉結果；反轉後若開頭出現零則移除，輸入為 0 時輸出 0；支援正負數。
 * Author: 你的姓名
 * Date: 2025-11-15
 */
import java.util.Scanner;

public class ReverseDigits {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取單一整數輸入
        if (!sc.hasNextLong()) {
            return;
        }
        long n = sc.nextLong();

        // 特殊情況：0
        if (n == 0) {
            System.out.println(0);
            return;
        }

        boolean negative = n < 0;
        long m = Math.abs(n);

        // 逐位反轉，透過 StringBuilder 累積結果
        StringBuilder sb = new StringBuilder();
        while (m > 0) {
            sb.append(m % 10);
            m /= 10;
        }

        // 去除前導零（反轉後的左側可能出現 '0'，需移除，直到只剩下一位為止）
        int i = 0;
        while (i < sb.length() && sb.charAt(i) == '0') i++;
        String res = (i == sb.length()) ? "0" : sb.substring(i);

        if (negative) {
            System.out.print("-" + res);
        } else {
            System.out.print(res);
        }
    }
}
