/*
 * 檔案名稱: LeibnizPi.java
 * 功能說明: 使用萊布尼茨級數計算圓周率近似值，輸出
 *          i 為 10000, 20000, ..., 100000 時的 pi 值，格式為：
 *          i = 6位數, pi = 10位小數
 *          另找出最小的 i 使 pi >= 3.14159，並輸出該 i
 * 作者: 你的姓名
 * 日期: 2025-11-12
 */

public class LeibnizPi {
    public static void main(String[] args) {
        // 輸出 i 為 10000, 20000, ..., 100000 對應的 pi 值
        System.out.printf("i = %6d, pi = %.10f%n", 10000, computePiByTerms(10000));
        for (int i = 20000; i <= 100000; i += 10000) {
            System.out.printf("i = %6d, pi = %.10f%n", i, computePiByTerms(i));
        }

        // 找出最小的 i 使 pi >= 3.14159
        int minI = findMinIForPiAtLeast(3.14159);
        System.out.println("i = " + minI);
    }

    // 計算使用 i 項的萊布尼茨級數 pi 值
    private static double computePiByTerms(int terms) {
        double sum = 0.0;
        for (int k = 0; k < terms; k++) {
            double term = ((k % 2 == 0) ? 1.0 : -1.0) / (2.0 * k + 1.0);
            sum += term;
        }
        return 4.0 * sum;
    }

    // 漸進計算直到 pi >= target，回傳最小的項數 i
    private static int findMinIForPiAtLeast(double target) {
        double sum = 0.0;
        int terms = 0;
        while (true) {
            double term = ((terms % 2 == 0) ? 1.0 : -1.0) / (2.0 * terms + 1.0);
            sum += term;
            double pi = 4.0 * sum;
            terms++;
            if (pi >= target) {
                break;
            }
            // 安全上限，避免極端情況造成無窮迴圈
            if (terms > 2000000) {
                break;
            }
        }
        return terms;
    }
}
