/*
 文件名稱: LeibnizPi.java
 功能說明: 使用 Leibniz 公式近似 π，輸出指定 i 的 pi 值（i = 10000, 20000, ..., 100000），
           以及找出最小的 i 使 pi 四捨五入至小數點第5位為 3.14159。
 作者: 你的名字
 日期: 2025-11-12
*/

public class LeibnizPi {
    public static void main(String[] args) {
        // Part (a): 輸出 i = 10000, 20000, ..., 100000 的 pi 值
        for (int i = 10000; i <= 100000; i += 10000) {
            double pi = computePi(i);
            // i 以寬度 6 對齊，pi 以小數點後 10 位輸出
            System.out.println("i = " + String.format("%6d", i) + ", pi = " + String.format("%.10f", pi));
        }

        // 空行分隔兩部分
        System.out.println();

        // Part (b): 找出最小的 i，使 pi 四捨五入至小數點第5位為 3.14159
        double sum = 0.0;
        int k = 0;
        int chosenI = -1;

        // 以逐步累加的方式尋找符合條件的最小 i
        while (true) {
            sum += ((k % 2 == 0) ? 1.0 : -1.0) / (2.0 * k + 1.0);
            double pi = 4.0 * sum;
            int roundedTo5 = (int) Math.round(pi * 100000.0);
            if (roundedTo5 == 314159) { // 四捨五入後等於 3.14159
                chosenI = k + 1; // i 是項數
                break;
            }
            k++;
            // 為避免極端情況可設定上限，例如 if (k > 2000000) break;
        }

        System.out.println("i = " + String.format("%6d", chosenI));
    }

    // 計算 pi 的近似值，給定 terms 表示使用的項數
    private static double computePi(int terms) {
        double sum = 0.0;
        int sign = 1;
        for (int t = 0; t < terms; t++) {
            sum += sign / (2.0 * t + 1.0);
            sign = -sign;
        }
        return 4.0 * sum;
    }
}
