/*
 * 檔案名稱: CSIE_1017_41343252_1.java
 * 功能說明: 將使用者輸入的整數數字反轉，輸出反轉後的整數。支援 0、正負數。
 * 作者: 教師
 * 日期: 2025-11-15
 */

import java.util.Scanner;

public class CSIE_1017_41343252_1 {
    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);
        int n = in.nextInt();
        int sign = n < 0 ? -1 : 1;
        int temp = Math.abs(n);
        int rev = 0;

        // 特殊情況：0 直接輸出 0
        if (temp == 0) {
            System.out.println(0);
        } else {
            while (temp > 0) {
                rev = rev * 10 + (temp % 10);
                temp /= 10;
            }
            System.out.println(rev * sign);
        }

        in.close();
    }
}
