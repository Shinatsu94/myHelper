/*
檔案名稱: LeibnizPiApprox.java
功能: 使用勒布尼茲級數計算 pi，輸出 i=10000,20000,...,100000 的近似值，格式符合題目要求
作者: 設計者
日期: 2025-11-12
說明: 使用累積方式避免重複計算，並以 printf 控制輸出格式
*/
public class LeibnizPiApprox {
    public static void main(String[] args) {
        final int step = 10000;
        final int maxI = 100000;

        double sum = 0.0; // 累積的級數和
        int terms = 0;    // 已計算的項數（對應到 k）

        // 依序輸出 i=10000, 20000, ..., 100000 的 pi 值
        for (int i = step; i <= maxI; i += step) {
            // 增加新項，直到計算到第 i 項
            while (terms < i) {
                terms++;
                double sign = (terms % 2 == 1) ? 1.0 : -1.0;
                sum += sign / (2.0 * terms - 1.0);
            }
            double pi = 4.0 * sum;
            System.out.printf("i = %6d, pi = %.10f%n", i, pi);
        }
    }
}
