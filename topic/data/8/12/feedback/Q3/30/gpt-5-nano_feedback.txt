import java.io.PrintStream;

/*
檔案名稱: LeibnizPi.java
功能: 計算 pi 值 using Leibniz 公式，輸出 i = 10000, 20000, ..., 100000
作者: 你的教授
日期: 2025-11-12
說明: pi = 4 * (1 - 1/3 + 1/5 - 1/7 + ... + (-1)^(i-1) / (2i-1))
輸出格式: i 使用寬度 6，pi 使用小數點後 10 位
*/
public class LeibnizPi {
    public static void main(String[] args) {
        PrintStream out = System.out;

        double sum = 0.0;
        int nextTarget = 10000;
        int idx = 0; // 已計算的項數上限為 idx-1

        for (int i = 10000; i <= 100000; i += 10000) {
            // 增量加入從 idx 到 i-1 的項
            for (int k = idx; k < i; k++) {
                double term = ((k % 2 == 0) ? 1.0 : -1.0) / (2.0 * k + 1.0);
                sum += term;
            }
            idx = i;
            double pi = 4.0 * sum;
            out.printf("i = %6d, pi = %.10f%n", i, pi);
        }
    }
}
