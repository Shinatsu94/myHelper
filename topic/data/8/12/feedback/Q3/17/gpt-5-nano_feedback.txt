/*
檔案名稱: ReverseDigits.java
功能: 將使用者輸入的整數的位數反轉，負數保留符號，零回傳 0。
作者: 王世豪
學號: 41343107
日期: 2025/11/15
*/
import java.util.Scanner;

public class ReverseDigits {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        long n;
        if (scanner.hasNextLong()) {
            n = scanner.nextLong();
        } else {
            scanner.close();
            return;
        }

        long sign = n < 0 ? -1 : 1;
        long m = Math.abs(n);
        long rev = 0;

        while (m > 0) {
            rev = rev * 10 + (m % 10);
            m /= 10;
        }

        rev *= sign;
        System.out.println(rev);
        scanner.close();
    }
}
