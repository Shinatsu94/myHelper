/**
 * 程式名稱 : 數字顛倒
 * 功能說明： 讀入一個整數，輸出其數字顛倒的結果（負數也支援）。
 * 日期: 2025/11/15
 * 姓名: 張竣傑
 * 學號: 41343224
 */
import java.util.Scanner;

public class CSIE_41343224_1017_3 {

    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);

        if (!input.hasNextLong()) {
            return;
        }

        long n = input.nextLong();
        long sign = (n < 0) ? -1 : 1;
        long m = (n < 0) ? -n : n;

        // 特別處理 0 的情況
        if (m == 0) {
            System.out.println(0);
            return;
        }

        long rev = 0;
        while (m > 0) {
            rev = rev * 10 + (m % 10);
            m /= 10;
        }

        System.out.println(sign * rev);
    }
}
