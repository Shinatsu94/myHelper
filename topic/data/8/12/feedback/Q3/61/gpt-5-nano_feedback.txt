import java.util.Scanner;

public class CSIE_41343247_1017_3 {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        long n = scanner.nextLong();

        if (n == 0) {
            System.out.print(0);
            return;
        }

        boolean isNeg = n < 0;
        if (isNeg) {
            System.out.print("-");
            n = -n;
        }

        long rev = 0;
        long m = n;
        while (m > 0) {
            rev = rev * 10 + (m % 10);
            m /= 10;
        }

        System.out.print(rev);
    }
}
