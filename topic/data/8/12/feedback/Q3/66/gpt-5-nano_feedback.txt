/*
 * File: ReverseDigits.java
 * Description: 讀入整數，輸出其數字反轉的結果，零及末尾零會自動去除；支援正負數的反轉。
 * Author: 教師姓名
 * Date: 2025-11-15
 */

import java.util.Scanner;

public class ReverseDigits {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }

        int n = sc.nextInt();
        boolean negative = n < 0;
        int x = Math.abs(n);

        int rev = 0;
        // 將數字逐位反轉
        while (x > 0) {
            rev = rev * 10 + (x % 10);
            x /= 10;
        }

        if (negative) rev = -rev;

        System.out.println(rev);
        sc.close();
    }
}
