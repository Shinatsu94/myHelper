        char[] s = {'|', '/', '-', '\\'};

        long endTime = startTime + 10_000; // 執行時間 10s
        double progress; // 宣告變數儲存目前進度

        while (System.currentTimeMillis() < endTime) {
            progress = (double) (System.currentTimeMillis() - startTime) / (endTime - startTime); // 計算進度百分比
            int percentage = (int) (progress * 100); //轉換成整數
            System.out.printf("\r%c %d%%", s[i % s.length], percentage); // 使用 printf 格式化輸出，棒子的動畫
            System.out.flush();
            Thread.sleep(100); // 暫停 100ms

            i += 1;

        }
        System.out.println();

    }
}
