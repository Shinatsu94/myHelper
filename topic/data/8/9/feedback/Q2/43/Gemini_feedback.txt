```java
public class CSIE_41343221_0926_2 {

    public static void main(String[] args) {
        // 定義旋轉棒子的樣式
        char[] animationChars = {'-', '\\', '|', '/'};

        System.out.println("程式開始運行...");

        int progress = 0; // 進度計數器

        while (true) { // 使用while迴圈來持續旋轉
            // 根據進度選擇旋轉棒子的樣式
            char currentAnimationChar = animationChars[progress % animationChars.length];

            // 格式化輸出，顯示旋轉棒子和進度百分比
            String outputString = String.format("%c %d%%", currentAnimationChar, progress);

            // 使用\r實現覆蓋輸出
            System.out.print("\r" + outputString);
            System.out.flush();

            // 增加進度
            progress++;

            try {
                // 暫停100毫秒，控制旋轉速度
                Thread.sleep(100);
            } catch (InterruptedException e) {
                // 如果中斷異常發生，則退出迴圈
                return;
            }
        }
    }
}
```
