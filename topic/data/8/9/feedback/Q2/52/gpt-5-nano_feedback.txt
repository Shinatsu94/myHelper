/* 檔案名稱: CSIE_41343233_0926_2.java
   作者: 你的名字
   日期: 2025-11-12
   程式說明: 印出「虎科大」與學生姓名的每個字及其對應的內碼(Unicode碼點)，每行輸出「字 空格 內碼」。
*/
public class CSIE_41343233_0926_2 {
    public static void main(String[] args) {
        String first = "虎科大";        // 先印虎科大
        String name  = "林武杰";        // 再印學生中文姓名

        printCharsWithCode(first);
        printCharsWithCode(name);
    }

    // 輸出字與其對應的Unicode碼點，支援Unicode字元（包含非BMP字元的代理對）
    private static void printCharsWithCode(String s) {
        for (int offset = 0; offset < s.length(); ) {
            int codePoint = s.codePointAt(offset);                 // 取得碼點
            String ch = new String(Character.toChars(codePoint));   // 將碼點轉回字元字串
            System.out.println(ch + " " + codePoint);               // 輸出 "字 內碼"
            offset += Character.charCount(codePoint);               // 移動至下一個字元
        }
    }
}
