        char[] spinner = {'|', '/', '-', '\\'};
        int index = 0;
        int percent = 0;

        while (percent <= 100) {
            System.out.print("\r" + spinner[index] + " " + percent + "%");
            index = (index + 1) % spinner.length;
            Thread.sleep(100); // 每 0.1 秒更新

            // 模擬進度增加 (實際應用中應根據實際情況更新)

        System.out.println("\n完成!");
    }
}
