    // 使用 enum 定義旋轉棒子的樣式，增加可讀性
    enum Spinner {
        DASH('-'),
        SLASH('/'),
        PIPE('|'),
        BACKSLASH('\\');

        private final char symbol;

        Spinner(char symbol) {
            this.symbol = symbol;
        }

        public char getSymbol() {
            return symbol;
        }
    }

    public static void main(String args[]) throws Exception {
        //宣告常數，減少Magic Number
        final int SLEEP_TIME = 200;
        final int MAX_PROGRESS = 100;
        Spinner[] spinners = Spinner.values(); // 取得所有 Spinner 樣式

        for (int i = 0; i <= MAX_PROGRESS; i++) {
            // 計算當前應該顯示的旋轉棒子樣式
            Spinner currentSpinner = spinners[i % spinners.length];

            // 輸出旋轉棒子和進度
            System.out.print("\r" + currentSpinner.getSymbol() + "\t" + i + "%");

            Thread.sleep(SLEEP_TIME); // 統一睡眠時間
        }

        System.out.println("\n" + "加載完成!");
    }
}
