            // 清除上一行輸出，並輸出當前旋轉符號和進度

            // 更新旋轉符號的索引，實現旋轉效果

            // 進度增加1
            progress++;

            // 暫停50毫秒，控制動畫速度

                Thread.sleep(50);

            }
        }

        // 迴圈結束，輸出完成提示
        System.out.println("\n完成！");
