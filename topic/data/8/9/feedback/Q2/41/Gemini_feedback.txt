        double progress = 0;
        // icon 字串包含旋轉棒子的各種狀態
        String icon = "/-\\|/-\\|";
        // iconIndex 用於追蹤目前icon字串的索引
        int iconIndex = 0;

        // 當進度小於等於100時，迴圈繼續執行
        while (progress <= 100) {
            // 顯示旋轉棒子的當前狀態和進度百分比

            // 暫停100毫秒，以便於觀察動畫效果

            // 增加進度百分比，每次增加1%
            progress += 1;

            // 更新icon字串的索引，使其在0到7之間循環
            iconIndex = (iconIndex + 1) % 8;
        }
        // 迴圈結束，顯示 "Success!!"
        System.out.print("\rSuccess!!");
    }
}
