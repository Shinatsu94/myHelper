public class CSIE_41343226_0926_2 {
    public static void main(String[] args) throws Exception {
        int n = 100; // 進度總量
        String[] spinner = {"\\", "|", "/", "-"}; // 旋轉棒子的樣式
        int spinnerIndex = 0; // 旋轉棒子的索引

        for (int i = 0; i <= n; i++) {
            System.out.print("\r"); // 回到行首

            int p = i; // 當前進度
            int l = 40; // 進度條長度
            int f = (p * l) / 100; // 已完成的進度條長度

            System.out.print(spinner[spinnerIndex] + " "); // 輸出旋轉棒子

            System.out.print("進度: [");
            for (int j = 0; j < l; j++) {
                if (j < f) {
                    System.out.print("█"); // 已完成部分
                } else {
                    System.out.print("░"); // 未完成部分
                }
            }
            System.out.printf("] %d%% (%d/%d)", p, i, n); // 輸出進度百分比

            System.out.flush(); // 強制輸出

            Thread.sleep(50); // 暫停50毫秒

            spinnerIndex = (spinnerIndex + 1) % spinner.length; // 更新旋轉棒子的索引
        }
        System.out.println("\nＯＫ!"); // 輸出完成訊息
    }
}
