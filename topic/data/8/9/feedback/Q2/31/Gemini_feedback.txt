```java
public class CSIE_41343203_0926_2{
    public static void main(String[] args) throws Exception {
        char[] spinner = {'|', '/', '-', '\\'};
        int progress = 0; // 使用更直觀的變數名稱
        int maxProgress = 100;
        int step = 2;

        while (progress <= maxProgress) { // 添加迴圈停止條件
            int displayProgress = Math.min(progress, maxProgress);
            System.out.print("\r" + spinner[(progress/step) % spinner.length] + " " + displayProgress + "%"); //顯示正確的棒子

            if (progress < maxProgress) {
                progress += step;
            }
            Thread.sleep(100);
        }
        System.out.println("\n完成!"); // 提示完成
    }
}
```
