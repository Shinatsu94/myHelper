```java
public class CSIE_41343131_0926_2 {
    public static void main(String args[]) throws Exception {
        // 模擬刪除 C:/ 的過程，顯示旋轉棒子和進度百分比
        double 任務 = 0;
        System.out.println("confirm to delete C: , Enter \"Y\" or \"N\":\nY\nDeleting whole C:/");
        while (true) {
            char[] 旋轉棒子 = { '\\', '|', '/', '-' };
            for (int 索引 = 0; 索引 < 旋轉棒子.length; 索引++) {
                System.out.printf("\r" + 旋轉棒子[索引] + '\t' + "%.3f" + "%%", 任務);
                try {
                    Thread.sleep(100);
                } catch (InterruptedException e) {
                    // 處理中斷異常
                    e.printStackTrace();
                }
            }
            //任務 += Math.random(); // 原本的隨機進度
            任務 += 0.5; // 改為每次增加0.5，模擬線性進度
            if (任務 > 100.0) {
                任務 = 100.0; // 確保任務不會超過 100
                break;
            }
        }
        System.out.printf("\r-\t%.3f%%\nDelete Completed", 任務); // 使用 printf 確保輸出格式一致
    }
}
```
