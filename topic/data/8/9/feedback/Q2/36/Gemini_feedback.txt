public class CSIE_41343212_0926_2 {
    public static void main(String[] args) throws Exception {
        char [] spinChars = {'/', '-', '\\', '|'};
        int percentage = 0;
        int spinIndex = 0;  // 新增 spinIndex 追蹤 spinChars 的索引

        while (percentage <= 100) { // 修改迴圈條件，避免 break
            System.out.print("\r" + spinChars[spinIndex] + " " +  percentage + "%");
            Thread.sleep(100);

            percentage++; // 在迴圈外部累加進度

            spinIndex++; // 增加 spinIndex
            if (spinIndex >= spinChars.length) {
                spinIndex = 0; // 索引回到開頭，實現循環
            }

        }
    }
}
