```java
public class RotatingBar { // 更直觀的類別名稱
    public static void main(String[] args) throws Exception {
        char[] animationChars = {'|', '/', '-', '\\'}; // 使用陣列儲存旋轉字符
        int animationDelay = 40; // 延遲時間，單位毫秒
        int progress = 0; // 進度百分比
        int animationIndex = 0; // 旋轉字符的索引

        while (progress <= 100) {
            // 使用StringBuilder來構建輸出字串，提高效率
            StringBuilder output = new StringBuilder();
            output.append("\t").append(progress).append("%"); // 進度條顯示
            output.append("\r").append(animationChars[animationIndex]); // 旋轉棒顯示

            System.out.print(output.toString()); // 輸出

            Thread.sleep(animationDelay); // 延遲
            progress++; // 進度條增長
            animationIndex = (animationIndex + 1) % animationChars.length; // 更新旋轉字符的索引，循環使用
        }
    }
}
```
