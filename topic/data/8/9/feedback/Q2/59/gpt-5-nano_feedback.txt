 * 作者: 教師
 * 日期: 2025-11-12

public class PrintChineseInternalCode {

    public static void main(String[] args) {
        // 先定義虎科大，再定學生姓名
        char[] tigerUniversity = {'虎','科','大'};
        char[] studentName = {'林','武','杰'};

        // 逐字輸出：字元與其對應的內碼（Unicode 編碼點，十進位表示）
        printCharsWithCodePoints(tigerUniversity);
        printCharsWithCodePoints(studentName);
    }

    // 輔助方法：輸出每個字元及其編碼點
    private static void printCharsWithCodePoints(char[] chars) {
        for (char c : chars) {
            int codePoint = (int) c; // Unicode 編碼點的十進位表示
            System.out.println(c + " " + codePoint);
        }
    }
}
