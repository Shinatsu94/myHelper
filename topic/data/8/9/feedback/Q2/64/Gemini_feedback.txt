```java
public class CSIE_51015105_0926_2 {
    public static void main(String[] args) throws Exception {
        // 定義旋轉棒子的樣式
        char[] spinner = {'|', '/', '-', '\\'};
        // 初始化進度
        int progress = 0;
        // 迴圈直到進度達到100%
        while (progress <= 100) {
            // 遍歷旋轉棒子的每個樣式
            for (char c : spinner) {
                // 印出旋轉棒子的當前樣式和進度百分比，使用\r回到行首覆蓋之前的輸出
                System.out.print("\r" + c + " " + progress + "%");
                // 延遲 0.1 秒
                Thread.sleep(100);
            }
            // 進度增加
            progress++;
        }
        // 印出完成訊息
        System.out.println("\n完成!");
    }
}
```
