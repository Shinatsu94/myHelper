// NameEncodingPrinter.java
/*
 * 檔案名稱: NameEncodingPrinter.java
 * 功能說明: 印出「虎 科 大 林 武 杰」六字及其對應的 Unicode 內碼（十進位）
 * 作者: 教授
 * 日期: 2025-11-12
 */
public class NameEncodingPrinter {
    public static void main(String[] args) {
        // 先印虎、科、大三個字及其內碼
        printCharWithCode('虎');
        printCharWithCode('科');
        printCharWithCode('大');
        // 再印林、武、杰三個字及其內碼
        printCharWithCode('林');
        printCharWithCode('武');
        printCharWithCode('杰');
    }

    // 輔助方法：輸出「字元 空格 內碼(十進位)」
    private static void printCharWithCode(char ch) {
        System.out.println(ch + " " + (int) ch);
    }
}
