public class CSIE_41343205_0926_2 {
    public static void main(String[] args) throws InterruptedException {
        // 定義旋轉棒的符號
        char[] spinner = {'|', '/', '-', '\\'};
        // 旋轉棒的索引
        int i = 0;
        // 目前的進度
        int prog = 0;
        // 最大進度
        int max = 100;
        // 進度增加的幅度
        int increment = 2; //調整這個變數來控制進度更新頻率，避免hard code

        // 當進度小於 100 時，持續迴圈
        while (prog < max) {
            // 顯示旋轉棒的目前狀態
            System.out.print("\r" + spinner[i % spinner.length] + " " + prog + "%");
            i++;

            // 增加進度
            prog += increment;
            // 確保進度不超過 100
            if (prog > max) {
                prog = max;
            }

            // 暫停 100 毫秒
            Thread.sleep(100);
        }
        System.out.println("\n完成!"); //印出完成，避免最後停留在98%
    }
}
