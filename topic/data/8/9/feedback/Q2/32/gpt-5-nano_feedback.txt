public class CSIE_41343205_0926_2 {

	// 功能: 印出 "虎科大" 與學生中文姓名每個字及其對應的內碼（Unicode 編碼點）
	// 作者: 你可以填入姓名
	// 日期: 2025-11-12
	//
	// 說明: 逐字輸出字元與其 Unicode 編碼點，格式為 "<字元> <編碼點(十進制)>"

	public static void main(String[] args) {
		String header = "虎科大";      // 校名與學校名稱
		String studentName = "林武杰"; // 學生姓名，實作時請改為實際姓名

		// 先輸出頭部字串的每個字及其編碼點
		printCharactersWithCode(header);
		// 再輸出學生姓名的每個字及其編碼點
		printCharactersWithCode(studentName);
	}

	// 輔助方法：逐字輸出字元及其 Unicode 編碼點（十進制表示）
	private static void printCharactersWithCode(String s) {
		for (int i = 0; i < s.length(); i++) {
			char ch = s.charAt(i);
			System.out.println(ch + " " + (int) ch);
		}
	}
}
