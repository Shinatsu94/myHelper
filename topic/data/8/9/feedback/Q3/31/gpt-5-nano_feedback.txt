/*
檔案名稱: CSIE_41343203_0926_3.java
功能說明: 印出虎科大與學生姓名的每個字以及對應的Unicode內碼，格式為 "<字> <內碼>"
作者: 教授
日期: 2025-11-12
*/
public class CSIE_41343203_0926_3 {
    public static void main(String[] args) {
        String institution = "虎科大";
        String studentName = "林杰武";

        // 先輸出虎科大中的每個字及其內碼
        printCharsWithCode(institution);
        // 再輸出學生姓名中的每個字及其內碼
        printCharsWithCode(studentName);
    }

    // 輔助方法：給定字串，逐字輸出 "<字> <內碼>"
    private static void printCharsWithCode(String s) {
        for (int i = 0; i < s.length(); i++) {
            char ch = s.charAt(i);
            int codePoint = (int) ch; // 取得該字的Unicode內碼 (十進位)
            System.out.println(ch + " " + codePoint);
        }
    }
}
