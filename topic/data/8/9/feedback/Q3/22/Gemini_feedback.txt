        Scanner scn = new Scanner(System.in);

                break;
            }

            if(score >= 60){            // 判斷成績
                System.out.println("及格");

                System.out.println("補考");
            } else {
                System.out.println("不及格");
            }
        }
        scn.close(); // 關閉Scanner，釋放資源

}
