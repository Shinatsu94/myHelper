    public static void main(String[] args) {
        // 創建 Scanner 物件以讀取使用者輸入
        Scanner scanner = new Scanner(System.in);

        // 提示使用者輸入成績，並限制輸入範圍

        int score = scanner.nextInt();

        // 判斷成績等級

            System.out.println("及格");

            System.out.println("補考");

        }

        // 關閉 Scanner 物件
        scanner.close();
    }
}
