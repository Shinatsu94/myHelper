public class CSIE_41343128_0926_3 {
    public static void main(String[] args) {
        // 依序印出「虎 科 大 林 武 杰」及其對應的內碼（十進位）
        char[] chars = {'虎', '科', '大', '林', '武', '杰'};
        for (char ch : chars) {
            System.out.println(ch + " " + (int) ch);
        }
    }
}
