修改後完整程式碼
/*
File: PrintUnicodeCodePoints.java
Purpose: 印出「虎科大」與「林武杰」的每個字及其對應的 Unicode 編碼點
Author: 你的姓名
Date: 2025-11-12
*/
public class PrintUnicodeCodePoints {
    public static void main(String[] args) {
        printCharsWithCode("虎科大");
        printCharsWithCode("林武杰");
    }

    private static void printCharsWithCode(String s) {
        for (int i = 0; i < s.length(); i++) {
            char ch = s.charAt(i);
            System.out.println(ch + " " + (int) ch);
        }
    }
}
