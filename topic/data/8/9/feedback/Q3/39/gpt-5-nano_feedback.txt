修改後完整程式碼（TigerNameCodePrinter.java）
/*
 檔案名稱: TigerNameCodePrinter.java
 功能: 印出「虎科大」與學生中文姓名的每個字及其 Unicode 內碼（十進位表示）
 作者: 教師姓名
 日期: 2025-11-12
*/
public class TigerNameCodePrinter {
    public static void main(String[] args) {
        // 先印「虎 科 大」三個字及其 Unicode 內碼
        char[] chars = {'虎','科','大','林','武','杰'};

        for (int i = 0; i < 3; i++) {
            char ch = chars[i];
            System.out.println(ch + " " + (int) ch);
        }

        // 再印學生的中文姓名「林 武 杰」三個字及其 Unicode 內碼
        for (int i = 3; i < chars.length; i++) {
            char ch = chars[i];
            System.out.println(ch + " " + (int) ch);
        }
    }
}
