修改後完整程式碼（符合題意的示例實作）
// File: PrintChineseWithUnicode.java
/*
  文件名稱: PrintChineseWithUnicode.java
  功能: 印出虎科大及學生姓名每個字的 Unicode 內碼
  作者: 教師姓名
  日期: 2025-11-12
*/
public class PrintChineseWithUnicode {
    public static void main(String[] args) {
        // 先印出「虎科大」每個字及其 Unicode 碼點
        printCharWithCode('虎');
        printCharWithCode('科');
        printCharWithCode('大');

        // 學生姓名（示範用，實際可改為動態輸入或替換為該生姓名）
        String name = "林武杰";
        for (int i = 0; i < name.length(); i++) {
            char ch = name.charAt(i);
            System.out.println(ch + " " + (int) ch);
        }
    }

    private static void printCharWithCode(char ch) {
        System.out.println(ch + " " + (int) ch);
    }
}
