```java
import java.util.Scanner;

public class CSIE_41343111_0926_3 {
    public static void main(String[] args) {
        // 建立 Scanner 物件，用於讀取使用者輸入
        Scanner scanner = new Scanner(System.in);

        // 迴圈讀取使用者輸入的分數，直到沒有更多輸入
        while (scanner.hasNextInt()) {
            int num = scanner.nextInt(); // 讀取使用者輸入的分數

            // 判斷分數等級
            if (num >= 60) {
                System.out.println("及格"); // 分數大於等於 60，輸出 "及格"
            } else if (num >= 50) {
                System.out.println("補考"); // 分數大於等於 50 且小於 60，輸出 "補考"
            } else {
                System.out.println("不及格"); // 分數小於 50，輸出 "不及格"
            }
        }

        // 關閉 Scanner 物件，釋放資源
        scanner.close();
    }
}
```
