/************************************************************
 * File: TigerCodePrinter.java
 * 功能: 依序輸出「虎科大」三字及學生姓名「林武杰」三字，並列出每個字對應的內碼（Unicode碼點，十進位）。
 * 作者: 教學用範例
 * 日期: 2025-11-12
 * 說明: 本程式直接輸出固定內容，符合題意的輸出格式。
 ************************************************************/
public class TigerCodePrinter {
    public static void main(String[] args) {
        // 虎科大
        char[] first = {'虎', '科', '大'};
        // 學生姓名林武杰
        char[] last = {'林', '武', '杰'};

        // 先印虎科大三字及對應內碼
        for (char ch : first) {
            System.out.println(ch + " " + (int) ch);
        }

        // 再印學生中文姓名三字及對應內碼
        for (char ch : last) {
            System.out.println(ch + " " + (int) ch);
        }
    }
}
