```java
import java.util.Scanner;

public class ScoreGrading { //修改程式名稱

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        try { //加入try catch finally
            // 讀取使用者輸入的分數，根據分數輸出此分數為：及格、補考、不及格
            while (sc.hasNextInt()) {
                int score = sc.nextInt();

                if (score >= 60) {
                    System.out.println("及格");
                } else if (score >= 50) {
                    System.out.println("補考");
                } else {
                    System.out.println("不及格");
                }
            }
        } catch (Exception e) {
            System.out.println("輸入錯誤，請輸入整數分數。");
        } finally {
            sc.close();
        }
    }
}
```
