```java
import java.util.Scanner;

/**
 * 檔案名稱: Main.java
 * 功能: 印出虎科大及學生中文姓名的內碼。輸入六個整數，分別對應
 *       虎、科、大、林、武、杰六個字的內碼，逐行印出字與內碼。
 * 作者: 教師
 * 日期: 2025-11-12
 */

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 順序：虎、科、大、林、武、杰 六個字
        String[] chars = {"虎", "科", "大", "林", "武", "杰"};
        int[] codes = new int[6];

        // 讀取六個內碼
        for (int i = 0; i < codes.length; i++) {
            if (sc.hasNextInt()) {
                codes[i] = sc.nextInt();
            } else {
                // 若輸入不足，預設為 0，避免例外
                codes[i] = 0;
            }
        }

        // 逐字輸出「字 空格 內碼」
        for (int i = 0; i < chars.length; i++) {
            System.out.println(chars[i] + " " + codes[i]);
        }

        sc.close();
    }
}
```
