```java
import java.util.Scanner;

public class ScoreAnalyzer {  // 修改類別名稱

    public static void main(String[] args) {
        // 使用 try-with-resources 確保 Scanner 物件在使用完後關閉
        try (Scanner sc = new Scanner(System.in)) {
            System.out.print("請輸入要分析的分數數量："); // 提示使用者輸入分數數量
            int numScores = sc.nextInt();

            // 讀取並分析指定數量的分數
            for (int i = 0; i < numScores; i++) {
                System.out.print("請輸入第 " + (i + 1) + " 個分數：");
                int score = sc.nextInt();

                if (score >= 60) {
                    System.out.println("及格");
                } else if (score >= 50) {
                    System.out.println("補考");
                } else {
                    System.out.println("不及格");
                }
            }
        } catch (Exception e) {
            System.out.println("輸入錯誤：" + e.getMessage());
        }
    }
}
```
