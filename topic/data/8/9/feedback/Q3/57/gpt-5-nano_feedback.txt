  // 功能: 印出「虎科大」與「林武杰」三字的對應內碼，格式為「字 XXX」。
  // 作者: 教授
  // 日期: 2025-11-12

  public class CSIE_41343239_0926_4 {
      public static void main(String[] args) {
          // 各字與其對應的 Unicode 內碼（十進位表示）
          String[] chars = {"虎", "科", "大", "林", "武", "杰"};
          int[] codes = {34350, 31185, 22823, 26519, 27494, 26480};

          for (int i = 0; i < chars.length; i++) {
              System.out.println(chars[i] + " " + codes[i]);
          }
      }
  }
