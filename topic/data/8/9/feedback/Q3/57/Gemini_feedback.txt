public class CSIE_41343239_0926_3 {
    public static void main(String[] args) {
        // 本程式讀取使用者輸入的分數，並根據分數輸出 "及格"、"補考" 或 "不及格"。
        // 分數 >= 60 : 及格
        // 60 > 分數 >= 50 : 補考
        // 50 > 分數 : 不及格
        // 使用者可輸入多個分數，每輸入一個分數後按Enter。輸入非數字結束程式。

        Scanner sca = new Scanner(System.in);

        System.out.println("請輸入你的成績 (輸入非數字結束):"); // 修改提示訊息
        while (sca.hasNextInt()) { // 使用迴圈處理多個輸入
            int score = sca.nextInt();

            if (score >= 60) {
                System.out.println("及格");
            } else if (score >= 50) {
                System.out.println("補考");
            } else {
                System.out.println("不及格");
            }
        }

        System.out.println("程式結束"); // 提示程式結束
        sca.close(); // 關閉 Scanner
    }
}
