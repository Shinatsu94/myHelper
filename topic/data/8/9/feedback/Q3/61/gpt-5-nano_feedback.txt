    public static void main(String[] args) {
        printCharsWithCode("虎科大");
        printCharsWithCode("林武杰");
    }

    // 輸出字元及其 Unicode 編碼（十進位）
    private static void printCharsWithCode(String s) {
        for (int i = 0; i < s.length(); i++) {
            char ch = s.charAt(i);
            int code = (int) ch; // 取得該字的內碼（十進位）
            System.out.println(ch + " " + code);
        }
    }
}
