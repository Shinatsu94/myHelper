作者: 你
日期: 2025-11-12

public class OutputNameCode {
    public static void main(String[] args) {
        // 虎科大三字
        char[] group = { '虎', '科', '大' };
        // 學生姓名三字：林 武 杰
        char[] student = { '林', '武', '杰' };

        // 印出虎科大三字及對應內碼
        for (char ch : group) {
            System.out.println(ch + " " + (int) ch);
        }

        // 印出學生姓名三字及對應內碼
        for (char ch : student) {
            System.out.println(ch + " " + (int) ch);
        }
    }
}
