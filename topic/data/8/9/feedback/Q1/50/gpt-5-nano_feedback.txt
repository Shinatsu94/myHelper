修改後完整程式碼（已符合題意與格式要求）
/*
 * 文件名稱: ChrOne.java
 * 功能: 印出虎科大及學生姓名的中文字與對應的 Unicode 編碼
 * 作者: 你的學生
 * 日期: 2025-11-12
 * 版本: 1.0
 */
public class ChrOne {
    public static void main(String[] args) {
        char school1 = '虎';
        char school2 = '科';
        char school3 = '大';

        // 學生中文姓名三個字，此處示例為林武杰
        char name1 = '林';
        char name2 = '武';
        char name3 = '杰';

        // 逐字輸出：字元 + 空白 + 對應的內碼
        System.out.println("虎 " + (int) school1);
        System.out.println("科 " + (int) school2);
        System.out.println("大 " + (int) school3);
        System.out.println("林 " + (int) name1);
        System.out.println("武 " + (int) name2);
        System.out.println("杰 " + (int) name3);
    }
}
