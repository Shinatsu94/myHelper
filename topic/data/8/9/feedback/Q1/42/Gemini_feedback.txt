```java
public class CSIE_41343220_0926_1 {
    public static void main(String[] args) {
        // 學校名稱與學生姓名
        String schoolName = "虎科大";
        String studentName = "洪楷珉";

        // 先印出學校名稱
        printCharacterAndCode(schoolName);

        // 再印出學生姓名
        printCharacterAndCode(studentName);
    }

    // 輔助函式：印出字串中每個字元及其內碼
    public static void printCharacterAndCode(String text) {
        for (int i = 0; i < text.length(); i++) {
            char character = text.charAt(i);
            System.out.println(character + " " + (int)character); // (int) 將char轉為int，顯示內碼
        }
    }
}
```
