        // 先印出虎科大三字及其對應的內碼
        printCharCode("虎", '虎');
        printCharCode("科", '科');
        printCharCode("大", '大');
        // 再印出學生姓名三字及其對應的內碼
        printCharCode("林", '林');
        printCharCode("武", '武');
        printCharCode("杰", '杰');
    }

    // 輔助方法：輸出指定字與其內碼，方便未來擴充

        System.out.println(chName + " " + (int) ch);
