/**
 * 文件名稱: CSIE_41343203_0926_1.java
 * 功能說明: 印出「虎科大」與學生中文姓名的每個字及對應的 Unicode 內碼。
 * 作者: 教師
 * 日期: 2025-11-12
 */

public class CSIE_41343203_0926_1 {
    public static void main(String[] args) {
        // 虎科大三個字
        char[] school = {'虎', '科', '大'};
        // 學生姓名三個字，依題意設定為「林武杰」
        char[] student = {'林', '武', '杰'};

        // 先印虎科大，再印學生姓名，每行一個字與其內碼
        for (char c : school) {
            System.out.println(c + " " + (int) c);
        }
        for (char c : student) {
            System.out.println(c + " " + (int) c);
        }
    }
}
