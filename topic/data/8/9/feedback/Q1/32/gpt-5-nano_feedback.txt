/*
 * File: PrintChineseCodes.java
 * Description: 依序輸出「虎科大」三字及對應的內碼，接著輸出學生姓名三字及對應的內碼。
 * Author: [你的姓名]
 * Date: 2025-11-12
 */

public class PrintChineseCodes {
    public static void main(String[] args) {
        // 虎科大三字
        char[] school = {'虎', '科', '大'};
        // 學生姓名三字，請依實際姓名修改此處
        char[] name = {'林', '武', '杰'};

        // 先輸出虎科大三字與對應的內碼
        for (char ch : school) {
            System.out.println(ch + " " + (int) ch);
        }

        // 再輸出學生姓名三字與對應的內碼
        for (char ch : name) {
            System.out.println(ch + " " + (int) ch);
        }
    }
}
