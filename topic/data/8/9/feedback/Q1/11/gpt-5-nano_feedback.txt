    // 將字元轉為16位元二進位字串，並輸出 "字元 空格 內碼"

        String binary = Integer.toBinaryString(c);
        // 補滿至16位，若不足以16位，左側補0
        binary = String.format("%16s", binary).replace(' ', '0');
        System.out.println(c + " " + binary);
    }
