**3. 改進的程式碼**

```java
public class CSIE_41343239_0926_1 {
    public static void main(String[] args) {
        // 程式目的：印出「虎科大」與「黃致翔」的中文姓名與對應的內碼。

        String schoolName = "虎科大";
        String studentName = "黃致翔";

        // 先印出虎科大
        printNameAndCode(schoolName);

        // 再印出學生姓名
        printNameAndCode(studentName);

    }

    //將文字拆解為字元並印出
    public static void printNameAndCode(String name) {
        for (int i = 0; i < name.length(); i++) {
            char character = name.charAt(i);
            System.out.println(character + " " + (int) character);
        }
    }
}
```
