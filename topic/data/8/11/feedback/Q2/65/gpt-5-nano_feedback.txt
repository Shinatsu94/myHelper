功能: 輸入三個整數，輸出最小值、中間值與最大值；使用兩種不同的方法計算最大、最小值，並以中位數公式計算中間值；要求不得使用 if-else。

import java.util.Scanner;

public class CSIE_51015108_1003_2 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int first = sc.nextInt();
        int second = sc.nextInt();
        int third = sc.nextInt();

        int max = maxOfThree(first, second, third);
        int min = minOfThree(first, second, third);
        int mid = first + second + third - max - min;

        System.out.println(min);
        System.out.println(mid);
        System.out.println(max);

        sc.close();
    }

    // 使用三元運算子計算三個數的最大值
    private static int maxOfThree(int a, int b, int c) {
        return (a > b ? (a > c ? a : c) : (b > c ? b : c));
    }

    // 使用三元運算子計算三個數的最小值
    private static int minOfThree(int a, int b, int c) {
        return (a < b ? (a < c ? a : c) : (b < c ? b : c));
    }
}
