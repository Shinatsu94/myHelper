/*
 * File: MaxMidMinThreeNumbers.java
 * Description: 讀取三個整數，使用兩種不同的方式求出最大值、最小值與中間值(mid)。
 *              方式1：使用三元運算子直接求最大值與最小值。
 *              方式2：中間值用 mid = a + b + c - max - min 的公式求出。
 * Author: 你的名字
 * Date: 2025-11-15
 */

import java.util.Scanner;

public class MaxMidMinThreeNumbers {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int a = sc.nextInt();
        int b = sc.nextInt();
        int c = sc.nextInt();

        // 方式1：使用三元運算子直接求最大值與最小值
        int max = (a > b && a > c) ? a : ((b > c) ? b : c);
        int min = (a < b && a < c) ? a : ((b < c) ? b : c);

        // 方式2：用中間值公式 mid = a + b + c - max - min
        int mid = a + b + c - max - min;

        // 輸出：最小值、中間值、最大值，各自換行
        System.out.println(min);
        System.out.println(mid);
        System.out.println(max);

        sc.close();
    }
}
