import java.util.Scanner;

/*
 * 檔名: CSIE_41343215_1003_2.java
 * 功能: 輸入三個整數，輸出最小值、介於兩者之間的值(mid)、最大值。使用兩個方法分別計算最大與最小，皆採用三元運算子。
 * 作者: 你的姓名
 * 日期: 2025-11-15
 * 備註: 不使用 if-else，並以三元運算子實作最大/最小值的判定。
 */

public class CSIE_41343215_1003_2 {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int a = scanner.nextInt();
        int b = scanner.nextInt();
        int c = scanner.nextInt();

        int maxVal = maxOfThree(a, b, c);
        int minVal = minOfThree(a, b, c);
        int midVal = (a + b + c) - maxVal - minVal;

        System.out.println(minVal);
        System.out.println(midVal);
        System.out.println(maxVal);
        // scanner.close(); // 保留是否關閉 Scanner 取決於執行環境
    }

    // 使用三元運算子找出三個數中的最大值
    private static int maxOfThree(int x, int y, int z) {
        return x > y ? (x > z ? x : z) : (y > z ? y : z);
    }

    // 使用三元運算子找出三個數中的最小值
    private static int minOfThree(int x, int y, int z) {
        return x < y ? (x < z ? x : z) : (y < z ? y : z);
    }
}
