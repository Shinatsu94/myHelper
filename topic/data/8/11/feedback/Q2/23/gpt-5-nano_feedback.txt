import java.util.Scanner;

public class js13 {
    public static void main(String[] args) {
        Scanner s = new Scanner(System.in);
        int A = s.nextInt();
        int B = s.nextInt();

        // 輸出原值
        System.out.println(A);
        System.out.println(B);

        // 不使用額外變數進行交換（位元異或交換法）
        A = A ^ B;
        B = A ^ B;
        A = A ^ B;

        // 輸出交換後的值
        System.out.println(A);
        System.out.println(B);
        s.close();
    }
}
