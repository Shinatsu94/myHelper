/*
 檔案名稱：Main.java
 功能：讀取三個整數，輸出它們的最小值、中間值、最大值。使用兩種不同的方式求解：
       1) 使用排序（Arrays.sort）不使用 if-else。
       2) 使用三元運算子找最大值與最小值，與中間值的計算式 sum - max - min。
 作者：您的教師助理
 日期：YYYY-MM-DD
*/
import java.util.Arrays;
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int num1 = scanner.nextInt();
        int num2 = scanner.nextInt();
        int num3 = scanner.nextInt();
        scanner.close();

        // 方法一：使用排序找出最小、中間、最大值（不使用 if-else）
        int[] arr = new int[]{num1, num2, num3};
        Arrays.sort(arr);
        int min = arr[0];
        int mid = arr[1];
        int max = arr[2];

        // 方法二：使用三元運算子找出最大值、最小值，再以總和減去最大與最小得到中間值
        int max_b = (num1 > num2) ? ((num1 > num3) ? num1 : num3) : ((num2 > num3) ? num2 : num3);
        int min_b = (num1 < num2) ? ((num1 < num3) ? num1 : num3) : ((num2 < num3) ? num2 : num3);
        int mid_b = num1 + num2 + num3 - max_b - min_b;

        // 輸出格式：最小值、中間值、最大值，各新行輸出
        System.out.println(min);
        System.out.println(mid);
        System.out.println(max);

        // 註：上述兩種方法皆可正確求得結果，實作上保留方法二以符合「使用三元運算子找最大與最小值」的要求。
    }
}
