/*
檔案名稱: CSIE_41343128_1003_2.java
功能: 找出三個整數的最大值、最小值與中間值。使用兩種不同的方法計算最大值與最小值，且不使用 if-else。
作者: 陳氏學生
日期: 2025-11-15
說明: 輸入三個整數，輸出最小值、中間值、最大值，各占一行。
*/
import java.util.Scanner;

public class CSIE_41343128_1003_2 {
    public static void main(String[] args) {
        int a, b, c;
        int max, min, mid;

        Scanner scn = new Scanner(System.in);
        a = scn.nextInt();
        b = scn.nextInt();
        c = scn.nextInt();

        // 使用兩種不同的方法求最大與最小值，避免使用 if-else
        max = Math.max(a, Math.max(b, c)); // 方法1：Math.max
        min = Math.min(a, Math.min(b, c)); // 方法2：Math.min

        // 中間值 = a + b + c - max - min
        mid = a + b + c - max - min;

        System.out.println(min);
        System.out.println(mid);
        System.out.println(max);

        scn.close();
    }
}
