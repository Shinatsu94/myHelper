/*
 * 檔案名稱: Main.java
 * 功能: 讀入三個整數，輸出最小值、中間值、最大值（以兩種方法求解）
 * 作者: 請自行填寫
 * 日期: 請自行填寫
 * 說明: 使用三元運算子求最大與最小，並以三數之和減去最大與最小得到中位數
 */

import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int num1 = sc.nextInt();
        int num2 = sc.nextInt();
        int num3 = sc.nextInt();

        // 方法1：使用三元運算子求最大值，考慮重複值情況
        int max = (num1 >= num2 && num1 >= num3) ? num1
                : (num2 >= num1 && num2 >= num3) ? num2 : num3;

        // 方法2：使用三元運算子求最小值，考慮重複值情況
        int min = (num1 <= num2 && num1 <= num3) ? num1
                : (num2 <= num1 && num2 <= num3) ? num2 : num3;

        // 中位數：三數總和減去最大值與最小值
        int mid = num1 + num2 + num3 - max - min;

        System.out.printf("%d\n%d\n%d\n", min, mid, max);

        sc.close();
    }
}
