/*
檔案名稱: CSIE_41343220_1003_2.java
功能: 從使用者輸入的三個整數中，輸出最小值、中間值、最大值。使用兩種不同的方法：
      1) 以三元運算子分別找出最大值與最小值。
      2) 以 mid = x + y + z - max - min 的方式求中間值。
作者: 教師姓名
日期: 2025-11-15
說明: 不使用 if-else 條件敘述。
*/
import java.util.Scanner;

public class CSIE_41343220_1003_2 {
    public static void main(String[] args) {
        Scanner scn = new Scanner(System.in);

        int x = scn.nextInt();
        int y = scn.nextInt();
        int z = scn.nextInt();

        // 使用兩種不同的方式找出最大值與最小值
        int max = (x > y) ? x : y;
        max = (max > z) ? max : z;

        int min = (x < y) ? x : y;
        min = (min < z) ? min : z;

        // 中間值透過總和減去最大值與最小值得到
        int mid = x + y + z - max - min;

        System.out.println(min);
        System.out.println(mid);
        System.out.println(max);

        scn.close();
    }
}
