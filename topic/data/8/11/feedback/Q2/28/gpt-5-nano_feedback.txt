// 檔案名稱: CSIE_41343143_1003_2.java
// 功能說明: 讀取三個整數，使用三元運算子求最大與最小值，並以和減去最大與最小得到中間值，輸出順序為 min, mid, max。
// 作者: 資訊工程系教授
// 日期: 2025-11-15

import java.util.Scanner;

public class CSIE_41343143_1003_2 {
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);

		int num1 = sc.nextInt();
		int num2 = sc.nextInt();
		int num3 = sc.nextInt();

		// 最大值：使用兩層三元運算子
		int max = (num1 > num2) ? ((num1 > num3) ? num1 : num3)
		                        : ((num2 > num3) ? num2 : num3);

		// 最小值：使用兩層三元運算子
		int min = (num1 < num2) ? ((num1 < num3) ? num1 : num3)
		                        : ((num2 < num3) ? num2 : num3);

		// 中間值：總和減去最大值與最小值
		int mid = num1 + num2 + num3 - max - min;

		System.out.println(min);
		System.out.println(mid);
		System.out.println(max);

		sc.close();
	}
}
