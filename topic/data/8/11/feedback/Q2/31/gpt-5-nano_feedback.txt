import java.util.Scanner;

/**
 * CSIE_41343203_1003_2
 * 功能：讀入三個整數，使用兩種不同的方式找出最大值、最小值、以及中間值，且不使用 if-else。
 * 作者：改進示例
 * 日期：2025-11-15
 */
public class CSIE_41343203_1003_2 {
    public static void main(String[] args) {
        // 使用 try-with-resources 自動關閉 Scanner
        try (Scanner sc = new Scanner(System.in)) {
            long a = sc.nextLong();
            long b = sc.nextLong();
            long c = sc.nextLong();

            // 找最大值（使用兩次三元運算子）
            long max = (a > b) ? a : b;
            max = (c > max) ? c : max;

            // 找最小值（使用兩次三元運算子）
            long min = (a < b) ? a : b;
            min = (c < min) ? c : min;

            // 中間值等於 sum - max - min
            long mid = a + b + c - max - min;

            System.out.println(min);
            System.out.println(mid);
            System.out.println(max);
        }
    }
}
