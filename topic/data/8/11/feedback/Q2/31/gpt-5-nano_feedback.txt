  public class SwapTwoNumbers {
      public static void main(String[] args) {
          // 使用 Scanner 讀取使用者輸入
          java.util.Scanner sc = new java.util.Scanner(System.in);
          int A = sc.nextInt();
          int B = sc.nextInt();

          System.out.println(B);

          // 不使用額外變數，僅用 A、B 進行交換

          B = A - B;

          System.out.println(B);

          sc.close();
      }
