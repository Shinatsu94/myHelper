/* 檔案名稱: csie_41343233_1003_2.java

import java.util.Scanner;

public class csie_41343233_1003_2 {
    public static void main(String[] args) {
        Scanner scn = new Scanner(System.in);

        int a = scn.nextInt();
        int b = scn.nextInt();
        int c = scn.nextInt();

        // 最大值的計算（使用兩個三元運算子）
        int max = (a > b) ? a : b;
        max = (c > max) ? c : max;

        // 最小值的計算（使用兩個三元運算子）
        int min = (a < b) ? a : b;
        min = (c < min) ? c : min;

        // 中位數的計算：透過總和減去最大值與最小值
        long sum = (long) a + b + c;
        int mid = (int) (sum - max - min);

        // 依序輸出：最小值、中位值、最大值，每行一個
        System.out.println(min);
        System.out.println(mid);
        System.out.println(max);
    }
}
