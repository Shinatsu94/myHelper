/*
 * 檔案名稱: CSIE_41343112_1003_2.java
 * 功能說明: 輸入三個整數，使用兩種方法（以三元運算子分別計算最大值與最小值，並以總和減去最大與最小得到中間值）輸出最小值、中間值、最大值。
 * 作者: 你的名字
 * 日期: 2025-11-15
 */

import java.util.Scanner;

public class CSIE_41343112_1003_2 {
    public static void main(String[] args) {
        try (Scanner sc = new Scanner(System.in)) {
            int a = sc.nextInt();
            int b = sc.nextInt();
            int c = sc.nextInt();

            // 最大值與最小值使用三元運算子
            int max = (a > b ? (a > c ? a : c) : (b > c ? b : c));
            int min = (a < b ? (a < c ? a : c) : (b < c ? b : c));

            // 中間值透過總和減去最大與最小得到
            int mid = a + b + c - max - min;

            // 輸出：先最小值、再中間值、最後最大值，每行一個
            System.out.println(min);
            System.out.println(mid);
            System.out.println(max);
        }
    }
}
