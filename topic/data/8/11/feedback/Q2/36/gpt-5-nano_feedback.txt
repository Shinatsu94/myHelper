/*
 * File name: CSIE_41343212_1003_2.java
 * Description: 讀入三個整數，使用兩種不同的方法求出最小值、中間值、最大值（不使用 if-else）
 * Author: 李品劭
 * ID: 41343212
 * Date: 2025-11-15
 * Notes: 使用兩個三元運算子分別求 min 和 max，mid 使用總和減去 max 與 min
 */

import java.util.Scanner;

public class CSIE_41343212_1003_2 {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);

        int a = input.nextInt();
        int b = input.nextInt();
        int c = input.nextInt();

        // 使用條件運算子（ternary operator）求最小值
        int min = (a < b) ? ((a < c) ? a : c) : ((b < c) ? b : c);

        // 使用條件運算子（ternary operator）求最大值
        int max = (a > b) ? ((a > c) ? a : c) : ((b > c) ? b : c);

        // 中位數（Mid）為總和減去最大與最小
        int mid = a + b + c - max - min;

        System.out.println(min);
        System.out.println(mid);
        System.out.println(max);

        input.close();
    }
}
