import java.util.Scanner;

public class CSIE_41343212_1003_2 {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        int A = input.nextInt();
        int B = input.nextInt();

        // 先印出原值
        System.out.println(A);
        System.out.println(B);

        // 不使用額外變數，透過加減法交換 A 與 B

        B = A - B;
        A = A - B;

        // 印出交換後的值
        System.out.println(A);
        System.out.println(B);

        input.close();
    }
}
