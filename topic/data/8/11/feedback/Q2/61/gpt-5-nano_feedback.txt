/*
檔案名稱: CSIE_41343247_1003_2.java
功能說明: 讀取三個整數，輸出最小值、中間值與最大值。不使用 if-else，使用兩種方法找出最大與最小值，並以三者和計算中位數。
作者: 盧冠宇
日期: 2025-11-15
*/
import java.util.Scanner;

public class CSIE_41343247_1003_2 {
    public static void main(String[] args) {
        int a, b, c;
        int max, min, mid;

        Scanner scanner = new Scanner(System.in);

        a = scanner.nextInt();
        b = scanner.nextInt();
        c = scanner.nextInt();

        // 最大值：使用巢狀三元運算子
        max = (a > b) ? (a > c ? a : (b > c ? b : c)) : (b > c ? b : c);

        // 最小值：使用巢狀三元運算子
        min = (a < b) ? (a < c ? a : (b < c ? b : c)) : (b < c ? b : c);

        // 中間值（中位數）：避免 if-else，利用 sum - max - min
        long sum = (long) a + b + c;
        mid = (int) (sum - max - min);

        System.out.println(min);
        System.out.println(mid);
        System.out.println(max);

        scanner.close();
    }
}
