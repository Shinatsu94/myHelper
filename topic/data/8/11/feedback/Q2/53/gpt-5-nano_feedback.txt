/*
File: MaxMidMin.java
Description: 讀入三個整數，使用兩種不同的方式找出最大值、最小值與中間值，並輸出順序為 min, mid, max。
Author: Your Name
Date: 2025-11-15
*/

import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int a = sc.nextInt();
        int b = sc.nextInt();
        int c = sc.nextInt();

        // 使用三元運算子求最大值與最小值
        int max = (a > b && a > c) ? a : ((b > c) ? b : c);
        int min = (a < b && a < c) ? a : ((b < c) ? b : c);

        // 中間值透過總和減去最大值與最小值得到
        int mid = a + b + c - max - min;

        // 輸出：最小值、中間值、最大值，各自換行
        System.out.println(min);
        System.out.println(mid);
        System.out.println(max);

        sc.close();
    }
}
