/*
檔案名稱: ThreeNumbersMaxMidMin.java
功能: 讀入三個整數，計算最大值、最小值與中間值（中間值 = a + b + c - max - min），
      使用兩種不同的方式透過三元運算子找出最大值與最小值，避免使用 if-else。
作者: 教授
日期: 2025-11-15
*/
import java.util.Scanner;

public class ThreeNumbersMaxMidMin {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        long a = scanner.nextLong();
        long b = scanner.nextLong();
        long c = scanner.nextLong();

        // 最大值 max（使用三元運算子）
        long max = (a > b) ? ((a > c) ? a : c) : ((b > c) ? b : c);

        // 最小值 min（使用三元運算子）
        long min = (a < b) ? ((a < c) ? a : c) : ((b < c) ? b : c);

        // 中間值 mid，利用總和減去最大與最小值
        long mid = a + b + c - max - min;

        // 輸出結果：先最小值、再中間值、最後最大值
        System.out.println(min);
        System.out.println(mid);
        System.out.println(max);

        scanner.close();
    }
}
