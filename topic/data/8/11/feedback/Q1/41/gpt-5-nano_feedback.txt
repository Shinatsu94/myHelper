import java.util.Scanner;

public class CSIE_41343219_1003_1 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取兩個整數
        int a = sc.nextInt();
        int b = sc.nextInt();
        sc.close();

        // 印出原始值
        System.out.println(a);
        System.out.println(b);

        // 使用不使用額外變數的三步驟 XOR 交換
        a = a ^ b;
        b = a ^ b;
        a = a ^ b;

        // 印出交換後的值
        System.out.println(a);
        System.out.println(b);
    }
