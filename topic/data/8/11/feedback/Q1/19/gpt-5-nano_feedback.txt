import java.util.Scanner;

public class CSIE_41343111_1003_1 {
    public static void main(String[] args) {
        Scanner s = new Scanner(System.in);
        int A = s.nextInt();
        int B = s.nextInt();

        // 使用位元異或交換，避免使用第三變數
        A = A ^ B;
        B = A ^ B;
        A = A ^ B;

        System.out.println(A);
        System.out.println(B);

        s.close();
    }
}
