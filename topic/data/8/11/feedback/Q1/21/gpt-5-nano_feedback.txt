import java.util.Scanner;

public class CSIE_41343119_1003_1 {
  public static void main(String[] args) {
    try (Scanner scn = new Scanner(System.in)) {
      int a, b;
      a = scn.nextInt();
      b = scn.nextInt();

      // 使用 XOR 交換法，避免可能的整數溢位
      a = a ^ b;
      b = a ^ b;
      a = a ^ b;

      System.out.println(a);
      System.out.println(b);
    }
  }
}
