/*
 * 檔案名稱: CSIE_41343216_1003_1.java
 * 功能說明: 讀入兩個整數 A、B，交換後分別印出新值（不使用額外變數，使用位元交換法）。
 * 作者: 資訊工程系教授
 * 日期: 2025-11-15
 */
import java.util.Scanner;

public class CSIE_41343216_1003_1 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int A = sc.nextInt();
        int B = sc.nextInt();

        // 使用位元交換法，避免加法導致的溢位，且不使用額外變數
        A = A ^ B;
        B = A ^ B;
        A = A ^ B;

        System.out.println(A);
        System.out.println(B);

        sc.close();
    }
}
