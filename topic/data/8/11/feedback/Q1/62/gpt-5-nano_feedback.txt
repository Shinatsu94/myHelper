import java.util.Scanner;

public class SwapTwoNumbers {
    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);
        int A = in.nextInt();
        int B = in.nextInt();

        // 不使用額外變數的交換方式
        A = A + B;
        B = A - B;
        A = A - B;

        System.out.println(A);
        System.out.println(B);
    }
}
