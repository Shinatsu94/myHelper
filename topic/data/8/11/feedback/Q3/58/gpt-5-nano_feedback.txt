        int A = sc.nextInt();
        int B = sc.nextInt();

        // 印出初始值
        System.out.println(A);
        System.out.println(B);

        // 不使用額外變數，使用 XOR 交換法

        B = A ^ B;
        A = A ^ B;

        // 印出交換後的值

        System.out.println(B);

        sc.close();
    }
}
