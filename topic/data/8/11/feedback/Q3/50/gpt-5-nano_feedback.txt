import java.util.Scanner;

/*
 * File: CSIE_41343231_1003_3.java
 * 功能: 輸入一個數字 A，找出 1 到 100000 之間，所有數字的各位數總和等於 A 的數量。
 * 作者: 你的名字
 * 日期: 2025-11-15
 * 說明: 使用動態規劃式計算每個數字的位數總和，避免重複運算，並輸出符合條件的數字個數。
 */

public class CSIE_41343231_1003_3 {
    public static void main(String[] args) {
        int targetSum;
        try (Scanner sc = new Scanner(System.in)) {
            targetSum = sc.nextInt();
        }

        int count = 0;
        int[] digitSum = new int[100001]; // 0..100000

        // 逐一計算每個數字的位數總和，並直接比對
        for (int i = 1; i <= 100000; i++) {
            digitSum[i] = digitSum[i / 10] + (i % 10);
            if (digitSum[i] == targetSum) {
                count++;
            }
        }

        System.out.println(count);
    }
}
