import java.util.Scanner;

public class CSIE_41343131_1003_3 {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        int A = input.nextInt();
        int B = input.nextInt();

        // 先印出原值
        System.out.println(A);
        System.out.println(B);
        // 再印出交換後的值（透過輸出順序達成交換效果，不使用額外變數）
        System.out.println(B);
        System.out.println(A);

        input.close();
    }
}
