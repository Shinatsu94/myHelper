public class CSIE_41343111_1003_3 {

        long a = sc.nextLong();
        long b = sc.nextLong();

        // 印出原始值
        System.out.println(a);
        System.out.println(b);

        // 不使用額外變數的交換法：a = a + b; b = a - b; a = a - b;

        b = a - b;
        a = a - b;

        // 印出交換後的值
        System.out.println(a);
        System.out.println(b);

        sc.close();
    }
