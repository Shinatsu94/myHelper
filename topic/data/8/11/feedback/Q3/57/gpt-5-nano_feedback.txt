/*
檔案名稱: SumOfDigitsMatch.java
功能: 統計 1..100000 範圍內，數字各位數字和等於輸入值 A 的數字個數
作者: 盡量提供作者資訊
日期: 2025-11-15
說明: 讀入整數 A，輸出 1 至 100000 中，數字各位數字和等於 A 的數字個數
*/
import java.util.Scanner;

public class SumOfDigitsMatch {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int targetSum = sc.nextInt();
        int count = 0;

        for (int i = 1; i <= 100000; i++) {
            int t = i;
            int sum = 0;
            // 逐位相加直到數字結束
            while (t > 0) {
                sum += t % 10;
                t /= 10;
            }
            if (sum == targetSum) {
                count++;
            }
        }

        System.out.print(count);
        sc.close();
    }
}
