修改後完整程式碼
```java
/**
 * File: DigitSumCount.java
 * 功能: 計算 1..100000 中，數字各位數總和等於指定值的數字個數
 * 作者: 調整為較清晰的實作
 * 日期: 2025-11-15
 * 備註: 內含 digitSum 方法，並加入輸入邊界檢查與資源管理
 */

import java.util.Scanner;

public class DigitSumCount {
    private static final int MAX = 100000;

    public static void main(String[] args) {
        try (Scanner scanner = new Scanner(System.in)) {
            int target = scanner.nextInt();
            int count = 0;

            // 最大位數總和為 45，若 target 超過 45，結果必為 0
            if (target < 0 || target > 45) {
                System.out.println(0);
                return;
            }

            for (int i = 1; i <= MAX; i++) {
                if (digitSum(i) == target) {
                    count++;
                }
            }

            System.out.println(count);
        }
    }

    // 計算整數 n 的各位數總和
    private static int digitSum(int n) {
        int sum = 0;
        while (n > 0) {
            sum += n % 10;
            n /= 10;
        }
        return sum;
    }
}
```
