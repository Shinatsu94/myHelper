修改後完整程式碼如下：

```java
/**
 * File: CSIE_41343109_1003_03.java
 * 功能: 計算在 1~100000 之間，數字各位數字的和等於使用者輸入 A 的數字個數
 * 作者: 教師
 * 日期: 2025-11-15
 * 修改紀錄: 使用函式 digitSum 以提升可讀性與維護性
 */
import java.util.Scanner;

public class CSIE_41343109_1003_03 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int target = sc.nextInt();
        int count = 0;
        for (int i = 1; i <= 100000; i++) {
            if (digitSum(i) == target) {
                count++;
            }
        }
        System.out.println(count);
        sc.close();
    }

    // 計算整數 n 的各位數字之和（不包含前導零，因為其值為 0）
    private static int digitSum(int n) {
        int sum = 0;
        while (n > 0) {
            sum += n % 10;
            n /= 10;
        }
        return sum;
    }
}
```
