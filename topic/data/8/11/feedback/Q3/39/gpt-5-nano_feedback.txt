- 使用常數 MAX 與有意義的變數名稱，例如 target、count，讓程式意圖更清楚。

/*
個人檔案名稱: CSIE_41343216_1003_3.java
功能說明  : 計算 1~100000 之間，數字各位數字總和等於輸入 A 的數量。
作者　　  : （請填寫作者姓名）
日期　　  : 2025-11-15

import java.util.Scanner;

public class CSIE_41343216_1003_3 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int target = sc.nextInt();
        int count = 0;
        final int MAX = 100000;

        for (int i = 1; i <= MAX; i++) {
            if (digitSum(i) == target) {
                count++;
            }
        }

        System.out.println(count);
        sc.close();
    }

    // 計算正整數 n 的各位數字總和
    private static int digitSum(int n) {
        int sum = 0;
        while (n > 0) {
            sum += n % 10;
            n /= 10;
        }
        return sum;
    }
}
