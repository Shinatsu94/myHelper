/*
  File: CSIE_41343226_1003_3.java
  Title: 讀取兩個整數並在交換後輸出
  Author: 論文機器人
  Date: 2025-11-11
  Description: 讀入兩個整數 A 與 B，先印出 A、B，接著不使用第三變數交換兩數，最後印出交換後的結果。
*/
import java.util.Scanner;

public class CSIE_41343226_1003_3 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int A = sc.nextInt();
        int B = sc.nextInt();

        // 原值輸出
        System.out.println(A);
        System.out.println(B);

        // 不使用額外變數，利用 XOR 交換法
        A = A ^ B;
        B = A ^ B;
        A = A ^ B;

        // 交換後輸出
        System.out.println(A);
        System.out.println(B);

        sc.close();
    }
}
