修改後的完整程式碼
```java
/*
File: DigitSumCounter.java
Description: 對於輸入的整數 A，計算 1..100000 的數字各位數字和等於 A 的數字個數
Author: YourName
Date: 2025-11-15
*/

import java.io.*;

public class DigitSumCounter {
    private static final int LIMIT = 100000;
    private static final int MAX_DIGIT_SUM = 45; // 9 * 5 ︱ 1..99999 的最大位數和

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        String line = br.readLine();
        if (line == null) return;

        line = line.trim();
        if (line.isEmpty()) {
            System.out.println(0);
            return;
        }

        int A = Integer.parseInt(line);

        // 若超出實際可能的位數和，直接輸出 0
        if (A < 0 || A > MAX_DIGIT_SUM) {
            System.out.println(0);
            return;
        }

        int count = countNumbersWithDigitSum(A, LIMIT);
        System.out.println(count);
    }

    // 回傳一個數字的各位數字加總
    private static int digitSum(int num) {
        int sum = 0;
        while (num > 0) {
            sum += num % 10;
            num /= 10;
        }
        return sum;
    }

    // 計算 1..limit 中，其各位數字和等於 target 的數量
    private static int countNumbersWithDigitSum(int target, int limit) {
        int cnt = 0;
        for (int i = 1; i <= limit; i++) {
            if (digitSum(i) == target) cnt++;
        }
        return cnt;
    }
}
```
