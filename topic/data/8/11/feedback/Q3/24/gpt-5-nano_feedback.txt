- 把計算位數和的邏輯提取為靜態方法 digitSum(int) 提高可重用性與測試性。
- 增加檔案頭註解，描述檔案名稱、功能、作者與日期，並在必要位置加入簡要註解。

修改後完整程式碼
import java.util.Scanner;

/**
 * 檔案名稱: CSIE_41343128_1003_3.java
 * 功能: 輸入整數 A; 計算 1..100000 之間，數字各位數的和等於 A 的數字個數。
 * 作者: 陳廷鑫
 * 日期: 2025-10-03
 */
public class CSIE_41343128_1003_3 {
    public static void main(String[] args) {
        Scanner scn = new Scanner(System.in);
        int target = scn.nextInt();
        int count = 0;

        // 遍歷範圍改為 1..100000，嚴格符合題意
        for (int i = 1; i <= 100000; i++) {
            if (digitSum(i) == target) {
                count++;
            }
        }

        System.out.println(count);
        scn.close();
    }

    // 計算整數 x 的各位數字總和
    private static int digitSum(int x) {
        int sum = 0;
        while (x > 0) {
            sum += x % 10;
            x /= 10;
        }
        return sum;
    }
}
