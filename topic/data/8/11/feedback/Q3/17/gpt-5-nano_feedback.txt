檔案名稱: CSIE_41343107_1003_3_optimized.java
功能: 讀入兩個整數，印出交換後的結果。不使用額外變數進行交換。
作者: 王世豪
日期: 2025-11-11
*/
import java.util.Scanner;

public class CSIE_41343107_1003_3_optimized {
    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);

        int A = in.nextInt();
        int B = in.nextInt();

        // 不使用額外變數進行交換
        A = A + B;
        B = A - B;
        A = A - B;

        // 輸出交換後的結果，符合題目之輸出樣例
        System.out.println(A);
        System.out.println(B);

        in.close();
    }
}
