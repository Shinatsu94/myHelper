- 使用直覺且快速的列印方式：直接用 new String(triangle[row]) 一次輸出整行，避免逐字 Console.print，提升效能與可讀性。

修改後完整程式碼：
/*
    filename: CSIE_41143122_1031_3a.java
    description: 建立一個下三角形二維陣列，高度 n 由使用者輸入，範圍 1..20，並輸出由 '*' 組成的下三角形。
    author: 林柏儒 資工四甲 41143122
    date: 2025/11/14
    version: 1.1
*/

public class CSIE_41143122_1031_3a {
    // 建立並輸出下三角形的函式
    private static void buildTriangle(int n) {
        // 輸入驗證：範圍必須在 1..20
        if (n <= 0 || n > 20) {
            System.err.println("Invalid n: must be in [1,20]");
            return;
        }

        // 使用 jagged array 儲存每列的 '*'，第 r 列長度為 r+1
        char[][] triangle = new char[n][];
        for (int row = 0; row < n; row++) {
            triangle[row] = new char[row + 1];
            for (int col = 0; col <= row; col++) {
                triangle[row][col] = '*';
            }
        }

        // 逐列輸出，整列不含空白符號
        for (int row = 0; row < n; row++) {
            System.out.println(new String(triangle[row]));
        }
    }

    public static void main(String[] args) {
        java.util.Scanner scanner = new java.util.Scanner(System.in);
        int n;

        // 讀取輸入
        if (scanner.hasNextInt()) {
            n = scanner.nextInt();
        } else {
            System.err.println("Invalid input");
            scanner.close();
            return;
        }

        // 進行輸出
        buildTriangle(n);
        scanner.close();
    }
}
