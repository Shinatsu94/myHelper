import java.util.Scanner;
import java.util.Arrays;

public class CSIE_41343247_1031_3a {
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int n = 0;
		// 取得陣列長度
		if (sc.hasNextInt()) {
			n = sc.nextInt();
		} else {
			sc.close();
			return;
		}

		int[] arr = new int[n];
		// 讀入 n 個整數
		for (int i = 0; i < n; i++) {
			if (sc.hasNextInt()) {
				arr[i] = sc.nextInt();
			} else {
				arr[i] = 0; // 若輸入不足，預設為 0，避免例外
			}
		}
		sc.close();

		// 進行升序排序
		Arrays.sort(arr);

		// 輸出排序後的整數，元素之間以空格分隔
		StringBuilder sb = new StringBuilder();
		for (int i = 0; i < n; i++) {
			if (i > 0) sb.append(' ');
			sb.append(arr[i]);
		}
		System.out.println(sb.toString());
	}
}
