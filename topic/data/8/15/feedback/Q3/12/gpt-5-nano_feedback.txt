// ArraySortAscending.java
/*
檔案名稱: ArraySortAscending.java
功能: 讀取使用者輸入的長度 n 與 n 個整數，將陣列內容由小到大排序後輸出
作者: 課程助教
日期: 2025-11-14
說明: 不使用 EOF，適當處理輸入與輸出格式
*/

import java.util.Arrays;
import java.util.Scanner;

public class ArraySortAscending {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        int n = 0;
        // 讀取正整數 n，且 n > 0
        while (true) {
            if (scanner.hasNextInt()) {
                n = scanner.nextInt();
                if (n > 0) {
                    break;
                }
            } else {
                // 若遇到非整數，略過
                scanner.next();
            }
        }

        int[] arr = new int[n];

        // 讀入 n 個整數，並存入陣列
        for (int i = 0; i < n; i++) {
            // 直接取得下一個整數，假設輸入會提供 n 個整數
            while (!scanner.hasNextInt()) {
                scanner.next();
            }
            arr[i] = scanner.nextInt();
        }

        // 排序：由小到大
        Arrays.sort(arr);

        // 輸出結果：一行，數字之間以單一空格分隔
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        scanner.close();
    }
}
