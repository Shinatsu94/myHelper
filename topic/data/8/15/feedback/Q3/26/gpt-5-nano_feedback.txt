/**
 * 檔案名稱: CSIE_41343136_1031_3a.java
 * 功能說明: 建立並列印下三角形的二維陣列，大小由使用者輸入 n (0 < n <= 20)
 * 作者: 您的名字
 * 日期: 2025-11-14
 * 備註: 使用 ragged array 儲存 '*'，每列長度為 i+1；輸出不含空白符號
 */

import java.util.Scanner;

public class CSIE_41343136_1031_3a {
    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);

        // 讀取輸入並基本驗證
        int n;
        if (!in.hasNextInt()) {
            System.out.println("輸入格式錯誤");
            in.close();
            return;
        }
        n = in.nextInt();
        if (n <= 0 || n > 20) {
            System.out.println(n > 20 ? "超過20" : "輸入錯誤");
            in.close();
            return;
        }

        // 建立下三角形的 ragged array，並填入 '*'
        char[][] a = new char[n][];
        for (int i = 0; i < n; i++) {
            a[i] = new char[i + 1];
            for (int j = 0; j < a[i].length; j++) {
                a[i][j] = '*';
            }
        }

        // 輸出每列字串，避免逐字迴圈的輸出開銷
        for (int i = 0; i < n; i++) {
            System.out.println(new String(a[i]));
        }

        in.close();
    }
}
