說明:

- 不使用 EOF，程式執行一次即結束

import java.util.Arrays;
import java.util.Scanner;

public class CSIE_41343242_1031_3a {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取陣列長度
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();
        if (n <= 0) {
            sc.close();
            return;
        }

        // 讀取 n 個整數到陣列
        int[] a = new int[n];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                a[i] = sc.nextInt();
            } else {
                // 若輸入不足，預設為 0 以避免例外；實務上可改為異常處理
                a[i] = 0;
            }
        }

        // 排序並輸出
        Arrays.sort(a);
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(a[i]);
        }
        System.out.println(sb.toString());

        sc.close();
    }
}
