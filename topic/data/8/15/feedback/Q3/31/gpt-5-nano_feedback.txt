import java.util.Scanner;

public class CSIE_41343203_1031_3a {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();

        // 輸入驗證：0 < n <= 20
        if (n <= 0 || n > 20) {
            sc.close();
            return;
        }

        // 使用二維陣列儲存下三角形的星號
        char[][] tri = new char[n][n];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= i; j++) {
                tri[i][j] = '*';
            }
        }

        // 輸出：逐行列印前 i+1 個星號，無空格
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= i; j++) {
                System.out.print(tri[i][j]);
            }
            System.out.println();
        }

        sc.close();
    }
}
