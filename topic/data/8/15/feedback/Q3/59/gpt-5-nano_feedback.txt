// 檔案名稱: LowerTriangleStars.java
// 功能: 輸入高度 n (0 < n <= 20)，建立並列印下三角形的二維陣列，內容為 '*'
// 作者: (未提供)
// 日期: (以實際日期為准)

import java.util.Scanner;

public class LowerTriangleStars {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();

        // 輸入驗證：若不在範圍內，結束程式
        if (n < 1 || n > 20) {
            sc.close();
            return;
        }

        // 建立 n x n 二維陣列，僅在下三角填 '*'，其他位置保持未使用
        char[][] arr = new char[n][n];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= i; j++) {
                arr[i][j] = '*';
            }
        }

        // 印出下三角形：每列印出 0..i 的元素
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= i; j++) {
                System.out.print(arr[i][j]);
            }
            System.out.println();
        }

        sc.close();
    }
}
