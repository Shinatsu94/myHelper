/*
Filename: SortArray.java
Description: 讀取使用者輸入的長度 n，接著讀取 n 個整數，將它們升序排序後輸出，元素以空格分隔。
Author: 你的名字
Date: 2025-11-12
*/

import java.util.Arrays;
import java.util.Scanner;

public class SortArray {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取長度
        if (!sc.hasNextInt()) {
            return;
        }
        int n = sc.nextInt();
        if (n < 1) {
            return;
        }

        // 讀取 n 個整數
        int[] a = new int[n];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                a[i] = sc.nextInt();
            } else {
                // 若輸入不足，尚未提供足夠數字，預設為 0 並結束後續處理
                a[i] = 0;
            }
        }

        // 進行排序
        Arrays.sort(a);

        // 輸出排序後的結果，元素之間以單一空格分隔
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(a[i]);
        }
        System.out.print(sb.toString());
    }
}
