/*
File: CSIE_41343112_1031_3a.java
功能: 建立一個高度為 n 的下三角形二維陣列，並列印星號 '*'
作者: 李宗懌
日期: 2025-11-14
說明: 使用 2D 陣列儲存下三角形的星號，輸出時僅輸出每行的前 i+1 個星號，避免輸出多餘的空白
*/
import java.util.Scanner;

public class CSIE_41343112_1031_3a {
    public static void main(String[] args) {
        try (Scanner sc = new Scanner(System.in)) {
            int n = sc.nextInt(); // 輸入高度

            // 輸入範圍檢查，符合題意 0 < n <= 20
            if (n <= 0 || n > 20) {
                return;
            }

            // 建立並填入下三角形的二維陣列
            char[][] tri = new char[n][n];
            for (int i = 0; i < n; i++) {
                for (int j = 0; j <= i; j++) {
                    tri[i][j] = '*';
                }
            }

            // 輸出下三角形（每行輸出前 i+1 個星號）
            for (int i = 0; i < n; i++) {
                for (int j = 0; j <= i; j++) {
                    System.out.print(tri[i][j]);
                }
                System.out.println();
            }
        }
    }
}
