/*
 * 文件名稱: ArrayAscendingSort.java
 * 功能簡述: 讀取長度 n 與 n 個整數，將它們升冪排序並輸出為單行，中間以空格分隔
 * 作者: 教授
 * 日期: 2025-11-12
 */

import java.util.Arrays;
import java.util.Scanner;

public class ArrayAscendingSort {
    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);

        // 讀取陣列長度 n
        if (!in.hasNextInt()) {
            in.close();
            return;
        }
        int n = in.nextInt();

        // 讀取 n 個整數放入陣列
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (in.hasNextInt()) {
                arr[i] = in.nextInt();
            } else {
                // 若輸入不足，以 0 做為預設值；通常測資會提供完整資料，此處為防護處理
                arr[i] = 0;
            }
        }

        // 排序（升冪）
        Arrays.sort(arr);

        // 輸出排序結果，元素以單個空格分隔，最後不加多餘的空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.print(sb.toString());

        in.close();
    }
}

- 使用合適的資料結構與方法：改用整數陣列 int[] 與 Arrays.sort 進行升冪排序，簡單且高效，能正確處理含負數的情況。
- 輸入輸出穩定性：沿用 Scanner 讀取整數，依題意「不使用 EOF、執行一次程式結束」的要求，讀取固定數量的入力即可滿足測資。
- 程式可讀性與維護性提升：加入程式頭註解、使用具意義的變數名稱、整理縮排與空格、移除未使用的變數與不相關輸出；最後輸出以單行、以空格分隔，與題意相符。
