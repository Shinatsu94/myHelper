// 依題意建立低三角形（下三角形）之二維陣列並輸出
import java.util.Scanner;

public class PrintStar {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int n = 0;

        // 讀取輸入
        if (scanner.hasNextInt()) {
            n = scanner.nextInt();
        }

        // 有效範圍檢查
        if (n > 0 && n <= 20) {
            // 使用 2D 陣列儲存下三角形的星號
            char[][] tri = new char[n][n];
            for (int i = 0; i < n; i++) {
                for (int j = 0; j <= i; j++) {
                    tri[i][j] = '*';
                }
            }

            // 輸出整個下三角形陣列，未使用空白分隔字元
            for (int i = 0; i < n; i++) {
                for (int j = 0; j <= i; j++) {
                    System.out.print(tri[i][j]);
                }
                System.out.println();
            }
        }

        scanner.close();
    }
}
