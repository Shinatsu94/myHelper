/*
File: CSIE_41343232_1031_3a.java
功能: 建立並列印下三角形二維陣列，內容以 '*' 填充，輸出格式不包含空白符號。
作者: 你的姓名
日期: 2025-11-14
*/

import java.util.Scanner;

public class CSIE_41343232_1031_3a {
    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);

        int n = 0;
        if (in.hasNextInt()) {
            n = in.nextInt();
        } else {
            in.close();
            return;
        }

        // 輸入範圍檢驗，題目要求 0 < n <= 20
        if (n <= 0) {
            in.close();
            return;
        }
        if (n > 20) {
            n = 20; // 超過上限時限制為 20，保持與題意相符
        }

        // 建立下三角形的二維陣列並填充 '*'
        char[][] tri = new char[n][n];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= i; j++) {
                tri[i][j] = '*';
            }
        }

        // 輸出下三角形，逐行拼接避免多次 IO
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= i; j++) {
                sb.append(tri[i][j]);
            }
            sb.append('\n');
        }

        System.out.print(sb.toString());
        in.close();
    }
}
