/*
 * 檔案名稱: SortArrayAscending.java
 * 功能: 建立一個任意長度的一維陣列，將內容由小到大排序後輸出。
 * 作者: 自動批改系統
 * 日期: 2025-11-12
 */
import java.util.Scanner;
import java.util.Arrays;

public class SortArrayAscending {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);

        // 讀取長度 n
        if (!input.hasNextInt()) {
            System.out.println("invalid input.");
            input.close();
            return;
        }
        int n = input.nextInt();
        if (n <= 0) {
            System.out.println("invalid input.");
            input.close();
            return;
        }

        // 讀取 n 個整數
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (input.hasNextInt()) {
                arr[i] = input.nextInt();
            } else {
                // 若輸入不足，預設為 0，避免失敗
                arr[i] = 0;
            }
        }

        // 升冪排序
        Arrays.sort(arr);

        // 輸出排序後的內容，元素間以單一空格分隔
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        input.close();
    }
}
