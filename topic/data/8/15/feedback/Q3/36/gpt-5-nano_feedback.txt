import java.util.Scanner;
import java.util.Arrays;

/**
 * 檔案名稱: CSIE_41343212_1031_3a.java
 * 功能說明: 建立並列印高度為 n 的下三角形二維陣列，n 需滿足 0 < n <= 20
 * 作者: 重新命名於此
 * 日期: 2025-11-14
 */
public class CSIE_41343212_1031_3a {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        int height = input.nextInt();

        if (height > 0 && height <= 20) {
            char[][] triMatrix = new char[height][];
            for (int i = 0; i < height; i++) {
                triMatrix[i] = new char[i + 1];
                Arrays.fill(triMatrix[i], '*');
            }

            StringBuilder sb = new StringBuilder();
            for (int i = 0; i < height; i++) {
                for (int j = 0; j <= i; j++) {
                    sb.append(triMatrix[i][j]);
                }
                sb.append('\n');
            }
            System.out.print(sb.toString());
        } else {
            System.out.println("invalid input.");
        }

        input.close();
    }
}
