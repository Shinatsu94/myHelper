/**
 * File: LowerTriangle.java
 * Description: 讀入高度 n (0 < n <= 20)，輸出下三角形星號矩陣，無空格分隔。
 * Author: 未知
 * Date: 2025-11-14
 */
import java.util.Scanner;

public class LowerTriangle {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int n;
        // 讀取高度，限制在 1..20
        do {
            n = sc.nextInt();
        } while (n <= 0 || n > 20);

        // 使用 StringBuilder 一次性輸出，減少 I/O 次數
        StringBuilder sb = new StringBuilder();
        for (int i = 1; i <= n; i++) {
            for (int j = 0; j < i; j++) {
                sb.append('*');
            }
            sb.append('\n');
        }

        System.out.print(sb.toString());
        sc.close();
    }
}
