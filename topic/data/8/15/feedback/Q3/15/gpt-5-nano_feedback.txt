```java
/**

 * 作者: 你的教授
 * 日期: 2025-11-12
 * 說明: 使用 Arrays.sort 直接排序，輸出避免尾端多餘空格，並加入適當註解
 */

import java.util.Scanner;
import java.util.Arrays;

public class CSIE_41343101_1031_3a {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();            // 讀取陣列長度
        int[] arr = new int[n];

        for (int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();       // 依序讀入 n 個整數
        }

        Arrays.sort(arr);                // 排序（從小到大）

        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');   // 以空格分隔，避免尾端空格
            sb.append(arr[i]);
        }

        System.out.println(sb.toString()); // 輸出排序結果
        sc.close();
    }
}
```

- 使用 Arrays.sort 提升排序效率與穩定性，較適合處理任意長度陣列。
