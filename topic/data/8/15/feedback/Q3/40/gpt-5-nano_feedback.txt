修改後完整程式碼
/*
 * 檔案名稱: LowerTrianglePrinter.java
 * 功能: 建立並列印高度為 n 的下三角形二維陣列，輸出每行連續的 '*'，且無空格。
 * 作者: 教師
 * 日期: 2025-11-14
 */

import java.util.Scanner;

public class LowerTrianglePrinter {

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        sc.close();

        // 0 < n <= 20 的限制檢查
        if (n <= 0 || n > 20) {
            return;
        }

        // 以二維陣列方式建立下三角形：grid[i][j] = '*' 只在 j <= i 的位置
        char[][] grid = new char[n][n];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= i; j++) {
                grid[i][j] = '*';
            }
        }

        // 逐行輸出：每行輸出從 j=0 到 j=i 的 '*'，不輸出其他位置
        StringBuilder line = new StringBuilder();
        for (int i = 0; i < n; i++) {
            line.setLength(0);
            for (int j = 0; j <= i; j++) {
                line.append(grid[i][j]);
            }
            System.out.println(line.toString());
        }
    }
}
