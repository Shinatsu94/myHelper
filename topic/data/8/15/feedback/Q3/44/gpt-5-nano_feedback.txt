/*
File: CSIE_41343222_1031_3.java
Description: 建立並列印高度為 n 的下三角形二維陣列，元素為 '*'
Author: 你/教授
Date: 2025-11-14

import java.util.Scanner;

public class CSIE_41343222_1031_3 {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        int n = scanner.nextInt();

        // 邊界處理：符合題目條件範圍，若超出回傳值截斷
        if (n <= 0) {
            scanner.close();
            return;
        }
        if (n > 20) {
            n = 20;
        }

        // 使用二維陣列存放下三角形的 '*'，其餘位置不需要使用
        char[][] arr = new char[n][n];

        // 填充下三角形部分
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= i; j++) {
                arr[i][j] = '*';
            }
        }

        // 逐行輸出，只印出每行前 i+1 個字元
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= i; j++) {
                System.out.print(arr[i][j]);
            }
            System.out.println();
        }

        scanner.close();
    }
}
