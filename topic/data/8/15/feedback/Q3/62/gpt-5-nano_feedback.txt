/*
檔案名稱: SortArray.java
功能: 讀取長度 n 與接著的 n 個整數，對其升序排序，輸出結果（以空白分隔，僅輸出一行）。
作者: 你的教授
日期: 2025-11-12
*/
import java.util.Arrays;
import java.util.Scanner;

public class SortArray {
    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);
        if (!in.hasNextInt()) {
            in.close();
            return;
        }
        int n = in.nextInt();
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (in.hasNextInt()) {
                arr[i] = in.nextInt();
            } else {
                arr[i] = 0; // 若輸入不足，預設為 0
            }
        }
        in.close();

        Arrays.sort(arr);

        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.print(sb.toString());
    }
}

- 使用 Arrays.sort 對整數陣列排序，邏輯清晰、穩定且效能高。
- 讀入流程改為先取得 n，再逐一讀取 n 個整數，確保一次執行結束，避免依賴 EOF。
