/*
File: CSIE_41343211_1031_3_a_2.java
功能: 依題目需求，讀取長度 n、再讀取 n 個整數，並輸出由小到大的排序結果。
作者: [你的姓名]
日期: 2025-11-12
說明: 使用 Arrays.sort 直接對整數陣列排序，輸出為單行，元素以空格分隔。
*/
import java.util.Arrays;
import java.util.Scanner;

public class CSIE_41343211_1031_3_a_2 {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);

        // 讀取長度 n
        if (!input.hasNextInt()) {
            input.close();
            return;
        }
        int n = input.nextInt();

        // 基本輸入驗證
        if (n <= 0) {
            input.close();
            return;
        }

        // 讀取 n 個整數
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (input.hasNextInt()) {
                arr[i] = input.nextInt();
            } else {
                // 若輸入不足，補零以避免例外；實務上應該避免這種情況，或改為異常處理
                arr[i] = 0;
            }
        }

        // 排序
        Arrays.sort(arr);

        // 輸出結果（單行，元素間以空格分隔）
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        input.close();
    }
}
