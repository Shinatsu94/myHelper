```java
/**
 * File: CSIE_41343226_1031_3A.java
 * 功能: 輸入高度 n，輸出下三角形 '*' 矩陣
 * 作者: 你的名字
 * 日期: 2025-11-14
 * 內容: 使用 Scanner 讀取整數 n，輸出長度為 n 的下三角形 '*' 行列
 */
import java.util.Scanner;

public class CSIE_41343226_1031_3A {
    public static void main(String[] args) {
        try (Scanner sc = new Scanner(System.in)) {
            int n = sc.nextInt();
            if (n <= 0 || n > 20) {
                // 若輸入不在允許範圍，直接結束
                return;
            }
            StringBuilder sb = new StringBuilder();
            for (int i = 1; i <= n; i++) {
                for (int j = 0; j < i; j++) {
                    sb.append('*');
                }
                sb.append('\n');
            }
            System.out.print(sb.toString());
        }
    }
}
```
