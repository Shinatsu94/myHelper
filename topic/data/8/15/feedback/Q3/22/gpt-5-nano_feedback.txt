```java
/*
 * 檔案名稱: CSIE_41343121_1031_3a.java
 * 功能: 讀取長度 n 與 n 個整數，進行升冪排序，輸出排序後的結果（以空格分隔）
 * 作者: 徐唯祐
 * 日期: 2023/10/31
 */

import java.util.Arrays;
import java.util.Scanner;

public class CSIE_41343121_1031_3a {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取陣列長度
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();
        if (n <= 0) {
            System.out.println();
            sc.close();
            return;
        }

        // 讀取 n 個整數
        int[] a = new int[n];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                a[i] = sc.nextInt();
            } else {
                a[i] = 0; // 若輸入不足，預設為 0
            }
        }

        // 排序
        Arrays.sort(a);

        // 輸出，避免尾端空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(a[i]);
        }
        System.out.println(sb.toString());

        sc.close();
    }
}
```
