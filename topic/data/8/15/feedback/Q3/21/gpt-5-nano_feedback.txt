 * 作者: 金宗翰
 * 日期: 2025-11-12
 */
import java.util.*;

public class CSIE_41343119_1031_3 {
    public static void main(String[] args) {
        Scanner scn = new Scanner(System.in);
        int n;
        if (!scn.hasNextInt()) {
            scn.close();
            return;
        }
        n = scn.nextInt();
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (scn.hasNextInt()) {
                arr[i] = scn.nextInt();
            } else {
                arr[i] = 0; // 若輸入數量不足，預設為 0
            }
        }
        Arrays.sort(arr);

        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.print(sb.toString());
        scn.close();
    }
}
