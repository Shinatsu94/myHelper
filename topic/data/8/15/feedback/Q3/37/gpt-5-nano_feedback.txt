import java.util.Scanner;
import java.util.Arrays;

/*
檔案名稱: Main.java
功能: 讀取一個整數 n，接著讀入 n 個整數，輸出升序排序結果，元素可為負數。
作者: 教師
日期: 2025-11-12
說明: 使用 Arrays.sort 將整數陣列排序，輸出時以單空格分隔，最後換行。
*/
public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取長度 n，並確保為正整數
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();
        while (n <= 0) {
            if (!sc.hasNextInt()) {
                sc.close();
                return;
            }
            n = sc.nextInt();
        }

        // 讀入 n 個整數，並存入陣列
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                // 若輸入數量不足，預設為 0（在此題情境通常會有足夠的數字，因此此分支不常發生）
                arr[i] = 0;
            }
        }

        // 升序排序
        Arrays.sort(arr);

        // 輸出，元素以單一空格分隔
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        sc.close();
    }
}

- 格式：改用有意義的變數名稱（n、arr），採用較標準的程式結構與註解，提升可讀性與維護性。
