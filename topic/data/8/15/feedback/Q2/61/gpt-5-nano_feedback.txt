import java.util.Scanner;

public class CSIE_41343247_1031_2 {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        // 讀取陣列長度
        int n;
        if (!scanner.hasNextInt()) {
            scanner.close();
            return;
        }
        n = scanner.nextInt();

        // 讀取陣列元素
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (scanner.hasNextInt()) {
                arr[i] = scanner.nextInt();
            } else {
                arr[i] = 0; // 若不足元素，預設為0
            }
        }

        // 選擇排序（正確實作：找最小值放在前面，並做交換）
        for (int i = 0; i < n - 1; i++) {
            int minIdx = i;
            for (int j = i + 1; j < n; j++) {
                if (arr[j] < arr[minIdx]) {
                    minIdx = j;
                }
            }
            // 交換 arr[i] 與 arr[minIdx]
            int tmp = arr[i];
            arr[i] = arr[minIdx];
            arr[minIdx] = tmp;
        }

        // 輸出排序結果
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.print(sb.toString());

        scanner.close();
    }
}
