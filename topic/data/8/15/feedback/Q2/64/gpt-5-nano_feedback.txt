/*
檔案名稱: CSIE_51015105_1031_2.java
功能: 建立任意長度的一維陣列，輸入長度與元素，輸出升序排序結果
作者: 教授
日期: 2025-11-14
說明: 使用選擇排序（selection sort）將陣列內容由小到大排序，並以空格分隔輸出。
*/
import java.util.Scanner;

public class CSIE_51015105_1031_2 {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        // 讀取陣列長度
        if (!scanner.hasNextInt()) {
            scanner.close();
            return;
        }
        int n = scanner.nextInt();
        int[] arr = new int[n];

        // 讀取陣列元素
        for (int i = 0; i < n; i++) {
            arr[i] = scanner.nextInt();
        }

        // 選擇排序：由小到大
        for (int i = 0; i < n - 1; i++) {
            int minIndex = i;
            for (int j = i + 1; j < n; j++) {
                if (arr[j] < arr[minIndex]) {
                    minIndex = j;
                }
            }
            // 交換 arr[i] 與 arr[minIndex]
            if (minIndex != i) {
                int tmp = arr[i];
                arr[i] = arr[minIndex];
                arr[minIndex] = tmp;
            }
        }

        // 輸出排序結果，元素以空格分隔
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.print(sb.toString());
        scanner.close();
    }
}
