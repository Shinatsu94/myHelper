```java
/*
檔案名稱: SortArrayAsc.java
功能: 讀入長度與元素，將整數陣列升冪排序，輸出排序後的陣列
作者: 你的名字
日期: 2025-11-12
說明: 不再進行不必要的二次轉換，直接輸出排序結果
*/
import java.util.Arrays;
import java.util.Scanner;

public class SortArrayAsc {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀入陣列長度
        int n = sc.nextInt();
        int[] arr = new int[n];

        // 讀入陣列元素
        for (int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();
        }

        // 升冪排序
        Arrays.sort(arr);

        // 輸出結果，元素以空格分隔，末尾不多餘空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.print(sb.toString());

        sc.close();
    }
}
```
