import java.util.Scanner;

/*
檔案名稱: RightMaxToRight.java
功能: 依題意輸出陣列中每個元素右側的最大值，
      最後一個元素為 -1。優化為單次從右至左的線性演算法。
作者: 你的姓名
日期: 2025-11-14
*/
public class Main {
    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);

        int n;
        if (in.hasNextInt()) {
            n = in.nextInt();
        } else {
            in.close();
            return;
        }

        int[] original = new int[Math.max(n, 0)];
        for (int i = 0; i < n; i++) {
            if (in.hasNextInt()) {
                original[i] = in.nextInt();
            } else {
                original[i] = 0;
            }
        }

        int[] result = new int[n];
        if (n > 0) {
            result[n - 1] = -1;
            int maxRight = original[n - 1];
            for (int i = n - 2; i >= 0; i--) {
                result[i] = maxRight;
                if (original[i] > maxRight) {
                    maxRight = original[i];
                }
            }
        }

        for (int i = 0; i < n; i++) {
            if (i > 0) System.out.print(" ");
            System.out.print(result[i]);
        }

        in.close();
    }
}
