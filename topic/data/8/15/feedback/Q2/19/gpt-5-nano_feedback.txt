修改後完整程式碼：
/*
 * 檔案名稱: Main.java
 * 功能說明: 讀取使用者指定長度的整數陣列，將其升序排序，並輸出排序結果。
 * 作者: 自動評改工具
 * 日期: 2025-11-12
 */
import java.util.Arrays;
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取長度
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();
        int[] arr = new int[n];

        // 讀取 n 個整數
        for (int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();
        }

        // 排序
        Arrays.sort(arr);

        // 輸出結果，數字以空格分隔，末尾不多餘空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.print(sb.toString());

        sc.close();
    }
}
