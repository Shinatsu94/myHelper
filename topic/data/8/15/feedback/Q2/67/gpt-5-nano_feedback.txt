import java.util.Scanner;
import java.util.Arrays;

public class CSIE_51015120_1031_2 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int n = sc.nextInt();
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();
        }

        // 使用內建排序確保正確性與效率
        Arrays.sort(arr);

        // 以空格分隔輸出，避免尾端多餘空格
        for (int i = 0; i < n; i++) {
            if (i > 0) System.out.print(" ");
            System.out.print(arr[i]);
        }
        System.out.println();

        sc.close();
    }
}

- 將排序邏輯改為使用 Arrays.sort(arr) 直接對整個陣列排序，能保證正確性與效率，且對負數亦正確處理。
- 導入 java.util.Arrays 並保留同樣的輸入方式（先讀 n，再讀 n 個整數），符合題意且不需額外輸入結束條件。
