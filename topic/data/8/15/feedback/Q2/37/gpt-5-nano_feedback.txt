- 原因與修改重點：

import java.util.Arrays;
import java.util.Scanner;

/*
檔案名稱: csie_1031_41343214_2.java
功能說明: 將讀入的任意長度整數陣列進行升序排序，輸出排序結果
作者: 你的姓名
日期: 2025-11-12
說明: 使用 Arrays.sort 以正確處理負數與任意長度的輸入，輸出以單行空格分隔的排序結果。
*/
public class csie_1031_41343214_2 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }

        int n = sc.nextInt();
        int[] arr = new int[n];

        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                arr[i] = 0;
            }
        }

        Arrays.sort(arr);

        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }

        System.out.println(sb.toString());
        sc.close();
    }
}
