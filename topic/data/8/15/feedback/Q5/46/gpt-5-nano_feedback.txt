/**
 * 程式名稱 : ArraySortAscending
 *
 * 功能說明： 讀入整數個數 n，接著讀入 n 個整數，將它們由小到大排序並輸出。
 *
 * 日期: 2025/11/14
 *
 * 作者: 張竣傑
 * 學號: 41343224
 */
import java.util.Arrays;
import java.util.Scanner;

public class ArraySortAscending {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);

        // 讀取陣列長度
        int n;
        if (input.hasNextInt()) {
            n = input.nextInt();
        } else {
            input.close();
            return;
        }

        // 讀取 n 個整數
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (input.hasNextInt()) {
                arr[i] = input.nextInt();
            } else {
                // 若輸入不足，以預設值 0 代替
                arr[i] = 0;
            }
        }

        // 由小到大排序
        Arrays.sort(arr);

        // 輸出排序後的結果，避免尾端空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        input.close();
    }
}
