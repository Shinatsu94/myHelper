修改後完整程式碼
/*
File: SortArrayAscending.java
功能: 依題目需求，讀取整數長度 n 與 n 個整數，輸出升序排序結果。支援負數。
作者: 教師
日期: 2025-11-12
*/
import java.util.Scanner;
import java.util.Arrays;

public class SortArrayAscending {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取陣列長度
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();

        // 讀取 n 個整數
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                // 若輸入不足，預設為 0，避免例外
                arr[i] = 0;
            }
        }

        // 進行升序排序
        Arrays.sort(arr);

        // 輸出結果，元素以單一空格分隔
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        sc.close();
    }
}
