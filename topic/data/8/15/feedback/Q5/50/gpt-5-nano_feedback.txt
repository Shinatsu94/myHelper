import java.util.Scanner;
import java.util.Arrays;

/*
檔案名稱: ArraySortAscending.java
功能說明: 讀取一個整數 n，接著讀取 n 個整數，將它們由小到大排序後輸出成一行以空格分隔的字串。
作者: 提交者
日期: 2025-11-12
*/

public class ArraySortAscending {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int n = sc.nextInt();
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();
        }
        sc.close();

        Arrays.sort(arr);

        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());
    }
}

修改原因說明
- 將原先與題意無關的矩陣與求逆運算移除，改為符合題意的排序實作。
- 使用動態長度陣列與 Arrays.sort 提升正確性與可維護性，且能正確處理包含負數的情形。
- 輸出格式改為單行、空格分隔，符合輸出樣例。
