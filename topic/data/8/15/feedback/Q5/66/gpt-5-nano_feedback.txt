/*
 * File: SortArrayAsc.java
 * Description: 讀入任意長度的整數陣列，將內容由小到大排序後輸出
 * Author: 你的姓名
 * Date: 2025-11-12
 * Note: 不使用 EOF，讀取一次後結束。
 */
import java.util.Scanner;
import java.util.Arrays;

public class SortArrayAsc {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取陣列長度
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();
        int[] arr = new int[n];

        // 依序讀入陣列元素
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                arr[i] = 0; // 若輸入不足，補回 0，避免例外
            }
        }

        sc.close();

        // 由小到大排序
        Arrays.sort(arr);

        // 輸出結果，避免末尾多餘空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());
    }
}

- 使用 Arrays.sort 進行整數陣列排序，簡潔且效率良好，支援負數與任意長度。
