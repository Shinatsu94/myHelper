修改與優化

以下提供符合題意的完整、可執行修改後程式，功能為：
- 讀取使用者指定的長度 n
- 依序讀入 n 個整數
- 將陣列排序（由小到大，包含負數）
- 輸出排序後的陣列，元素以空格分隔，結尾換行

修改後完整程式碼（新檔案名稱可自訂，例如 CSIE_41343240_1031_5_SortArray.java）：
// Date: 2025/10/31
// Creator: 41343240 黃榮裕
// Function: 讀入使用者指定長度的一維整數陣列，將其升序排序並輸出

import java.util.Arrays;
import java.util.Scanner;

public class CSIE_41343240_1031_5_SortArray {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取陣列長度
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();
        if (n <= 0) {
            // 若長度為 0 或負數，直接輸出空行並結束
            System.out.println();
            sc.close();
            return;
        }

        // 讀取 n 個整數
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                // 若輸入不足，預設為 0，避免跑出例外
                arr[i] = 0;
            }
        }

        // 排序
        Arrays.sort(arr);

        // 輸出，元素以空格分隔，結尾換行
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        sc.close();
    }
}
