修改後完整程式碼
```java
/*
File: Main.java
Description: 讀取整數 n，接著讀入 n 個整數，將它們升序排序後輸出為一行，數字間以空格分隔
Author: 你的名字
Date: 2025-11-14
*/

import java.util.Arrays;
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n;

        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }

        n = sc.nextInt();
        int[] arr = new int[n];

        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                // 若輸入不足，預設為 0 避免例外
                arr[i] = 0;
            }
        }
        sc.close();

        Arrays.sort(arr);

        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }

        System.out.println(sb.toString());
    }
}
```
