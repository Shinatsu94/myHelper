/*
檔案名稱: SortArray.java
功能說明: 讀入長度 n 與 n 個整數，將陣列由小到大排序後輸出。
作者: 你的姓名
日期: 2025-11-12
註解: 使用 Bubble Sort，支援負數，輸出格式為以單一空格分隔的排序結果。
*/
import java.util.Scanner;

public class SortArray {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取陣列長度
        int n = sc.nextInt();
        int[] arr = new int[n];

        // 讀取 n 個整數
        for (int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();
        }

        // Bubble Sort: 由小到大
        for (int i = 0; i < n - 1; i++) {
            boolean swapped = false;
            for (int j = 0; j < n - 1 - i; j++) {
                if (arr[j] > arr[j + 1]) {
                    int tmp = arr[j];
                    arr[j] = arr[j + 1];
                    arr[j + 1] = tmp;
                    swapped = true;
                }
            }
            if (!swapped) break; // 已排序完成
        }

        // 輸出排序結果，元素以空格分隔，末尾換行
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        sc.close();
    }
}
