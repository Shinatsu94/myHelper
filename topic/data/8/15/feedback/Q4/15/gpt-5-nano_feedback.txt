import java.util.Scanner;

/**
 * File: LowerTriangularArray.java
 * Description: 建立下三角形二維陣列，填入遞增數字，尋找指定數字的位置，輸出 (row, col)（皆從1開始）。
 * Author: 你的姓名
 * Date: 2025-11-14
 */
public class LowerTriangularArray {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int height = sc.nextInt();
        int target = sc.nextInt();

        // 建立下三角形二維陣列並填值
        int[][] tri = new int[height][];
        int value = 1;
        for (int i = 0; i < height; i++) {
            tri[i] = new int[i + 1];
            for (int j = 0; j <= i; j++) {
                tri[i][j] = value++;
            }
        }

        // 尋找 target 的位置
        int foundRow = -1; // 1-based
        int foundCol = -1; // 1-based
        outer:
        for (int i = 0; i < height; i++) {
            for (int j = 0; j <= i; j++) {
                if (tri[i][j] == target) {
                    foundRow = i + 1;
                    foundCol = j + 1;
                    break outer;
                }
            }
        }

        if (foundRow != -1) {
            System.out.println(target + "[" + foundRow + ", " + foundCol + "]");
        } else {
            System.out.println("not find " + target + " in array.");
        }
    }
}
