/*
檔案名稱: LowerTriangularSearch.java
功能: 建立下三角形二維陣列，找出指定數字在陣列中的位置
作者: 徐唯祐
日期: 2025-11-14
描述: 根據輸入的高度 n 與要尋找的數字 tar，若 tar 在下三角形中，輸出 tar [row, col]，否則輸出 not find tar in array.
*/
import java.util.Scanner;

public class LowerTriangularSearch {
    public static void main(String[] args) {
        Scanner scn = new Scanner(System.in);
        int n = scn.nextInt();
        int tar = scn.nextInt();

        long total = (long) n * (n + 1) / 2; // 下三角形總元素數
        if (tar <= 0 || tar > total) {
            System.out.printf("not find %d in array.", tar);
            return;
        }

        for (int row = 1; row <= n; row++) {
            long endOfRow = (long) row * (row + 1) / 2;       // 第 row 行的最後一個數字
            long prevEnd = (long) (row - 1) * row / 2;        // 第 row-1 行的最後一個數字
            if (tar <= endOfRow) {
                int col = (int) (tar - prevEnd); // 第 row 行中的列號 (從 1 開始)
                System.out.printf("%d[%d, %d]", tar, row, col);
                return;
            }
        }

        // 理論上不會到這裡，但作為保險
        System.out.printf("not find %d in array.", tar);
    }
}
