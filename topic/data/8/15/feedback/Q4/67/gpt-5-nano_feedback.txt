/*
 * 檔案名稱: LowerTriangularArray.java
 * 功能: 建立下三角形二維陣列，並找出指定數字的位置
 * 作者: 教師
 * 日期: 2025-11-14
 */

import java.util.Scanner;

public class LowerTriangularArray {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();

        // 建立下三角形二維陣列
        int[][] arr = new int[n][];
        int value = 1;
        for (int i = 0; i < n; i++) {
            arr[i] = new int[i + 1];
            for (int j = 0; j <= i; j++) {
                arr[i][j] = value++;
            }
        }

        int target = sc.nextInt();

        int maxValue = n * (n + 1) / 2;
        if (target <= 0 || target > maxValue) {
            System.out.println("not find " + target + " in array.");
            sc.close();
            return;
        }

        // 使用數學公式直接定位目標值的位置
        int row = 1;
        // 找到最小的 row 使 target <= row*(row+1)/2
        while (row <= n && target > row * (row + 1) / 2) {
            row++;
        }
        // 1-based 的欄位
        int col = target - (row - 1) * row / 2;

        System.out.println(target + "[" + row + ", " + col + "]");

        sc.close();
    }
}
