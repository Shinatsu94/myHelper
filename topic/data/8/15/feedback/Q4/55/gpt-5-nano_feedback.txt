import java.util.Scanner;
import java.util.Arrays;

/**
 * 檔案名稱: ArraySortAscending.java
 * 功能說明: 讀取長度 n，接著讀入 n 個整數，將整數以升冪排序，最後輸出排序結果（以空格分隔）。
 * 作者: 說明者
 * 日期: 2025-11-12
 */
public class ArraySortAscending {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取陣列長度
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();
        int[] arr = new int[n];

        // 依序讀入 n 個整數
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                // 若輸入不足，補 0 並結束
                arr[i] = 0;
            }
        }

        // 升冪排序
        Arrays.sort(arr);

        // 輸出結果，數字以空格分隔，結尾換行
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        sc.close();
    }
}
