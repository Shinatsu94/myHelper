/*
 * File: CSIE_41343128_1031_3b.java
 * 功能: 讀取整數 n，接著讀入 n 個整數，將它們由小到大排序後輸出。
 * 作者: 教師
 * 日期: 2025-11-12
 * 備註: 不使用 EOF，程式執行一次即結束；支援包含負數的輸入。
 */

import java.util.Arrays;
import java.util.Scanner;

public class CSIE_41343128_1031_3b {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取陣列長度
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();

        // 讀取 n 個整數並存入一維陣列
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                // 若輸入不足，預設為 0 以避免 ArrayIndexOutOfBounds
                arr[i] = 0;
            }
        }
        sc.close();

        // 排序
        Arrays.sort(arr);

        // 輸出，元素以單一空白區隔，末尾不留額外空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());
    }
}

- 將原本的複雜且不符合題意的 2D 陣列結構與尋找機制移除，改以 1D 陣列直接處理。
- 使用 Arrays.sort 以確保排序正確且支援負數、任意長度，提升穩定性與可維護性。
- 增加適當的程式註解與標頭，提升可讀性與維護性。
