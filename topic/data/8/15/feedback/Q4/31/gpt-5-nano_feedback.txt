// LowerTriangularMatrix.java
// 功能: 建立下三角形二維陣列，輸入高度 n 與尋找的數字 target，輸出該數字的位置，若未找到則給出提示。
// 作者: 以教學用為主
// 日期: 2025-11-14

import java.util.Scanner;

public class LowerTriangularMatrix {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();
        if (n <= 0) {
            sc.close();
            return;
        }

        // 建立下三角形二維陣列
        int[][] tri = new int[n][];
        int num = 1;
        for (int i = 0; i < n; i++) {
            tri[i] = new int[i + 1];
            for (int j = 0; j <= i; j++) {
                tri[i][j] = num++;
            }
        }

        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int target = sc.nextInt();

        // 搜尋 target 的位置
        boolean found = false;
        int foundRow = -1;
        int foundCol = -1;
        for (int i = 0; i < n && !found; i++) {
            for (int j = 0; j < tri[i].length; j++) {
                if (tri[i][j] == target) {
                    found = true;
                    foundRow = i + 1; // 1-based
                    foundCol = j + 1; // 1-based
                    break;
                }
            }
        }

        if (found) {
            System.out.println(target + "[" + foundRow + ", " + foundCol + "]");
        } else {
            System.out.println("not find " + target + " in array.");
        }

        sc.close();
    }
}
