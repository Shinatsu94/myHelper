修改後程式碼（請以原始檔保存為 SortArrayAsc.java）

- 程式碼內容：
class SortArrayAsc {
    /*
     檔案名稱: SortArrayAsc.java
     功能: 建立一維整數陣列，讀取 n 個整數，將其升序排序後印出
     作者: 教授
     日期: 2025-11-12
    */
    public static void main(String[] args) {
        java.util.Scanner sc = new java.util.Scanner(System.in);

        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();
        if (n <= 0) {
            sc.close();
            return;
        }

        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (!sc.hasNextInt()) {
                // 若輸入數量不足，提前結束
                sc.close();
                return;
            }
            arr[i] = sc.nextInt();
        }

        java.util.Arrays.sort(arr);

        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < arr.length; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());
        sc.close();
    }
}
