修改後完整程式碼（可直接執行）
import java.util.Arrays;
import java.util.Scanner;

/*
檔案名稱：Main.java
功能：讀取一個整數長度 n，以及 n 個整數，將其升冪排序後輸出
作者：教學助理
日期：2025-11-12
說明：支援負數，使用 Arrays.sort 進行排序，輸出以空格分隔的一行。
*/
public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }

        int n = sc.nextInt();
        int[] a = new int[n];

        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                a[i] = sc.nextInt();
            } else {
                // 若輸入數不足，以 0 填補以避免例外，實務上題目通常會保證數量正確
                a[i] = 0;
            }
        }

        Arrays.sort(a);

        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(a[i]);
        }

        System.out.println(sb.toString());
        sc.close();
    }
}
