/*
 * File: LowerTriangularArray.java
 * Description: 建立下三角形二維陣列，並找出指定數字的位置（行、列，皆以 1 為起點）
 * Author: [Your Name]
 * Date: 2025-11-14
 * Notes: 使用實際的 2D 陣列儲存下三角形內容，遇到指定數字時輸出其位置；若不存在則給出提示。
 */

import java.util.Scanner;

public class LowerTriangularArray {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        int n = scanner.nextInt();       // 陣列高度
        int target = scanner.nextInt();  // 要尋找的數字

        // 建立下三角形二維陣列，索引從 1 開始使用
        int[][] tri = new int[n + 1][n + 1];
        int value = 1;
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= i; j++) {
                tri[i][j] = value++;
            }
        }

        // 尋找 target 的位置
        int foundRow = -1;
        int foundCol = -1;
        outer:
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= i; j++) {
                if (tri[i][j] == target) {
                    foundRow = i;
                    foundCol = j;
                    break outer;
                }
            }
        }

        if (foundRow != -1) {
            System.out.println(target + "[" + foundRow + ", " + foundCol + "]");
        } else {
            System.out.println("not find " + target + " in array.");
        }

        scanner.close();
    }
}
