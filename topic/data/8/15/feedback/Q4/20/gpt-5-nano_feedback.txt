/*
檔案名稱: CSIE_41343112_1031_3b.java
功能: 建立下三角形二維陣列，尋找指定數字的位置，輸出座標；若不在陣列中，輸出提示
作者: 李宗懌
日期: 2025-11-14
*/
import java.util.Scanner;

public class CSIE_41343112_1031_3b {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int n = sc.nextInt();
        int target = sc.nextInt();

        if (n <= 0) {
            System.out.printf("not find %d in array.", target);
            sc.close();
            return;
        }

        // 建立實際的下三角形二維陣列，並填入 1, 2, 3, ...
        int[][] tri = new int[n][];
        int value = 1;
        for (int i = 0; i < n; i++) {
            tri[i] = new int[i + 1];
            for (int j = 0; j <= i; j++) {
                tri[i][j] = value++;
            }
        }

        // 尋找 target 的位置（行、列皆以 1 為起點）
        boolean found = false;
        int foundRow = -1;
        int foundCol = -1;

        outer:
        for (int i = 0; i < tri.length; i++) {
            for (int j = 0; j < tri[i].length; j++) {
                if (tri[i][j] == target) {
                    found = true;
                    foundRow = i + 1;
                    foundCol = j + 1;
                    break outer;
                }
            }
        }

        if (found) {
            System.out.printf("%d[%d, %d]", target, foundRow, foundCol);
        } else {
            System.out.printf("not find %d in array.", target);
        }

        sc.close();
    }
}
