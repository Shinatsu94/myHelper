/*
 * File: LowerTriangularSearch.java
 * Description: 計算下三角形陣列中 num 的位置，若超出範圍則輸出 not find。
 * Author: 系統自動評分
 * Date: 2025-11-14

import java.util.Scanner;

public class LowerTriangularSearch {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int num = sc.nextInt();
        sc.close();

        int total = n * (n + 1) / 2;
        if (num <= 0 || num > total) {
            System.out.println("not find " + num + " in array.");
            return;
        }

        // 計算 num 所在的列(row) using triangular numbers: find smallest row such that row*(row+1)/2 >= num
        int row = (int) Math.ceil((Math.sqrt(8.0 * num + 1) - 1) / 2.0);
        int endOfRow = row * (row + 1) / 2;
        int startOfRow = endOfRow - row + 1;
        int col = num - startOfRow + 1;

        System.out.println(num + "[" + row + ", " + col + "]");
    }
}
