```java
import java.util.*;

public class CSIE_41343227_1031_3b {
    /*
     * File: CSIE_41343227_1031_3b.java
     * Description: 讀取長度 n 與 n 個整數，排序並輸出升序
     * Author: 修改後提交
     * Date: 2025-11-12
     */
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取長度 n
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();
        int[] arr = new int[n];

        // 讀取 n 個整數
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                arr[i] = 0; // 若輸入不足，補 0，避免例外
            }
        }
        sc.close();

        // 排序
        Arrays.sort(arr);

        // 輸出，空格分隔，結尾換行
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());
    }
}
```
