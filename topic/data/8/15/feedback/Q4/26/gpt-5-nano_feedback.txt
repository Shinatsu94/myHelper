/*
 * File: CSIE_41343136_1031_3b.java
 * Description: 讀取一個長度 n，接著讀取 n 個整數，將它們升序排序並輸出結果。
 * Author: 請在此填寫作者姓名
 * Date: 2025-11-12
 */

import java.util.Arrays;
import java.util.Scanner;

public class CSIE_41343136_1031_3b {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取陣列長度
        if (!sc.hasNextInt()) {
            return;
        }
        int n = sc.nextInt();
        if (n <= 0) {
            // 若長度為0或負，直接結束
            return;
        }

        // 讀取 n 個整數，存入陣列
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                // 若輸入不足，預設為 0（但實務上應確保輸入正確）
                arr[i] = 0;
            }
        }

        // 排序
        Arrays.sort(arr);

        // 輸出：一行，元素以單一空格分隔
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.print(sb.toString());
    }
}
