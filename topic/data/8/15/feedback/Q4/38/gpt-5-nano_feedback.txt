/*
 檔案名稱: CSIE_41343215_1031_3b.java
 功能: 建立下三角形二維陣列，填入連續數字，並找出指定數字的位置
 作者: 教師
 日期: 2025-11-14
 說明: 讀入高度 n 及欲尋找的數字 num，若 num 在陣列範圍內，輸出其位置 [row, col]，否則輸出 not find ...
*/
import java.util.Scanner;

public class CSIE_41343215_1031_3b {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        // 讀取高度 n 與欲尋找的數字 num
        int n = scanner.nextInt();
        int num = scanner.nextInt();
        scanner.close();

        int maxNumInArray = (n * (n + 1)) / 2;

        // 若 num 超出下三角陣列的範圍，直接輸出找不到的訊息
        if (num < 1 || num > maxNumInArray) {
            System.out.println("not find " + num + " in array.");
            return;
        }

        // 建立真正的下三角形二維陣列，並同時尋找 num 的位置
        int[][] tri = new int[n][];
        int counter = 1;
        int foundRow = -1;
        int foundCol = -1;

        for (int i = 0; i < n; i++) {
            tri[i] = new int[i + 1];
            for (int j = 0; j <= i; j++) {
                tri[i][j] = counter;
                if (counter == num) {
                    // 位置以 1-based 為輸出
                    foundRow = i + 1;
                    foundCol = j + 1;
                }
                counter++;
            }
        }

        // 輸出結果，符合題目格式
        if (foundRow != -1) {
            System.out.println(num + "[" + foundRow + ", " + foundCol + "]");
        } else {
            System.out.println("not find " + num + " in array.");
        }
    }
}
