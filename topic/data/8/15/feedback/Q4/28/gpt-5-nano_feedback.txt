修改後的完整程式碼
以下程式會建立一個下三角二維陣列，從左上角往下填充數字，若 num 在陣列範圍內則輸出位置，否則輸出 not find 的訊息。

/**
 * 檔名: LowerTriangularArray.java
 * 功能: 建立下三角形二維陣列，輸入高度 n，填入從 1 開始的連續數字，找出 num 的位置（列與行，皆從 1 開始）。
 * 作者: 教學評分系統
 * 日期: 2025-11-14
 * 說明: 未印出三角內容，僅輸出 num 的位置或找不到的訊息。
 */

import java.util.Scanner;

public class LowerTriangularArray {
    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);
        int n = in.nextInt();
        int num = in.nextInt();

        // 若高度或數字不符合基本條件，直接回傳找不到的訊息
        if (n <= 0) {
            System.out.println("not find " + num + " in array.");
            return;
        }

        int max = n * (n + 1) / 2; // 下三角陣列的總元素數
        if (num <= 0 || num > max) {
            System.out.println("not find " + num + " in array.");
            return;
        }

        // 建立並填充下三角二維陣列
        int[][] tri = new int[n][n];
        int value = 1;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= i; j++) {
                tri[i][j] = value++;
            }
        }

        // 尋找 num 的位置
        int foundRow = -1;
        int foundCol = -1;
        outer:
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= i; j++) {
                if (tri[i][j] == num) {
                    foundRow = i + 1; // 轉換為從 1 開始的列
                    foundCol = j + 1; // 轉換為從 1 開始的欄
                    break outer;
                }
            }
        }

        if (foundRow != -1) {
            System.out.println(num + "[" + foundRow + ", " + foundCol + "]");
        } else {
            // 理論上不會發生，因為 num 已在範圍內，但仍作安全處理
            System.out.println("not find " + num + " in array.");
        }
    }
}
