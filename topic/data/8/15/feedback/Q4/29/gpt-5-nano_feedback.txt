// 檔案名稱：SortArrayAscending.java
// 功能：讀取長度 n 與 n 個整數，將它們由小到大排序，輸出排序結果（以空格分隔）。
// 作者：請填寫
// 日期：請填寫
import java.util.Arrays;
import java.util.Scanner;

public class SortArrayAscending {
    public static void main(String[] args) {
        Scanner scn = new Scanner(System.in);

        // 讀取長度 n
        if (!scn.hasNextInt()) {
            scn.close();
            return;
        }
        int n = scn.nextInt();
        int[] arr = new int[n > 0 ? n : 0];

        // 讀取 n 個整數
        for (int i = 0; i < n; i++) {
            if (scn.hasNextInt()) {
                arr[i] = scn.nextInt();
            } else {
                // 若輸入不足，預設為 0，避免例外
                arr[i] = 0;
            }
        }
        scn.close();

        // 排序
        Arrays.sort(arr);

        // 輸出，避免尾端多餘空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < arr.length; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());
    }
}
