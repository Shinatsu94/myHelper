/*
 * 檔案名稱: TriangularArrayFind.java
 * 功能: 建立一個下三角形的二維陣列，填入 1,2,...，找出指定數字的位置並輸出
 * 作者: 輸入者
 * 日期: 2025-11-17
 */
import java.util.Scanner;

public class TriangularArrayFind {
    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);

        if (!in.hasNextInt()) return;
        int n = in.nextInt(); // 三角形高度

        if (!in.hasNextInt()) return;
        int num = in.nextInt(); // 要尋找的數字

        // 基本輸入與範圍驗證
        if (n <= 0) {
            System.out.println("not find " + num + " in array.");
            return;
        }
        int total = n * (n + 1) / 2;
        if (num <= 0 || num > total) {
            System.out.println("not find " + num + " in array.");
            return;
        }

        // 建立並填入下三角形二維陣列
        int[][] tri = new int[n][];
        int value = 1;
        for (int r = 0; r < n; r++) {
            tri[r] = new int[r + 1];
            for (int c = 0; c <= r; c++) {
                tri[r][c] = value++;
            }
        }

        // 查找 num 的位置
        int foundRow = -1;
        int foundCol = -1;
        outer:
        for (int r = 0; r < n; r++) {
            for (int c = 0; c <= r; c++) {
                if (tri[r][c] == num) {
                    foundRow = r;
                    foundCol = c;
                    break outer;
                }
            }
        }

        if (foundRow != -1) {
            // 1-based 輸出
            System.out.println(num + "[" + (foundRow + 1) + ", " + (foundCol + 1) + "]");
        } else {
            System.out.println("not find " + num + " in array.");
        }
    }
}
