 檔案名稱: Main.java
 功能: 讀入整數個數 n，接著讀入 n 個整數，將它們由小到大排序後印出
 作者: 你
 日期: 2025-11-12

import java.util.Arrays;
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();
        int[] a = new int[n];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                a[i] = sc.nextInt();
            } else {
                a[i] = 0; // 若輸入不足，預設為 0（實務可改為拋出例外）
            }
        }
        sc.close();

        Arrays.sort(a);

        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(a[i]);
        }
        System.out.println(sb.toString());
    }
}
