public class CSIE_41343229_1031_4 {
    public static void main(String[] args) {
        // 使用 try-with-resources 自動關閉 Scanner
        try (Scanner sc = new Scanner(System.in)) {
            // 讀取高度 n 與要查找的數字
            int n = sc.nextInt();
            int target = sc.nextInt();

            // 建立下三角形的 ragged 二維陣列，並填入 1, 2, 3, ...
            int[][] tri = new int[n][];
            int value = 1;
            for (int r = 0; r < n; r++) {
                tri[r] = new int[r + 1];
                for (int c = 0; c <= r; c++) {
                    tri[r][c] = value++;
                }
            }

            // 總數量上限
            long total = (long) n * (n + 1) / 2;

                return;
            }

            // 計算 target 在哪一列與什麼位置（列、欄皆從 1 開始）

            int col = target - prevSum;

        }
