優化後的完整程式碼：
public class CSIE_41143122_1031_3b {

    // 建立一個高度為 n 的下三角形二維陣列，內容依序為 1, 2, 3, ...
    // 找出指定數字 target 在陣列中的位置，輸出格式為: target[col,row]，欄與列皆從 1 開始。
    public static void buildAndFind(int n, int target) {
        int[][] tri = new int[n][];
        int foundRow = -1; // 1-based
        int foundCol = -1; // 1-based
        int current = 0;

        for (int row = 0; row < n; row++) {
            tri[row] = new int[row + 1];
            for (int col = 0; col <= row; col++) {
                current++;
                tri[row][col] = current;
                if (current == target) {
                    foundRow = row + 1;
                    foundCol = col + 1;
                }
            }
        }

        if (foundRow != -1) {
            System.out.println(target + "[" + foundCol + "," + foundRow + "]");
        } else {
            System.out.println("not find " + target + " in array.");
        }
    }

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int n = scanner.nextInt();
        int m = scanner.nextInt();
        buildAndFind(n, m);
        scanner.close();
    }
}
