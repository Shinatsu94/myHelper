/*
 * File: CSIE_41343203_1031_1.java
 * 功能: 讀取整數 n 與 n 個整數，將它們由小到大排序後輸出，元素以空格分隔。
 * 作者: (請自行填寫)
 * 日期: 2025-11-12
 */

import java.util.Scanner;

public class CSIE_41343203_1031_1 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = 0;
        if (sc.hasNextInt()) {
            n = sc.nextInt();
        } else {
            sc.close();
            return;
        }

        int[] arr = new int[n];
        for (int i = 0; i < n && sc.hasNextInt(); i++) {
            arr[i] = sc.nextInt();
        }

        sortArray(arr);

        // 輸出時避免尾端空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < arr.length; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.print(sb.toString());

        sc.close();
    }

    // 將整數陣列做由小到大排序（選擇排序）
    private static void sortArray(int[] arr) {
        int n = arr.length;
        for (int i = 0; i < n - 1; i++) {
            int min = i;
            for (int j = i + 1; j < n; j++) {
                if (arr[j] < arr[min]) {
                    min = j;
                }
            }
            if (min != i) {
                int temp = arr[min];
                arr[min] = arr[i];
                arr[i] = temp;
            }
        }
    }
}
