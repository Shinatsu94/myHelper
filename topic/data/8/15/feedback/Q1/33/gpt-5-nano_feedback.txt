/*
檔案名稱：CSIE_41343208_1031_1.java
功能：讀取整數 n 與 n 個整數，將陣列內容由小到大排序，並輸出排序結果
作者：資訊工程系教授
日期：2025-11-14
*/
import java.util.Scanner;
import java.util.Arrays;

public class CSIE_41343208_1031_1 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        if (!sc.hasNextInt()) {
            // 無有效輸入，僅輸出換行
            System.out.println();
            return;
        }

        int n = sc.nextInt();
        if (n <= 0) {
            System.out.println();
            return;
        }

        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                arr[i] = 0; // 輸入不足時以 0 作為預設
            }
        }

        // 使用內建排序提升效率與穩定性
        Arrays.sort(arr);

        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }

        System.out.println(sb.toString());
    }
}
