/*
 * 檔案名稱: CSIE_51015114_1031_1.java
 * 功能: 讀取使用者輸入的長度與對應整數，對陣列進行升冪排序，並輸出排序結果
 * 作者: 學號 CSIE_51015114
 * 日期: 2025-11-12
 */

import java.util.Arrays;
import java.util.Scanner;

public class CSIE_51015114_1031_1 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int n = sc.nextInt();
        int[] nums = new int[n];

        for (int i = 0; i < n; i++) {
            nums[i] = sc.nextInt();
        }

        // 使用內建排序，簡潔且正確
        Arrays.sort(nums);

        // 輸出結果，避免尾端多餘空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(nums[i]);
        }
        System.out.print(sb.toString());

        sc.close();
    }
}

修改說明
- 透過 Arrays.sort 替代自寫排序迴圈，降低錯誤機率並提高可讀性與維護性。
- 移除不必要的陣列與多餘變數，讓程式結構更清晰。
- 強化輸出格式，確保輸出符合題目要求且在不同測資下穩定。
