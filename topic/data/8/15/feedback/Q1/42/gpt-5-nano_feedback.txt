```java
/*
檔案名稱: CSIE_41343220_1031_1.java
功能: 建立任意長度的一維陣列，將內容由小到大排序，輸出排序結果
作者: 計算機科學系 教師
日期: 2025-11-14
*/
import java.util.Scanner;
import java.util.Arrays;

public class CSIE_41343220_1031_1 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取陣列長度
        int n;
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        n = sc.nextInt();

        // 讀取陣列元素
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();
        }

        // 排序（使用 Java 內建排序提高穩定性與效能）
        Arrays.sort(arr);

        // 輸出排序結果
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.print(sb.toString());

        sc.close();
    }
}
```
