import java.util.Scanner;
import java.util.Arrays;

/*
 姓名: 王世豪
 學號: 41343107
 功能: 讀取長度與整數陣列，輸出升序排序後的內容
 日期: 2025/10/31
*/

public class CSIE_41343107_1031_1
{
    public static void main(String[] args)
    {
        Scanner scanner = new Scanner(System.in);
        int n;
        if (scanner.hasNextInt()) {
            n = scanner.nextInt();
        } else {
            scanner.close();
            return;
        }

        if (n <= 0) {
            // 沒有元素要輸出，直接換行並結束
            System.out.println();
            scanner.close();
            return;
        }

        int[] a = new int[n];
        for (int i = 0; i < n; i++) {
            if (scanner.hasNextInt()) {
                a[i] = scanner.nextInt();
            } else {
                a[i] = 0; // 若輸入不足，補 0；實務可依需求調整
            }
        }

        // 使用內建排序提升效率與穩定性
        Arrays.sort(a);

        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(a[i]);
        }
        System.out.println(sb.toString());

        scanner.close();
    }
}

說明修改該部分的原因：
- Arrays.sort 提升可讀性與執行效率，避免自訂排序邏輯的錯誤與複雜度，且對負數與整數排序天然支援。
- 對 n <= 0 做處理，避免空陣列時的索引存取問題，提升魯棒性。
- 改善輸出格式，統一使用換行結尾，符合常見輸出規範。
- 增加註解與統一的程式風格，提升可維護性。
