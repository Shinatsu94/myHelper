修改後完整程式碼
```java
/*
 * 檔案名稱: CSIE_41343101_1031_1.java
 * 功能: 讀取長度 n 與 n 個整數，輸出升序排列結果
 * 作者: 教師
 * 日期: 2025-11-12
 */

import java.util.Arrays;
import java.util.Scanner;

public class CSIE_41343101_1031_1 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();
        if (n <= 0) {
            // 輸出空行，結束
            System.out.println();
            sc.close();
            return;
        }
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();
        }
        Arrays.sort(arr);

        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());
        sc.close();
    }
}
```

- 使用 Arrays.sort 提升可讀性與穩定性，並確保負數與各種整數的排序正確。
