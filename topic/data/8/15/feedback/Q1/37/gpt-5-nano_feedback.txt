/*
檔案名稱: csie_1031_41343214.java
功能: 建立任意長度的一維陣列，將內容進行升序排序 (由小到大)，輸入長度與元素，輸出排序結果
作者: 教授
日期: 2025-11-14
說明: 使用選擇排序，輸出使用空格分隔且不留 trailing space
*/
public class csie_1031_41343214 {
    public static void main(String[] args) {
        java.util.Scanner sc = new java.util.Scanner(System.in);

        int n = 0;
        if (sc.hasNextInt()) {
            n = sc.nextInt();
        }

        int[] arr = new int[Math.max(0, n)];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                arr[i] = 0;
            }
        }

        selectionSort(arr);

        // 輸出：避免尾端多餘空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < arr.length; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        sc.close();
    }

    // 需要的排序演算法：選擇排序
    private static void selectionSort(int[] a) {
        int n = a.length;
        for (int i = 0; i < n - 1; i++) {
            int minIndex = i;
            for (int j = i + 1; j < n; j++) {
                if (a[j] < a[minIndex]) minIndex = j;
            }
            int t = a[i];
            a[i] = a[minIndex];
            a[minIndex] = t;
        }
    }
}
