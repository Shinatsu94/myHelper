// File: CSIE_41343202_1031_1.java
// Description: 依題目要求，讀取長度與整數陣列，將其升序排序後輸出。
// Author: 你的名字或學號
// Date: 2025-11-14
// Version: 1.0
// Notes: 使用選擇排序實作，輸入僅允許一輪測資輸入，符合題目要求。

import java.util.Scanner;

public class CSIE_41343202_1031_1 {

    // 選擇排序：將 arr 內的元素由小到大排列
    public static void selectionSort(int[] arr) {
        int n = arr.length;
        for (int i = 0; i < n - 1; i++) {
            int minIndex = i;
            for (int j = i + 1; j < n; j++) {
                if (arr[j] < arr[minIndex]) {
                    minIndex = j;
                }
            }
            // 交換 arr[i] 與 arr[minIndex]
            int tmp = arr[i];
            arr[i] = arr[minIndex];
            arr[minIndex] = tmp;
        }
    }

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        int length = scanner.nextInt();
        int[] array = new int[length];

        for (int i = 0; i < length; i++) {
            array[i] = scanner.nextInt();
        }

        // 排序
        selectionSort(array);

        // 輸出，避免尾端空格
        for (int i = 0; i < length; i++) {
            if (i > 0) System.out.print(" ");
            System.out.print(array[i]);
        }
        System.out.println();

        scanner.close();
    }
}
