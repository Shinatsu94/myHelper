/*
 * 檔案名稱: Main.java
 * 功能說明: 讀取一個整數 n，接著讀入 n 個整數，將其升序排序後輸出，輸出以空格分隔。
 * 作者: 你的名字
 * 日期: 2025-11-14
 */

import java.util.Scanner;
import java.util.Arrays;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 步驟 1: 讀取陣列長度
        int len = sc.nextInt();

        int[] nums = new int[len];

        // 步驟 2: 讀取陣列元素
        for (int i = 0; i < len; i++) {
            nums[i] = sc.nextInt();
        }

        // 步驟 4: 對陣列進行排序
        Arrays.sort(nums);

        // 步驟 5: 印出排序後的陣列，使用 StringBuilder 優化輸出
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < len; i++) {
            if (i > 0) sb.append(' ');
            sb.append(nums[i]);
        }
        System.out.println(sb.toString());

        // 不要關閉 System.in，以避免對測資環境造成影響
        // sc.close();
    }
}
