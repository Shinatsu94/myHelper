/*
 * File: CSIE_41343221_1031_1.java
 * Title: 一維整數陣列排序（升序）
 * Description: 讀取長度與元素，將陣列內容升序排序，輸出結果。
 * Author: 你的姓名
 * Date: 2025-11-14
 */

import java.util.Arrays;
import java.util.Scanner;

public class CSIE_41343221_1031_1 {

    public static void main(String[] args) {
        // 使用 try-with-resources 自動關閉 Scanner
        try (Scanner scanner = new Scanner(System.in)) {
            int n = scanner.nextInt();
            int[] arr = new int[n];
            for (int i = 0; i < n; i++) {
                arr[i] = scanner.nextInt();
            }

            // 使用內建排序提升效率與穩定性
            Arrays.sort(arr);

            // 使用 StringBuilder 輸出，避免多次 I/O
            StringBuilder sb = new StringBuilder();
            for (int i = 0; i < n; i++) {
                if (i > 0) sb.append(' ');
                sb.append(arr[i]);
            }
            System.out.println(sb.toString());
        }
    }
}
