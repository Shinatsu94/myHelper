import java.util.Scanner;

public class CSIE_41343109_1031_01 {
    public static void main(String[] args) {
        Scanner scn = new Scanner(System.in);
        int n = scn.nextInt();
        int[] a = new int[n];
        for (int i = 0; i < n; i++) {
            a[i] = scn.nextInt();
        }

        // 使用簡單的泡沫排序（Bubble Sort）做昇冪排序
        for (int i = 0; i < n - 1; i++) {
            for (int j = 0; j < n - 1 - i; j++) {
                if (a[j] > a[j + 1]) {
                    int t = a[j];
                    a[j] = a[j + 1];
                    a[j + 1] = t;
                }
            }
        }

        // 輸出：單一空格分隔，結尾換行
        if (n > 0) {
            System.out.print(a[0]);
            for (int i = 1; i < n; i++) {
                System.out.print(" " + a[i]);
            }
        }
        System.out.println();
        scn.close();
    }
}
