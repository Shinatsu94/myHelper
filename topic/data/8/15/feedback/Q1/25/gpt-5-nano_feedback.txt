/*
檔案名稱: CSIE_41343131_1031_1.java
功能: 讀取整數長度 n 與 n 個整數，並輸出由小到大排序的結果
作者: 教學改良版
日期: 2025-11-14
*/

import java.util.Scanner;

public class CSIE_41343131_1031_1 {
    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);
        int n = 0;
        if (in.hasNextInt()) {
            n = in.nextInt();
        }

        int[] a = new int[Math.max(0, n)];
        for (int i = 0; i < a.length; i++) {
            if (in.hasNextInt()) {
                a[i] = in.nextInt();
            }
        }

        // 使用清晰的選擇排序，僅在需要時才交換
        selectionSort(a);

        // 輸出排序結果，元素以單一空格分隔，末尾換行
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < a.length; i++) {
            if (i > 0) sb.append(' ');
            sb.append(a[i]);
        }
        System.out.println(sb.toString());

        in.close();
    }

    // 選擇排序：將陣列分成已排序與未排序兩部分，逐步選出最小值放到前方
    static void selectionSort(int[] arr) {
        int n = arr.length;
        for (int i = 0; i < n; i++) {
            int min = i;
            for (int j = i + 1; j < n; j++) {
                if (arr[j] < arr[min]) min = j;
            }
            if (min != i) {
                int tmp = arr[i];
                arr[i] = arr[min];
                arr[min] = tmp;
            }
        }
    }
}
