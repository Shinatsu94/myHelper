// 文件名稱: CSIE_41343131_1031_1.java
// 功能: 讀取整數 n 與 n 個整數，將陣列內容升序排序，輸出排序結果
// 作者: 教授
// 日期: 2025-11-12
// 說明: 改用 Arrays.sort 提升穩定性與可讀性，並加上適當註解與統一命名。
import java.util.Arrays;
import java.util.Scanner;

public class CSIE_41343131_1031_1 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = 0;
        if (sc.hasNextInt()) n = sc.nextInt();

        int[] a = new int[n];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) a[i] = sc.nextInt();
        }

        Arrays.sort(a);

        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(a[i]);
        }
        System.out.print(sb.toString());

        sc.close();
    }
}
