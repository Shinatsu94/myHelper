import java.util.Scanner;

/*
 * 檔案名稱: ArraySortAscending.java
 * 功能說明: 讀取整數長度 n 與 n 個整數，將陣列內容由小到大排序，並輸出排序後的結果。
 * 作者: 你的姓名
 * 日期: 2025-11-14
 * 備註: 不使用 EOF；支援負數；輸出不含尾端多餘空格，並於結尾換行。
 */

public class ArraySortAscending {
    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);

        // 讀取長度
        if (!in.hasNextInt()) {
            in.close();
            return;
        }
        int n = in.nextInt();

        // 讀取陣列內容
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (in.hasNextInt()) {
                arr[i] = in.nextInt();
            } else {
                // 若輸入不足，預設為 0
                arr[i] = 0;
            }
        }

        // 選擇排序（Selection Sort）
        for (int i = 0; i < n - 1; i++) {
            int minIdx = i;
            for (int j = i + 1; j < n; j++) {
                if (arr[j] < arr[minIdx]) {
                    minIdx = j;
                }
            }
            // 交換
            int tmp = arr[i];
            arr[i] = arr[minIdx];
            arr[minIdx] = tmp;
        }

        // 輸出，避免尾端有多餘空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.print(sb.toString());

        in.close();
    }
}
