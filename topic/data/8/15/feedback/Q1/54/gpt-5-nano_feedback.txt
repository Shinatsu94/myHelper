/*

功能: 讀取長度 n 與 n 個整數，對其進行升序排序，並輸出排序結果
作者: 教師
日期: 2025-11-12
說明: 使用選擇排序，輸出以空格分隔且不會在末尾附加多餘空格，並在適當時關閉 Scanner。
*/
import java.util.Scanner;

public class csie_41343235_1031_1 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取長度
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();

        // 輸入長度為非正數時，結束程式
        if (n <= 0) {
            sc.close();
            return;
        }

        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                // 若輸入不足，預設為 0（不影響排序邏輯，避免例外）
                arr[i] = 0;
            }
        }
        sc.close();

        // 選擇排序：由小到大
        for (int i = 0; i < n - 1; i++) {
            int minIdx = i;
            for (int j = i + 1; j < n; j++) {
                if (arr[j] < arr[minIdx]) {
                    minIdx = j;
                }
            }
            int tmp = arr[i];
            arr[i] = arr[minIdx];
            arr[minIdx] = tmp;
        }

        // 輸出：以單一字串輸出，元素之間以單一空格分隔，末尾不多餘空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.print(sb.toString());
    }
}
