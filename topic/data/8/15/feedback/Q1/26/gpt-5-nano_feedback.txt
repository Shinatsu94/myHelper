```java
import java.util.Scanner;

/**
 * 檔案名稱: CSIE_41343136_1031_1.java
 * 功能: 讀入長度 n，接著讀入 n 個整數，將其升冪排序後輸出。
 * 作者: 教授
 * 日期: 2025-11-14
 * 內容註解: 使用標準的選擇排序，避免溢位風險，並提供適當註解與輸出格式。
 */
public class CSIE_41343136_1031_1 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }

        int n = sc.nextInt();
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();
        }

        // 標準選擇排序：每次找出最小元素放到前面
        for (int i = 0; i < n - 1; i++) {
            int minIndex = i;
            for (int j = i + 1; j < n; j++) {
                if (arr[j] < arr[minIndex]) {
                    minIndex = j;
                }
            }
            // 交換 arr[i] 與 arr[minIndex]
            int tmp = arr[i];
            arr[i] = arr[minIndex];
            arr[minIndex] = tmp;
        }

        // 輸出
        for (int i = 0; i < n; i++) {
            System.out.print(arr[i]);
            if (i < n - 1) System.out.print(" ");
        }
        System.out.println();

        sc.close();
    }
}
```
