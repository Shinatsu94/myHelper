/*
 * 文件名稱: SortAscending.java
 * 功能說明: 讀取陣列長度與元素，將元素由小到大排序，輸出排序結果（以空格分隔）
 * 作者: 您的名字
 * 日期: 2025-11-12
 * 備註: 不使用 EOF，單次執行結束；支援負數
 */

import java.util.Arrays;
import java.util.Scanner;

public class SortAscending {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取長度
        int length = 0;
        if (sc.hasNextInt()) {
            length = sc.nextInt();
        }

        // 讀取陣列元素
        int[] a = new int[length];
        for (int i = 0; i < length; i++) {
            if (sc.hasNextInt()) {
                a[i] = sc.nextInt();
            } else {
                a[i] = 0; // 若輸入不足，預設為 0
            }
        }

        // 排序：使用內建 Arrays.sort 以提升穩定性與性能
        Arrays.sort(a);

        // 輸出，元素以空格分隔
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < length; i++) {
            if (i > 0) sb.append(' ');
            sb.append(a[i]);
        }
        System.out.print(sb.toString());

        sc.close();
    }
}
