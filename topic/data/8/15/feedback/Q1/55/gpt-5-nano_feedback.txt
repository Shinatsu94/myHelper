/**
 * 檔案名稱: CSIE_41343236_1031_1.java
 * 功能: 建立任意長度的一維陣列，將內容由小到大排序，並輸出排序結果。
 * 作者: 請填寫
 * 日期: 請填寫
 * 備註: 使用系統輸入讀取資料，支援負數，避免直接依靠 EOF。
 */
import java.util.Scanner;
import java.util.Arrays;

public class CSIE_41343236_1031_1 {
    public static void main(String[] args) {
        // 使用 try-with-resources 以自動關閉 Scanner，穩定資源管理
        try (Scanner scanner = new Scanner(System.in)) {
            int n = 0;
            if (scanner.hasNextInt()) {
                n = scanner.nextInt();
            }

            // 基本輸入驗證：長度不可為負
            if (n < 0) {
                System.out.println();
                return;
            }

            int[] arr = new int[n];
            for (int i = 0; i < n; i++) {
                if (scanner.hasNextInt()) {
                    arr[i] = scanner.nextInt();
                } else {
                    // 若輸入不足，補上 0，避免 ArrayIndexOutOfBounds
                    arr[i] = 0;
                }
            }

            // 升冪排序
            Arrays.sort(arr);

            // 使用 StringBuilder 組裝輸出，避免多次 I/O 呼叫
            StringBuilder sb = new StringBuilder();
            for (int i = 0; i < n; i++) {
                if (i > 0) sb.append(' ');
                sb.append(arr[i]);
            }

            System.out.println(sb.toString());
        }
    }
}
