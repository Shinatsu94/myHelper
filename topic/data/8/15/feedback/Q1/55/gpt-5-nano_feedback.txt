/*
 * 檔案名稱: CSIE_41343236_1031_1.java
 * 功能簡述: 讀取整數長度 n 與 n 個整數，將它們排序後輸出（升冪）
 * 作者: 教授
 * 日期: 2025-11-14
 * 版本: 1.0
 */
import java.util.Arrays;
import java.util.Scanner;

public class CSIE_41343236_1031_1 {
    public static void main(String[] args) {
        // 使用 try-with-resources 以自動關閉 Scanner
        try (Scanner scanner = new Scanner(System.in)) {
            // 若沒有輸入，直接結束
            if (!scanner.hasNextInt()) {
                return;
            }

            int n = scanner.nextInt();
            if (n < 0) {
                // 負數長度視為 0，避免產生例外
                n = 0;
            }

            int[] arr = new int[n];

            // 讀取 n 個整數到陣列
            for (int i = 0; i < n; i++) {
                arr[i] = scanner.nextInt();
            }

            // 排序：升冪
            Arrays.sort(arr);

            // 使用 StringBuilder 組裝輸出，避免最後多餘空格
            StringBuilder sb = new StringBuilder();
            for (int i = 0; i < n; i++) {
                if (i > 0) sb.append(' ');
                sb.append(arr[i]);
            }

            System.out.println(sb.toString());
        }
    }
}
