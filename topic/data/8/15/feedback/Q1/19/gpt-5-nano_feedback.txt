 */

import java.util.Scanner;
import java.util.Arrays;

public class CSIE_41343111_1031_1 {
    public static void main(String[] args) {
        // 建立 Scanner 以讀取標準輸入
        Scanner s = new Scanner(System.in);

        // 讀取陣列長度
        int n = s.nextInt();
        int[] arr = new int[n];

        // 讀取 n 個整數
        for (int i = 0; i < n; i++) {
            arr[i] = s.nextInt();
        }

        // 進行升冪排序
        Arrays.sort(arr);

        // 使用 StringBuilder 組裝輸出，避免中間直接輸出導致格式問題
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }

        // 輸出排序結果，並換行
        System.out.println(sb.toString());

        // 關閉資源
        s.close();
    }
}

修改說明：
- 將變數名稱由 in、D 改為更具語意的 n、arr，提升可讀性。
- 以 StringBuilder 組裝輸出，並在輸出時只產生一次空格分隔，避免首尾空格問題。
- 最後輸出加上換行，符合大多數測資對輸出格式的預期。
- 加入完整的檔案頭註解，說明檔案名稱、功能、作者與日期，提升維護性。
- 將資源釋放自動化：關閉 Scanner。
