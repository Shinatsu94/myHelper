/*
 * File: CSIE_51015120_1031_1.java
 * 功能: 對任意長度的一維整數陣列進行升序排序，並輸出結果（以空格分隔，末尾不留空格）
 * 作者: 教授姓名
 * 日期: 2025-11-12
 * 說明: 使用 Java 的內建 Arrays.sort 進行排序，提高效能與穩定性。
 */
import java.util.Arrays;
import java.util.Scanner;

public class CSIE_51015120_1031_1 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int n;
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        n = sc.nextInt();

        if (n <= 0) {
            // 輸入長度為0或負數時，直接輸出空行並結束
            System.out.println();
            sc.close();
            return;
        }

        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();
        }

        // 使用內建排序提升效能
        Arrays.sort(arr);

        // 使用簡潔輸出，避免最後多餘空格
        for (int i = 0; i < n; i++) {
            if (i > 0) System.out.print(' ');
            System.out.print(arr[i]);
        }
        System.out.println();

        sc.close();
    }
}
