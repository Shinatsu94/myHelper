檔案名稱: CSIE_41343220_1031_3a.java
功能說明: 讀入長度 n，接著讀入 n 個整數，將它們升序排序後輸出，元素以空格分隔。
作者: 你的姓名
日期: 2025-11-12
*/
import java.util.*;

public class CSIE_41343220_1031_3a {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n;
        if (sc.hasNextInt()) {
            n = sc.nextInt();
        } else {
            // 沒有輸入長度時，直接結束
            return;
        }

        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                arr[i] = 0; // 若輸入不足，預設為 0
            }
        }

        Arrays.sort(arr);

        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }

        System.out.print(sb.toString());
    }
}
