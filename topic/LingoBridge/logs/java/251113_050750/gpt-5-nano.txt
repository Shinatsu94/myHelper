/**
 * 程式名稱 : Unicode 內碼輸出
 * 功能說明：
 *   輸出虎科大與學生中文姓名的每個字的 Unicode 內碼，格式為 "<字> <內碼>"，先印虎科大三個字，再印學生姓名三個字。
 * 日期: 2025/09/26
 * 作者: 張竣傑
 * 學號: 41343224
 */
public class CSIE_41343224_0926_3 {
    public static void main(String[] args) {
        // 先印出虎科大
        String[] chars = {"虎", "科", "大", "林", "武", "杰"}; // -> 林武杰為學生姓名
        for (String ch : chars) {
            int code = ch.codePointAt(0); // 取得Unicode內碼（十進位表示）
            System.out.println(ch + " " + code);
        }
    }
}
