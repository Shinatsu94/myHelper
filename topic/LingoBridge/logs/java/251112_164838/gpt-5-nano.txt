import java.util.Scanner;
import java.util.Arrays;

/*
檔案名稱: CSIE_41343136_1031_1.java
功能: 讀取整數 n 與 n 個整數，將其由小到大排序，並輸出排序結果
作者: 你的名字
日期: 2025-11-12
*/
public class CSIE_41343136_1031_1 {
    public static void main(String[] args) {
        Scanner s = new Scanner(System.in);
        int n = s.nextInt();
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            arr[i] = s.nextInt();
        }
        s.close();

        // 使用標準排序，避免自訂排序的溢位風險並提升可讀性
        Arrays.sort(arr);

        // 使用 StringBuilder 輸出，避免多次輸出造成效能問題
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.print(sb.toString());
    }
}

- 將自訂的兩兩交換與加減法交換改為使用 Arrays.sort，避免可能的整數溢位風險，提升穩定性與可讀性。
- 增加標頭註解，符合作業需求的標頭說明與文件資訊。
