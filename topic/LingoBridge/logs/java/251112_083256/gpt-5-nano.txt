// 檔案名稱: LeibnizPi.java
// 功能: 計算 pi = 4 * (1 - 1/3 + 1/5 - 1/7 + ...，共 i 項)，
//       輸出 i = 10000, 20000, ..., 100000 時的 pi 值，pi 保留小數點後10位。
// 作者: 教師
// 日期: 2025-11-12
// 程式內容含適當註解，且輸出對齊符合題目格式。

public class LeibnizPi {
    public static void main(String[] args) {
        // 逐步增加項數 i，輸出 i 與對應的 pi 值
        for (int i = 10000; i <= 100000; i += 10000) {
            double sum = 0.0;
            // Leibniz 求和：sum_{k=0}^{i-1} (-1)^k / (2k+1)
            for (int k = 0; k < i; k++) {
                double term = 1.0 / (2.0 * k + 1.0);
                if ((k & 1) == 0) { // k 偶數，正項
                    sum += term;
                } else {            // k 奇數，負項
                    sum -= term;
                }
            }
            double pi = 4.0 * sum;
            // 輸出格式：i 以六位數右對齊，pi 保留小數點後10位
            System.out.printf("i = %6d, pi = %.10f%n", i, pi);
        }
    }
}
