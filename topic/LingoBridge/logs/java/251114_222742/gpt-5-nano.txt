/*
檔案名稱: ArraySortAsc.java
功能說明: 讀取任意長度的一維整數陣列，將內容由小到大排序，輸出排序結果
作者: 你的名字
日期: 2025-11-14
說明: 採用標準輸入方式讀取 n 與 n 個整數，使用 Arrays.sort 進行排序，輸出以空格分隔的結果
*/
import java.util.Arrays;
import java.util.Scanner;

public class ArraySortAsc {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);

        // 讀取陣列長度
        if (!input.hasNextInt()) {
            input.close();
            return;
        }
        int n = input.nextInt();

        // 讀取 n 個整數
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (input.hasNextInt()) {
                arr[i] = input.nextInt();
            } else {
                // 若輸入不足，預設為 0，以避免執行時錯誤
                arr[i] = 0;
            }
        }

        input.close();

        // 升序排序
        Arrays.sort(arr);

        // 輸出排序結果，空格分隔，最後換行
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());
    }
}
