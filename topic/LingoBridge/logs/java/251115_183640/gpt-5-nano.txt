檔案名稱: WellOrderedNumber.java
功能說明: 判斷使用者輸入的數字是否為「嚴格遞增的數字」（自左至右每一位皆嚴格遞增）。
作者: 你的名字
日期: 2025-11-15
*/
import java.util.Scanner;

public class WellOrderedNumber {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        String inputStr = sc.nextLine().trim();
        sc.close();

        boolean isWellOrdered = true;
        if (inputStr.length() == 0) {
            isWellOrdered = false;
        } else {
            int prevDigit = inputStr.charAt(0) - '0';
            for (int idx = 1; idx < inputStr.length(); idx++) {
                int currDigit = inputStr.charAt(idx) - '0';
                if (prevDigit >= currDigit) {
                    isWellOrdered = false;
                    break;
                }
                prevDigit = currDigit;
            }
        }

        System.out.print(inputStr);
        if (isWellOrdered) {
            System.out.println(" is well-ordered number.");
        } else {
            System.out.println(" is not well-ordered number.");
        }
    }
}
