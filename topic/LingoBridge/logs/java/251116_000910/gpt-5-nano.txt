/* 
 * 檔案名稱: CSIE_41343217_1003_3.java
 * 功能: 計算在 1..100000 之間，數字各位數之和等於指定值 A 的數量
 * 作者: 參考用戶
 * 日期: 2025-11-15
 * 說明: 透過逐個數字計算位數和，判斷是否等於 A，輸出符合條件的數量。
 * 註解: 內部對關鍵步驟均有註解，提升可讀性與維護性。
 */

import java.util.Scanner;

public class CSIE_41343217_1003_3 {

    // 計算整數 number 的各位數字和
    private static int sumDigits(int number) {
        int sum = 0;
        int temp = number;
        while (temp > 0) {
            sum += temp % 10;
            temp /= 10;
        }
        return sum;
    }

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        // 目標和
        int targetSum = scanner.nextInt();

        // 邊界檢查：在 1..100000 的數字中，最多只有 99999 其位數和為 45，因此若 targetSum 不在 0..45 之間，結果為 0
        if (targetSum < 0 || targetSum > 45) {
            System.out.println(0);
            scanner.close();
            return;
        }

        int count = 0;
        final int LIMIT = 100000; // 上界

        for (int i = 1; i <= LIMIT; i++) {
            if (sumDigits(i) == targetSum) {
                count++;
            }
        }

        System.out.println(count);
        scanner.close();
    }
}
