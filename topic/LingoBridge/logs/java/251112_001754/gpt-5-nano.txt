import java.util.Scanner;

public class CSIE_41343208_1003_1 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int a = sc.nextInt();
        int b = sc.nextInt();

        // 交換前的值
        System.out.println(a);
        System.out.println(b);

        // 使用位元異或交換法，僅使用兩個變數 a, b，避免溢位風險
        a = a ^ b;
        b = a ^ b;
        a = a ^ b;

        // 交換後的值
        System.out.println(a);
        System.out.println(b);

        sc.close();
    }
}
