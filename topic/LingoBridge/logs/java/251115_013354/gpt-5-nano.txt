- 使用快速輸入與 StringBuilder 輸出：原本的 Scanner 會較慢，若題目輸入量大可能影響執行時間；快速輸入與累積輸出能提升效能並避免尾端空格問題。

import java.io.*;
import java.util.Arrays;

/**
 * File: CSIE_41343219_1031_1.java
 * 功能: 建立任意長度的一維陣列，對內容做升序排序，並輸出排序結果。
 * 作者: 教師
 * 日期: 2025-11-14
 * 版本: 1.0
 */
public class CSIE_41343219_1031_1 {
    // 快速輸入內建類別
    private static class FastScanner {
        private final InputStream in;
        private final byte[] buffer = new byte[1 << 16];
        private int ptr = 0, len = 0;

        FastScanner(InputStream is) {
            in = is;
        }

        private int read() throws IOException {
            if (ptr >= len) {
                len = in.read(buffer);
                ptr = 0;
                if (len <= 0) return -1;
            }
            return buffer[ptr++];
        }

        int nextInt() throws IOException {
            int c;
            do {
                c = read();
            } while (c <= ' ' && c != -1);
            boolean neg = false;
            if (c == '-') {
                neg = true;
                c = read();
            }
            int val = 0;
            while (c > ' ') {
                val = val * 10 + (c - '0');
                c = read();
            }
            return neg ? -val : val;
        }
    }

    public static void main(String[] args) throws Exception {
        FastScanner fs = new FastScanner(System.in);
        int n;
        try {
            n = fs.nextInt();
        } catch (Exception e) {
            return;
        }

        int[] ary = new int[n];
        for (int i = 0; i < n; i++) {
            ary[i] = fs.nextInt();
        }

        // 使用內建排序提升穩定性與效能
        Arrays.sort(ary);

        // 輸出，避免尾端多餘空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(ary[i]);
        }
        if (sb.length() > 0) System.out.print(sb.toString());
        // 根據題目若需要換行可在此加入 System.out.println();
    }
}
