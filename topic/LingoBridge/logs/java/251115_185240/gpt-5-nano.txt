/*
File: ReverseDigits.java
Description: 讀取一個整數，反轉其數字順序；若原數為負，輸出相對應的負數反轉值。
Author: 教師
Date: 2025-11-15
*/

import java.util.Scanner;

public class ReverseDigits {
    public static void main(String[] args) {
        // 建立輸入掃描器，僅讀取單一數字
        Scanner sc = new Scanner(System.in);
        long n = sc.nextLong();
        sc.close();

        // 判斷正負號，並取絕對值做反轉
        boolean isNegative = n < 0;
        long absN = isNegative ? -n : n;

        long rev = 0;
        // 單次遍歷完成數字反轉
        while (absN > 0) {
            long digit = absN % 10;
            rev = rev * 10 + digit;
            absN /= 10;
        }

        long result = isNegative ? -rev : rev;
        System.out.println(result);
    }
}
