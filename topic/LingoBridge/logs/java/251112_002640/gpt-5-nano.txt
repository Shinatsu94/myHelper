        int a = input.nextInt();
        int b = input.nextInt();

        // 使用不產生第三變數的位元運算交換，若相等則不需要交換
        if (a != b) {
            a = a ^ b;
            b = a ^ b;

        System.out.println(a);
        System.out.println(b);

        input.close();
    }
}
