/*
File: CSIE_41343101_1031_2.java
Description: 將一維陣列中每個元素替換為該元素右側的最大值，最後一個元素設為 -1，並輸出結果。
Author: ChatGPT
Date: 2025-11-14
*/

import java.util.Scanner;

public class CSIE_41343101_1031_2 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int arrSize = sc.nextInt();
        int[] arr = new int[Math.max(arrSize, 0)];

        for (int i = 0; i < arrSize; i++) {
            arr[i] = sc.nextInt();
        }

        // 將每個元素替換為其右側的最大值，最後一個為 -1
        for (int i = 0; i < arrSize - 1; i++) {
            int max = arr[i + 1];
            for (int j = i + 2; j < arrSize; j++) {
                if (arr[j] > max) {
                    max = arr[j];
                }
            }
            arr[i] = max;
        }

        if (arrSize > 0) {
            arr[arrSize - 1] = -1;
        }

        // 輸出結果，避免尾端多餘空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < arrSize; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        sc.close();
    }
}
