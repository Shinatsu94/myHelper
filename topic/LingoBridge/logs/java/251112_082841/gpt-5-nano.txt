```java
/*
 檔案名稱: PiLeibniz.java
 功能: 使用 Leibniz 公式逐步計算 pi 的近似值，輸出 i = 10000, 20000, ..., 100000 以及對應的 pi 值
 作者: 你的名字
 日期: 2025-11-12
 說明: 以累加的方式計算 pi，使用 i 為項數，並輸出格式符合題目要求
*/
public class PiLeibniz {
    public static void main(String[] args) {
        final int MAX_I = 100000; // 最大項數
        final int STEP = 10000;   // 步階
        double sum = 0.0;         // Leibniz 累加和
        double term = 1.0;        // 當前項的分母分子符號，初始為 +1

        // 從第1項開始累加到 MAX_I 項
        for (int i = 1; i <= MAX_I; i++) {
            sum += term / (2.0 * i - 1);
            term = -term; // 下一項符號取反
            if (i % STEP == 0) {
                double piApprox = 4.0 * sum;
                System.out.printf("i = %6d, pi = %.10f%n", i, piApprox);
            }
        }
    }
}
```
