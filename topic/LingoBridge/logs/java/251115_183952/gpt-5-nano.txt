/*
檔案名稱: CSIE_41343226_1017_2_Optimized.java
功能: 判斷輸入的十進位數字的數字是否嚴格遞增
作者: 教授
日期: 2025-11-15
*/
public class CSIE_41343226_1017_2_Optimized {
    public static void main(String[] args) {
        java.util.Scanner in = new java.util.Scanner(System.in);
        String s = "";
        if (in.hasNextLine()) {
            s = in.nextLine().trim();
        }
        String original = s;

        boolean wellOrdered = true;

        if (s.length() == 0) {
            wellOrdered = false;
        } else {
            int start = 0;
            // 若有前導符號，視為非正整數的情況
            if (s.charAt(0) == '-') {
                wellOrdered = false;
            }
            if (wellOrdered && s.charAt(0) == '+') {
                start = 1;
            }
            for (int i = start; i < s.length() - 1; i++) {
                char a = s.charAt(i);
                char b = s.charAt(i + 1);
                // 確保都是數字
                if (a < '0' || a > '9' || b < '0' || b > '9') {
                    wellOrdered = false;
                    break;
                }
                // 檢查嚴格遞增
                if (a >= b) {
                    wellOrdered = false;
                    break;
                }
            }
        }

        if (wellOrdered) {
            System.out.printf("%s is well-ordered number.%n", original);
        } else {
            System.out.printf("%s is not well-ordered number.%n", original);
        }
        in.close();
    }
}
