完整修改後程式碼
/* 
檔案名稱: CSIE_41343219_1031_2.java
功能說明: 從使用者讀取一個整數長度 n，接著讀入 n 個整數，輸出由小到大的排序結果（以空格分隔）。
作者: 教師姓名 (示意)
日期: 2025-11-14
註解: 使用 Arrays.sort 進行排序，確保正確性與支援負數；輸出使用 StringBuilder 組裝，避免尾端多餘空白。
*/
import java.util.Scanner;
import java.util.Arrays;

public class CSIE_41343219_1031_2 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int n = sc.nextInt();
        int[] ary = new int[n];
        for (int i = 0; i < n; i++) {
            ary[i] = sc.nextInt();
        }
        sc.close();

        // 使用內建排序，確保正確性與對負數的支援
        Arrays.sort(ary);

        // 輸出，避免尾端多餘的空白
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(ary[i]);
        }
        System.out.println(sb.toString());
    }
}
