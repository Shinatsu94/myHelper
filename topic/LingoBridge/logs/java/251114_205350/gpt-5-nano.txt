/* 
檔案名稱: SortArrayAscending.java
功能說明: 讀取整數長度 n 與 n 個整數，將陣列由小到大排序，輸出排序結果
作者: 徐唯祐 (示例)
日期: 2025-11-14
*/
import java.util.Arrays;
import java.util.Scanner;

public class SortArrayAscending {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }

        int n = sc.nextInt();
        int[] arr = new int[n];

        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                // 若輸入不足，預設為 0
                arr[i] = 0;
            }
        }
        sc.close();

        // 使用內建排序，提升穩定性與效率
        Arrays.sort(arr);

        // 輸出：數字間用單一空格分隔，最後換行
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < arr.length; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());
    }
}
