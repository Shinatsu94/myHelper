public class CSIE_41343234_0926_3 {
    public static void main(String[] args) {
        String[] first = {"虎","科","大"};
        String[] second = {"林","武","杰"};

        printPairs(first);
        printPairs(second);
    }

    private static void printPairs(String[] chars) {
        for (String ch : chars) {
            int code = ch.codePointAt(0);
            System.out.println(ch + " " + code);
        }
    }
}
