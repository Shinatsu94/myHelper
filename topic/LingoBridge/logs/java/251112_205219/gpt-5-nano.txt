/*
File: CSIE_41343247_1031_3b.java
Name: 盧冠宇
Student_ID: 41343247
功能: 讀入一個整數 n，接著讀入 n 個整數，將它們升冪排序，最後輸出排序後的陣列內容（以空格分隔）。
日期: 2025-11-12
*/
import java.util.Arrays;
import java.util.Scanner;

public class CSIE_41343247_1031_3b {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        // 讀取陣列長度
        if (!scanner.hasNextInt()) {
            scanner.close();
            return;
        }
        int n = scanner.nextInt();

        // 讀取 n 個整數到陣列
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (scanner.hasNextInt()) {
                arr[i] = scanner.nextInt();
            } else {
                // 若輸入不足，以 0 佔位，避免例外
                arr[i] = 0;
            }
        }

        // 排序（升冪）
        Arrays.sort(arr);

        // 輸出排序後的陣列內容，空格分隔
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        scanner.close();
    }
}

- 使用 Arrays.sort 以確保排序正確且效率穩定，時間複雜度 O(n log n)。
