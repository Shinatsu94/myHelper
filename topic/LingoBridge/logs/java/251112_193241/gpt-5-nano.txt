import java.util.*;

public class CSIE_41343229_1031_2 {
    /*
     * 檔案名稱: CSIE_41343229_1031_2.java
     * 功能: 讀取整數長度 n 與 n 個整數，輸出升序排序的結果（由小到大）
     * 作者: 你的名字
     * 日期: 2025-11-12
     */
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        // 讀取長度
        int n = sc.nextInt();
        int[] a = new int[n];
        for (int i = 0; i < n; i++) {
            a[i] = sc.nextInt();
        }

        // 升序排序
        Arrays.sort(a);

        // 輸出結果，元素以空格分隔
        for (int i = 0; i < n; i++) {
            if (i > 0) System.out.print(" ");
            System.out.print(a[i]);
        }
        // System.out.println(); // 可選：輸出結尾換行
        sc.close();
    }
}
