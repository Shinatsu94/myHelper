/**
 * File: CSIE_51015114_1031_3.java
 * 功能: 建立下三角形二維陣列，並尋找指定數字在陣列中的位置
 * 作者: 你的名字
 * 日期: 2025-11-14
 * 說明: 本程式先建立高度為 n 的下三角形，內容為自增整數 1..n(n+1)/2。
 *       若輸入的 num 在陣列範圍內，則以公式計算其所在行列（均從 1 開始）。
 *       若不在範圍，輸出提示訊息。
 */

import java.util.Scanner;

class CSIE_51015114_1031_3 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int n = sc.nextInt();     // 下三角形高度
        int num = sc.nextInt();   // 要尋找的數字

        // 建立完整下三角形二維陣列
        int[][] arr = new int[n][];
        int cnt = 1;
        for (int i = 0; i < n; i++) {
            arr[i] = new int[i + 1]; // 每列 i+1 個元素
            for (int j = 0; j <= i; j++) {
                arr[i][j] = cnt++;
            }
        }

        // 透過數學公式直接定位 num 在陣列中的位置
        int max = n * (n + 1) / 2;
        if (num < 1 || num > max) {
            System.out.printf("not find %d in array.", num);
        } else {
            // r 為 num 所在的列（1-based），c 為行中位置（1-based）
            int r = (int) Math.ceil((Math.sqrt(8.0 * num + 1) - 1) / 2.0);
            int c = num - (r - 1) * r / 2;
            if (r > n) {
                System.out.printf("not find %d in array.", num);
            } else {
                System.out.printf("%d[%d, %d]\n", num, r, c);
            }
        }

        sc.close();
    }
}
