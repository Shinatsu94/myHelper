import java.util.Arrays;

/*
 檔案名稱: Main.java
 功能: 依輸入長度與對應整數，輸出由小到大排序後的結果
 作者: 教師
 日期: 2025-11-12
 使用說明: 讀取第一個整數為陣列長度 n，接著讀入 n 個整數，將其由小到大排序後輸出，元素以空格分隔。
*/
public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取陣列長度
        if (!sc.hasNextInt()) {
            return;
        }
        int n = sc.nextInt();

        int[] arr = new int[Math.max(0, n)];

        // 依序讀取 n 個整數
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                arr[i] = 0; // 若輸入不足，補 0，避免例外
            }
        }

        // 排序
        Arrays.sort(arr);

        // 輸出，空格分隔，末端不留多餘空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < arr.length; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        sc.close();
    }
}
