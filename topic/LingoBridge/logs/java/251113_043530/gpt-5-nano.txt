修改後完整程式碼
// 檔案名稱: CSIE_<YourID>_<Date>_PrintChineseCode.java
// 功能: 依題意輸出虎科大及學生姓名的每個字的 Unicode 內碼，格式為：字元 內碼 (十六進位)。
// 作者: 你自己的名字
// 日期: 你提交的日期
// 說明: 使用字元陣列控制輸出順序，避免不必要的迴圈與動畫。內碼以 0xXXXX 的十六進位格式呈現。

public class CSIE_PrintChineseCode {
    public static void main(String[] args) {
        // 依題意先印出虎科大，後印學生姓名 林武杰
        char[] chars = {'虎', '科', '大', '林', '武', '杰'};

        for (char ch : chars) {
            int codePoint = (int) ch;
            // 以十六進位顯示 Unicode 編碼點，格式為 0xXXXX
            System.out.println(ch + " 0x" + String.format("%04X", codePoint));
        }
    }
}
