/*
Filename: CSIE_41343203_1031_3b.java
Description: 建立一個任意長度的一維整數陣列，讀入長度與數值，將陣列內容由小到大排序後輸出。
Author: 教師
Date: 2025-11-14
*/
import java.util.Arrays;
import java.util.Scanner;

public class CSIE_41343203_1031_3b {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取陣列長度
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();
        if (n <= 0) {
            sc.close();
            return;
        }

        // 讀取 n 個整數元素
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                // 若輸入不足，預設為 0，避免程式異常
                arr[i] = 0;
            }
        }

        // 排序
        Arrays.sort(arr);

        // 輸出，空格分隔，尾端不留空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        sc.close();
    }
}
