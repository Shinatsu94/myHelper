/*
檔案名稱: WellOrderedNumber.java
功能: 判斷輸入整數的數字是否左至右嚴格遞增，並輸出對應訊息
作者: 教師姓名
日期: 2025-11-15
*/

import java.util.Scanner;

public class WellOrderedNumber {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int number = scanner.nextInt();
        scanner.close();

        int n = Math.abs(number);
        boolean isWellOrdered = true;

        int prev = n % 10;
        n /= 10;

        while (n > 0) {
            int curr = n % 10;
            // 左至右為嚴格遞增，現在的位數(curr)必須小於先前的位數(prev)
            if (!(curr < prev)) {
                isWellOrdered = false;
                break;
            }
            prev = curr;
            n /= 10;
        }

        if (isWellOrdered) {
            System.out.println(number + " is well-ordered number.");
        } else {
            System.out.println(number + " is not well-ordered number.");
        }
    }
}
