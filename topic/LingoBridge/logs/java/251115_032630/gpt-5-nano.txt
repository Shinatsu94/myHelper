import java.util.Scanner;

public class TriIndexFinder {
    /*
     * 檔案名稱: TriIndexFinder.java
     * 功能說明: 建立下三角形二維陣列，依序填入數字 1..n(n+1)/2，
     *           找出使用者輸入的 num 在陣列中的位置，輸出格式為: num[row, col]
     * 作者: 
     * 日期: 
     */
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀入高度 n 與要找的數字 num
        if (!sc.hasNextInt()) return;
        int n = sc.nextInt();
        if (!sc.hasNextInt()) return;
        int num = sc.nextInt();

        // 建立完整的下三角形二維陣列，內容依 1,2,3,... 填入
        int[][] tri = new int[n][];
        int val = 1;
        for (int i = 0; i < n; i++) {
            tri[i] = new int[i + 1];
            for (int j = 0; j <= i; j++) {
                tri[i][j] = val++;
            }
        }

        int total = n * (n + 1) / 2;
        if (num <= 0 || num > total) {
            System.out.println("not find " + num + " in array.");
            return;
        }

        // 計算位置：行 r 與列 c，皆從 1 開始
        long r = (long) Math.ceil((Math.sqrt(1.0 + 8.0 * (long) num) - 1.0) / 2.0);
        long prev = (r - 1) * r / 2;
        long c = num - prev;
        System.out.printf("%d[%d, %d]%n", num, (int) r, (int) c);
    }
}
