/**
 * 檔案名稱: SelectionSortExample.java
 * 功能說明: 讀取一維整數陣列長度與元素，將陣列內容由小到大排序，輸出排序結果（以空格分隔的一行）。
 * 作者: 教師
 * 日期: 2025-11-12
 */
import java.util.Scanner;

public class SelectionSortExample {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);

        // 讀取陣列長度
        int n = input.nextInt();
        int[] arr = new int[n];

        // 逐一讀入陣列內容
        for (int i = 0; i < n; i++) {
            arr[i] = input.nextInt();
        }

        // 使用選擇排序將陣列由小到大排序
        for (int i = 0; i < n - 1; i++) {
            int minIndex = i;
            for (int j = i + 1; j < n; j++) {
                if (arr[j] < arr[minIndex]) {
                    minIndex = j;
                }
            }
            // 交換
            int tmp = arr[i];
            arr[i] = arr[minIndex];
            arr[minIndex] = tmp;
        }

        // 輸出排序結果：以空格分隔，單行輸出
        for (int i = 0; i < n; i++) {
            if (i > 0) System.out.print(" ");
            System.out.print(arr[i]);
        }
        System.out.println();

        input.close();
    }
}
