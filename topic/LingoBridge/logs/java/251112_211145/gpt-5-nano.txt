```java
/*
 * File: Main.java
 * 功能: 讀取第一個整數作為長度 n，接著讀入 n 個整數，將其升序排序後輸出，元素以空格分隔。
 * 作者: 你的名字
 * 日期: 2025-11-12
 */

import java.util.Arrays;
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取長度
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();
        if (n <= 0) {
            // 無元素時不輸出內容
            sc.close();
            return;
        }

        // 讀取 n 個整數
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            arr[i] = sc.hasNextInt() ? sc.nextInt() : 0;
        }
        sc.close();

        // 排序
        Arrays.sort(arr);

        // 輸出，元素以空格分隔，尾端不留空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());
    }
}
```

修改部分的原因：
- 使用一維整數陣列與 Arrays.sort，直接且有效地完成題目需求，且支援負數輸入。
