// File: Sorting1DArray.java
// 功能: 讀取使用者輸入的長度 n 與 n 個整數，將其由小到大排序，輸出排序後的陣列
// 作者: 你的名字
// 日期: 2025-11-14

import java.util.*;

public class Sorting1DArray {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取陣列長度
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();
        int[] arr = new int[n];

        // 讀入 n 個整數
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                arr[i] = 0; // 若輸入不足，補零以避免例外
            }
        }
        sc.close();

        // 進行升冪排序
        Arrays.sort(arr);

        // 輸出排序後的結果，元素以空格分隔
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());
    }
}
