import java.util.Scanner;

public class CSIE_41343229_1003_3 {
    // 使用 XOR 交換法，無需額外變數
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int A = sc.nextInt();
        int B = sc.nextInt();

        // 不使用額外變數，直接在 A、B 之間交換
        A = A ^ B;
        B = A ^ B;
        A = A ^ B;

        System.out.println(A);
        System.out.println(B);

        sc.close();
    }
}
