// File: CSIE_41343242_1031_3a.java
// 功能: 讀取整數 n 與 n 個整數，輸出升序排序後的陣列內容。
// 作者: 設計者姓名
// 日期: 2025-11-14
// 說明: 使用 Arrays.sort() 進行排序，支援負數，輸出為單行以空格分隔。
import java.util.Scanner;
import java.util.Arrays;

public class CSIE_41343242_1031_3a {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }

        int n = sc.nextInt();
        if (n <= 0) {
            sc.close();
            return;
        }

        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                // 若輸入不足，預設為 0（可根據需求調整为提前結束或抛出例外）
                arr[i] = 0;
            }
        }

        Arrays.sort(arr);

        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        sc.close();
    }
}
