/*
檔案名稱: UnicodePrinter.java
功能: 印出虎科大三個字與學生姓名三個字的內碼
作者: 教師
日期: 2025-11-12
說明: 依題意輸出每個字與其對應的 Unicode 編碼，格式為 "<字> <內碼>"，
並先印出虎科大三字，再印出學生姓名三字。
*/
public class UnicodePrinter {
    public static void main(String[] args) {
        // 先印虎科大三字及其內碼
        printCharCode('虎');
        printCharCode('科');
        printCharCode('大');

        // 再印學生中文姓名三字及其內碼
        // 這裡以常見的姓名範例「林武杰」為例，若實際姓名不同請自行調整字元
        printCharCode('林');
        printCharCode('武');
        printCharCode('杰');
    }

    // 輔助方法：輸出單一字及其對應的內碼
    private static void printCharCode(char ch) {
        System.out.println(ch + " " + (int) ch);
    }
}
