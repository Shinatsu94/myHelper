public class CSIE_41343220_1017_1 {
    /*
     * 檔案名稱: CSIE_41343220_1017_1.java
     * 功能說明: 計算對於公式 pi = 4(1 - 1/3 + 1/5 - 1/7 + ...) 的兩個需求
     *        (a) 当 i = 10000, 20000, ..., 100000 時輸出 pi 值
     *        (b) 找出使 pi 的近似值等於/大於 3.14159 的最小 i 值並輸出
     * 作者: 學生
     * 日期: 2025-11-15
     */

    public static void main(String[] args) {
        double sum = 0.0;

        // (a) i = 1 ~ 100000，並在每 10000 時輸出 pi 的近似值
        int nextPrint = 10000;
        for (int i = 1; i <= 100000; i++) {
            // 奇數 i 加，偶數 i 減
            sum += ((i & 1) == 1) ? 1.0 / (2 * i - 1) : -1.0 / (2 * i - 1);

            if (i == nextPrint) {
                System.out.printf("i = %6d, pi = %.10f%n", i, sum * 4.0);
                nextPrint += 10000;
            }
        }

        System.out.println();

        // (b) 從 i >= 100000 開始找最小 i，使 pi * 4 >= 3.14159
        sum = 0.0;
        int i = 1;

        while (true) {
            sum += ((i & 1) == 1) ? 1.0 / (2 * i - 1) : -1.0 / (2 * i - 1);

            if (i >= 100000 && sum * 4.0 >= 3.14159) {
                System.out.printf("i = %6d%n", i);
                break;
            }
            i++;
        }
    }
}
