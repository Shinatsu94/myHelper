 * Author: 教師

import java.util.Scanner;

public class CSIE_51015130_1003_1 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int A = sc.nextInt();
        int B = sc.nextInt();

        // 使用 XOR 交換兩個變數的值，且不使用額外變數
        A = A ^ B;
        B = A ^ B;
        A = A ^ B;

        System.out.println(A);
        System.out.println(B);

        sc.close();
    }
}
