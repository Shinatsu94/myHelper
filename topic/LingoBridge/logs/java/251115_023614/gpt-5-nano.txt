修改後完整程式碼
/* 
檔案名稱: LowerTriangularMatrix.java
功能: 建立並列印高度為 n 的下三角形二維陣列，元素為 '*'
作者: 徐唯祐
日期: 2025-11-14
描述: 讀取輸入 n（0 < n <= 20），在 2D 陣列中填入下三角星號，然後逐行輸出三角形。
*/
import java.util.Scanner;

public class LowerTriangularMatrix {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        int n = scanner.nextInt();

        // 基本輸入檢查，符合題意的範圍才執行
        if (n <= 0 || n > 20) {
            scanner.close();
            return;
        }

        // 建立並填入下三角形的二維陣列
        char[][] tri = new char[n][n];
        for (int row = 0; row < n; row++) {
            for (int col = 0; col <= row; col++) {
                tri[row][col] = '*';
            }
        }

        // 輸出結果：每列輸出前 (row+1) 個 '*'，其餘位置不輸出
        for (int row = 0; row < n; row++) {
            for (int col = 0; col <= row; col++) {
                System.out.print(tri[row][col]);
            }
            System.out.println();
        }

        scanner.close();
    }
}
