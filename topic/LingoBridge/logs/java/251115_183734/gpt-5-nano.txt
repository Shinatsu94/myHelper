/*
 * File: CSIE_41343221_1017_2.java
 * 功能: 判斷輸入的整數各位數是否嚴格遞增
 * 作者: 教師
 * 日期: 2025-11-15
 * 備註: 程式以字串方式處理，避免整數轉換造成的問題，並對輸入做前後空白修剪與基本驗證。
 */

import java.util.Scanner;

public class CSIE_41343221_1017_2 {

    // 判斷字串表示的非負整數各位數是否嚴格遞增
    public static boolean isWellOrdered(String s) {
        if (s == null || s.length() == 0) return false;

        int idx = 0;
        if (s.charAt(0) == '-') return false;
        if (s.charAt(0) == '+') idx = 1;

        if (s.length() - idx <= 1) {
            // 一位或零位數，同樣需要驗證其餘字符皆為數字
            for (int j = idx; j < s.length(); j++) {
                if (!Character.isDigit(s.charAt(j))) return false;
            }
            return true;
        }

        // 檢查後續是否為數字並比較相鄰位數
        for (int i = idx; i < s.length() - 1; i++) {
            char a = s.charAt(i);
            char b = s.charAt(i + 1);
            if (!Character.isDigit(a) || !Character.isDigit(b)) return false;
            if (a >= b) return false;
        }

        return true;
    }

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        String input = scanner.nextLine().trim();
        boolean ok = isWellOrdered(input);

        if (ok) {
            System.out.println(input + " is well-ordered number.");
        } else {
            System.out.println(input + " is not well-ordered number.");
        }

        scanner.close();
    }
}
