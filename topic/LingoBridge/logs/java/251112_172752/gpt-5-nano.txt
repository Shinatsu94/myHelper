/*
 檔案名稱: CSIE_41343150_1031_2.java
 功能: 讀入一維整數陣列，對其進行升序排序，並輸出排序結果
 作者: 教師姓名
 日期: 2025-11-12
*/
import java.util.Scanner;

public class CSIE_41343150_1031_2 {
    public static void main(String[] args) {
        Scanner scn = new Scanner(System.in);

        // 讀取長度
        if (!scn.hasNextInt()) {
            scn.close();
            return;
        }
        int n = scn.nextInt();
        int[] arr = new int[n];

        // 讀取陣列元素
        for (int i = 0; i < n; i++) {
            if (scn.hasNextInt()) {
                arr[i] = scn.nextInt();
            } else {
                arr[i] = 0; // 若輸入不足，給予預設值
            }
        }

        // 選取排序法 (升序)
        for (int i = 0; i < n - 1; i++) {
            int minIndex = i;
            for (int j = i + 1; j < n; j++) {
                if (arr[j] < arr[minIndex]) {
                    minIndex = j;
                }
            }
            // 交換 arr[i] 與 arr[minIndex]
            int tmp = arr[i];
            arr[i] = arr[minIndex];
            arr[minIndex] = tmp;
        }

        // 輸出，元素以空格分隔，尾端 no trailing space
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.print(sb.toString());

        scn.close();
    }
}

修改部分原因說明
- 以選取排序確保每一步都能找出當前未排序部分的最小值，避免覆蓋原始資料並導致最終排序錯亂，符合題意的「由小到大」排序需求。
- 使用具描述性的變數名稱與清晰註解，提高可讀性與維護性，減少未來維護成本。
- 標頭註解補足檔案基本資訊，符合「程序檔案註解」的要求。
