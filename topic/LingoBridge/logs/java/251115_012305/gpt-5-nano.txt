```java
/*
檔案名稱: CSIE_41343119_1031_1.java
功能: 將輸入任意長度的一維整數陣列，進行升序排序 (使用 Selection Sort)，並輸出結果。
作者: 金宗翰
日期: 2025-11-14
*/

import java.util.Scanner;

public class CSIE_41343119_1031_1 {

    public static void main(String[] args) {
        Scanner scn = new Scanner(System.in);

        int n = 0;
        if (scn.hasNextInt()) {
            n = scn.nextInt();
        }

        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (scn.hasNextInt()) {
                arr[i] = scn.nextInt();
            } else {
                arr[i] = 0; // 預設值，若輸入不足，避免例外
            }
        }

        // Selection sort: sort arr ascending
        for (int i = 0; i < n - 1; i++) {
            int minIndex = i;
            for (int j = i + 1; j < n; j++) {
                if (arr[j] < arr[minIndex]) {
                    minIndex = j;
                }
            }
            // swap
            if (minIndex != i) {
                int tmp = arr[i];
                arr[i] = arr[minIndex];
                arr[minIndex] = tmp;
            }
        }

        // Output
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        scn.close();
    }
}
```
