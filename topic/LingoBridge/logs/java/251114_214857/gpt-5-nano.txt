檔案名稱: CSIE_51015108_1031_2.java
功能說明: 讀取使用者輸入的長度 n 與 n 個整數，將它們由小到大排序並輸出

import java.util.Arrays;
import java.util.Scanner;

public class CSIE_51015108_1031_2 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int[] array = new int[n];
        for (int i = 0; i < n; i++) {
            array[i] = sc.nextInt();
        }

        Arrays.sort(array);

        for (int i = 0; i < n; i++) {
            if (i > 0) System.out.print(' ');
            System.out.print(array[i]);
        }
        System.out.println();
        sc.close();
    }
}
