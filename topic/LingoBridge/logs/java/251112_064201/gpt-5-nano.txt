/*
  檔案名稱: LeibnizPi.java
  功能描述: 計算勒布尼茲級數近似圓周率。
           (a) 輸出 i = 10000, 20000, ..., 100000 的 pi 值
           (b) 找出最小的 i 使 pi 四捨五入到五位小數等於 3.14159
  作者: 教師姓名
  日期: 2025-11-11
*/

public class LeibnizPi {
    public static void main(String[] args) {
        // Part (a): 輸出 i = 10000, 20000, ..., 100000 的 pi 值
        for (int i = 10000; i <= 100000; i += 10000) {
            double sum = 0.0;
            for (int k = 1; k <= i; k++) {
                double term = (k % 2 == 1) ? (1.0 / (2 * k - 1)) : (-1.0 / (2 * k - 1));
                sum += term;
            }
            double pi = sum * 4.0;
            System.out.printf("i = %6d, pi = %.10f%n", i, pi);
        }

        // Part (b): 找到最小的 i 使 pi 四捨五入到五位小數等於 3.14159
        double target = 3.14159;
        double sum = 0.0;
        int iMin = -1;
        int limit = 2000000; // 安全上限，避免無窮迴圈
        for (int k = 1; k <= limit; k++) {
            double term = (k % 2 == 1) ? (1.0 / (2 * k - 1)) : (-1.0 / (2 * k - 1));
            sum += term;
            double pi = sum * 4.0;
            double rounded = Math.round(pi * 100000.0) / 100000.0;
            // 比對四捨五入後的值與 3.14159，允許微小誤差
            if (Math.abs(rounded - target) < 1e-9) {
                iMin = k;
                break;
            }
        }
        if (iMin != -1) {
            System.out.println("i = " + iMin);
        } else {
            System.out.println("i = not found within limit");
        }
    }
}
