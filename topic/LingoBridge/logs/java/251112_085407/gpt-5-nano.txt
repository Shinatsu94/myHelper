/*
File: LeibnizPiCalculator.java
功能: 使用勒布尼茲級數估算圓周率，輸出指定的 i 與 pi 值，並找出最小的 i 使 pi ≥ 3.14159
作者: 你
日期: 2025-11-12
*/
public class LeibnizPiCalculator {
    public static void main(String[] args) {
        final int STEP = 10000;     // 每次輸出的 i 步進
        final int MAX_I = 100000;   // 最大輸出 i
        double sum = 0.0;
        
        // 從第一項開始累加，逐步到 MAX_I
        for (int k = 1; k <= MAX_I; k++) {
            // 公式：pi ≈ 4 * sum_{t=1..k} [(-1)^{t-1} / (2t - 1)]
            double term = ((k % 2 == 1) ? 1.0 : -1.0) / (2.0 * k - 1.0);
            sum += term;

            if (k % STEP == 0) {
                int i = k;
                double pi = 4.0 * sum;
                // 輸出格式：i 以寬度 6 對齊，pi 保留小數點後 10 位
                System.out.printf("i = %6d, pi = %.10f%n", i, pi);
            }
        }

        // 計算使 pi ≥ 3.14159 的最小 i
        // 重新從頭開始累加，直到符合條件
        sum = 0.0;
        int iMin = -1;
        for (int k = 1; ; k++) {
            double term = ((k % 2 == 1) ? 1.0 : -1.0) / (2.0 * k - 1.0);
            sum += term;
            if (4.0 * sum >= 3.14159) {
                iMin = k;
                break;
            }
            // 安全保護，實務上不太可能到此極限，但避免無窮迴圈
            if (k > 20000000) break;
        }

        System.out.println();
        // 輸出最小 i，使 pi >= 3.14159，格式同上
        System.out.printf("i = %6d%n", iMin);
    }
}
