/*
檔案名稱: LeibnizPi.java
功能: 計算 pi 值，當 i 為 10000, 20000, ..., 100000 時，輸出
      i 與 pi 的值，i 以6位寬對齊，pi 小數點後顯示 10 位。
作者: 使用者
日期: 2025-11-12
說明: 使用 Leibniz 無窮級數 pi = 4 * sum_{t=0}^{i-1} (-1)^t / (2t+1)
      其中 i 為項數，對應於題目中的 2i-1 分母項。
*/
public class LeibnizPi {
    public static void main(String[] args) {
        // 依題目要求，i 為 10000, 20000, ..., 100000
        for (int i = 10000; i <= 100000; i += 10000) {
            double sum = 0.0;
            // 李普尼茲公式的前 i 項
            for (int t = 0; t < i; t++) {
                double term = 1.0 / (2 * t + 1);
                if ((t & 1) == 0) { // 偶數項為正
                    sum += term;
                } else {            // 奇數項為負
                    sum -= term;
                }
            }
            double pi = 4.0 * sum;
            // 輸出格式：i 6 位寬對齊，pi 保留小數點後 10 位
            System.out.printf("i = %6d, pi = %.10f%n", i, pi);
        }
    }
}
