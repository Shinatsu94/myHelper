這是一個符合題意的簡單實作，使用 Scanner 讀取長度與對應數字，排序後輸出。若要進一步提高效能，可改以 BufferedInputStream 與 StringTokenizer 作為輸入方式。

import java.util.Arrays;
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取陣列長度 n
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();

        // 讀取 n 個整數，存入陣列
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();
        }

        // 進行升序排序
        Arrays.sort(arr);

        // 輸出排序結果，使用空格分隔，結尾換行
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        sc.close();
    }
}
