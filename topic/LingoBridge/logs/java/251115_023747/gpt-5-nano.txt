修改後完整程式碼
import java.util.Scanner;

/**
 * File: CSIE_41343131_1031_3a.java
 * 功能: 建立並印出下三角形的二維陣列
 * 作者: 你的姓名
 * 日期: 2025-11-14
 */
public class CSIE_41343131_1031_3a {
    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);
        int n = 0;

        if (in.hasNextInt()) {
            n = in.nextInt();
        }

        // 簡單的輸入邊界防護，符合題意 0 < n <= 20
        if (n <= 0 || n > 20) {
            in.close();
            return;
        }

        // 建立下三角形的二維陣列，並填入 '*'
        char[][] a = new char[n][n];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= i; j++) {
                a[i][j] = '*';
            }
        }

        // 以字串構建的方式輸出，避免不必要的 I/O 呼叫
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= i; j++) {
                sb.append(a[i][j]);
            }
            sb.append('\n');
        }

        System.out.print(sb.toString());
        in.close();
    }
}
