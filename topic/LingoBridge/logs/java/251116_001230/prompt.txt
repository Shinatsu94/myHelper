題目：
【題目】
輸入一個數字A, 找尋 1 到 100000之間有多少數字每一個位數的加總結果等於A。
例如：當A=6, 則數字 123 -> 1+2+3=6 符合要求。
程式輸出為1-100000之間，符合要求的數字數量。

【INPUT】
6

【OUTPUT】
210

學生作答程式：
import java.util.Scanner;

public class CSIE_41343229_1003_3 {
    // 回傳正整數 n 的各位數字總和
    static int digitSum(int n) {
        int s = 0;
        while (n > 0) {
            s += n % 10;
            n /= 10;
        }
        return s;
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int A = sc.nextInt();   // 讀入目標數位和
        sc.close();

        int count = 0;
        for (int i = 1; i <= 100000; i++) {   // 1 到 100000（含）
            if (digitSum(i) == A) count++;
        }
        System.out.println(count);
    }
}
