- 演算法與效能：使用 Arrays.sort 進行排序，時間複雜度 O(n log n)，適合此題需求。

修改後的完整程式碼
/* 
 * File: Main.java
 * 功能: 讀取整數 n，接著讀入 n 個整數，將陣列內容由小到大排序後輸出。
 * 作者: 你之協助 AI
 * 日期: 2025-11-14
 * 說明: 此版本符合題目需求，輸出以空格分隔，尾端不多餘空格。
 */

import java.util.Scanner;
import java.util.Arrays;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取陣列長度
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();
        int[] arr = new int[n];

        // 逐一讀入 n 個整數
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                // 若輸入不足，預設為 0
                arr[i] = 0;
            }
        }
        sc.close();

        // 由小到大排序
        Arrays.sort(arr);

        // 輸出排序結果，元素以空格分隔，避免結尾多餘空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.print(sb.toString());
    }
}
