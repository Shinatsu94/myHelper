/*
 檔案名稱: ReverseInteger.java
 功能: 將輸入的整數的位元順序翻轉後輸出
 作者: 教師
 日期: 2025-11-15
 說明: 直接對 x 進行逐位反轉，支援正整數與負整數，輸出不保留前導零。
*/

import java.util.*;

public class ReverseInteger {
    // 直接對數字逐位反轉，保留符號（負數會自動對應負號輸出）
    public static long reverse(long x) {
        long rev = 0;
        while (x != 0) {
            long d = x % 10;     // 取得最低位數字
            rev = rev * 10 + d;  // 將數字追加到結果尾端
            x /= 10;              // 去除最低位
        }
        return rev;
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        long n = sc.nextLong();
        long r = reverse(n);
        System.out.println(r);
        sc.close();
    }
}
