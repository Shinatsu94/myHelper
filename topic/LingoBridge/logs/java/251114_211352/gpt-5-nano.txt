import java.util.Scanner;

/*
檔案名稱: Main.java
功能: 讀入整數長度與元素，對陣列進行升冪排序，輸出排序後結果
作者: 請自行填寫
日期: 2025-11-14
說明: 透過 selection sort 將輸入的整數陣列由小到大排序，輸出同一行、以空白分隔的結果，結尾加換行。
*/
public class Main {
    // 將陣列就地以升冪排序
    static void selectionSort(int[] arr) {
        int n = arr.length;
        for (int i = 0; i < n - 1; i++) {
            int minIdx = i;
            for (int j = i + 1; j < n; j++) {
                if (arr[j] < arr[minIdx]) minIdx = j;
            }
            if (minIdx != i) {
                int tmp = arr[i];
                arr[i] = arr[minIdx];
                arr[minIdx] = tmp;
            }
        }
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        try {
            int size = sc.nextInt();          // 讀入陣列長度
            if (size < 0) size = 0;           // 安全處理負數長度
            int[] arr = new int[size];
            for (int i = 0; i < size; i++) {  // 讀入 size 個整數（可含負數）
                arr[i] = sc.nextInt();
            }

            selectionSort(arr);               // 升冪排序

            // 輸出（同一行、以空白分隔、無多餘提示）
            for (int i = 0; i < size; i++) {
                if (i > 0) System.out.print(" ");
                System.out.print(arr[i]);
            }
            System.out.println(); // 加上換行，符合常見輸出慣例
        } finally {
            sc.close();
        }
    }
}
