/*
 * 檔案名稱: CSIE_41343209_1031_3a.java
 * 功能: 建立並輸出下三角形的二維陣列，元素為 '*'
 * 作者: 你的名字
 * 日期: 2025-11-14
 * 備註: 測資格式僅含 '*' 組成的下三角形，不輸出多餘空白
 */

import java.util.Scanner;

public class CSIE_41343209_1031_3a {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n;
        if (sc.hasNextInt()) {
            n = sc.nextInt();
        } else {
            sc.close();
            return;
        }

        // 驗證高度範圍，符合題意 (0 < n <= 20)
        if (n <= 0 || n > 20) {
            sc.close();
            return;
        }

        // 使用二維陣列填充下三角形的 '*'
        char[][] tri = new char[n][n];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= i; j++) {
                tri[i][j] = '*';
            }
        }

        // 輸出時僅輸出有效的部分，避免多餘的空白字元
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= i; j++) {
                System.out.print(tri[i][j]);
            }
            System.out.println();
        }

        sc.close();
    }
}
