修改後完整程式碼
// 檔案名稱: CSIE_41343221_1031_3b.java
// 功能: 讀取整數 N，接著讀入 N 個整數，將內容升序排序，並印出排序結果
// 作者: 教授
// 日期: 2025-11-12

import java.util.Scanner;
import java.util.Arrays;

public class CSIE_41343221_1031_3b {

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        // 讀取陣列長度 N
        if (!scanner.hasNextInt()) {
            scanner.close();
            return;
        }
        int N;
        try {
            N = scanner.nextInt();
        } catch (Exception e) {
            scanner.close();
            return;
        }

        if (N < 0) {
            // 不允許負長度
            scanner.close();
            return;
        }

        int[] arr = new int[N];

        // 依序讀入 N 個整數
        for (int i = 0; i < N; i++) {
            if (!scanner.hasNextInt()) {
                // 若輸入不足，直接結束
                scanner.close();
                return;
            }
            arr[i] = scanner.nextInt();
        }

        // 排序
        Arrays.sort(arr);

        // 印出排序結果，元素以空格分隔，末尾不留多餘空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < N; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }

        System.out.println(sb.toString());

        scanner.close();
    }
}
