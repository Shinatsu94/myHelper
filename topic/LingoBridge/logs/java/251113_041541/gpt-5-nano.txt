    /*
     檔案名稱: NameCodePrinter.java
     功能: 依序印出虎科大三字與學生姓名每個字之 Unicode 內碼
     作者: 檔案撰寫者 (請自行填寫)
     日期: 2025-11-12
    */
    public static void main(String[] args) {
        // 先印出虎科大三字及其對應的內碼
        char[] headerChars = {'虎', '科', '大'};
        for (char ch : headerChars) {
            System.out.println(ch + " " + (int) ch);
        }

        // 再印出學生的中文姓名每個字及內碼
        String studentName = "林武杰";
        for (char ch : studentName.toCharArray()) {
            System.out.println(ch + " " + (int) ch);
        }
    }
