修改後完整程式碼如下：

/*
檔案名稱: MaxMidMin.java
功能: 輸入三個整數，輸出最小值、中間值與最大值。使用兩種不同方式找出最大、最小值，中間值透過 sum-max-min 計算。
作者: 你的名字
日期: <日期>
*/

import java.util.Scanner;

public class MaxMidMin {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int a = sc.nextInt();
        int b = sc.nextInt();
        int c = sc.nextInt();

        // 最大值：若 a 大於 b 且 a 大於 c，則為 a，否則在 b 與 c 中取較大者
        int max = (a > b && a > c) ? a : (b > c ? b : c);

        // 最小值：若 a 小於 b 且 a 小於 c，則為 a，否則在 b 與 c 中取較小者
        int min = (a < b && a < c) ? a : (b < c ? b : c);

        // 中間值：總和減去最大值與最小值
        int mid = a + b + c - max - min;

        System.out.println(min);
        System.out.println(mid);
        System.out.println(max);

        sc.close();
    }
}
