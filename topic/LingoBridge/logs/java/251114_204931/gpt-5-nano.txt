- 使用有意義的變數命名（n、arr），提高可讀性。

- 關閉 Scanner，培養良好資源管理習慣。

/**
 * 檔案名稱: CSIE_41343101_1031_1.java
 * 功能說明: 建立任意長度的一維整數陣列，讀入元素，並以升序排序後輸出
 * 作者: 你的姓名
 * 日期: 2025-11-14
 */

import java.util.Arrays;
import java.util.Scanner;

public class CSIE_41343101_1031_1 {
    public static void main(String[] args) {
        Scanner sc = null;
        try {
            sc = new Scanner(System.in);
            if (!sc.hasNextInt()) {
                System.out.println();
                return;
            }
            int n = sc.nextInt();
            int[] arr = new int[n];
            for (int i = 0; i < n; i++) {
                arr[i] = sc.nextInt();
            }

            // 使用內建排序提升穩定性與效率
            Arrays.sort(arr);

            // 輸出結果，元素以單一空格分隔，末尾換行
            StringBuilder sb = new StringBuilder();
            for (int i = 0; i < n; i++) {
                if (i > 0) sb.append(' ');
                sb.append(arr[i]);
            }
            System.out.println(sb.toString());
        } finally {
            if (sc != null) sc.close();
        }
    }
}
