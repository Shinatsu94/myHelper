/*
檔案名稱: WellOrderedNumber.java
功能Description: 判斷一個整數的數位是否嚴格遞增（從左到右的數位需逐位遞增），
            輸入為單一整數，輸出 "<n> is well-ordered number." 或
            "<n> is not well-ordered number."。
作者: 你的名字
日期: 2025-11-15
*/
import java.io.BufferedReader;
import java.io.InputStreamReader;

public class WellOrderedNumber {
    public static void main(String[] args) {
        try (BufferedReader br = new BufferedReader(new InputStreamReader(System.in))) {
            String line = br.readLine();
            if (line == null) return;
            line = line.trim();
            if (line.isEmpty()) return;

            int n = Integer.parseInt(line);
            boolean ok = isWellNumber(n);
            System.out.println(n + " is " + (ok ? "" : "not ") + "well-ordered number.");
        } catch (Exception e) {
            // 若輸入格式不正確，避免程式崩潰；此案例僅回傳空輸出以符合單次輸入的情境
        }
    }

    // 使用字串方法檢查數字的數位是否嚴格遞增
    private static boolean isWellNumber(int n) {
        if (n < 0) n = -n;
        String s = Integer.toString(n);
        for (int i = 1; i < s.length(); i++) {
            if (s.charAt(i) <= s.charAt(i - 1)) return false;
        }
        return true;
    }
}
