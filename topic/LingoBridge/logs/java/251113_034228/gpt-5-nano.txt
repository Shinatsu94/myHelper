```java
/**
 * 檔案名稱: CSIE_41343252_0926_1.java
 * 功能描述: 印出「虎科大」及「你的中文姓名」對應的字元內碼
 * 作者: 教授
 * 日期: 2025/11/12
 * 版本: 1.0
 */
public class CSIE_41343252_0926_1 {
    public static void main(String[] args) {
        // 先印出「虎科大」
        char[] head = {'虎', '科', '大'};

        // 後印出學生姓名，假設為「林武杰」
        String name = "林武杰";

        // 輸出虎科大三個字及對應的碼
        for (char c : head) {
            System.out.println(c + " " + (int) c);
        }

        // 輸出學生姓名的三個字及對應的碼
        for (int i = 0; i < name.length(); i++) {
            char ch = name.charAt(i);
            System.out.println(ch + " " + (int) ch);
        }
    }
}
```
