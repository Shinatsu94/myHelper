/*
 * 檔案名稱: CSIE_41343136_1031_3b.java
 * 功能說明: 讀取整數長度 n，接著讀入 n 個整數，將陣列內容升序排序，並輸出排序後的結果。
 * 作者: 資訊工程系教授
 * 日期: 2025-11-14
 */

import java.util.Arrays;
import java.util.Scanner;

public class CSIE_41343136_1031_3b {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        if (!scanner.hasNextInt()) {
            // 沒有輸入長度，直接結束
            scanner.close();
            return;
        }

        int n = scanner.nextInt();
        if (n <= 0) {
            // 長度為非正數，結束
            scanner.close();
            System.out.println();
            return;
        }

        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (scanner.hasNextInt()) {
                arr[i] = scanner.nextInt();
            } else {
                // 少於 n 個輸入時，預設為 0，以穩定結束
                arr[i] = 0;
            }
        }
        scanner.close();

        // 進行升序排序
        Arrays.sort(arr);

        // 輸出排序結果，元素以空格分隔，結尾換行
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());
    }
}
