/*
檔案名稱: Main.java
功能    : 讀入第一個整數為 n，接著讀入 n 個整數，將它們
          進行升序排序，並輸出排序後的結果（以空格分隔）。
作者    : 你的姓名
日期    : 2025-11-14
說明    : 改用直接讀入與排序，符合題目需求，並增加註解與可讀性。
*/

import java.util.Arrays;
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        // 讀入陣列長度
        if (!scanner.hasNextInt()) {
            return;
        }
        int n = scanner.nextInt();

        // 讀入 n 個整數
        int[] numbers = new int[n];
        for (int i = 0; i < n; i++) {
            if (scanner.hasNextInt()) {
                numbers[i] = scanner.nextInt();
            } else {
                // 若輸入不足，預設為 0（保守處理）
                numbers[i] = 0;
            }
        }

        // 進行升冪排序
        Arrays.sort(numbers);

        // 輸出排序結果，元素之間以單個空格分隔
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(numbers[i]);
        }
        System.out.println(sb.toString());
    }
}
