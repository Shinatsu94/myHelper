import java.util.Scanner;

/**
 * 檔案名稱: CSIE_41343119_1031_3.java
 * 功能說明: 建立並列印下三角形的二維陣列（每行有 1..n 顆 '*'，不在元素間加入空白）
 * 作者: 金宗翰
 * 日期: 2025-10-31
 */
public class CSIE_41343119_1031_3 {
    public static void main(String[] args) {
        Scanner scn = new Scanner(System.in);
        int n = scn.nextInt();
        // 輸入範圍檢查，符合題意 0 < n <= 20
        if (n <= 0 || n > 20) {
            scn.close();
            return;
        }

        // 建立並填入下三角形的二維陣列
        char[][] triangle = new char[n][n];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= i; j++) {
                triangle[i][j] = '*';
            }
        }

        // 輸出：逐行列出 triangle[i][0..i]，每行後換行
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= i; j++) {
                sb.append(triangle[i][j]);
            }
            sb.append('\n');
        }
        System.out.print(sb.toString());
        scn.close();
    }
}
