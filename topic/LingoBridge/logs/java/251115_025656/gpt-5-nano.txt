/*
檔案名稱: LowerTriangularStars.java
功能說明: 輸入高度 n (0 < n <= 20)，建立 n x n 的二維陣列，將下三角形位置填入 '*',
          並印出對應的下三角星號圖樣。要求輸出中元素之間無空白符號。
作者: 你的名字
日期: 2025-11-14
*/

import java.util.Scanner;

public class LowerTriangularStars {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();

        // 驗證輸入範圍
        if (n <= 0 || n > 20) {
            sc.close();
            return;
        }

        // 使用 n x n 的二維陣列，填入下三角的 '*'
        char[][] arr = new char[n][n];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= i; j++) {
                arr[i][j] = '*';
            }
        }

        // 逐行輸出下三角陣列內容
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= i; j++) {
                sb.append(arr[i][j]);
            }
            sb.append('\n');
        }

        System.out.print(sb.toString());
        sc.close();
    }
}
