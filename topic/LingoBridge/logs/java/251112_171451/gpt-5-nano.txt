import java.util.Scanner;
import java.util.Arrays;

/**
 * 檔案名稱: CSIE_51015105_1031_1.java
 * 功能: 讀取整數長度 n 與 n 個整數，將其升序排序，輸出結果（以空格分隔）
 * 作者: 教授
 * 日期: 2025-11-12
 */
public class CSIE_51015105_1031_1 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();
        }

        // 使用內建排序提升穩定性與可維護性
        Arrays.sort(arr);

        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.print(sb.toString());

        sc.close();
    }
}
