/*
 * File: Main.java
 * Description: 讀入長度 n 與 n 個整數，輸出升序排序的結果（選擇排序）
 * Author: 教師
 * Date: 2025-11-14
 */

import java.util.Scanner;

public class Main {
    // 將整數陣列 a 進行選擇排序（由小到大）
    static void selectionSort(int[] a) {
        int n = a.length;
        if (n <= 1) return;
        for (int i = 0; i < n - 1; i++) {
            int minIdx = i;
            for (int j = i + 1; j < n; j++) {
                if (a[j] < a[minIdx]) minIdx = j;
            }
            if (minIdx != i) {
                int tmp = a[i];
                a[i] = a[minIdx];
                a[minIdx] = tmp;
            }
        }
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        // 讀入陣列長度
        int n = sc.nextInt();
        int[] arr = new int[n];
        // 讀入 n 個整數
        for (int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();
        }

        selectionSort(arr); // 升冪排序

        // 組裝輸出以提升效能
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.print(sb.toString());
        System.out.println(); // 加上換行，符合一般輸出慣例

        sc.close();
    }
}
