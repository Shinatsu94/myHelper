程式修改與優化說明

- 輸入讀取與輸出格式處理為單一輸出行，元素之間以空格分隔。

// Date: 2025/10/31

// Function: 讀取一個整數 n，接著讀入 n 個整數，將它們升序排序，最後輸出排序後的內容
import java.util.Scanner;
import java.util.Arrays;

public class SortOneDimensionalArrayAscending {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取陣列長度
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();
        if (n <= 0) {
            sc.close();
            return;
        }

        // 讀取 n 個整數
        int[] a = new int[n];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                a[i] = sc.nextInt();
            } else {
                // 若輸入不足，預設為 0（但一般測資應該提供足夠的數字）
                a[i] = 0;
            }
        }

        // 排序
        Arrays.sort(a);

        // 輸出結果，元素之間以空格分隔，結尾不留多餘空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(a[i]);
        }
        System.out.print(sb.toString());

        sc.close();
    }
}
