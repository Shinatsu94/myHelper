public class LeibnizPi {
    public static void main(String[] args) {
        // 逐步計算並輸出指定的 i 值與 pi 值
        for (int i = 10000; i <= 100000; i += 10000) {
            double sum = 0.0;
            for (int k = 0; k < i; k++) {
                double term = 1.0 / (2.0 * k + 1.0);
                if ((k & 1) == 0) { // k 偶數，符號為 +
                    sum += term;
                } else {            // k 奇數，符號為 -
                    sum -= term;
                }
            }
            double pi = 4.0 * sum;
            // i 使用六位寬度輸出，pi 保留小數點後十位
            System.out.printf("i = %6d, pi = %.10f%n", i, pi);
        }
    }
}
