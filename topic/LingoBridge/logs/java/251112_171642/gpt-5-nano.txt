import java.util.Scanner;
import java.util.Arrays;

/**
 * File: CSIE_51015120_1031_1.java
 * Title: 由使用者輸入長度與元素，輸出升序排序後的結果
 * Author: （作者姓名或教學用名字）
 * Date: 2025-11-12
 * Description: 讀入整數長度 n 與 n 個整數，將其排序為升序並輸出成一行，元素以空格區隔。
 */
public class CSIE_51015120_1031_1 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();
        }

        // 使用內建排序提升穩定性與效率
        Arrays.sort(arr);

        // 輸出，元素之間以單一空格分隔，結尾換行
        for (int i = 0; i < n; i++) {
            if (i > 0) System.out.print(" ");
            System.out.print(arr[i]);
        }
        System.out.println();

        sc.close();
    }
}
