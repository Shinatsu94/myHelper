import java.util.Scanner;
import java.util.Arrays;

/*
 * 檔案：CSIE_51015130_1031_3b.java
 * 功能：讀入一個整數 n，接著讀入 n 個整數，輸出排序後的內容（由小到大）
 * 作者：改良版
 * 日期：2025-11-14
 */

public class CSIE_51015130_1031_3b {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取長度
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();
        if (n <= 0) {
            // 若長度為0或負值，直接輸出空行並結束
            System.out.println();
            sc.close();
            return;
        }

        // 讀取 n 個整數
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                // 若輸入不足，預設為 0
                arr[i] = 0;
            }
        }
        sc.close();

        // 進行升序排序
        Arrays.sort(arr);

        // 輸出排序結果，元素之間以單一空格分隔
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());
    }
}
