import java.util.Scanner;

public class CSIE_41343219_1003_1 {
    public static void main(String[] args) {
        int a, b;
        Scanner sc = new Scanner(System.in);
        a = sc.nextInt();
        b = sc.nextInt();

        // 使用三步驟 XOR 互換，避免額外變數
        a = a ^ b;
        b = a ^ b;
        a = a ^ b;

        sc.close();
        System.out.println(a);
        System.out.println(b);
    }
}
