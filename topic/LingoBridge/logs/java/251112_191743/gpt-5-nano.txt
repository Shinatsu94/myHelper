import java.util.Arrays;
import java.util.Scanner;

/**
 * File: CSIE_51015130_1031_1.java
 * 功能: 讀取長度 n 與 n 個整數，升冪排序並輸出結果（以空格分隔，末尾不留空格）。
 * 作者: AI 助手
 * 日期: 2025-11-12
 * 備註: 使用 try-with-resources 以確保 Scanner 關閉，輸出使用單一空格分隔，避免尾端空格。
 */
public class CSIE_51015130_1031_1 {
    public static void main(String[] args) {
        try (Scanner sc = new Scanner(System.in)) {
            // 讀取長度
            if (!sc.hasNextInt()) return;
            int n = sc.nextInt();
            int[] nums = new int[n];

            // 讀入 n 個整數
            for (int i = 0; i < n; i++) {
                nums[i] = sc.nextInt();
            }

            // 升冪排序
            Arrays.sort(nums);

            // 輸出，避免尾端多餘空格
            StringBuilder sb = new StringBuilder();
            for (int i = 0; i < nums.length; i++) {
                if (i > 0) sb.append(' ');
                sb.append(nums[i]);
            }
            System.out.println(sb.toString());
        }
    }
}
