Author: 你的姓名

public class WellOrderedNumber {
    public static void main(String[] args) {
        java.util.Scanner sc = new java.util.Scanner(System.in);
        int number = sc.nextInt();
        int absVal = Math.abs(number);
        boolean ok = isWellOrdered(absVal);
        System.out.printf("%d is %swell-ordered number.%n", number, ok ? "" : "not ");
        sc.close();
    }

    // 判斷 n 的十進位數字是否嚴格由左至右遞增
    private static boolean isWellOrdered(int n) {
        if (n < 10) return true; // 單一位數視為 well-ordered
        int lastDigit = n % 10;
        n /= 10;
        while (n > 0) {
            int current = n % 10;
            if (current >= lastDigit) return false;
            lastDigit = current;
            n /= 10;
        }
        return true;
    }
}
