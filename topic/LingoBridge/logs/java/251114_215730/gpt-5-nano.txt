/*
檔案名稱: CSIE_41343150_1031_3a.java
功能: 讀入長度 n 的整數陣列，排序後以空格分隔輸出
作者: 教授
日期: 2025-11-14
說明: 讀取 n 與 n 個整數，排序並印出結果，支援負數。
*/
import java.util.Arrays;
import java.util.Scanner;

public class CSIE_41343150_1031_3a {
    public static void main(String[] args) {
        Scanner scn = new Scanner(System.in);

        // 讀取陣列長度
        int n = scn.nextInt();
        int[] arr = new int[n];

        // 讀取陣列元素
        for (int i = 0; i < n; i++) {
            arr[i] = scn.nextInt();
        }

        // 排序（升冪）
        Arrays.sort(arr);

        // 輸出排序後的陣列，元素以空格分隔
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        scn.close();
    }
}
