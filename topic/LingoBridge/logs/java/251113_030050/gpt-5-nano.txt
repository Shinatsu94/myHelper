public class CSIE_41343101_0926_1 {
    /*
     * 檔案名稱: CSIE_41343101_0926_1.java
     * 功能說明: 印出虎科大與學生姓名每個中文字及其 Unicode 內碼，輸出順序為「虎科大」先於學生姓名。
     * 作者: 自訂
     * 日期: 2025-11-12
     * 備註: 內碼使用四位十六進位表示，逐字輸出，支援代理區字元。
     */

    public static void main(String[] args) {
        String studentName = "王薇詒";  // 學生姓名
        String schoolName = "虎科大"; // 學校名稱

        // 先輸出虎科大的每個字及其 Unicode 內碼
        printUnicodePairs(schoolName);

        // 再輸出學生姓名的每個字及其 Unicode 內碼
        printUnicodePairs(studentName);
    }

    // 印出字串中每個字元及其對應的 Unicode 內碼（四位十六進位），每行一字
    private static void printUnicodePairs(String text) {
        for (int i = 0; i < text.length();) {
            int codePoint = text.codePointAt(i);
            // 以四位十六進位表示內碼，並同時輸出對應的字元
            System.out.printf("%c %04X%n", codePoint, codePoint);
            i += Character.charCount(codePoint);
        }
    }
}
