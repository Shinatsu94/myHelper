/**
 * 檔案名稱: CSIE_41343210_0926_1.java
 * 功能: 印出「虎科大」與「你的中文姓名」，並印出對應的內碼。先印虎、科、大再印姓名三字。
 * 作者: 吳翌銘
 * 日期: 2025-11-12
 * 說明: 使用 char 陣列與共用印出函式，減少重複程式碼，便於未來修改姓名或學校名稱。
 */
public class CSIE_41343210_0926_1 {

    public static void main(String[] args) {
        // 學校名稱三個字
        char[] school = {'虎', '科', '大'};
        // 學生姓名三個字（可依實際姓名修改）
        char[] name = {'林', '武', '杰'};

        for (char c : school) {
            printCharCode(c);
        }
        for (char c : name) {
            printCharCode(c);
        }
    }

    // 統一輸出字元與其內碼
    private static void printCharCode(char ch) {
        System.out.println(ch + " " + (int) ch);
    }
}
