/*
 * 檔案名稱: CSIE_41343215_1031_3b_Optimized.java
 * 功能說明: 讀取使用者輸入的長度 n，接著讀入 n 個整數，
 *           將陣列內容由小到大排序後輸出，並支援負數。
 * 作者: 你的名字
 * 日期: 2025-11-14
 * 註解: 程式內包含必要的註解以說明各區段功能
 */

import java.util.Arrays;
import java.util.Scanner;

public class CSIE_41343215_1031_3b_Optimized {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 先讀取陣列長度
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();

        // 讀入 n 個整數，存入陣列
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                // 若輸入不足，預設為 0
                arr[i] = 0;
            }
        }
        sc.close();

        // 將陣列排序（由小到大）
        Arrays.sort(arr);

        // 輸出排序後的陣列，元素以單一空格分隔
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());
    }
}

- 程式可正確處理包含負數的情況，並支援跨多行的輸入。
- 若輸入數量不足，為了不發生例外，程式會將缺失的元素預設為 0；若你希望嚴格要求輸入數量，亦可在讀取不足時拋出例外或提示錯誤。
