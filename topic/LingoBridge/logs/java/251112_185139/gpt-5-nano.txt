/*
 * File: CSIE_41343202_1031_1.java
 * 功能: 依序輸入長度與整數，建立一維陣列並進行升序排序，輸出排序後結果。
 * 作者: 教師
 * 日期: 2025-11-12
 */

import java.util.Scanner;

public class CSIE_41343202_1031_1 {

    // 使用簡單的選擇排序算法，將陣列 arr 由小到大排序
    public static void selectionSort(int[] arr) {
        int n = arr.length;
        for (int i = 0; i < n - 1; i++) {
            int min = i;
            for (int j = i + 1; j < n; j++) {
                if (arr[j] < arr[min]) {
                    min = j;
                }
            }
            // 交換 arr[i] 與 arr[min]
            int temp = arr[min];
            arr[min] = arr[i];
            arr[i] = temp;
        }
    }

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        int length = 0;
        if (scanner.hasNextInt()) {
            length = scanner.nextInt();
        }

        // 避免負數長度造成異常
        if (length < 0) {
            length = 0;
        }

        int[] array = new int[length];

        for (int i = 0; i < length; i++) {
            if (scanner.hasNextInt()) {
                array[i] = scanner.nextInt();
            } else {
                // 若輸入不足，缺省為 0
                array[i] = 0;
            }
        }

        selectionSort(array);

        // 輸出排序後的陣列
        for (int i = 0; i < length; i++) {
            if (i > 0) System.out.print(" ");
            System.out.print(array[i]);
        }
        System.out.println();

        scanner.close();
    }
}
