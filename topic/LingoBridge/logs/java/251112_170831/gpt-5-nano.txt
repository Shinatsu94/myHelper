import java.util.Scanner;

public class csie_41343235_1031_1
{
    /*
     * 檔案名稱: csie_41343235_1031_1.java
     * 功能: 讀取一個整數 n，接著讀入 n 個整數，將它們升冪排序，並輸出排序結果（以空格分隔）
     * 作者: 教師
     * 日期: 2025-11-12
     * 註解: 使用清晰變數名稱、完整區塊語句與資源自動關閉，並於輸出新增換行
     */
    public static void main(String[] args) {
        // 使用 try-with-resources 以確保 Scanner 能正確關閉
        try (Scanner sc = new Scanner(System.in)) {
            if (!sc.hasNextInt()) return;
            int n = sc.nextInt();
            int[] arr = new int[n];

            for (int i = 0; i < n; i++) {
                arr[i] = sc.nextInt();
            }

            // 選擇排序（Selection Sort）
            for (int i = 0; i < n - 1; i++) {
                int min = i;
                for (int j = i + 1; j < n; j++) {
                    if (arr[j] < arr[min]) min = j;
                }
                int t = arr[i];
                arr[i] = arr[min];
                arr[min] = t;
            }

            // 輸出結果，避免尾端多餘空白，並加上換行
            StringBuilder sb = new StringBuilder();
            for (int i = 0; i < n; i++) {
                if (i > 0) sb.append(' ');
                sb.append(arr[i]);
            }
            System.out.println(sb.toString());
        }
    }
}
