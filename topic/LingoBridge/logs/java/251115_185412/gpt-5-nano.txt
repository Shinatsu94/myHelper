/*
  檔案名稱: CSIE_41343121_1017_3.java
  功能: 將輸入整數的數字反轉後輸出
  作者: 未知
  日期: 2025-11-15
  說明: 輸入一個整數，輸出其數字反轉的結果。支持 0、正整數、負整數。
*/

import java.util.Scanner;

public class CSIE_41343121_1017_3 {
  public static void main(String[] args) {
    Scanner scn = new Scanner(System.in);
    long number = 0;

    if (scn.hasNextLong()) {
      number = scn.nextLong();
    } else {
      // 非數字輸入時輸出 0 並結束
      System.out.println(0);
      scn.close();
      return;
    }

    long result = reverseDigits(number);
    System.out.println(result);
    scn.close();
  }

  // 將整數的數字反轉，保留正負號
  private static long reverseDigits(long n) {
    if (n == 0) return 0;

    int sign = (n < 0) ? -1 : 1;
    long m = Math.abs(n);
    long rev = 0;

    while (m > 0) {
      long digit = m % 10;
      rev = rev * 10 + digit;
      m /= 10;
    }
    return rev * sign;
  }
}
