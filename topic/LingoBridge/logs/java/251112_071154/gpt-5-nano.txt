public class CSIE_41343121_1017_1 {
    public static void main(String[] args) {
        // 目的：計算 pi 的近似值並輸出指定 i 的結果（a），以及找出使 pi ~ 3.14159 的最小 i（b）
        // 輸出格式：i = %6d, pi = %.10f，pi 取小數點後10位，i 6位寬度

        // Part (a): 逐步累計求和，並在 i 為 10000 的倍數時輸出 pi
        double sum = 0.0;
        int sign = 1;
        for (int i = 1; i <= 100000; i++) {
            sum += sign * (1.0 / (2 * i - 1));
            sign = -sign;
            if (i % 10000 == 0) {
                double pi = sum * 4.0;
                System.out.printf("i = %6d, pi = %.10f%n", i, pi);
            }
        }

        // Part (b): 找出使 pi 落在 [3.14159, 3.14160) 的最小 i（以累加方式逐步逼近）
        sum = 0.0;
        sign = 1;
        int iMin = 0;
        double pi = 0.0;
        int t = 0;
        while (true) {
            t++;
            sum += sign * (1.0 / (2 * t - 1));
            sign = -sign;
            pi = sum * 4.0;
            if (pi >= 3.14159 && pi < 3.14160) {
                iMin = t;
                break;
            }
        }

        // 顯示與範例相同的換行風格，並輸出最小 i 值
        System.out.println();
        System.out.printf("i = %d%n", iMin);
