import java.util.Scanner;

public class CSIE_41343247_1031_1 {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        // 讀取陣列長度
        int n;
        if (scanner.hasNextInt()) {
            n = scanner.nextInt();
        } else {
            System.out.println();
            scanner.close();
            return;
        }

        if (n <= 0) {
            System.out.println();
            scanner.close();
            return;
        }

        // 讀入陣列元素
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (scanner.hasNextInt()) {
                arr[i] = scanner.nextInt();
            } else {
                arr[i] = 0; // 若輸入不足，預設為 0
            }
        }

        // 選擇排序（升序）
        for (int i = 0; i < n; i++) {
            int minIndex = i;
            for (int j = i; j < n; j++) {
                if (arr[j] < arr[minIndex]) {
                    minIndex = j;
                }
            }
            if (minIndex != i) {
                int tmp = arr[i];
                arr[i] = arr[minIndex];
                arr[minIndex] = tmp;
            }
        }

        // 輸出結果，元素之間以空格分隔
        for (int i = 0; i < n; i++) {
            if (i > 0) System.out.print(" ");
            System.out.print(arr[i]);
        }
        System.out.println(); // 結尾換行

        scanner.close();
    }
}
