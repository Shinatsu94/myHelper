/**
 * 檔案名稱: Main.java
 * 功能: 印出「虎科大」與「丁祥誌」每個字及其對應的內碼（Unicode 代碼點），
 *       先印虎科大再印學生姓名，每行一個字與其內碼。
 * 作者: 你的名字
 * 日期: 2025-11-12
 */
public class Main {
    public static void main(String[] args) {
        String chineseName = "丁祥誌";
        String school = "虎科大";

        // 先印虎科大，再印學生中文姓名，每個字一行，後面為該字的內碼（Unicode 代碼點）
        printCharsWithCode(school);
        printCharsWithCode(chineseName);
    }

    // 輸出字元與對應的 Unicode 代碼點，並以字元本身顯示於左側
    private static void printCharsWithCode(String s) {
        for (int i = 0; i < s.length();) {
            int codePoint = s.codePointAt(i);
            int charCount = Character.charCount(codePoint);
            String ch = new String(Character.toChars(codePoint));
            System.out.println(ch + " " + codePoint);
            i += charCount;
        }
    }
}
