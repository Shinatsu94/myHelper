修改後完整程式碼
/* 檔案名稱: Main.java
 * 功能: 讀取整數長度 n 與 n 個整數，將其升序排序後輸出，元素以空格分隔。
 * 作者: 學生
 * 日期: 2025-11-14
 * 註解: 使用 Arrays.sort 進行排序，支援負數，輸出格式為以單一空格分隔的行。
 */
import java.util.Arrays;
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int n = sc.nextInt();       // 讀取陣列長度
        int[] arr = new int[n];       // 建立整數陣列

        for (int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();    // 讀入每個整數，支援負數
        }

        Arrays.sort(arr);             // 升序排序

        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }

        System.out.print(sb.toString()); // 輸出排序後的陣列
        sc.close();
    }
}
