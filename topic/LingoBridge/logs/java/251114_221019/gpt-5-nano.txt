/*
 * File: SortArrayAscending.java
 * 功能: 讀取使用者輸入的長度 n 與 n 個整數，並以升序輸出
 * 作者: [請填寫作者姓名]
 * 日期: 2025-11-14
 * 備註: 不使用 EOF，單次執行結束
 */

import java.util.Scanner;
import java.util.Arrays;

public class SortArrayAscending {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取長度 n
        if (!sc.hasNextInt()) {
            return;
        }
        int n = sc.nextInt();

        // 讀取 n 個整數
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                // 若輸入不足，預設為 0
                arr[i] = 0;
            }
        }

        // 進行升序排序
        Arrays.sort(arr);

        // 輸出結果，元素之間以空格分隔，末尾不留多餘空格，最後換行
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());
    }
}
