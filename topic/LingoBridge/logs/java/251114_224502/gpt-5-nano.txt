修改後完整程式碼：

/*
 File: Main.java
 功能: 讀入長度 n 與 n 個整數，將陣列內容由小到大排序，輸出排序後的結果。
 作者: 教師姓名
 日期: 2025-11-14
*/

import java.util.Arrays;
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀入陣列長度
        int n = sc.nextInt();
        int[] arr = new int[n];

        // 讀入 n 個整數
        for (int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();
        }

        sc.close();

        // 升序排序
        Arrays.sort(arr);

        // 輸出排序後的陣列，元素以空格分隔
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());
    }
}
