功能: 建立任意長度的一維陣列，並對內容進行由小到大排序輸出

說明: 使用選擇排序實作，支援負數，並輸出以空格分隔的排序結果

import java.util.Scanner;

public class CSIE_41343119_1031_1 {
  public static void main(String[] args) {
    Scanner scn = new Scanner(System.in);

    int n = scn.nextInt();
    int[] arr = new int[n];

    for (int i = 0; i < n; i++) {
      arr[i] = scn.nextInt();
    }

    // 選擇排序：每一輪找出當前區段的最小值，與起始位置交換
    for (int i = 0; i < n - 1; i++) {
      int minIdx = i;
      for (int j = i + 1; j < n; j++) {
        if (arr[j] < arr[minIdx]) {
          minIdx = j;
        }
      }
      int tmp = arr[i];
      arr[i] = arr[minIdx];
      arr[minIdx] = tmp;
    }

    // 輸出：避免尾端多餘空格
    for (int i = 0; i < n; i++) {
      if (i > 0) System.out.print(" ");
      System.out.print(arr[i]);
    }
    System.out.println();

    scn.close();
  }
}
