修改後完整程式碼
// 檔案名稱: CSIE_41343233_0926_3.java
// 功能說明: 依序輸出「虎科大」及學生姓名「林武杰」的每一個字及其對應的內碼（Unicode Code Point）。
// 作者/日期等註解可依實際需求補充。

public class CSIE_41343233_0926_3 {
    public static void main(String[] args) {
        String org = "虎科大";
        String name = "林武杰";

        printWithCode(org);
        printWithCode(name);
    }

    // 印出字串中每個字元及其 Unicode 內碼（以十進位表示）
    private static void printWithCode(String s) {
        for (int i = 0; i < s.length();) {
            int cp = s.codePointAt(i);
            String ch = new String(Character.toChars(cp)); // 取得字元字串表示
            System.out.println(ch + " " + cp);
            i += Character.charCount(cp);
        }
    }
}
