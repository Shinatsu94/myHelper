/**
 * File: LeibnizPiCalculator.java
 * 功能: 計算 pi 的近似值，輸出 i = 10000, 20000, ..., 100000 對應的 pi 值，
 *       i 顯示為寬度為 6 的整數，pi 顯示至小數點後 10 位。
 * 作者: [你的姓名]
 * 日期: [日期]
 * 說明: 使用累加方式逐項累積 Leibniz 公式的前 i 項，避免重複計算。
 */
public class LeibnizPiCalculator {
    public static void main(String[] args) {
        final int MAX_I = 100000; // 最大項數（總共 100000 項分母為 2k+1 的項）
        double sum = 0.0;

        for (int k = 0; k < MAX_I; k++) {
            // 第 k 項: (-1)^k / (2k + 1)
            double term = ((k % 2 == 0) ? 1.0 : -1.0) / (2.0 * k + 1.0);
            sum += term;

            int i = k + 1; // 已累積的項數，對應公式中的 i
            if (i % 10000 == 0) {
                double pi = 4.0 * sum;
                System.out.printf("i = %6d, pi = %.10f%n", i, pi);
            }
        }
    }
}
