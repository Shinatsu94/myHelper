/*
File: Main.java
功能: 判斷輸入的數字各位數是否嚴格遞增，若是則輸出 "<digits> is well-ordered number."，否則輸出 "<digits> is not well-ordered number."
作者: 你的姓名
日期: 2025-11-15
說明: 以單次輸入為主，檢查數字各位是否嚴格遞增，並輸出結果。
*/

public class Main {
    public static void main(String[] args) throws Exception {
        java.io.BufferedReader br = new java.io.BufferedReader(new java.io.InputStreamReader(System.in));
        String s = br.readLine();
        if (s != null) s = s.trim(); else s = "";

        boolean ok = isWellOrdered(s);
        if (ok) {
            System.out.println(s + " is well-ordered number.");
        } else {
            System.out.println(s + " is not well-ordered number.");
        }
    }

    // 判斷字串中各位數字是否嚴格遞增
    private static boolean isWellOrdered(String s) {
        if (s == null) return false;
        // 空字串或單一位數視為良序
        if (s.length() <= 1) return true;

        for (int i = 1; i < s.length(); i++) {
            char curChar = s.charAt(i);
            char prevChar = s.charAt(i - 1);

            // 非數字字元視為無效輸入，回傳非良序
            if (!Character.isDigit(curChar) || !Character.isDigit(prevChar)) {
                return false;
            }

            int cur = curChar - '0';
            int prev = prevChar - '0';
            if (cur <= prev) return false; // 未嚴格遞增
        }
        return true;
    }
}
