/*
檔案名稱: LowerTriangleArray.java
功能: 建立並輸出高度為 n 的下三角形二維陣列，元素為 '*'
作者: 教師
日期: 2025-11-14
說明: 讀取 0 < n <= 20 的整數 n，建立 n x n 的二維陣列，將 arr[i][j] 在 i >= j 時設為 '*', 其餘位置不印出
*/

import java.util.Scanner;

public class LowerTriangleArray {

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        int n = 0;
        // 讀取並驗證輸入，確保 0 < n <= 20
        while (true) {
            if (scanner.hasNextInt()) {
                n = scanner.nextInt();
                if (n > 0 && n <= 20) {
                    break;
                }
            } else {
                // 遇到非整數，移除該輸入並繼續等待有效數字
                scanner.next();
            }
        }

        // 建立並填充二維陣列
        char[][] arr = new char[n][n];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= i; j++) {
                arr[i][j] = '*';
            }
        }

        // 輸出下三角形，僅印出有星號的位置
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= i; j++) {
                System.out.print(arr[i][j]);
            }
            System.out.println();
        }

        scanner.close();
    }
}
