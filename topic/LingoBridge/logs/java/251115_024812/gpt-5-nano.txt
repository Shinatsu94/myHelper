import java.util.Scanner;

/*
 檔案名稱: CSIE_41343221_1031_3.java
 功能: 建立並列印下三角形的二維陣列，高度 n 由使用者輸入，範圍 0 < n <= 20
 作者: 教師
 日期: 2025-11-14
*/

public class CSIE_41343221_1031_3 {

    public static void main(String[] args) {
        int n;
        StringBuilder sb = new StringBuilder();

        // 使用 try-with-resources 以確保 Scanner 在使用完畢後自動關閉
        try (Scanner scanner = new Scanner(System.in)) {
            if (!scanner.hasNextInt()) {
                return;
            }
            n = scanner.nextInt();

            // 驗證高度範圍
            if (n <= 0 || n > 20) {
                return;
            }

            // 建立下三角形的二維陣列，並填入 '*'
            char[][] tri = new char[n][n];
            for (int i = 0; i < n; i++) {
                for (int j = 0; j <= i; j++) {
                    tri[i][j] = '*';
                }
            }

            // 輸出下三角形，避免在每次輸出時呼叫 System.out.print
            for (int i = 0; i < n; i++) {
                for (int j = 0; j <= i; j++) {
                    sb.append(tri[i][j]);
                }
                if (i < n - 1) {
                    sb.append('\n');
                }
            }

            System.out.print(sb.toString());
        }
    }
}
