修改後完整程式碼
/*
檔案名稱: Main.java
功能: 輸入整數 targetSum，統計在 1..100000 範圍內，有多少個數字的各位數字和等於 targetSum。
作者: 教師
日期: 2025-11-15
*/

import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);
        int targetSum = in.nextInt();
        in.close();

        int count = 0;
        // 0 <= targetSum <= 45 為可行範圍，超出則答案為 0
        if (targetSum >= 0 && targetSum <= 45) {
            for (int i = 1; i <= 100000; i++) {
                if (digitSum(i) == targetSum) {
                    count++;
                }
            }
        }

        System.out.println(count);
    }

    // 計算整數 n 的各位數字和
    private static int digitSum(int n) {
        int sum = 0;
        while (n > 0) {
            sum += n % 10;
            n /= 10;
        }
        return sum;
    }
}
