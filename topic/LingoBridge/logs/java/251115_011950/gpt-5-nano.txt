/**
 * File: CSIE_41343101_1031_1.java
 * 功能: 輸入任意長度整數陣列，並由小到大排序，輸出排序後的內容
 * 作者: 你的名字
 * 日期: 2025-11-14
 * 版本: 1.0
 */

import java.util.Arrays;
import java.util.Scanner;

public class CSIE_41343101_1031_1 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();
        if (n <= 0) {
            sc.close();
            return;
        }

        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                arr[i] = 0; // 若不足輸入，填補為 0，避免程式異常
            }
        }
        sc.close();

        Arrays.sort(arr);

        for (int i = 0; i < n; i++) {
            if (i > 0) System.out.print(" ");
            System.out.print(arr[i]);
        }
        // System.out.println(); // 如需換行，可取消註解
    }
}
