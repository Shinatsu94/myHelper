/*
Filename: SortArrayAscending.java
功能: 讀入整數陣列長度與元素，排序並輸出升冪序列
作者: 你的姓名
日期: 2025-11-14
*/

import java.util.Scanner;
import java.util.Arrays;

public class SortArrayAscending {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int n;
        // 讀取陣列長度
        if (sc.hasNextInt()) {
            n = sc.nextInt();
        } else {
            sc.close();
            return;
        }

        // 避免不合理長度
        if (n <= 0) {
            System.out.println();
            sc.close();
            return;
        }

        int[] arr = new int[n];
        // 依序讀入 n 個整數
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                // 不足時以 0 補齊，避免 ArrayIndexOutOfBounds
                arr[i] = 0;
            }
        }

        // 升冪排序
        Arrays.sort(arr);

        // 輸出排序結果，元素以空格分隔，末尾不留空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        sc.close();
    }
}
