/*
 檔案名稱: CSIE_41343119_1031_2.java
 功能: 讀取長度與整數序列，輸出升序排序結果
 作者: 金宗翰
 日期: 2025/10/31
 說明: 使用 Java 內建排序 Arrays.sort 以確保正確性與簡潔性；輸出格式為單行、元素間空格分隔。
*/
import java.util.Scanner;
import java.util.Arrays;

public class CSIE_41343119_1031_2 {
    public static void main(String[] args) {
        Scanner scn = new Scanner(System.in);

        // 讀取長度
        if (!scn.hasNextInt()) {
            scn.close();
            return;
        }
        int n = scn.nextInt();
        int[] A = new int[n];

        // 讀取陣列元素
        for (int i = 0; i < n; i++) {
            if (scn.hasNextInt()) {
                A[i] = scn.nextInt();
            } else {
                A[i] = 0; // 若輸入不足，預設為 0
            }
        }

        // 排序（升序）
        Arrays.sort(A);

        // 輸出，避免尾端多餘空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(A[i]);
        }
        System.out.println(sb.toString());

        scn.close();
    }
}
