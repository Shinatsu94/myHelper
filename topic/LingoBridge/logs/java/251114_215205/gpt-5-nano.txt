/* 
檔案名稱: CSIE_41343107_1031_3_a.java
功能: 讀取整數長度 n，接著 n 個整數，將它們升序排序並輸出
作者: 王世豪
日期: 2025/11/14
*/

import java.util.Arrays;
import java.util.Scanner;

public class CSIE_41343107_1031_3_a {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取長度 n
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();
        int[] arr = new int[n];

        // 讀入 n 個整數
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                arr[i] = 0; // 若輸入不足，預設為 0，防止例外
            }
        }

        // 升序排序
        Arrays.sort(arr);

        // 輸出，數值之間以單一空格分隔，末尾換行
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        sc.close();
    }
}
