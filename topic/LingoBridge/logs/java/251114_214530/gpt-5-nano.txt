/*

import java.util.Scanner;

public class CSIE_41343240_1031_2 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int n = sc.nextInt();
        int[] arr = new int[n];

        // 讀取陣列元素
        for (int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();
        }

        // 使用插入排序進行升序排序（適用於任意長度陣列，且可處理負數）
        for (int i = 1; i < n; i++) {
            int key = arr[i];
            int j = i - 1;
            // 將比 key 大的元素向右移動
            while (j >= 0 && arr[j] > key) {
                arr[j + 1] = arr[j];
                j--;
            }
            arr[j + 1] = key;
        }

        // 輸出排序結果，元素之間以單一空格分隔
        for (int i = 0; i < n; i++) {
            if (i > 0) System.out.print(" ");
            System.out.print(arr[i]);
        }
        System.out.println();

        sc.close();
    }
}
