public class LeibnizPi {
    public static void main(String[] args) {
        // 逐步累加，避免重複計算，輸出 i 為 10000 的倍數直到 100000
        double sum = 0.0;
        int denom = 1; // 分母為 1,3,5,...
        int sign = 1;  // 符號，正負交替
        for (int i = 1; i <= 100000; i++) {
            sum += sign * (4.0 / denom);
            denom += 2;
            sign = -sign;
            if (i % 10000 == 0) {
                System.out.printf("i = %6d, pi = %.10f%n", i, sum);
            }
        }
    }
}
