/*
File: CSIE_41343208_1031_1.java
Description: 讀取長度 n，接著讀入 n 個整數，並以升序輸出排序結果。
Author: ChatGPT
Date: 2025-11-14
*/

import java.util.Arrays;
import java.util.Scanner;

public class CSIE_41343208_1031_1 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取長度
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();

        if (n <= 0) {
            // 無元素時輸出空行以符合輸出格式
            System.out.println();
            sc.close();
            return;
        }

        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                arr[i] = 0; // 預設值，避免讀取中斷
            }
        }

        // 排序，這裡採用 Java 內建排序以穩定且高效
        Arrays.sort(arr);

        // 輸出排序結果，避免末尾多餘空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        sc.close();
    }
}
