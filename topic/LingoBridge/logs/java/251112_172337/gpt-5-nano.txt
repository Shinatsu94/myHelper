/*
檔案名稱: CSIE_41343119_1031_2.java
功能: 讀取一個整數 n 與 n 個整數，將它們升冪排序後輸出
作者: 金宗翰
日期: 2025-11-12
說明: 使用內建排序 Arrays.sort 提高正確性與效率，輸出結果以空格分隔，單行輸出。
*/

import java.util.Scanner;
import java.util.Arrays;

public class CSIE_41343119_1031_2 {
    public static void main(String[] args) {
        Scanner scn = new Scanner(System.in);
        if (!scn.hasNextInt()) {
            scn.close();
            return;
        }

        int n = scn.nextInt();
        if (n <= 0) {
            System.out.println();
            scn.close();
            return;
        }

        int[] A = new int[n];
        for (int i = 0; i < n; i++) {
            A[i] = scn.nextInt();
        }

        Arrays.sort(A); // 升序排序

        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(A[i]);
        }
        System.out.println(sb.toString());

        scn.close();
    }
}

- 輸出格式改為單行空格分隔，符合題目輸出範例與常見競賽測資格式。
