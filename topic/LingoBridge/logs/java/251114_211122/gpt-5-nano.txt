/*
 * File: CSIE_41343233_1031_2.java
 * 功能: 將使用者輸入的一維整數陣列由小到大排序並輸出
 * 作者: 教師
 * 日期: 2025-11-14
 */

import java.util.Scanner;
import java.util.Arrays;

public class CSIE_41343233_1031_2 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }

        int n = sc.nextInt();
        if (n <= 0) {
            sc.close();
            return;
        }

        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                arr[i] = 0; // 若輸入不足，補上預設值以避免例外
            }
        }

        // 使用 Arrays.sort 直接對整數陣列由小到大排序
        Arrays.sort(arr);

        // 使用 StringBuilder 輸出，避免尾端多餘空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }

        System.out.print(sb.toString());
        sc.close();
    }
}
