/*
 * 檔案名稱: SortArrayAscending.java
 * 功能描述: 讀取整數長度 n 與 n 個整數，將其升序排序，最後輸出排序結果
 * 作者: 教師
 * 日期: 2025-11-14
 */

import java.util.Arrays;
import java.util.Scanner;

public class SortArrayAscending {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 1. 讀取陣列長度 n
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();
        if (n <= 0) {
            sc.close();
            return;
        }

        // 2. 讀取 n 個整數
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                // 不足的輸入情況，預設為 0（題目未說明此情況，這裡提供保守的預設）
                arr[i] = 0;
            }
        }

        sc.close();

        // 3. 進行升序排序
        Arrays.sort(arr);

        // 4. 輸出排序結果，數字之間以空白分隔，末尾不留空白
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());
    }
}
