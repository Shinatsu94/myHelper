import java.util.Arrays;
import java.util.Scanner;

/**
 * CSIE_41343215_1031_2.java
 * 功能: 讀取一個整數 n，接著讀入 n 個整數，將其升冪排序後輸出（以空格分隔）
 * 作者: 教師
 * 日期: 2025-11-14
 * 備註: 不使用 EOF，執行一次即可結束
 */
public class CSIE_41343215_1031_2 {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        // 先讀取陣列長度
        if (!scanner.hasNextInt()) {
            scanner.close();
            return;
        }
        int n = scanner.nextInt();

        if (n <= 0) {
            // 無元素可排序，輸出換行並結束
            System.out.println();
            scanner.close();
            return;
        }

        // 讀取 n 個整數
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (scanner.hasNextInt()) {
                arr[i] = scanner.nextInt();
            } else {
                arr[i] = 0; // 若輸入不足，預設為 0；實務上應避免此情況
            }
        }

        // 升冪排序
        Arrays.sort(arr);

        // 輸出：以單一行，元素以空格分隔
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        scanner.close();
    }
}
