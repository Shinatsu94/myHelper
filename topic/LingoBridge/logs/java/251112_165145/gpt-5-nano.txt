/*
 * 檔案名稱: CSIE_41343203_1031_1.java
 * 功能說明: 依序讀入長度 n 與 n 個整數，將陣列內容升序排序，輸出排序後的結果（以空格分隔，不在結尾留空格）。
 * 作者: 你的姓名
 * 日期: 2025-11-12
 */

public class CSIE_41343203_1031_1 {
    public static void main(String[] args) {
        java.util.Scanner sc = new java.util.Scanner(System.in);

        int n = sc.nextInt();
        int[] arr = new int[n];

        for (int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();
        }

        sortArray(arr);

        // 輸出：不在最後留尾隨空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < arr.length; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.print(sb.toString());

        sc.close();
    }

    // 選擇排序 - 由小到大
    private static void sortArray(int[] arr) {
        int n = arr.length;
        for (int i = 0; i < n - 1; i++) {
            int min = i;
            for (int j = i + 1; j < n; j++) {
                if (arr[j] < arr[min]) {
                    min = j;
                }
            }
            if (min != i) {
                int tmp = arr[min];
                arr[min] = arr[i];
                arr[i] = tmp;
            }
        }
    }
}
