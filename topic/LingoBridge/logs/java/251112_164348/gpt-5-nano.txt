Description: 讀取整數個數 n，接著讀入 n 個整數，將它們升序排序，並以空格分隔輸出。
Author: 你可以填入作者姓名
Date: 2025-11-12

import java.util.Scanner;
import java.util.Arrays;

public class CSIE_41343111_1031_1 {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        if (!scanner.hasNextInt()) {
            scanner.close();
            return;
        }

        int n = scanner.nextInt();
        if (n <= 0) {
            // 輸出換行符以符合一般輸出格式，並結束
            System.out.println();
            scanner.close();
            return;
        }

        int[] a = new int[n];
        for (int i = 0; i < n; i++) {
            a[i] = scanner.nextInt();
        }

        Arrays.sort(a);

        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(a[i]);
        }

        System.out.println(sb.toString());
        scanner.close();
    }
}
