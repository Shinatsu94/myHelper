/*
File: ReverseDigits.java
Description: Read a single integer from input and print its digits reversed. Supports negative numbers.
Author: YourName
Date: 2025-11-15
*/

import java.util.Scanner;

public class ReverseDigits {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        long n = sc.nextLong();

        long sign = (n < 0) ? -1L : 1L;
        long val = Math.abs(n);
        long rev = 0;

        while (val != 0) {
            rev = rev * 10 + (val % 10);
            val /= 10;
        }

        rev *= sign;
        System.out.println(rev);
        sc.close();
    }
}
