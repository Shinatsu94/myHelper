/*
檔案名稱: SortArrayAsc.java
功能: 讀入長度 n 並接著 n 個整數，輸出升序排序結果
作者: 王世豪
日期: 2025-10-31
*/
import java.util.Scanner;
import java.util.Arrays;

public class SortArrayAsc {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀入長度 n
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();

        // 讀入 n 個整數
        int[] a = new int[n];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                a[i] = sc.nextInt();
            } else {
                // 若數量不足，預設為 0（依需求，此情況應避免）
                a[i] = 0;
            }
        }
        sc.close();

        // 升序排序
        Arrays.sort(a);

        // 輸出，避免尾端多餘空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(a[i]);
        }
        System.out.print(sb.toString());
    }
}
