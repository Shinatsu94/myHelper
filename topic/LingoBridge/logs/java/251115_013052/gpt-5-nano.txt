/*
 * 檔案名稱: CSIE_41343211_1031_2.java
 * 功能: 讀取整數個數與內容，對內容進行升序排序，輸出結果
 * 作者: 學生
 * 日期: 2025-11-14
 *
 * 使用 Arrays.sort 進行升序排序，支援負數與任意長度的輸入。
 */

import java.util.Scanner;
import java.util.Arrays;

public class CSIE_41343211_1031_2 {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);

        // 讀取陣列長度
        int n;
        if (!input.hasNextInt()) {
            input.close();
            return;
        }
        n = input.nextInt();

        if (n <= 0) {
            System.out.println("陣列長度必須大於 0");
            input.close();
            return;
        }

        int[] arr = new int[n];

        // 讀取陣列元素
        for (int i = 0; i < n; i++) {
            if (input.hasNextInt()) {
                arr[i] = input.nextInt();
            } else {
                // 若輸入不足，預設為 0，避免例外
                arr[i] = 0;
            }
        }

        // 將陣列進行升序排序
        Arrays.sort(arr);

        // 輸出結果，元素以單一空格分隔
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        input.close();
    }
}
