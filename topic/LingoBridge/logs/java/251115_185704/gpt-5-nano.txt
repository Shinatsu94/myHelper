import java.util.Scanner;

public class CSIE_41343202_1017_3 {
    // Returns the integer formed by reversing the digits of number.
    // Works for negative numbers as well by preserving the sign.
    private static long reverseNumber(long number) {
        long sign = number < 0 ? -1 : 1;
        long n = Math.abs(number);
        long reversed = 0;

        while (n != 0) {
            long digit = n % 10;
            reversed = reversed * 10 + digit;
            n /= 10;
        }

        return sign * reversed;
    }

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        long number = scanner.nextLong();
        long result = reverseNumber(number);
        System.out.println(result);
        scanner.close();
    }
}
