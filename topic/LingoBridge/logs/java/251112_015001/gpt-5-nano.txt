import java.util.Scanner;

public class SwapTwoNumbers {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int A = sc.nextInt();
        int B = sc.nextInt();

        // 不使用額外變數的交換（XOR 交換法）
        A = A ^ B;
        B = A ^ B;
        A = A ^ B;

        // 輸出交換後的值，每行一個
        System.out.println(A);
        System.out.println(B);

        sc.close();
    }
}
