public class CSIE_41343124_0926_2 {
    public static void main(String[] args) {
        // 定義需要輸出的字串
        String school = "虎科大";
        String studentName = "林武杰";

        // 逐字印出及其對應的 Unicode 內碼
        printCharsWithCodePoints(school);
        printCharsWithCodePoints(studentName);
    }

    // 輔助方法：對給定字串中的每個字元，印出字元與其整數型的 code point
    private static void printCharsWithCodePoints(String s) {
        for (int i = 0; i < s.length(); i++) {
            char ch = s.charAt(i);
            System.out.println(ch + " " + (int) ch);
        }
    }
}
