/*
 * Filename: DigitSumCounter.java
 * Description: 計算 1 到 100000 之間，有多少個數字的各位數字總和等於輸入的 A。
 * Author: 你的教師
 * Date: 2025-11-15
 * Notes: 以模組化方式實作，提供更清晰的邏輯與註解。
 */

import java.io.BufferedReader;
import java.io.InputStreamReader;

public class DigitSumCounter {
    public static void main(String[] args) throws Exception {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        String line = br.readLine();
        int target;
        try {
            target = Integer.parseInt(line.trim());
        } catch (Exception e) {
            // 非法輸入視為無解
            System.out.println(0);
            return;
        }

        int count = 0;
        // 5位數字（最大 99999）之位數總和最大為 45；100000 的位數之和為 1，因此範圍仍在 [0,45]
        if (target < 0 || target > 45) {
            System.out.println(0);
            return;
        }

        for (int i = 1; i <= 100000; i++) {
            if (digitSum(i) == target) {
                count++;
            }
        }

        System.out.println(count);
    }

    // 計算整數 n 的各位數字總和
    private static int digitSum(int n) {
        int sum = 0;
        while (n > 0) {
            sum += n % 10;
            n /= 10;
        }
        return sum;
    }
}
