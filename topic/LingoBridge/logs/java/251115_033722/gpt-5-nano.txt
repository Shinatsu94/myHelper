File: CSIE_41343241_1031_3b.java
功能: 建立高度為 n 的下三角形二維陣列，內容以 1 開始遞增填入，尋找指定 num 的位置，輸出為 [row, col]，若找不到則輸出相應訊息。
作者: 自動批改系統
日期: 2025-11-14
*/

import java.util.Scanner;

public class CSIE_41343241_1031_3b {
    public static void main(String[] args) {
        int n;
        int num;

        // 使用 try-with-resources 確保 Scanner 會正確關閉
        try (Scanner sc = new Scanner(System.in)) {
            if (!sc.hasNextInt()) {
                return;
            }
            n = sc.nextInt();

            if (!sc.hasNextInt()) {
                return;
            }
            num = sc.nextInt();
        }

        // 建立下三角形二維陣列
        int[][] arr = new int[n][];
        int nextValue = 1;
        for (int i = 0; i < n; i++) {
            arr[i] = new int[i + 1];
            for (int j = 0; j <= i; j++) {
                arr[i][j] = nextValue++;
            }
        }

        // 找出 num 位置
        int foundRow = -1;
        int foundCol = -1;
        outer:
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= i; j++) {
                if (arr[i][j] == num) {
                    foundRow = i;
                    foundCol = j;
                    break outer;
                }
            }
        }

        if (foundRow != -1) {
            System.out.println(num + "[" + (foundRow + 1) + ", " + (foundCol + 1) + "]");
        } else {
            System.out.println("not find " + num + " in array.");
        }
    }
}
