修改後完整程式碼（LeibnizPi.java）
/import java.io.*;  // 如需改用 I/O 以便測試，可保留
/***************************************
 * File: LeibnizPi.java
 * 功能: 使用 Leibniz 公式近似圓周率 π，
 *      輸出 i = 10000, 20000, ..., 100000 對應的 π 值，
 *      π 輸出小數點後 10 位，i 輸出為 6 位寬度。
 *      另外找出最小的 i 使 pi >= 3.14159，並輸出該 i。
 * 作者: 教師
 * 日期: 2025-11-11
 ***************************************/
public class LeibnizPi {
    public static void main(String[] args) {
        // 第1部份：計算並輸出 i = 10000 到 100000，步長 10000 的 pi 值
        double sum = 0.0;
        for (int i = 1; i <= 100000; i++) {
            // 第 i 項對應的分母為 2*i - 1，符號為 (-1)^(i-1)
            double term = ((i % 2 == 1) ? 1.0 : -1.0) / (2.0 * i - 1.0);
            sum += term;
            if (i % 10000 == 0) {
                double pi = 4.0 * sum;
                // i 使用6位寬度輸出，pi 顯示小數點後10位
                System.out.printf("i = %6d, pi = %.10f%n", i, pi);
            }
        }

        // 第2部份：找出最小的 i 使 pi >= 3.14159
        double sum2 = 0.0;
        int iMin = -1;
        for (int i = 1; ; i++) {
            double term = ((i % 2 == 1) ? 1.0 : -1.0) / (2.0 * i - 1.0);
            sum2 += term;
            double pi = 4.0 * sum2;
            if (pi >= 3.14159) {
                iMin = i;
                break;
            }
        }

        System.out.println();
        System.out.println("i = " + iMin);
    }
}
