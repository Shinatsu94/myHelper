完整程式碼（修改後）
/*
檔案名稱: LeibnizPi.java
功能: 使用莱布尼茨公式計算 pi，輸出在 i = 10000, 20000, ..., 100000 的近似值，並找出最小 i 使 pi 的近似值符合 3.14159
作者: 教授
日期: 2025-11-11
*/
public class LeibnizPi {
    public static void main(String[] args) {
        int maxI = 100000;
        int step = 10000;
        double sum = 0.0;
        int foundI = -1;

        // 逐步累加，逐步輸出指定 i 對應的 pi 值
        for (int i = 1; i <= maxI; i++) {
            double term = ((i % 2 == 1) ? 1.0 : -1.0) / (2.0 * i - 1.0);
            sum += term;

            if (i % step == 0) {
                double pi = 4.0 * sum;
                System.out.printf("i = %6d, pi = %.10f%n", i, pi);
            }

            // 尋找最小的 i 使 pi 的五位小數為 3.14159
            if (foundI == -1) {
                double pi = 4.0 * sum;
                double pi5 = Math.round(pi * 100000.0) / 100000.0;
                if (pi5 == 3.14159) {
                    foundI = i;
                }
            }
        }

        // 額外行：題目要求的最小 i 值
        System.out.println();
        if (foundI != -1) {
            System.out.println("i = " + foundI);
        } else {
            // 若尚未在給定範圍內找到，給出範圍外的提示
            System.out.println("i = " + maxI);
        }
    }
}
