import java.util.Scanner;

/*
檔案名稱: CSIE_41343150_1003_3.java
功能: 讀入兩個整數 A 與 B，使用僅有 A、B 的算術交換法互換後印出結果。
作者: 你的名字
日期: 2025-11-11
*/
public class CSIE_41343150_1003_3 {
    public static void main(String[] args) {
        Scanner s = new Scanner(System.in);
        int A = s.nextInt();
        int B = s.nextInt();

        // 不使用額外變數，使用算術方法互換 A 與 B
        A = A + B;
        B = A - B;
        A = A - B;

        // 輸出：互換後的 A 與 B，符合題意與範例輸出
        System.out.println(A);
        System.out.println(B);

        s.close();
    }
}
