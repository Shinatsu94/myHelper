修改後完整程式碼
```java
/**
 * LowerTriStar.java
 * 功能: 建立並列印下三角形的二維陣列（n x n）的前半部分，此部分填 '*'，形成下三角形。
 * 使用者輸入高度 n (0 < n <= 20)
 * 作者: 教師
 * 日期: 2025-11-14
 */

import java.util.Scanner;

public class LowerTriStar {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();

        // 邊界檢查：0 < n <= 20
        if (n <= 0 || n > 20) {
            sc.close();
            return;
        }

        // 建立並填充二維陣列：下三角部分為 '*'
        char[][] arr = new char[n][n];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= i; j++) {
                arr[i][j] = '*';
            }
        }

        // 輸出下三角形：每行輸出前 i+1 個星號
        for (int i = 0; i < n; i++) {
            StringBuilder sb = new StringBuilder();
            for (int j = 0; j <= i; j++) {
                sb.append(arr[i][j]);
            }
            System.out.println(sb.toString());
        }

        sc.close();
    }
}
```
