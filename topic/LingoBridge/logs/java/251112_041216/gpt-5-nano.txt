/*
 * File: Main.java
 * Description: 讀入兩個整數，存入變數 a、b；交換兩個變數的值（不使用額外變數），
 *              並印出交換後的結果（每行一個數字）。
 * Author: 教師
 * Date: 2025-11-11
 */

import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int a = sc.nextInt();
        int b = sc.nextInt();

        // 使用不需要額外變數的位元互換法（XOR swap）
        a = a ^ b;
        b = a ^ b;
        a = a ^ b;

        System.out.println(a);
        System.out.println(b);

        sc.close();
    }
}
