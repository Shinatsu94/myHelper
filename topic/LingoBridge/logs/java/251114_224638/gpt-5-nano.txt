/**
 * File: SortArray.java
 * 功能: 讀取整數陣列長度 n，接著讀入 n 個整數，將陣列排序並輸出排序結果（由小到大）。
 * 作者: 你的姓名
 * 日期: 2025-11-14
 * 備註: 避免使用 EOF，程式執行一次後結束。
 */

import java.util.Arrays;
import java.util.Scanner;

public class SortArray {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);

        // 讀取陣列長度
        if (!input.hasNextInt()) {
            return;
        }
        int n = input.nextInt();
        if (n <= 0) {
            // 無效長度直接結束
            return;
        }

        // 讀入 n 個整數
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (input.hasNextInt()) {
                arr[i] = input.nextInt();
            } else {
                // 不足的輸入以 0 代替，避免程式崩動
                arr[i] = 0;
            }
        }

        // 排序
        Arrays.sort(arr);

        // 輸出：同一行，數字以空格分隔
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.print(sb.toString());
    }
}
