```java
/*
檔案名稱: CSIE_41343205_1031_1.java
功能: 讀取長度與整數陣列，將內容由小到大排序，並輸出
作者: 教授
日期: 2025-11-12
*/
import java.io.InputStream;
import java.io.IOException;
import java.util.Arrays;

public class CSIE_41343205_1031_1 {
    public static void main(String[] args) {
        FastScanner fs = new FastScanner(System.in);
        int n = fs.nextInt();
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            arr[i] = fs.nextInt();
        }

        // 使用系統內建排序，提升效率與穩定性
        Arrays.sort(arr);

        // 不輸出尾端多餘空格，輸出整體格式更穩定
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());
    }

    // 簡易快速輸入解析器，避免使用 Scanner 的效能問題
    private static class FastScanner {
        private final InputStream in;
        private final byte[] buffer = new byte[1 << 16];
        private int ptr = 0, len = 0;

        FastScanner(InputStream is) { in = is; }

        private int read() throws IOException {
            if (ptr >= len) {
                len = in.read(buffer);
                ptr = 0;
                if (len <= 0) return -1;
            }
            return buffer[ptr++];
        }

        int nextInt() {
            int c, sign = 1, val = 0;
            try {
                do {
                    c = read();
                } while (c <= ' ');
                if (c == '-') {
                    sign = -1;
                    c = read();
                }
                while (c > ' ') {
                    val = val * 10 + (c - '0');
                    c = read();
                }
            } catch (IOException e) {
                throw new RuntimeException(e);
            }
            return val * sign;
        }
    }
}
```

修改說明
- 使用 Arrays.sort(arr) 取代手動選擇排序，時間複雜度降至 O(n log n)，提升效率並簡化程式邏輯。
- 引入快速輸入 FastScanner，避免 Scanner 在大量資料輸入時的效能瓶頸，符合「一次執行完畢」的需求與測資穩定性。
- 輸出改為不產生尾端多餘空格，改善輸出格式穩定性，與題目期望輸出更一致。
- 追加標頭註解，並使用有意義的變數命名與適當的縮排，提升可讀性與維護性。
