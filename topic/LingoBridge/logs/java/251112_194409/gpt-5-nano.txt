import java.util.Scanner;

public class CSIE_41343242_1031_2 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int n = sc.nextInt();
        int[] ary = new int[n];
        for (int i = 0; i < n; i++) {
            ary[i] = sc.nextInt();
        }
        sc.close();

        // 使用簡單的選擇排序，將陣列由小到大排序
        for (int i = 0; i < n - 1; i++) {
            int minIndex = i;
            for (int j = i + 1; j < n; j++) {
                if (ary[j] < ary[minIndex]) {
                    minIndex = j;
                }
            }
            if (minIndex != i) {
                int tmp = ary[i];
                ary[i] = ary[minIndex];
                ary[minIndex] = tmp;
            }
        }

        // 輸出排序後的結果，元素以空白分隔，無尾部多餘空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(ary[i]);
        }
        System.out.print(sb.toString());
    }
}
