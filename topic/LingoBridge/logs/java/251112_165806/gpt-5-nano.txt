 * 作者: (你的姓名)
 * 日期: (日期)
 */

import java.util.Scanner;

public class CSIE_41343217_1031_1 {

    public static void main(String[] args) {
        // 建立 Scanner 物件，用於讀取使用者輸入
        Scanner sc = new Scanner(System.in);

        // 讀取陣列長度 n
        int n = sc.nextInt();

        // 只使用一個陣列來儲存輸入的整數
        int[] a = new int[n];
        for (int i = 0; i < n; i++) {
            a[i] = sc.nextInt();
        }

        // 對陣列 a 進行排序
        selectionSort(a);

        // 印出排序後的陣列
        printArray(a);

        // 關閉 Scanner
        sc.close();
    }

    // 選擇排序 (Selection Sort) 的實作方法
    public static void selectionSort(int[] a) {
        int n = a.length;

        for (int i = 0; i < n - 1; i++) {
            int minIdx = i;
            for (int j = i + 1; j < n; j++) {
                if (a[j] < a[minIdx]) {
                    minIdx = j;
                }
            }
            // 交換 a[i] 與 a[minIdx]
            int tmp = a[i];
            a[i] = a[minIdx];
            a[minIdx] = tmp;
        }
    }

    // 印出陣列內容的方法，元素之間以單一空格分隔
    public static void printArray(int[] a) {
        for (int i = 0; i < a.length; i++) {
            if (i > 0) System.out.print(" ");
            System.out.print(a[i]);
        }
        System.out.println();
    }
}
