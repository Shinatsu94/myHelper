import java.util.*;

public class CSIE_51015105_1031_3
{
    /*
     * 檔案名稱: CSIE_51015105_1031_3.java
     * 功能: 讀入長度 n，接著 n 個整數；將它們升序排序並輸出。
     * 作者: 自訂
     * 日期: 2025-11-14
     */
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        // 讀取陣列長度
        if (!scanner.hasNextInt()) {
            return;
        }
        int n = scanner.nextInt();

        // 讀取 n 個整數
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (scanner.hasNextInt()) {
                arr[i] = scanner.nextInt();
            } else {
                arr[i] = 0; // 若輸入不足，預設為 0
            }
        }

        // 排序
        Arrays.sort(arr);

        // 輸出，空格分隔，結尾換行
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());
    }
}
