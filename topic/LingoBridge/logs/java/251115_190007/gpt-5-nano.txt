import java.util.Scanner;

/*
 * 檔案名稱: CSIE_41343216_1017_3.java
 * 功能: 將使用者輸入的整數數字位元依序反轉，並輸出結果。
 * 作者: 教師
 * 日期: 2025-11-15
 * 說明:
 *  - 支援正負號，負數時反轉後仍為負數。
 *  - 使用字串處理以避免整數溢位問題，且對任意長度的輸入皆有效。
 *  - 輸入為 0 時輸出 0；輸入若含前導零，反轉後會移除反轉結果前方的多餘零。
 */

public class CSIE_41343216_1017_3 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        if (!sc.hasNext()) {
            sc.close();
            return;
        }

        String input = sc.next().trim();
        boolean isNegative = input.startsWith("-");
        if (isNegative) {
            input = input.substring(1);
        }

        // 反轉數字字串
        String reversed = new StringBuilder(input).reverse().toString();

        // 移除反轉後字串前方的多餘零，確保輸出為一個整數字串；若全為零，輸出 "0"
        reversed = reversed.replaceFirst("^0+(?!$)", "");
        if (reversed.isEmpty()) {
            reversed = "0";
        }

        if (isNegative) {
            System.out.println("-" + reversed);
        } else {
            System.out.println(reversed);
        }

        sc.close();
    }
}
