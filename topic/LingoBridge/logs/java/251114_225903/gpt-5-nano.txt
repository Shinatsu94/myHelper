```java
// File: SortOneDArray.java
// 功能: 讀取一維整數陣列，升序排序並輸出
// 作者: 教師
// 日期: 2025-11-14
// 說明: 本程式只執行一次，讀入長度與對應的整數，支援負數。

import java.util.Arrays;
import java.util.Scanner;

public class SortOneDArray {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                arr[i] = 0; // 若輸入不足，補 0
            }
        }
        sc.close();

        Arrays.sort(arr);

        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());
    }
}
```
