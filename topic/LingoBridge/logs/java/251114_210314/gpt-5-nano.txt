/*
檔案名稱: csie_1031_41343214.java
功能: 讀入陣列長度與元素，將元素升序排序，並輸出排序結果
作者: 設定中
日期: 2025-11-14
*/
import java.util.Scanner;

public class csie_1031_41343214 {
    public static void main(String[] args) {
        // 建立掃描器，以標準輸入取得資料
        Scanner sc = new Scanner(System.in);

        // 讀取陣列長度
        int n = sc.nextInt();

        int[] arr = new int[n];

        // 逐個讀入陣列元素
        for (int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();
        }

        // 使用選擇排序 (Selection Sort) 將陣列由小到大排序
        for (int i = 0; i < n - 1; i++) {
            int minIndex = i;
            for (int j = i + 1; j < n; j++) {
                if (arr[j] < arr[minIndex]) minIndex = j;
            }
            int temp = arr[i];
            arr[i] = arr[minIndex];
            arr[minIndex] = temp;
        }

        // 輸出排序結果，避免在最後多出一個空格
        for (int i = 0; i < n; i++) {
            System.out.print(arr[i]);
            if (i < n - 1) System.out.print(" ");
        }
        System.out.println();

        sc.close();
    }
}
