/*
File: Main.java
功能: 讀取整數 n 與 n 個整數，將它們從小到大排序後輸出，元素可為負數，且不使用 EOF。
作者: 教師（評分用版本）
日期: 2025-11-14
*/

import java.util.Scanner;
import java.util.Arrays;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取長度 n
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();

        // 讀取 n 個整數
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                // 若輸入不足，預設為 0，避免發生例外
                arr[i] = 0;
            }
        }

        // 排序
        Arrays.sort(arr);

        // 輸出排序後的數列，元素以空格分隔
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        sc.close();
    }
}
