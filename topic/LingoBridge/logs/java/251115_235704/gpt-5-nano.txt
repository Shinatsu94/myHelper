/**
 * File: CSIE_41243247_1003_3.java
 * 功能: 計算在 1..100000 之間，數字各位數字總和等於輸入 A 的數字個數
 * 作者: 李教授
 * 日期: 2025-11-15
 * 註解: 僅輸出符合條件的數字個數，未列出中間的符合數，符合題目輸出格式
 */

import java.util.Scanner;

public class CSIE_41243247_1003_3 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取目標的位數總和
        int targetSum = sc.nextInt();

        int count = 0; // 符合條件的數字數量

        // 條件：1 到 100000 之間，位數總和等於 targetSum
        for (int i = 1; i <= 100000; i++) {
            if (digitSum(i) == targetSum) {
                count++;
            }
        }

        // 只輸出符合條件的數字個數
        System.out.println(count);

        sc.close();
    }

    // 計算數字每一位的總和
    private static int digitSum(int n) {
        int sum = 0;
        while (n > 0) {
            sum += n % 10; // 取出個位數
            n /= 10;       // 去掉最後一位
        }
        return sum;
    }
}
