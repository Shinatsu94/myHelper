/*
檔案名稱: CSIE_41343128_1031_3b.java
功能說明: 讀取使用者輸入的一維整數陣列，將其從小到大排序，並輸出排序結果。
作者:（請填入學生姓名或代號）
日期: 2025-11-14
*/
import java.util.Scanner;
import java.util.Arrays;

public class CSIE_41343128_1031_3b {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取陣列長度
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();

        // 讀入一維陣列
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                // 如果輸入不足，預設為 0，避免例外
                arr[i] = 0;
            }
        }

        // 進行排序（升冪）
        Arrays.sort(arr);

        // 輸出排序後的陣列，元素以空格分隔
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.print(sb.toString());

        sc.close();
    }
}
