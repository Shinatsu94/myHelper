/*
 * 檔案名稱: SortArrayAscending.java
 * 功能說明: 建立任意長度的一維整數陣列，將內容由小到大排序，輸出排序結果
 * 作者: 你的名字
 * 日期: 2025-11-14
 */
import java.util.Scanner;
import java.util.Arrays;

public class SortArrayAscending {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);

        // 讀取陣列長度
        if (!input.hasNextInt()) {
            return;
        }
        int n = input.nextInt();
        if (n <= 0) {
            // 對於無效長度，直接結束輸出
            System.out.println();
            return;
        }

        // 讀取 n 個整數並存入陣列
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (input.hasNextInt()) {
                arr[i] = input.nextInt();
            } else {
                // 若輸入不足，預設為 0，避免發生例外
                arr[i] = 0;
            }
        }

        // 對陣列排序（由小到大）
        Arrays.sort(arr);

        // 輸出排序結果，元素以單一空格分隔
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());
    }
}
