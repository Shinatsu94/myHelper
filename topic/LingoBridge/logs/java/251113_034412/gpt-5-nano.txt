 * 檔案名稱: TigerCodePrinter.java
 * 功能: 印出虎科大及學生姓名的每個字及其對應的 Unicode 編碼
 * 作者: 你的姓名
 * 日期: 2025-11-12
 * 備註: 使用陣列與迴圈以提升可讀性與擴充性
 */
public class TigerCodePrinter {
    public static void main(String[] args) {
        // 先印虎科大
        char[] tiger = {'虎','科','大'};
        // 再印學生中文姓名，這裡以林武杰為例
        char[] studentName = {'林','武','杰'};

        printCharsWithCode(tiger);
        printCharsWithCode(studentName);
    }

    private static void printCharsWithCode(char[] chars) {
        for (char c : chars) {
            System.out.println(c + " " + (int) c);
        }
    }
}
