import java.util.Scanner;

/*
檔案名稱: csie_1003_41343214.java
功能說明: 讀入兩個整數，將它們的值互換並印出。要求在交換過程中僅使用兩個變數，使用 XOR 交換法。
作者: 你的名字
日期: 2025-11-15
*/

public class csie_1003_41343214 {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        // 讀入兩個整數，分別存入 a 與 b
        int a = scanner.nextInt();
        int b = scanner.nextInt();

        // 使用不需要暫存變數的 XOR 交換法：
        // a = a ^ b;  // 第一步將 a 與 b 做位元異或
        // b = a ^ b;  // 利用新的 a 與 b 計算出原始的 a
        // a = a ^ b;  // 計算出原始的 b，使兩變數完成互換
        a = a ^ b;
        b = a ^ b;
        a = a ^ b;

        // 輸出交換後的結果
        System.out.println(a);
        System.out.println(b);

        scanner.close();
    }
}
