// 檔案名稱: CSIE_41343221_1003_2.java
// 功能: 讀入三個整數，輸出最小值、中間值、最大值（分別一行輸出）
// 作者: 你的名字
// 日期: 2025-11-15
// 備註: 依需求使用兩種不同的方法計算最大值與最小值，中間值以總和減去最大值與最小值取得

import java.util.Scanner;

public class CSIE_41343221_1003_2 {

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        // 讀入三個整數
        int first  = scanner.nextInt();
        int second = scanner.nextInt();
        int third  = scanner.nextInt();

        scanner.close();

        // 使用三元運算子找出最大值與最小值
        int maxVal = (first > second) ? ((first > third) ? first : third) : ((second > third) ? second : third);
        int minVal = (first < second) ? ((first < third) ? first : third) : ((second < third) ? second : third);

        // 中間值以 sum - max - min 計算
        int midVal = first + second + third - maxVal - minVal;

        // 輸出順序：最小值、中間值、最大值
        System.out.println(minVal);
        System.out.println(midVal);
        System.out.println(maxVal);
    }
}
