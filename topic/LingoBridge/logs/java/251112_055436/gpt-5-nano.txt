/**

 * 功能說明：計算李布尼茲級數逼近 圓周率 pi，輸出 i = 10000, 20000, ..., 100000
 *            以及找出使 pi >= 3.14159 的最小 i。
 * 日期: 2025/11/11
 * 姓名: 不同實作示例
 * 學號: 00000000
 */
public class LeibnizPiCalculator {

    public static void main(String[] args) {
        final double TARGET_PI = 3.14159;
        final int STEP = 10000;
        final int MAX_I = 100000;

        double sum = 0.0;
        int k = 0; // k 為項數 minus 1，因為 term for k=0 对应 1/(2*0+1)

        // 逐步累加，輸出 i = 10000, 20000, ..., 100000 的 pi 值
        for (; k < MAX_I; k++) {
            double term = ((k % 2 == 0) ? 1.0 : -1.0) / (2.0 * k + 1.0);
            sum += 4.0 * term;
            if ((k + 1) % STEP == 0) {
                int currentI = k + 1;
                System.out.printf(Locale.US, "i = %6d, pi = %.10f%n", currentI, sum);
            }
        }

        // 找到使 pi >= 3.14159 的最小 i
        int minIForTarget = MAX_I;
        if (sum < TARGET_PI) {
            int iIndex = MAX_I; // 已經計算到 MAX_I 個項
            while (sum < TARGET_PI) {
                iIndex++;
                double term = ((iIndex % 2 == 0) ? 1.0 : -1.0) / (2.0 * iIndex + 1.0);
                sum += 4.0 * term;
            }
            minIForTarget = iIndex;
        } else {
            minIForTarget = MAX_I;
        }

        System.out.printf(Locale.US, "i = %6d%n", minIForTarget);
    }
}
