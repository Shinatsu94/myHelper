/*
 * 檔案名稱: CSIE_41343101_1003_2.java
 * 功能說明: 讀入兩個整數 A, B，交換後輸出新值（無額外變數交換）
 * 作者: 教授
 * 日期: 2025-11-11
 */

public class CSIE_41343101_1003_2 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int a = sc.nextInt();
        int b = sc.nextInt();

        // 使用算術交換法，且不使用額外變數
        a = a + b;
        b = a - b;
        a = a - b;

        System.out.println(a);
        System.out.println(b);

        sc.close();
    }
}
