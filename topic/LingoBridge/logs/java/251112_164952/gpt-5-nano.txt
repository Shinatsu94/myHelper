/**
 * File: BubbleSortArray.java
 * 功能: 讀取整數長度 n，接著讀入 n 個整數，將陣列由小到大排序，最後印出排序結果。
 * 作者: 你的名字
 * 日期: 2025-11-12
 */
import java.util.Scanner;

public class BubbleSortArray {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);

        int n = input.nextInt();
        int[] arr = new int[n];

        // 逐個讀取 n 個整數，容錯與格式更為健壯
        for (int i = 0; i < n; i++) {
            arr[i] = input.nextInt();
        }

        // 氣泡排序，加入早期結束優化
        for (int i = 0; i < n - 1; i++) {
            boolean swapped = false;
            for (int j = 0; j < n - 1 - i; j++) {
                if (arr[j] > arr[j + 1]) {
                    int tmp = arr[j];
                    arr[j] = arr[j + 1];
                    arr[j + 1] = tmp;
                    swapped = true;
                }
            }
            if (!swapped) break; // 沒有交換，已排序完畢
        }

        // 輸出：單一空格分隔，避免尾端空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        input.close();
    }
}
