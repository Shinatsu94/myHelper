import java.util.Scanner;

/*
檔案名稱: csie_41343235_1031_1.java
功能: 讀入一個整數長度 n、再讀入 n 個整數，輸出由小到大的排序結果（不使用 EOF，僅執行一次）。
作者: 您的教授
日期: 2025-11-14
說明: 使用選取排序（Selection Sort）實作，並有必要的程式註解與較佳的命名。
*/

public class csie_41343235_1031_1 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int n = sc.nextInt();

        if (n <= 0) {
            sc.close();
            System.out.println();
            return;
        }

        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();
        }

        // 選取排序 (升冪)
        for (int i = 0; i < n - 1; i++) {
            int minIndex = i;
            for (int j = i + 1; j < n; j++) {
                if (arr[j] < arr[minIndex]) {
                    minIndex = j;
                }
            }
            // 交換 arr[i] 與 arr[minIndex]
            int temp = arr[i];
            arr[i] = arr[minIndex];
            arr[minIndex] = temp;
        }

        // 輸出結果，以空格分隔，最後換行
        for (int i = 0; i < n; i++) {
            if (i > 0) System.out.print(" ");
            System.out.print(arr[i]);
        }
        System.out.println();

        sc.close();
    }
}
