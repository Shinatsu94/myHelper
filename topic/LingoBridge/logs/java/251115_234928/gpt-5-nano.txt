public class CSIE_41343241_1003_2 {
    public static void main(String[] args) {
        // 使用 try-with-resources 自動關閉 Scanner
        try (Scanner sc = new Scanner(System.in)) {
            int a = sc.nextInt();
            int b = sc.nextInt();
            int c = sc.nextInt();

            // 找最大值與最小值：使用三元運算子
            int largest = (a > b) ? ((a > c) ? a : c) : ((b > c) ? b : c);

            // 中位數：用三數和減去最大與最小得到中間值
