修改後完整程式碼
```java
/*
 * File: CSIE_41343216_1031_1.java
 * Description: 讀取一個整數 n，接著 n 個整數，將陣列內容升序排序後輸出，元素以空白分隔。
 * Author: 你的名字
 * Date: 2025-11-12
 */

import java.util.Scanner;
import java.util.Arrays;

public class CSIE_41343216_1031_1 {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);

        int n = input.nextInt();
        int[] arr = new int[n];

        for (int i = 0; i < n; i++) {
            arr[i] = input.nextInt();
        }

        Arrays.sort(arr);

        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }

        System.out.println(sb.toString());

        input.close();
    }
}
```
