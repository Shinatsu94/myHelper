import java.util.Scanner;

public class CSIE_41343219_1031_1 {
    // 檔案名稱: CSIE_41343219_1031_1.java
    // 功能: 讀入陣列長度與元素，對陣列內容進行升冪排序，並輸出排序後結果
    // 作者: (您的名字)
    // 日期: (日期)

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int n = sc.nextInt();
        int[] ary = new int[n];
        for (int i = 0; i < n; i++) {
            ary[i] = sc.nextInt();
        }

        // 選擇排序 (升冪)
        for (int i = 0; i < n - 1; i++) {
            int min = i;
            for (int j = i + 1; j < n; j++) {
                if (ary[j] < ary[min]) {
                    min = j;
                }
            }
            if (min != i) {
                int tmp = ary[i];
                ary[i] = ary[min];
                ary[min] = tmp;
            }
        }
        sc.close();

        // 輸出排序結果，元素之間以空格分隔，結尾不多印空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(ary[i]);
        }
        System.out.print(sb.toString());
    }
}
