import java.util.Scanner;

/*
 * 檔案名稱: CSIE_41245139_1003_3.java
 * 功能說明: 計算範圍 1..100000 之間，數字各位數字的加總等於輸入 A 的數字個數
 * 作者: [學生姓名]
 * 日期: 2025-11-15
 */

public class CSIE_41245139_1003_3 {
    public static void main(String[] args) {
        // 使用 try-with-resources 以自動關閉 Scanner
        try (Scanner sc = new Scanner(System.in)) {
            int A = sc.nextInt();
            int count = 0;

            // 範圍限定在 1..100000，避免將 0 包含進去
            for (int i = 1; i <= 100000; i++) {
                if (digitSum(i) == A) {
                    count++;
                }
            }

            System.out.println(count);
        }
    }

    // 計算數字 number 的各位數字總和
    private static int digitSum(int number) {
        int sum = 0;
        while (number > 0) {
            sum += number % 10;
            number /= 10;
        }
        return sum;
    }
}
