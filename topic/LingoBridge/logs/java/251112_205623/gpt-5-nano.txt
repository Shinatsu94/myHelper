修改後程式碼（SortArray.java）：
/*
檔案名稱: SortArray.java
功能: 讀取長度 n 與 n 個整數，將陣列內容由小到大排序並輸出。
作者: 學生
日期: 2025-11-12
版本: 1.0
*/
import java.util.Scanner;
import java.util.Arrays;

public class SortArray {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取長度 n
        int n = 0;
        if (sc.hasNextInt()) {
            n = sc.nextInt();
        } else {
            sc.close();
            return;
        }

        // 讀取 n 個整數
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                // 若輸入不足，預設為 0
                arr[i] = 0;
            }
        }
        sc.close();

        // 排序
        Arrays.sort(arr);

        // 輸出，避免尾端多餘空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());
    }
}
