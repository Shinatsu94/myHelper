修改後完整程式碼

/*
 * 檔案名稱: CSIE_41343109_1031_01.java
 * 功能描述: 讀入任意長度整數陣列，先輸入長度 n，接著輸入 n 個整數，輸出升冪排序後的陣列。
 * 作者: 吳育丞
 * 日期: 1031
 * 註解:
 * - 透過 Arrays.sort 進行排序，提升可讀性與穩定性。
 * - 輸出以單一空格分隔，結尾不多餘空格，並在需要時關閉 Scanner。
 */

import java.util.Scanner;
import java.util.Arrays;

public class CSIE_41343109_1031_01 {
    public static void main(String[] args) {
        Scanner scn = new Scanner(System.in);
        try {
            int n = scn.nextInt();
            int[] a = new int[n];
            for (int i = 0; i < n; i++) {
                a[i] = scn.nextInt();
            }

            // 使用內建排序提升可靠性與效能
            Arrays.sort(a);

            // 輸出：單一空格分隔，結尾不留空格
            StringBuilder sb = new StringBuilder();
            for (int i = 0; i < a.length; i++) {
                if (i > 0) sb.append(' ');
                sb.append(a[i]);
            }
            System.out.print(sb.toString());
        } finally {
            scn.close();
        }
    }
}
