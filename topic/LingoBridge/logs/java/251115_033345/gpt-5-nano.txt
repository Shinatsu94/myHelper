/*
 檔案名稱: LowerTriangularArray.java
 功能: 建立下三角形二維陣列，輸入高度 n 與要尋找的數字 num，輸出該數字在陣列中的位置，若找不到則輸出提示。
 作者: 教授
 日期: 2025-11-14
 版本: 1.0
 */

public class LowerTriangularArray {
    public static void main(String[] args) {
        java.util.Scanner sc = new java.util.Scanner(System.in);

        // 讀取高度 n，並做基本驗證
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();
        if (n <= 0) {
            sc.close();
            return;
        }

        // 建立下三角形二維陣列，並依序填入 1, 2, ...
        int[][] triangle = new int[n][];
        int count = 1;
        for (int i = 0; i < n; i++) {
            triangle[i] = new int[i + 1];
            for (int j = 0; j <= i; j++) {
                triangle[i][j] = count++;
            }
        }

        // 讀取要尋找的數字 num
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int num = sc.nextInt();

        // 在陣列中搜尋 num 的位置
        boolean found = false;
        int foundRow = -1;
        int foundCol = -1;

        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= i; j++) {
                if (triangle[i][j] == num) {
                    found = true;
                    foundRow = i;
                    foundCol = j;
                    break;
                }
            }
            if (found) {
                break;
            }
        }

        // 輸出結果
        if (found) {
            System.out.println(num + "[" + (foundRow + 1) + ", " + (foundCol + 1) + "]");
        } else {
            System.out.println("not find " + num + " in array.");
        }

        sc.close();
    }
}
