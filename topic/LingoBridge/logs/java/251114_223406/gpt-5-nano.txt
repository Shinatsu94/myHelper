/*
 * 檔案名稱: CSIE_41343227_1031_3b.java
 * 功能: 讀入整數長度 n，接著讀入 n 個整數，將陣列內容排序為升序並輸出
 * 作者: 教學助理
 * 日期: 2025-11-14
 */

import java.util.*;

public class CSIE_41343227_1031_3b {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取長度 n
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();
        if (n <= 0) {
            sc.close();
            return;
        }

        // 讀取 n 個整數，存入陣列
        int[] a = new int[n];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                a[i] = sc.nextInt();
            } else {
                // 若輸入不足，直接結束
                sc.close();
                return;
            }
        }
        sc.close();

        // 將陣列排序為升序
        Arrays.sort(a);

        // 輸出排序結果，元素以單一空格分隔
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(a[i]);
        }
        System.out.println(sb.toString());
    }
}
