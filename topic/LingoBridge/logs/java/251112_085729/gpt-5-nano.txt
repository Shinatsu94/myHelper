作者: 你的姓名

public class LeibnizPi {
    public static void main(String[] args) {
        double sum = 0.0;
        // 使用增量累加方式，避免每次都從頭計算
        for (int i = 1; i <= 100000; i++) {
            double term = ((i % 2 == 1) ? 1.0 : -1.0) / (2.0 * i - 1.0);
            sum += term;

            // 輸出要求：每到 10000 的倍數就輸出一次
            if (i % 10000 == 0) {
                double pi = sum * 4.0;
                System.out.printf("i = %6d, pi = %.10f%n", i, pi);
            }
        }
    }
}
