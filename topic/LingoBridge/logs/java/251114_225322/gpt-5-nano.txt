import java.util.Scanner;
import java.util.Arrays;

/**
 * 檔案名稱: SortArrayAscending.java
 * 功能: 讀取任意長度的一維整數陣列，將內容由小到大排序，輸出排序後的結果
 * 作者: 學生
 * 日期: 2025-11-14
 */
public class SortArrayAscending {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取陣列長度
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();
        int[] arr = new int[n];

        // 依序讀入 n 個整數
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                // 若輸入不足，預設為 0
                arr[i] = 0;
            }
        }
        sc.close();

        // 升序排序
        Arrays.sort(arr);

        // 輸出排序後的結果，使用單一空格分隔
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());
    }
}
