修改後完整程式碼：
/*
檔案名稱: CSIE_41343221_1031_3b.java
功能: 建立下三角形的二維陣列，填入數字 1..n(n+1)/2，找出指定數字的位置
作者: XXX
日期: 2025-11-14
*/
import java.util.Scanner;

public class CSIE_41343221_1031_3b {

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取高度 n
        int n;
        if (sc.hasNextInt()) {
            n = sc.nextInt();
        } else {
            sc.close();
            return;
        }

        // 讀取要尋找的數字 num
        int num;
        if (sc.hasNextInt()) {
            num = sc.nextInt();
        } else {
            sc.close();
            return;
        }

        sc.close();

        // 建立下三角形陣列，第 i 列長度為 i+1（1-based 說明對應 0-based 的 i+1） 
        int[][] tri = new int[n][];
        for (int i = 0; i < n; i++) {
            tri[i] = new int[i + 1];
        }

        // 依序填入 1, 2, ..., total
        int val = 1;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= i; j++) {
                tri[i][j] = val++;
            }
        }

        int total = n * (n + 1) / 2;
        if (num <= 0 || num > total) {
            System.out.println("not find " + num + " in array.");
            return;
        }

        // 找出 num 在陣列中的位置 (1-based)
        int foundRow = -1;
        int foundCol = -1;

        outer:
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= i; j++) {
                if (tri[i][j] == num) {
                    foundRow = i + 1;
                    foundCol = j + 1;
                    break outer;
                }
            }
        }

        if (foundRow != -1) {
            System.out.println(num + "[" + foundRow + ", " + foundCol + "]");
        } else {
            System.out.println("not find " + num + " in array.");
        }
    }
}
