/*
檔案名稱: SortArrayAscending.java
功能: 讀取整數長度 n 與後續 n 個整數，將其由小到大排序並以單行輸出
作者: 教授
日期: 2025-11-12
*/

import java.util.Scanner;
import java.util.Arrays;

public class SortArrayAscending {
    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);

        // 讀取陣列長度
        if (!in.hasNextInt()) {
            in.close();
            return;
        }
        int n = in.nextInt();

        // 讀取 n 個整數
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (in.hasNextInt()) {
                arr[i] = in.nextInt();
            } else {
                arr[i] = 0; // 若不足 n 個，預設為 0，避免例外
            }
        }

        // 排序
        Arrays.sort(arr);

        // 輸出：以空格分隔，避免尾端空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.print(sb.toString());

        in.close();
    }
}
