// 檔案名稱: SortArrayAsc.java
// 功能: 讀取整數長度與陣列內容，做升序排序，輸出排序後的陣列
// 作者: 由使用者提供
// 日期: 2025/11/14
import java.util.Scanner;
import java.util.Arrays;

public class SortArrayAsc {
    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);

        // 讀取陣列長度
        int n = 0;
        if (in.hasNextInt()) {
            n = in.nextInt();
        } else {
            in.close();
            return;
        }

        // 讀取陣列內容
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (in.hasNextInt()) {
                arr[i] = in.nextInt();
            } else {
                arr[i] = 0; // 若輸入不足，預設為 0，避免例外
            }
        }

        // 升序排序
        Arrays.sort(arr);

        // 輸出排序後的結果，空格分隔，結尾不附加額外換行
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.print(sb.toString());

        in.close();
    }
}
