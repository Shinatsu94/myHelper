```java
/*
 * File: Main.java
 * 功能: 讀取整數 n，再讀取 n 個整數，對陣列進行升序排序，輸出排序結果（以空格分隔）。
 * 作者: ChatGPT
 * 日期: 2025-11-14
 */

import java.util.Arrays;
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }

        int n = sc.nextInt();
        int[] arr = new int[n];

        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                // 若輸入不足，預設為 0，避免例外
                arr[i] = 0;
            }
        }

        sc.close();

        Arrays.sort(arr);

        // 輸出排序結果，元素之間以單一空格分隔，末尾不留空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());
    }
}
```
