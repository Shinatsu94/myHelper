import java.util.Scanner;

public class CSIE_41343150_1031_3a {
    public static void main(String[] args) {
        Scanner scn = new Scanner(System.in);
        int n = scn.nextInt();

        // 基本輸入驗證：若不符合範圍，程式直接結束，避免輸出非預期內容
        if (n <= 0 || n > 20) {
            scn.close();
            return;
        }

        // 直覺的下三角形列印：外層控制列數，內層控制每列的星星數
        for (int i = 1; i <= n; i++) {
            for (int j = 0; j < i; j++) {
                System.out.print('*');
            }
            System.out.println();
        }

        scn.close();
    }
}
