// 檔案名稱與功能說明：DigitSumCounter.java
// 功能：計算在區間 [1, 100000] 內，數字位數總和等於輸入值 target 的數字個數
// 作者：<作者姓名>
// 日期：2025-11-15

import java.util.Scanner;

public class DigitSumCounter {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int target = sc.nextInt(); // 需要的數字位數總和
        int count = 0;              // 符合條件的數字數量

        // 遍歷區間 [1, 100000]
        for (int num = 1; num <= 100000; num++) {
            int digitSum = 0;
            int x = num;
            // 計算該數字的各位數字總和
            while (x != 0) {
                digitSum += x % 10;
                x /= 10;
            }
            if (digitSum == target) {
                count++;
            }
        }

        System.out.println(count);
        sc.close();
    }
}
