/*
Filename: CSIE_51015108_1031_5.java
Description: 讀取使用者輸入的 3x3 矩陣，計算其反矩陣，並以小數點後四位輸出；若元素為零則輸出 0。
Author: 你的名字
Date: 2025-11-14
*/

import java.util.Locale;
import java.util.Scanner;

public class CSIE_51015108_1031_5 {
    private static final double EPS = 1e-9; // 判定浮點數近似於零的容忍度

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        double[][] mat = new double[3][3];

        // 讀入 3x3 矩陣
        for (int i = 0; i < 3; i++) {
            for (int j = 0; j < 3; j++) {
                if (sc.hasNextDouble()) {
                    mat[i][j] = sc.nextDouble();
                } else {
                    // 若輸入不足，預設為 0
                    mat[i][j] = 0.0;
                }
            }
        }
        sc.close();

        double det = determinant(mat);
        if (Math.abs(det) < EPS) {
            // 非反矩陣的處理，依原本程式風格給出訊息
            System.out.println("矩陣不可逆");
            return;
        }

        double[][] inv = inverse(mat, det);

        for (int i = 0; i < 3; i++) {
            for (int j = 0; j < 3; j++) {
                System.out.print(formatValue(inv[i][j]));
                if (j < 2) System.out.print(" ");
            }
            System.out.println();
        }
    }

    // 計算 3x3 矩陣的行列式
    public static double determinant(double[][] m) {
        return m[0][0]*(m[1][1]*m[2][2]-m[1][2]*m[2][1])
             - m[0][1]*(m[1][0]*m[2][2]-m[1][2]*m[2][0])
             + m[0][2]*(m[1][0]*m[2][1]-m[1][1]*m[2][0]);
    }

    // 計算反矩陣
    public static double[][] inverse(double[][] m, double det) {
        double[][] inv = new double[3][3];

        inv[0][0] =  (m[1][1]*m[2][2] - m[1][2]*m[2][1]) / det;
        inv[0][1] = -(m[0][1]*m[2][2] - m[0][2]*m[2][1]) / det;
        inv[0][2] =  (m[0][1]*m[1][2] - m[0][2]*m[1][1]) / det;

        inv[1][0] = -(m[1][0]*m[2][2] - m[1][2]*m[2][0]) / det;
        inv[1][1] =  (m[0][0]*m[2][2] - m[0][2]*m[2][0]) / det;
        inv[1][2] = -(m[0][0]*m[1][2] - m[0][2]*m[1][0]) / det;

        inv[2][0] =  (m[1][0]*m[2][1] - m[1][1]*m[2][0]) / det;
        inv[2][1] = -(m[0][0]*m[2][1] - m[0][1]*m[2][0]) / det;
        inv[2][2] =  (m[0][0]*m[1][1] - m[0][1]*m[1][0]) / det;

        return inv;
    }

    // 輸出格式化：近似於零的數輸出 0，其他輸出四位小數
    private static String formatValue(double v) {
        final double ZERO_EPS = 1e-8;
        if (Math.abs(v) < ZERO_EPS) return "0";
        return String.format(Locale.US, "%.4f", v);
    }
}
