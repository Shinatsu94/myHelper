/*
 * File: SwapTwoNumbers.java
 * Description: Reads two integers A and B, prints the swapped values using XOR swap without extra variables.
 * Author: 教授
 * Date: 2025-11-15
 */

import java.util.Scanner;

public class SwapTwoNumbers {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int A = sc.nextInt();
        int B = sc.nextInt();

        // 使用 XOR 交換，避免整數溢位且不使用額外變數
        A = A ^ B;
        B = A ^ B;
        A = A ^ B;

        System.out.println(A);
        System.out.println(B);

        sc.close();
    }
}
