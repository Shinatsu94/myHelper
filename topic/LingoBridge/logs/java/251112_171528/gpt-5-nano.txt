```java
/*
File: Main.java
Description: 讀入長度與元素，將元素升序排序，並以空格分隔輸出。
Author: 故筆者
Date: 2025-11-12
*/

import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);

        if (!in.hasNextInt()) {
            in.close();
            return;
        }

        int n = in.nextInt();
        if (n <= 0) {
            in.close();
            System.out.println();
            return;
        }

        int[] array = new int[n];
        for (int i = 0; i < n; i++) {
            array[i] = in.hasNextInt() ? in.nextInt() : 0;
        }

        // 選擇排序：由小到大
        for (int i = 0; i < n - 1; i++) {
            int minIdx = i;
            for (int j = i + 1; j < n; j++) {
                if (array[j] < array[minIdx]) {
                    minIdx = j;
                }
            }
            int tmp = array[minIdx];
            array[minIdx] = array[i];
            array[i] = tmp;
        }

        // 輸出結果，元素以空格分隔
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(array[i]);
        }
        System.out.println(sb.toString());

        in.close();
    }
}
```
