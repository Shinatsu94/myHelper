```java
/**
 * File: CSIE_51015120_0926_3.java
 * 功能: 印出「虎科大」以及使用者輸入的中文姓名每個字的內碼(Unicode 十進位)
 * 作者: 教師
 * 日期: 2025-11-12
 */

import java.util.Scanner;

public class CSIE_51015120_0926_3 {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);

        // 讀取使用者輸入的中文姓名；若無輸入，name 會為空字串
        String name = "";
        if (input.hasNextLine()) {
            name = input.nextLine().trim();
        }

        // 依序印出「虎科大」三個字及對應的內碼（Unicode 十進位）
        printWithCode("虎");
        printWithCode("科");
        printWithCode("大");

        // 逐字印出使用者姓名的每一個字及對應的內碼
        for (int i = 0; i < name.length(); i++) {
            char ch = name.charAt(i);
            if (Character.isWhitespace(ch)) continue; // 避免空白字元影響
            printWithCode(Character.toString(ch));
        }

        input.close();
    }

    // 輔助方法：輸出單個字的內容與其 Unicode 內碼（十進位）
    private static void printWithCode(String s) {
        int codePoint = s.codePointAt(0);
        System.out.println(s + " " + codePoint);
    }
}
```
