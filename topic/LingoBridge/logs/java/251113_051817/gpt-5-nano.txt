import java.util.*;

public class CSIE_41343240_0926_3 {
    public static void main(String[] args) {
        // 直接輸出「虎科大」及學生中文姓名每個字的對應編碼
        String[] uni = {"虎","科","大"};
        String[] name = {"林","武","杰"};

        // 印出虎科大每個字及其對應內碼
        for (String ch : uni) {
            int code = ch.codePointAt(0);
            System.out.println(ch + " " + code);
        }

        // 印出學生姓名每個字及其對應內碼
        for (String ch : name) {
            int code = ch.codePointAt(0);
            System.out.println(ch + " " + code);
        }
    }
}
