/*
檔案名稱: CSIE_51015130_1031_1.java
功能: 建立任意長度的一維整數陣列，讀入元素後升冪排序並輸出
作者: 教授
日期: 2025-11-14
說明: 不使用 EOF，只接受先輸入長度再輸入元素；支援負數；輸出以空格分隔，末尾不多餘空格
*/

import java.util.Scanner;
import java.util.Arrays;

public class CSIE_51015130_1031_1 {
    public static void main(String[] args) {
        // 讀取使用者輸入的長度
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();

        int[] nums = new int[n];
        // 讀入元素
        for (int i = 0; i < n; i++) {
            nums[i] = sc.nextInt();
        }

        // 進行升冪排序
        Arrays.sort(nums);

        // 輸出結果，避免尾端多餘的空格
        if (n > 0) {
            System.out.print(nums[0]);
            for (int i = 1; i < n; i++) {
                System.out.print(" " + nums[i]);
            }
        }
        System.out.println(); // 換行結束輸出
        sc.close();
    }
}
