/*
File: SortingArray.java
Description: 讀取長度 n，接著讀入 n 個整數，對它們由小到大排序，輸出排序後的內容，元素可為負數。
Author: 你的姓名
Date: 2025-11-12
*/

import java.util.Scanner;
import java.util.Arrays;

public class SortingArray {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取陣列長度
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();

        // 讀入 n 個整數
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                arr[i] = 0; // 若輸入不足，預設為 0
            }
        }

        // 由小到大排序
        Arrays.sort(arr);

        // 輸出排序結果，數字之間以空格分隔，末尾不多餘空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        sc.close();
    }
}
