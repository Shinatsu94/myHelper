【題目】
請撰寫一個 Java 程式，可以讀入使用者輸入的任意 3x3 矩陣，
並且列印出它的反矩陣，顯示到小數點後四位。
如果元素為零，請直接列印 '0'，無需顯示到小數點後四位

說明:
- 各元素使用空白符號(' ')進行換行，每三個進行換行('\n')
- 最後一行不得有空白符號


【input】
1 2 3
4 5 6
7 8 19

【output】
-1.5667 0.4667 0.1000
1.1333 0.0667 -0.2000
0.1000 -0.2000 0.1000

【input】
5 2 7
10 4 6
9 1 3

【output】
-0.0577 -0.0096 0.1538
-0.2308 0.4615 -0.3846
0.2500 -0.1250 0


學生作答程式：
import java.util.Scanner;
public class MatrixInverse_3x3_Format {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        double[][] A = new double[3][3];
        for (int i = 0; i < 3; i++) {
            for (int j = 0; j < 3; j++) {
                A[i][j] = input.nextDouble();
            }
        }
        double det = A[0][0] * (A[1][1] * A[2][2] - A[1][2] * A[2][1])
                   - A[0][1] * (A[1][0] * A[2][2] - A[1][2] * A[2][0])
                   + A[0][2] * (A[1][0] * A[2][1] - A[1][1] * A[2][0]);

        if (det == 0) {
            System.out.println("This matrix has no inverse (det=0)");
            return;
        }
        double[][] adj = new double[3][3];
        adj[0][0] =  (A[1][1]*A[2][2] - A[1][2]*A[2][1]);
        adj[0][1] = -(A[1][0]*A[2][2] - A[1][2]*A[2][0]);
        adj[0][2] =  (A[1][0]*A[2][1] - A[1][1]*A[2][0]);

        adj[1][0] = -(A[0][1]*A[2][2] - A[0][2]*A[2][1]);
        adj[1][1] =  (A[0][0]*A[2][2] - A[0][2]*A[2][0]);
        adj[1][2] = -(A[0][0]*A[2][1] - A[0][1]*A[2][0]);

        adj[2][0] =  (A[0][1]*A[1][2] - A[0][2]*A[1][1]);
        adj[2][1] = -(A[0][0]*A[1][2] - A[0][2]*A[1][0]);
        adj[2][2] =  (A[0][0]*A[1][1] - A[0][1]*A[1][0]);
        double[][] inv = new double[3][3];
        for (int i = 0; i < 3; i++) {
            for (int j = 0; j < 3; j++) {
                inv[i][j] = adj[i][j] / det;
            }
        }
        for (int i = 0; i < 3; i++) {
            for (int j = 0; j < 3; j++) {
                double val = inv[i][j];
                if (Math.abs(val) < 1e-6) {
                    System.out.print("0");
                } else {
                    System.out.printf("%.4f", val);
                }
                if (j < 2) System.out.print(" ");
            }
            if (i < 2) System.out.println();
        }
    }
}
