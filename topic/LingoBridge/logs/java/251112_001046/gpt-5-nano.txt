public class CSIE_41343124_1003_1 {
    public static void main(String[] args) {
        Scanner s = new Scanner(System.in);
        int a = s.nextInt();
        int b = s.nextInt();

        // 使用位元異或交換法，避免加減法可能造成的整數溢位
