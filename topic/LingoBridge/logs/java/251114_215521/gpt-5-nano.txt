```java
/**
 * File: CSIE_41343128_1031_3a.java
 * 功能: 讀取長度 n，接著讀入 n 個整數，將其升冪排序並輸出為一行以空格分隔。
 * 作者: 學生
 * 日期: 2025-11-14
 */
import java.util.Arrays;
import java.util.Scanner;

public class CSIE_41343128_1031_3a {
    public static void main(String[] args) {
        Scanner scn = new Scanner(System.in);
        if (!scn.hasNextInt()) {
            scn.close();
            return;
        }
        int n = scn.nextInt();
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (scn.hasNextInt()) {
                arr[i] = scn.nextInt();
            } else {
                arr[i] = 0;
            }
        }
        scn.close();

        Arrays.sort(arr);

        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < arr.length; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.print(sb.toString());
    }
}
```
