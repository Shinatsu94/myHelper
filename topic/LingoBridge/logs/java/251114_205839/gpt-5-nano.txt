/*
File: CSIE_41343150_1031_1.java
Description: 讀取整數個數與元素，對一維整數陣列進行升序排序，並輸出排序結果。
Author: 教授
Date: 2025-11-14

import java.util.Scanner;
import java.util.Arrays;

public class CSIE_41343150_1031_1 {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int n = scanner.nextInt();
        int[] arr = new int[n];

        for (int i = 0; i < n; i++) {
            arr[i] = scanner.nextInt();
        }

        // 使用內建排序提升穩定性與效率
        Arrays.sort(arr);

        // 輸出排序結果，元素之間以空格分隔，結尾換行
        for (int i = 0; i < n; i++) {
            System.out.print(arr[i]);
            if (i < n - 1) {
                System.out.print(" ");
            }
        }
        System.out.println();

        scanner.close();
    }
}
