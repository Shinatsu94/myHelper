修改後的完整程式碼：
/*
 * File: ReverseDigits.java
 * Description: 讀取一個整數，輸出其數字反轉後的值，保留符號，忽略前導零。
 * Author: 改進版本
 * Date: 2025-11-15
 */

import java.util.Scanner;

public class ReverseDigits {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        long n = scanner.nextLong();
        // 不關閉 System.in 以提高穩定性
        long sign = n < 0 ? -1 : 1;
        long abs = Math.abs(n);

        long reversedAbs = reverseDigits(abs);
        long result = sign * reversedAbs;
        System.out.println(result);
        // scanner.close(); // 避免關閉 System.in
    }

    // 將非負整數的數字反轉
    public static long reverseDigits(long x) {
        long r = 0;
        while (x != 0) {
            r = r * 10 + (x % 10);
            x /= 10;
        }
        return r;
    }
}
