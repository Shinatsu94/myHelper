/*
 * 檔案名稱: WellOrderedNumber.java
 * 功能說明: 判斷輸入的整數各位數字是否嚴格遞增
 * 作者: 教師
 * 日期: 2025-11-15
 */
public class WellOrderedNumber {
    // 將輸入字串轉為數字位元的嚴格遞增判斷；允許前導零與可選的負號
    private static boolean isWellOrdered(String s) {
        if (s == null) return false;

        // 移除前導負號，若是負數視為非遞增的數字
        if (s.startsWith("-")) return false;
        // 去除可能的前後空白
        s = s.trim();
        if (s.isEmpty()) return false;

        // 檢查是否全為數字
        for (int i = 0; i < s.length(); i++) {
            if (!Character.isDigit(s.charAt(i))) return false;
        }

        // 空字串或只有一位數，視為遞增
        if (s.length() <= 1) return true;

        // 檢查相鄰兩位是否嚴格遞增
        for (int i = 0; i < s.length() - 1; i++) {
            char a = s.charAt(i);
            char b = s.charAt(i + 1);
            if (a >= b) return false;
        }
        return true;
    }

    public static void main(String[] args) {
        java.util.Scanner scanner = new java.util.Scanner(System.in);
        String input = scanner.next();
        scanner.close();

        boolean ok = isWellOrdered(input);
        if (ok) {
            System.out.println(input + " is well-ordered number.");
        } else {
            System.out.println(input + " is not well-ordered number.");
        }
    }
}
