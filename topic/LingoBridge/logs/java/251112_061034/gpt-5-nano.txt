```java
/*
 * File: PiLeibniz.java
 * 功能: 使用 Leibniz 公式計算 π，輸出 i = 10000, 20000, ..., 100000 之 pi 值，並找出使 pi 四捨五入為 3.14159 的最小 i。
 * 作者: 教師
 * 日期: 2025-11-11
 * 內容說明: 以逐項累加的方式計算 π 的近似值，格式化輸出符合題目要求。
 */

public class PiLeibniz {
    public static void main(String[] args) {
        int maxI = 100000;
        double sum = 0.0;
        int sign = 1; // 第 i 項的符號，第一項為正

        // 先計算並輸出 i = 10000, 20000, ..., 100000 的結果
        StringBuilder output = new StringBuilder();
        for (int i = 1; i <= maxI; i++) {
            sum += sign * (1.0 / (2 * i - 1));
            sign = -sign;
            if (i % 10000 == 0) {
                double pi = 4.0 * sum;
                output.append(String.format("i = %6d, pi = %.10f%n", i, pi));
            }
        }

        // 尋找最小的 i，使 pi 四捨五入為 3.14159
        int iMinForPiRound = -1;
        // 繼續從 maxI + 1 開始累加，直到找到符合條件的 i
        for (int i = maxI + 1; i <= 2000000; i++) {
            sum += sign * (1.0 / (2 * i - 1));
            sign = -sign;
            double pi = 4.0 * sum;
            double rounded = Math.round(pi * 100000.0) / 100000.0;
            if (rounded == 3.14159) {
                iMinForPiRound = i;
                break;
            }
        }

        output.append(String.format("i = %d%n", iMinForPiRound));

        System.out.print(output.toString());
    }
}
```
