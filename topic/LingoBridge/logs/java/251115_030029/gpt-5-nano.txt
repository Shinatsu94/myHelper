/*
檔案名稱: CSIE_51015120_1031_3a_opt.java
功能: 輸入高度 n (0 < n <= 20)，輸出下三角形的 '*' 矩陣形式，且每行不含空格
作者: 你的名字
日期: 2025-11-14
說明: 不使用多餘陣列，直接產生每行的星號字串，提高效率並易於閱讀。
*/

import java.util.Scanner;

public class CSIE_51015120_1031_3a_opt {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n;
        // 僅接受 1..20 的整數
        do {
            n = sc.nextInt();
        } while (n <= 0 || n > 20);

        StringBuilder output = new StringBuilder();
        for (int i = 1; i <= n; i++) {
            // 在第 i 行印出 i 個 '*'
            for (int j = 0; j < i; j++) {
                output.append('*');
            }
            // 除了最後一行之外，加入換行
            if (i < n) {
                output.append('\n');
            }
        }

        System.out.print(output.toString());
        sc.close();
    }
}
