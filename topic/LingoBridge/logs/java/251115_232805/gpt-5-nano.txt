import java.util.Scanner;

/*
檔案名稱: CSIE_41343109_1003_02.java
功能說明: 讀入三個整數，使用兩種不同的方法計算最大值、最小值與中間值，並輸出順序為 min、mid、max。
作者: 請填入作者名稱
日期: 2025-11-15
*/

public class CSIE_41343109_1003_02 {
    public static void main(String[] args) {
        // 使用 try-with-resources 確保 Scanner 會被自動關閉
        try (Scanner scn = new Scanner(System.in)) {
            int a = scn.nextInt();
            int b = scn.nextInt();
            int c = scn.nextInt();

            // 最大值與最小值使用三元運算子計算，符合題意不可使用 if-else
            int max = (a > b) ? ((a > c) ? a : c) : ((b > c) ? b : c);
            int min = (a < b) ? ((a < c) ? a : c) : ((b < c) ? b : c);

            // 中間值透過 sum 減去最大與最小值得到
            int mid = a + b + c - max - min;

            // 輸出順序：最小值、中間值、最大值
            System.out.println(min);
            System.out.println(mid);
            System.out.println(max);
        }
    }
}
