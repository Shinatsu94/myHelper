```java
/**
 * File: PrintSortedArray.java
 * 功能: 讀取整數長度與整數陣列，輸出排序後的一維整數陣列（由小到大）。
 * 作者: your_name
 * 日期: 2025-11-14
 * 註解: 使用 Arrays.sort 進行排序，支援負數，避免使用 EOF。
 */
import java.util.Arrays;
import java.util.Scanner;

public class PrintSortedArray {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取陣列長度
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();

        if (n <= 0) {
            // 無效長度，輸出空行並結束
            System.out.println();
            sc.close();
            return;
        }

        // 讀取 n 個整數
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                // 若輸入不足，預設為 0
                arr[i] = 0;
            }
        }

        // 排序
        Arrays.sort(arr);

        // 輸出，元素以單一空格分隔
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        sc.close();
    }
}
```
