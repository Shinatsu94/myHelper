/*
 檔案名稱: Main.java
 功能說明: 讀入一維整數陣列的長度與元素，將陣列由小到大排序，輸出結果
 作者: 教師
 日期: 2025-11-12
 備註: 避免依賴 EOF 的輸入，支援負整數，輸出以空格分隔，結尾不多餘空格
*/
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);

        // 讀取長度
        if (!in.hasNextInt()) {
            in.close();
            return;
        }
        int n = in.nextInt();
        if (n <= 0) {
            in.close();
            return;
        }

        int[] a = new int[n];
        for (int i = 0; i < n; i++) {
            if (in.hasNextInt()) {
                a[i] = in.nextInt();
            } else {
                a[i] = 0; // 不足的輸入以 0 處理，避免例外
            }
        }
        in.close();

        // 冒泡排序，加入早停機制以提升效能
        for (int i = 0; i < n - 1; i++) {
            boolean swapped = false;
            for (int j = 0; j < n - 1 - i; j++) {
                if (a[j] > a[j + 1]) {
                    int tmp = a[j];
                    a[j] = a[j + 1];
                    a[j + 1] = tmp;
                    swapped = true;
                }
            }
            if (!swapped) break;
        }

        // 使用 StringBuilder 組裝輸出，避免多次 I/O
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(a[i]);
        }
        System.out.print(sb.toString());
    }
}
