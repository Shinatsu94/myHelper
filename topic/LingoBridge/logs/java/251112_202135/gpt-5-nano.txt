// 程式檔案名稱: Main.java
// 功能: 讀取整數長度 n，接著讀入 n 個整數，將它們升冪排序，最後輸出排序結果。
// 作者: 你
// 日期: 2025-11-12
// 說明: 使用整數陣列與 Arrays.sort 以符合題目需求，並提供適當註解。

import java.util.Arrays;
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取陣列長度
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();

        // 讀取 n 個整數並存入陣列
        int[] arr = new int[Math.max(0, n)];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                // 若輸入不足，預設為 0
                arr[i] = 0;
            }
        }

        // 升冪排序
        Arrays.sort(arr);

        // 輸出排序後的結果，元素以空格分隔
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < arr.length; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        sc.close();
    }
}
