/*
File: CSIE_41343212_1031_2.java
Description: 建立任意長度的一維陣列，將每個元素以其右側元素中的最大值取代，最後輸出結果；最後一個元素為-1。
Author: 使用者
Date: 2025-11-14
*/

import java.util.Scanner;

public class CSIE_41343212_1031_2 {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);

        if (!input.hasNextInt()) {
            // 未輸入長度，直接結束
            System.out.println();
            input.close();
            return;
        }

        int length = input.nextInt();
        if (length <= 0) {
            // 長度為0或負數，輸出空行並結束
            System.out.println();
            input.close();
            return;
        }

        int[] arr = new int[length];
        for (int i = 0; i < length; i++) {
            if (input.hasNextInt()) {
                arr[i] = input.nextInt();
            } else {
                // 若輸入不足，以預設值 0 代替
                arr[i] = 0;
            }
        }

        int max = arr[length - 1];
        arr[length - 1] = -1;
        for (int i = length - 2; i >= 0; i--) {
            int tmp = arr[i];
            arr[i] = max;
            if (tmp > max) {
                max = tmp;
            }
        }

        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < length; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        input.close();
    }
}
