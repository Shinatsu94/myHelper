import java.util.*;

public class CSIE_41343236_1031_3a_sorted {

    /**
     * File name: CSIE_41343236_1031_3a_sorted.java
     * 功能: 依題意接收一個長度 n 與 n 個整數，將陣列由小到大排序後輸出。
     * 作者: 你的姓名
     * 日期: 2025-11-14
     * 備註: 不使用 EOF，會在一次執行結束；支援負數。
     */

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        // 讀取長度 n
        if (!scanner.hasNextInt()) {
            scanner.close();
            return;
        }
        int n = scanner.nextInt();

        if (n <= 0) {
            scanner.close();
            return;
        }

        // 讀取 n 個整數
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (scanner.hasNextInt()) {
                arr[i] = scanner.nextInt();
            } else {
                arr[i] = 0;
            }
        }

        // 排序
        Arrays.sort(arr);

        // 輸出：一行，以空白分隔
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.print(sb.toString());

        scanner.close();
    }
}
