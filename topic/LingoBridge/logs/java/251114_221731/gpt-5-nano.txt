Description: 依題意讀取長度 n 與 n 個整數，將陣列內容由小到大排序後輸出。

Date: 2025-11-14

import java.util.Arrays;
import java.util.Scanner;

public class SortArrayAscending {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取長度 n
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();

        int[] arr = new int[n];

        // 讀取 n 個整數，支援負數
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                // 若輸入不足，設為 0 以避免例外；實務上可自行決定錯誤處理策略
                arr[i] = 0;
            }
        }

        sc.close();

        // 從小到大排序
        Arrays.sort(arr);

        // 輸出排序結果，以空格分隔
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < arr.length; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.print(sb.toString());
    }
}
