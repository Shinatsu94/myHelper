import java.util.Scanner;

/*
 File: CSIE_51015108_1003_2.java
 功能: 讀入兩個整數，交換兩個數字的值，並輸出交換後的結果。
 作者: 請自行填寫
 日期: 請自行填寫
 說明: 不使用任何額外變數進行數字交換，採用位元異或交換法。
*/

public class CSIE_51015108_1003_2 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int a = sc.nextInt();
        int b = sc.nextInt();

        // 使用 XOR 交換，無需第三個變數
        a = a ^ b;
        b = a ^ b;
        a = a ^ b;

        // 依題意輸出交換後的結果
        System.out.println(a);
        System.out.println(b);

        sc.close();
    }
}
