public class SwapTwoNumbers {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀入兩個整數，分別存放於 A 與 B
        int A = sc.nextInt();
        int B = sc.nextInt();

        System.out.println(A);

        sc.close();
    }
}
