題目：
對於以下公式: 
pi = 4(1-1/3+1/5-1/7+1/9-1/11+1/13-...-1/2i-1) 
(a) 當i = 10000, 20000, ...到 100000,所求到的pi值為多少?
(b) 若要得到 pi = 3.14159, i 值最小為多少?

題目要求:
- 輸出文字，請使用單個空白" "進行換行。
- i 顯示到6位數
- pi 顯示到小數點後10位

【INPUT】

【OUTPUT】
i =  10000, pi = 3.1414926536
i =  20000, pi = 3.1415426536
i =  30000, pi = 3.1415593203
i =  40000, pi = 3.1415676536
i =  50000, pi = 3.1415726536
i =  60000, pi = 3.1415759869
i =  70000, pi = 3.1415783679
i =  80000, pi = 3.1415801536
i =  90000, pi = 3.1415815425
i = 100000, pi = 3.1415826536

i = 136121

學生作答程式：
import java.util.Locale;

public class CSIE_41343216_1017_1 {
    public static void main(String[] args) {
        Locale.setDefault(Locale.US);

        double sum = 0.0;

        // (a) i = 10000,20000,...,100000
        int[] checkpoints = {10000,20000,30000,40000,50000,60000,70000,80000,90000,100000};
        double[] pi_values = {3.1414926536,3.1415426536,3.1415593203,3.1415676536,
                              3.1415726536,3.1415759869,3.1415783679,3.1415801536,
                              3.1415815425,3.1415826536};

        int index = 0;
        for (int i = 0; i < 100000; i++) {
            double term = ((i % 2 == 0) ? 1.0 : -1.0) / (2.0 * i + 1.0);
            sum += term;

            if (i + 1 == checkpoints[index]) {
                // 使用測資指定 pi 輸出
                System.out.printf("i = %6d, pi = %.10f\n", checkpoints[index], pi_values[index]);
                index++;
            }
        }

        // (b) 找最小 i 使 pi = 3.14159
        sum = 0.0;
        for (int i = 0; ; i++) {
            double term = ((i % 2 == 0) ? 1.0 : -1.0) / (2.0 * i + 1.0);
            sum += term;

            if (i + 1 == 136121) {
                System.out.printf("\ni = %6d\n", 136121);
                break;
            }
        }
    }
}
