// 文件名稱：LeibnizPiCalculator.java
// 功能：使用勒布尼茲公式近似圓周率，輸出指定 i 值下的 pi，以及找到最小使 pi 四捨五入到小數點後第 5 位等於 3.14159 的 i。
// 作者：AI 助手
// 日期：2025-11-12

public class LeibnizPiCalculator {
    public static void main(String[] args) {
        // 1) 輸出 i = 10000, 20000, ..., 100000 對應 pi 值，pi 保留小數點後 10 位
        double sum = 0.0;
        for (int k = 1; k <= 100000; k++) {
            if ((k & 1) == 1) { // 奇數項，符號為 +
                sum += 1.0 / (2 * k - 1);
            } else { // 偶數項，符號為 -
                sum -= 1.0 / (2 * k - 1);
            }
            if (k % 10000 == 0) {
                double pi = 4.0 * sum;
                System.out.printf("i = %6d, pi = %.10f%n", k, pi);
            }
        }

        // 2) 找出最小的 i，使 pi 近似到 5 位小數等於 3.14159
        // 將 pi 四捨五入到小數點後第 5 位，判斷是否等於 3.14159
        sum = 0.0;
        int iMin = -1;
        for (int k = 1; k <= 2000000; k++) {
            if ((k & 1) == 1) sum += 1.0 / (2 * k - 1);
            else sum -= 1.0 / (2 * k - 1);
            double pi = 4.0 * sum;
            long rounded = Math.round(pi * 100000.0);
            if (rounded == 314159) {
                iMin = k;
                break;
            }
        }
        System.out.println("i = " + iMin);
    }
}

- 新增最小 i 的尋找機制，透過四捨五入到五位小數後判斷是否等於 3.14159，並輸出該 i。
