import java.util.Scanner;

public class CSIE_41343208_1031_1 {
    // 檔案名稱: CSIE_41343208_1031_1.java
    // 功能: 讀入一個整數 n 與 n 個整數，將其內容升序排序，並輸出排序結果
    // 作者: [作者名稱]
    // 日期: 2025-11-12

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取陣列長度
        int n = 0;
        if (sc.hasNextInt()) {
            n = sc.nextInt();
        }

        int[] arr = new int[Math.max(0, n)];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            }
        }

        // 若長度為 0，直接輸出換行
        if (arr.length == 0) {
            System.out.println();
            sc.close();
            return;
        }

        // 選擇排序（Selection Sort）
        for (int i = 0; i < arr.length - 1; i++) {
            int minIndex = i;
            for (int j = i + 1; j < arr.length; j++) {
                if (arr[j] < arr[minIndex]) {
                    minIndex = j;
                }
            }
            int temp = arr[i];
            arr[i] = arr[minIndex];
            arr[minIndex] = temp;
        }

        // 輸出排序結果，避免尾端多餘空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < arr.length; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        sc.close();
    }
}
