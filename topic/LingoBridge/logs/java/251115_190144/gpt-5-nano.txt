優化說明與修改內容

- 支援負數輸入，保留符號；並正確處理輸入為 0 的情況。
- 使用長整數型別 (long) 以增加數值範圍，減少溢位風險。
- 加入完整的程式標頭註解及適當的行內註解，符合格式與備註要求。

修改後完整程式碼
/*
File: ReverseDigits.java
Description: 反轉整數的位數，去除反轉後開頭的多餘零；若原數為負數，保留符號。
Author: 教師
Date: 2025-11-15
*/

import java.util.Scanner;

public class ReverseDigits {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        // 讀取單一整數
        long n = scanner.nextLong();

        // 記錄符號並取絕對值以利反轉
        boolean negative = n < 0;
        if (negative) n = -n;

        // 數值反轉：逐位加到 rev，避免字串處理與前導 0 的問題
        long rev = 0;
        while (n > 0) {
            rev = rev * 10 + (n % 10);
            n /= 10;
        }

        // 若原數為負，恢復符號
        if (negative) rev = -rev;

        // 輸出結果
        System.out.println(rev);

        scanner.close();
    }
}
