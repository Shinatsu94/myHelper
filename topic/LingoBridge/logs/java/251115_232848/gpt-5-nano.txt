/**
 * 檔案名稱: CSIE_41343111_1003_2.java
 * 功能: 讀取三個整數，找出最大值、中間值、最小值，並輸出最小值、中間值、最大值（不使用 if-else，使用兩種方法）
 * 作者: 作者姓名
 * 日期: 2025-11-15
 * 說明: 使用三元運算子計算最大值與最小值；中間值透過 sum - max - min 計算。
 */

import java.util.Scanner;

public class CSIE_41343111_1003_2 {
    public static void main(String[] args) {
        Scanner s = new Scanner(System.in);

        // 讀入三個整數
        int n1 = s.nextInt();
        int n2 = s.nextInt();
        int n3 = s.nextInt();

        // 使用三元運算子找最大值與最小值
        int max = n1 > n2 && n1 > n3 ? n1 : (n2 > n3 ? n2 : n3);
        int min = n1 < n2 && n1 < n3 ? n1 : (n2 < n3 ? n2 : n3);

        // 中間值透過總和減去最大值與最小值計算
        long sum = (long) n1 + n2 + n3;
        int mid = (int) (sum - max - min);

        // 輸出順序與題目範例一致：最小值、中間值、最大值
        System.out.println(min);
        System.out.println(mid);
        System.out.println(max);

        // 關閉 Scanner，釋放資源
        s.close();
    }
}
