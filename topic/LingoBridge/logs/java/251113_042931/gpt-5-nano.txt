// function: 輸出「虎科大」與學生姓名的每個字及其 Unicode 內碼
public class CSIE_41343240_0926_2 {
    public static void main(String[] args) {
        // 先印虎科大，依序輸出每個字與對應的內碼
        printCharsWithCode("虎科大");
        // 再印學生姓名（假設為林武杰），依序輸出每個字與對應的內碼
        printCharsWithCode("林武杰");
    }

    // 將字串中的每個字元逐一輸出其 Unicode 碼點
    private static void printCharsWithCode(String s) {
        for (int i = 0; i < s.length(); ) {
            int codePoint = s.codePointAt(i);                 // 取得 Unicode 碼點
            String ch = new String(Character.toChars(codePoint)); // 將碼點轉回字元
            System.out.println(ch + " " + codePoint);          // 印出字元與內碼
            i += Character.charCount(codePoint);               // 移至下一字元
        }
    }
}
