import java.util.Scanner;

/*
 * File: csie_41343150_1017_3.java
 * 功能: 將使用者輸入的整數反轉其數字順序，支援正數、負數與零。
 * 作者: 教授
 * 日期: 2025-11-15
 * 說明: 程式僅執行一次，輸出僅為反轉後的整數，避免額外字串輸出以符合題目要求。
 */

public class csie_41343150_1017_3{
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        // 讀取長整數，避免溢位問題
        if (!scanner.hasNextLong()) {
            // 若輸入非數字，直接結束
            scanner.close();
            return;
        }

        long num = scanner.nextLong();
        boolean negative = num < 0;
        long n = Math.abs(num);

        long reversed = 0;
        while (n != 0) {
            long digit = n % 10;
            reversed = reversed * 10 + digit;
            n /= 10;
        }

        if (negative) reversed = -reversed;

        System.out.println(reversed);
        scanner.close();
    }
}
