/**
 * 檔案名稱: Main.java
 * 功能: 讀取長度 n 與 n 個整數，輸出升序排列的結果，元素以空格分隔。
 * 作者: 教師
 * 日期: 2025-11-14
 */

import java.util.*;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取陣列長度
        int n = sc.nextInt();

        // 儲存 n 個整數
        int[] arr = new int[n];

        // 讀取 n 個整數
        for (int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();
        }

        // 將陣列排序（由小到大）
        Arrays.sort(arr);

        // 輸出結果，元素以空格分隔
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());
    }
}
