修改後完整程式碼（NameCodePrinter.java）
/*
 * File: NameCodePrinter.java
 * Description: Print虎科大及林武杰等中文姓名及其Unicode內碼，每行輸出一字與其碼點，先印虎科大再印學生姓名。
 * Author: 教師
 * Date: 2025-11-12
 */

public class NameCodePrinter {
    public static void main(String[] args) {
        // 先印虎科大三字
        char[] firstGroup = {'虎', '科', '大'};
        // 再印學生姓名三字
        char[] secondGroup = {'林', '武', '杰'};

        // 印出第一組字與其Unicode碼點
        for (char ch : firstGroup) {
            int codePoint = (int) ch;
            System.out.println(ch + " " + codePoint);
        }

        // 印出第二組字與其Unicode碼點
        for (char ch : secondGroup) {
            int codePoint = (int) ch;
            System.out.println(ch + " " + codePoint);
        }
    }
}
