/*
檔案名稱: SortArrayAsc.java
功能: 讀取長度 n 與 n 個整數，將其排序為升冪並輸出，支援負數與任意長度
作者: 教授
日期: 2025-11-14
*/
import java.util.*;

public class SortArrayAsc {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        // 讀取陣列長度
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();
        int[] arr = new int[n];

        // 讀取 n 個整數
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                // 若輸入不足，預設為 0
                arr[i] = 0;
            }
        }
        sc.close();

        // 進行升冪排序
        Arrays.sort(arr);

        // 輸出排序後的陣列，元素以空白分隔，末尾不留空格，並換行
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());
    }
}
