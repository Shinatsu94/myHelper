/************************************************************
 * 檔案名稱：CSIE_41343209_1003_2.java
 * 功能說明：讀入三個整數，使用兩種不同的方式找出最大值、最小值與中間值，並分別輸出
 * 作者：教師
 * 日期：2025-11-15
 * 參考：使用三元運算子計算最大值與最小值，中間值以 sum - max - min 計算
 ************************************************************/
import java.util.Scanner;

public class CSIE_41343209_1003_2 {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        int num1 = scanner.nextInt();
        int num2 = scanner.nextInt();
        int num3 = scanner.nextInt();

        // 使用三元運算子求最大值
        int maxVal = (num1 >= num2) ? ((num1 >= num3) ? num1 : num3)
                                   : ((num2 >= num3) ? num2 : num3);

        // 使用三元運算子求最小值
        int minVal = (num1 <= num2) ? ((num1 <= num3) ? num1 : num3)
                                   : ((num2 <= num3) ? num2 : num3);

        // 中間值等於總和減去最大值與最小值
        int midVal = num1 + num2 + num3 - maxVal - minVal;

        // 輸出順序為：最小值、中間值、最大值
        System.out.println(minVal);
        System.out.println(midVal);
        System.out.println(maxVal);
    }
}
