/**
 * File: CSIE_41343150_1031_3b.java
 * 功能: 讀取正整數 n 及 n 個整數，將陣列內容由小到大排序輸出
 * 作者: 你的名字
 * 日期: 2025-11-14
 * 備註: 會處理負數，輸出為以空格分隔的排序結果
 */

import java.util.Scanner;
import java.util.Arrays;

public class CSIE_41343150_1031_3b {
    public static void main(String[] args) {
        Scanner scn = new Scanner(System.in);

        int n = scn.nextInt();
        int[] arr = new int[n];

        for (int i = 0; i < n; i++) {
            arr[i] = scn.nextInt();
        }

        Arrays.sort(arr);

        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }

        System.out.println(sb.toString());

        scn.close();
    }
}
