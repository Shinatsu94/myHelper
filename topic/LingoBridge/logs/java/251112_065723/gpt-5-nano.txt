/*
 文件名稱: LeibnizPi.java
 功能說明: 使用 Leibniz 公式近似圓周率 pi，輸出 i = 10000, 20000, ..., 100000 對應的 pi 值，並找出使 pi 接近 3.14159 的最小 i。
 作者: 你的姓名
 日期: 2025-11-11
 版本: 1.0
 說明: 采用逐項累加的方式計算，避免每次從頭計算，並符合輸出格式需求。
*/
public class LeibnizPi {
    public static void main(String[] args) {
        double sum = 0.0;
        int maxI = 100000;
        int outputStep = 10000;
        int minIForGoal = -1;

        // 累加計算 π/4 的級數：sum_{k=0}^{i-1} (-1)^k / (2k+1)
        for (int i = 0; i < maxI; i++) {
            double term = ((i % 2 == 0) ? 1.0 : -1.0) / (2.0 * i + 1.0);
            sum += term;

            // 每到 10000 的倍數輸出一次
            if ((i + 1) % outputStep == 0) {
                double pi = 4.0 * sum;
                System.out.printf("i = %6d, pi = %.10f%n", i + 1, pi);
            }

            // 找出使 pi 接近 3.14159 的最小 i
            if (minIForGoal == -1) {
                double piCurrent = 4.0 * sum;
                if (Math.abs(piCurrent - 3.14159) <= 0.00001) {
                    minIForGoal = i + 1;
                }
            }
        }

        // 將最小符合條件的 i 輸出
        System.out.println();
        System.out.println("i = " + minIForGoal);
    }
}
