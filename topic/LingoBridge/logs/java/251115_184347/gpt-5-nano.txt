```java
/**
 * File: CSIE_41343238_1017_2.java
 * Description: Determine whether a given number's digits are strictly increasing from left to right.
 * Author: 教師
 * Date: 2025-11-15
 */

import java.util.Scanner;

public class CSIE_41343238_1017_2 {
    public static void main(String[] args) {
        try (Scanner sc = new Scanner(System.in)) {
            String input = sc.nextLine().trim();
            boolean isWellOrdered = isWellOrdered(input);

            if (isWellOrdered) {
                System.out.println(input + " is well-ordered number.");
            } else {
                System.out.println(input + " is not well-ordered number.");
            }
        }
    }

    // 判斷字串是否為「數字且其數字從左至右嚴格遞增」
    private static boolean isWellOrdered(String s) {
        if (s == null || s.length() == 0) return false;

        // 檢查是否全為數字
        for (int i = 0; i < s.length(); i++) {
            if (!Character.isDigit(s.charAt(i))) return false;
        }

        // 檢查嚴格遞增
        for (int i = 1; i < s.length(); i++) {
            if (s.charAt(i - 1) >= s.charAt(i)) return false;
        }

        return true;
    }
}
```
