```java
/*
檔案名稱: SortingArrayAsc.java
功能: 讀入任意長度的一維整數陣列，將內容由小到大排序後輸出
作者: 你可以填入作者姓名
日期: 2025-11-12
說明: 適當註解，處理負數，避免使用 EOF，僅執行一次後結束。
*/

import java.util.Arrays;
import java.util.Scanner;

public class SortingArrayAsc {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取陣列長度
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();

        // 讀入 n 個整數到陣列
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                // 若輸入不足，預設為 0，避免拋出例外；實務可依需求調整
                arr[i] = 0;
            }
        }

        sc.close();

        // 排序
        Arrays.sort(arr);

        // 輸出，元素之間以單一空格分隔，末尾換行
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < arr.length; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());
    }
}
```
