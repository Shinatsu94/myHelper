/*
檔案名稱: CSIE_41343208_1031_3a.java
功能說明: 讀取整數長度 n，接著讀取 n 個整數，將它們升序排序後以空格分隔輸出
作者: 教授
日期: 2025-11-14
*/
import java.util.Arrays;
import java.util.Scanner;

public class CSIE_41343208_1031_3a {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }

        int n = sc.nextInt();
        int[] arr = new int[n];

        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                // 若輸入不足，預設為 0，避免 ArrayIndexOutOfBounds
                arr[i] = 0;
            }
        }

        Arrays.sort(arr);

        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }

        System.out.println(sb.toString());
        sc.close();
    }
}
