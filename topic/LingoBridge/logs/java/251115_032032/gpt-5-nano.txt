import java.util.Scanner;

/**
 * File: LowerTriangularMatrix.java
 * 功能: 依使用者輸入的高度 n，建立一個下三角形二維陣列，並找出指定數字的座標。
 * 作者: 調整後的版本
 * 日期: 2025-11-14
 */
public class LowerTriangularMatrix {
    public static void main(String[] args) {
        try (Scanner sc = new Scanner(System.in)) {
            int n = sc.nextInt();

            // 建立下三角形二維陣列並填數字
            int[][] arr = new int[n][];
            int numCount = 1;
            for (int i = 0; i < n; i++) {
                arr[i] = new int[i + 1];
                for (int j = 0; j <= i; j++) {
                    arr[i][j] = numCount++;
                }
            }

            // 取得要尋找的數字
            int target = sc.nextInt();

            boolean found = false;
            int foundRow = -1;
            int foundCol = -1;

            outer:
            for (int i = 0; i < n; i++) {
                for (int j = 0; j <= i; j++) {
                    if (arr[i][j] == target) {
                        foundRow = i + 1; // 1-based
                        foundCol = j + 1; // 1-based
                        found = true;
                        break outer;
                    }
                }
            }

            if (found) {
                System.out.printf("%d[%d, %d]%n", target, foundRow, foundCol);
            } else {
                System.out.printf("not find %d in array.%n", target);
            }
        }
    }
}
