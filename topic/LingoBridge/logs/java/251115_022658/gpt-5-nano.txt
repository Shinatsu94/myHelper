import java.util.Scanner;

class CSIE_51015114_1031_2 {
    /*
     * File: CSIE_51015114_1031_2.java
     * Description: 將陣列中每個元素替換為其右側元素的最大值，最後一個元素替換為 -1，最後輸出結果。
     * Author: 你的姓名
     * Date: 2025-11-14
     */
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.hasNextInt() ? sc.nextInt() : 0;
        if (n <= 0) {
            sc.close();
            return;
        }

        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            arr[i] = sc.hasNextInt() ? sc.nextInt() : 0;
        }

        // 從右向左單次掃描，將每個位置設為其右側的最大值
        int maxRight = arr[n - 1];
        arr[n - 1] = -1;
        for (int i = n - 2; i >= 0; i--) {
            int original = arr[i]; // 儲存原始值，避免被覆蓋的值影響後續計算
            arr[i] = maxRight;
            if (original > maxRight) {
                maxRight = original;
            }
        }

        // 輸出結果
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.print(sb.toString());

        sc.close();
    }
}
