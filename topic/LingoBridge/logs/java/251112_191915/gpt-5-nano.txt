/**
 * ArrayProcessor.java
 * 功能: 讀取長度與元素，將陣列內容由小到大排序後輸出
 * 檔案名稱: ArrayProcessor.java
 * 作者: [你的名字]
 * 日期: 2025-11-12
 * 說明: 使用內建排序 Arrays.sort，處理可能的負數與長度為 1 的情形，並在輸出時避免多餘空格。
 */

import java.util.Scanner;

public class ArrayProcessor {

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        // 取得陣列長度
        if (!scanner.hasNextInt()) {
            scanner.close();
            return;
        }
        int n = scanner.nextInt();

        // 處理非正長度情況
        if (n <= 0) {
            System.out.println();
            scanner.close();
            return;
        }

        int[] arr = new int[n];

        // 依序讀入 n 個整數
        for (int i = 0; i < n; i++) {
            if (scanner.hasNextInt()) {
                arr[i] = scanner.nextInt();
            } else {
                // 輸入不足，結束
                scanner.close();
                return;
            }
        }

        // 排序（由小到大）
        java.util.Arrays.sort(arr);

        // 輸出，元素以單一空格分隔，末尾不多餘空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        scanner.close();
    }
}
