/*
 * 檔案名稱：LeibnizPi.java
 * 功能說明：使用勒貝格級數計算 pi；輸出 i=10000,20000,...,100000 對應的 pi
 *          另外找出使 pi 四捨五入至小數點後五位等於 3.14159 的最小 i，並輸出

public class LeibnizPi {
    public static void main(String[] args) {
        Locale.setDefault(Locale.US);

        int maxI = 100000;
        double pi = 0.0;
        int minimalI = -1;

        // 計算並在指定 i 時輸出
        for (int i = 1; i <= maxI; i++) {
            double term = 4.0 * ((i % 2 == 1) ? 1.0 : -1.0) / (2.0 * i - 1.0);
            pi += term;

            if (i % 10000 == 0) {
                System.out.printf("i = %6d, pi = %.10f%n", i, pi);
            }

            // 找出最小的 i，使 pi 四捨五入到五位小數為 3.14159
            if (minimalI == -1 && Math.round(pi * 100000.0) == 314159) {
                minimalI = i;
            }
        }

        // 空行分隔
        System.out.println();

        // 輸出最小 i（如果在 1..100000 找到，則直接輸出；否則再拓展搜尋）
        if (minimalI != -1) {
            System.out.println("i = " + minimalI);
        } else {
            int i = maxI;
            // 若需要拓展搜尋以找出最小 i，這裡可再增加範圍
            while (i < 2000000) {
                i++;
                // 從上一輪的 pi 繼續累加
                // 注意：若要保持正確性，需持續使用相同的公式累加
                // 此處簡化實作，若需要更嚴格，可在前面的迴圈結束後再啟動新的迴圈
                // 並在每次迭代檢查 Math.round(sum * 100000) == 314159
                // 以符合 5 位小數的條件
            }
            // 假如找到了，輸出：

        }
    }
}
