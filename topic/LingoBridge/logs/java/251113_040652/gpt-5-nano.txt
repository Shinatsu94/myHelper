/*
檔案名稱: PrintUnicodeCodepoints.java
功能: 依序輸出虎科大及學生姓名每個字元與對應的 Unicode 內碼，格式為 "<字> <內碼>"
作者: [您的名字]
日期: 2025-11-12
說明: 使用 Unicode code point（int）作為內碼輸出，便於跨平台一致性。
*/
public class PrintUnicodeCodepoints {
    public static void main(String[] args) {
        // 先印出虎科大
        char[] first = {'虎', '科', '大'};
        // 再印出學生的名字，例如林武杰
        char[] studentName = {'林', '武', '杰'};

        // 輸出格式：每行 "<字元> <內碼(十進制)>"
        for (char ch : first) {
            System.out.println(ch + " " + (int) ch);
        }
        for (char ch : studentName) {
            System.out.println(ch + " " + (int) ch);
        }
    }
}
