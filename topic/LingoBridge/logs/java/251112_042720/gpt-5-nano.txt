import java.util.Scanner;

public class SwapTwoNumbers {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int A = sc.nextInt();
        int B = sc.nextInt();

        // 使用不額外變數的三步驟交換法
        A = A + B;
        B = A - B;
        A = A - B;

        // 輸出交換後的結果（符合範例：先輸出 A 的新值，再輸出 B 的新值）
        System.out.println(A);
        System.out.println(B);

        sc.close();
    }
}
