/*
檔案名稱: LeibnizPi.java
功能說明: 計算 pi 的近似值，使用萊布尼茲公式
作者: 你自行填寫
日期: 2025-11-12
備註: 輸出 i 從 10000 到 100000，以 10000 為步長；i 顯示寬度為 6，pi 顯示小數點後 10 位。
*/

public class LeibnizPi {
    public static void main(String[] args) {
        int maxI = 100000;
        int step = 10000;
        double sum = 0.0;

        for (int k = 0; k < maxI; k++) {
            // 第 k 項：符號(-1)^k，分母為 2k+1
            double term = ((k % 2 == 0) ? 1.0 : -1.0) / (2.0 * k + 1.0);
            sum += term;

            int iVal = k + 1;
            if (iVal % step == 0) {
                double pi = 4.0 * sum;
                System.out.printf("i = %6d, pi = %.10f%n", iVal, pi);
            }
        }
    }
}
