/*
檔案名稱: CSIE_41343128_1031_1.java
功能: 讀取長度 N 與 N 個整數，將其升序排序並輸出
作者: 你的名字
日期: 2025-11-12
*/
import java.util.Scanner;

public class CSIE_41343128_1031_1 {
    public static void main(String[] args) {
        Scanner scn = new Scanner(System.in);

        int n = scn.nextInt();
        if (n <= 0) {
            // 無元素可排序，直接結束
            scn.close();
            return;
        }

        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            arr[i] = scn.nextInt();
        }

        // 選擇排序（升序）
        for (int i = 0; i < arr.length - 1; i++) {
            int minIndex = i;
            for (int j = i + 1; j < arr.length; j++) {
                if (arr[j] < arr[minIndex]) {
                    minIndex = j;
                }
            }
            // 交換
            int tmp = arr[i];
            arr[i] = arr[minIndex];
            arr[minIndex] = tmp;
        }

        // 使用 StringBuilder 組裝輸出，最後再輸出整段文字
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < arr.length; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        scn.close();
    }
}
