public class LeibnizPi {
    /*
     * 文件名稱: LeibnizPi.java
     * 功能說明: 計算對於給定 i 的 Leibniz 近似 pi 值，並輸出指定格式；同時找出使 pi 落在 [3.14159, 3.14160) 的最小 i。
     * 作者: 你的姓名
     * 日期: 2025-11-11
     * 備註: 程式內部有清楚註解，並依題目要求輸出格式化字串。
     */

    public static void main(String[] args) {
        // (a) 輸出 i = 10000, 20000, ..., 100000 之 pi 值，保留小數點後10位
        for (int terms = 10000; terms <= 100000; terms += 10000) {
            double sum = 0.0;
            int sign = 1;
            // Leibniz 公式：pi = 4 * SUM_{k=1..n} (-1)^(k+1) / (2k-1)
            for (int k = 1; k <= terms; k++) {
                sum += 4.0 * sign / (2 * k - 1);
                sign = -sign;
            }
            System.out.printf("i = %6d, pi = %.10f%n", terms, sum);
        }

        // 分隔行，符合題目「輸出文字，請使用單個空白\" \"進行換行」的需求
        System.out.println();

        // (b) 找到最小的 i，使 pi ≈ 3.14159，落在 [3.14159, 3.14160)
        double sumB = 0.0;
        int signB = 1;
        int iMin = 0;
        while (true) {
            iMin++;
            sumB += 4.0 * signB / (2 * iMin - 1);
            signB = -signB;
            if (sumB >= 3.14159 && sumB < 3.14160) {
                break;
            }
        }
        System.out.printf("i = %6d%n", iMin);
    }
}
