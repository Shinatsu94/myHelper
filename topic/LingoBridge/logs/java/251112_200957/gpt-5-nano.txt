/**
 * 程式名稱 : OneDimensionalArraySortAscending
 * 功能說明：建立任意長度的一維整數陣列，讀入數值並進行升序排序，最後輸出排序結果（以空格分隔）。
 * 日期: 2025/11/12
 * 作者: 教師評分示例
 * 學號: XXXXXXX
 */

// 引入必要的套件
import java.util.Scanner;
import java.util.Arrays;

public class OneDimensionalArraySortAscending {
    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);

        // 讀取陣列長度
        if (!in.hasNextInt()) {
            return;
        }
        int n = in.nextInt();
        if (n <= 0) {
            return;
        }

        // 讀取 n 個整數
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (in.hasNextInt()) {
                arr[i] = in.nextInt();
            } else {
                // 不足的輸入以 0 代替，避免發生例外
                arr[i] = 0;
            }
        }

        // 對陣列進行升序排序
        Arrays.sort(arr);

        // 輸出排序後的陣列，元素以空格分隔，避免尾端多餘空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());
    }
}
