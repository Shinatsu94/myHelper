/*
 * 檔案名稱: SortArray.java
 * 功能: 依序輸入長度 n 與 n 個整數，輸出按升序排列且以空白分隔的結果
 * 作者: 你可以填寫
 * 日期: 2025-11-12
 * 備註: 不使用 EOF，讀取一次後輸出結果並結束
 */
import java.util.Arrays;
import java.util.Scanner;

public class SortArray {
    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);

        // 讀取長度
        if (!in.hasNextInt()) {
            in.close();
            return;
        }
        int n = in.nextInt();
        int[] arr = new int[n];

        // 讀取 n 個整數
        for (int i = 0; i < n; i++) {
            if (in.hasNextInt()) {
                arr[i] = in.nextInt();
            } else {
                // 若不足 n 個數，提前結束
                break;
            }
        }

        // 排序（升序）
        Arrays.sort(arr);

        // 輸出，數字之間以空格分隔，最後不多餘空格，且結束不加多餘換行符以符合題目輸出格式
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.print(sb.toString());

        in.close();
    }
}
