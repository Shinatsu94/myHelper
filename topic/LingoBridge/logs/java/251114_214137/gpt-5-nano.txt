/*
檔案名稱: CSIE_41343232_1031_2.java
功能: 讀入整數個數 n 與 n 個整數，將它們由小到大排序，並輸出排序結果。
作者: 教師
日期: 2025-11-14
說明: 使用 Arrays.sort 進行升序排序，並輸出結果，避免使用 EOF。
*/
import java.util.Scanner;
import java.util.Arrays;

public class CSIE_41343232_1031_2 {
    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);

        if (!in.hasNextInt()) {
            in.close();
            return;
        }

        int n = in.nextInt();
        int[] a = new int[Math.max(0, n)];

        for (int i = 0; i < n; i++) {
            if (in.hasNextInt()) {
                a[i] = in.nextInt();
            } else {
                a[i] = 0; // 若輸入不足，補 0 以避免例外
            }
        }

        Arrays.sort(a);

        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < a.length; i++) {
            if (i > 0) sb.append(' ');
            sb.append(a[i]);
        }

        System.out.println(sb.toString());
        in.close();
    }
}
