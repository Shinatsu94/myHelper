import java.util.Scanner;

/*
檔案名稱: CSIE_41343208_1031_3a.java
功能: 建立並輸出下三角形的二維陣列
作者: 學生
日期: 2025-11-14
說明: 會建立 n x n 的 char 陣列，僅填充下三角部分為 '*'，再印出下三角形
*/
public class CSIE_41343208_1031_3a {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = 0;
        if (sc.hasNextInt()) {
            n = sc.nextInt();
        }

        // 設定範圍，若不在 0 < n <= 20，直接結束
        if (n <= 0 || n > 20) {
            sc.close();
            return;
        }

        // 建立並填充下三角形的二維陣列
        char[][] tri = new char[n][n];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= i; j++) {
                tri[i][j] = '*';
            }
        }

        // 輸出下三角形
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= i; j++) {
                System.out.print(tri[i][j]);
            }
            System.out.println();
        }

        sc.close();
    }
}
