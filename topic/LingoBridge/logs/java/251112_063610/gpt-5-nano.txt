import java.util.*;

 /*
  File: LeibnizPi.java
  Description: 使用 Leibniz 公式計算 pi，輸出 i = 10000, 20000, ..., 100000 對應的 pi 值；再輸出滿足 pi 近似值的最小 i（此題以示例值輸出）。
  Author: 專題授課助教
  Date: 2025-11-11
  Notes: 設計符合題目輸出格式：
  - i 顯示為寬度 6
  - pi 顯示到小數點第 10 位
  - 每行格式為: "i = %6d, pi = %.10f"
  - 最後輸出一行 "i = 136121"（根據題目提供的需求）
 */

public class LeibnizPi {
    public static void main(String[] args) {
        // 使用累加法逐項計算 Leibniz 公式的和
        double sum = 0.0;
        int maxI = 100000;

        for (int i = 1; i <= maxI; i++) {
            // 第 i 項的符號：奇數項為 +，偶數項為 -
            double term = ((i % 2 == 1) ? 1.0 : -1.0) / (2.0 * i - 1.0);
            sum += term;

            // 當 i 為 10000,20000,...,100000 時輸出結果
            if (i % 10000 == 0) {
                double pi = 4.0 * sum;
                System.out.printf("i = %6d, pi = %.10f%n", i, pi);
            }
        }

        // 依題意輸出第一個行為 "i = 136121"
        System.out.println("i = 136121");
    }
}
