import java.util.Arrays;
import java.util.Scanner;

/*
File: CSIE_41343112_1031_1.java
Description: 讀取整數 n，接著讀取 n 個整數，將它們升序排序後以空格分隔輸出
Author: 李宗懌
Date: 2025-11-14
*/

public class CSIE_41343112_1031_1 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = 0;
        if (sc.hasNextInt()) {
            n = sc.nextInt();
        } else {
            sc.close();
            return;
        }

        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                arr[i] = 0; // 若輸入不足，預設為0
            }
        }

        // 排序（使用標準排序，適用於正負整數）
        Arrays.sort(arr);

        // 輸出，數字之間以單一空格分隔，末尾換行
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        sc.close();
    }
}
