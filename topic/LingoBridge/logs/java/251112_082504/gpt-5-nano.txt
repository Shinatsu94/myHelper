修改後的完整程式碼（可直接執行）：
/*
 檔案名稱: LeibnizPi.java
 功能: 計算使用 Leibniz 公式近似圓周率 pi；輸出 i=10000,20000,...,100000 的 pi，並找出最小 i 使 pi 四捨五入至小數點第五位為 3.14159。
 作者: AI 助手
 日期: 2025-11-12
*/
public class LeibnizPi {
    private static double computePi(int terms) {
        double sum = 0.0;
        int sign = 1;
        for (int k = 1; k <= terms; k++) {
            sum += sign * (1.0 / (2 * k - 1));
            sign = -sign;
        }
        return 4.0 * sum;
    }

    public static void main(String[] args) {
        // 輸出 i = 10000, 20000, ..., 100000 的 pi 值
        for (int i = 10000; i <= 100000; i += 10000) {
            double pi = computePi(i);
            System.out.println("i = " + String.format("%6d", i) + ", pi = " + String.format("%.10f", pi));
        }

        // 找出最小的 i 使 pi 四捨五入到小數點後五位為 3.14159
        int iMin = -1;
        for (int i = 1; i <= 2000000; i++) {
            double pi = computePi(i);
            long rounded = Math.round(pi * 100000); // 四捨五入到小數點後五位
            if (rounded == 314159) {
                iMin = i;
                break;
            }
        }
        System.out.println("i = " + String.format("%6d", iMin));
    }
}
