功能: 讀取整數陣列長度與內容，將內容由小到大排序後輸出
作者: [Your Name]
日期: 2025-11-14
說明: 使用 Arrays.sort 進行排序，輸出同一列且以空格分隔
*/
import java.util.Arrays;
import java.util.Scanner;

public class SortArrayAscending {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取陣列長度
        int n = sc.nextInt();

        // 讀取陣列元素
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();
        }

        // 由小到大排序
        Arrays.sort(arr);

        // 輸出排序結果，元素以空格分隔
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        sc.close();
    }
}
