/* 
 * File: CSIE_41343216_1031_1.java
 * Title: 一維陣列升序排序
 * Description: 讀取長度 n 與 n 個整數，輸出升序排序的結果。
 * Author: your name
 * Date: 2025-11-14
 */

import java.util.Scanner;
import java.util.Arrays;

public class CSIE_41343216_1031_1 {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);

        // 讀取陣列長度
        if (!input.hasNextInt()) {
            input.close();
            return;
        }
        int n = input.nextInt();

        // 長度檢查，避免非法情況
        if (n <= 0) {
            input.close();
            System.out.println();
            return;
        }

        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            arr[i] = input.nextInt();
        }

        // 升序排序
        Arrays.sort(arr);

        // 使用 StringBuilder 組合輸出，最後印出並換行
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        input.close();
    }
}
