/*
 * File: CSIE_41343222_1031_4.java
 * 功能: 建立任意長度的一維陣列，讀取元素並以升冪排序輸出
 * 作者: your_name
 * 日期: 2025-11-14
 * 備註: 讀入長度 n，接著讀入 n 個整數，排序後輸出結果，支援負數。
 */

import java.util.Arrays;
import java.util.Scanner;

public class CSIE_41343222_1031_4 {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        // 讀取陣列長度
        int n = scanner.nextInt();

        // 建立並填充陣列
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            arr[i] = scanner.nextInt();
        }

        // 對陣列進行升冪排序
        Arrays.sort(arr);

        // 輸出排序結果，使用空格分隔，末尾換行
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        scanner.close();
    }
}
