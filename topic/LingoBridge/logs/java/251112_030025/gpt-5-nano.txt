import java.util.Scanner;

public class SwapTwoNumbers {
  public static void main(String[] args) {
    Scanner sc = new Scanner(System.in);
    long A = sc.nextLong();
    long B = sc.nextLong();

    // 不使用額外變數進行兩數值交換
    A = A + B;
    B = A - B;
    A = A - B;

    System.out.println(A);
    System.out.println(B);

    sc.close();
  }
}
