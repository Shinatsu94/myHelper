/*
 * 檔名: CSIE_41343211_1031_2.java
 * 功能: 讀入一維整數陣列，將內容由小到大排序後輸出
 * 作者: 你的姓名
 * 日期: 2025-11-14
 */

import java.util.Arrays;
import java.util.Scanner;

public class CSIE_41343211_1031_2 {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);

        // 讀取陣列長度
        int n = input.nextInt();
        if (n <= 0) {
            System.out.println("陣列長度必須大於 0");
            input.close();
            return;
        }

        int[] arr = new int[n];

        // 讀取陣列元素
        for (int i = 0; i < n; i++) {
            arr[i] = input.nextInt();
        }

        // 使用內建排序，從小到大排序
        Arrays.sort(arr);

        // 輸出結果
        for (int i = 0; i < n; i++) {
            if (i > 0) System.out.print(" ");
            System.out.print(arr[i]);
        }
        System.out.println();

        input.close();
    }
}
