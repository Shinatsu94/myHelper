import java.util.Arrays;

/*
 * 檔案名稱: SortArray.java
 * 功能說明: 讀入整數陣列的長度與元素，輸出由小到大排序後的結果
 * 作者: 請填寫
 * 日期: 2025-11-14
 * 版本: 1.0
 * 備註: 使用 Arrays.sort 以正確實作排序，輸出格式為數字以單一空格分隔
 */

public class SortArray {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取陣列長度
        int length = 0;
        if (sc.hasNextInt()) {
            length = sc.nextInt();
        }

        // 讀取陣列元素
        int[] a = new int[length];
        for (int i = 0; i < length; i++) {
            if (sc.hasNextInt()) {
                a[i] = sc.nextInt();
            } else {
                // 若輸入不足，預先填充 0
                a[i] = 0;
            }
        }

        // 排序 (由小到大)
        Arrays.sort(a);

        // 輸出結果，數字之間以空格分隔，末尾無多餘空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < length; i++) {
            if (i > 0) sb.append(' ');
            sb.append(a[i]);
        }
        System.out.print(sb.toString());

        sc.close();
    }
}
