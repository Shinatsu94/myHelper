```java
/*
 * 檔名: LeibnizPi.java
 * 功能: 計算 pi 的近似值，根據題意輸出 i=10000,20000,...,100000 對應的 pi 值，
 *       並求得使 pi 四捨五入到小數點第六位為 3.14159 的最小 i 值。
 * 作者: 你可以填入名字
 * 日期: 2025-11-11
 *
 * 註解說明:
 * - 使用 Leibniz 公式 pi = 4 * sum_{k=1..i} (-1)^(k+1) / (2k-1)
 * - i 的值依題意分別輸出為 10000, 20000, ..., 100000，格式化輸出精度  
 * - 最終找出使 pi 四捨五入後等於 3.14159 的最小 i，輸出形式為 "i = <value>"
 */

public class LeibnizPi {
    public static void main(String[] args) {
        // Part (a): 計算 i = 10000, 20000, ..., 100000 時的 pi 值
        int maxI = 100000;
        double sum = 0.0;
        int sign = 1; // 符號，初始為正
        for (int i = 1; i <= maxI; i++) {
            sum += sign * (1.0 / (2 * i - 1));
            if (i % 10000 == 0) {
                double pi = 4.0 * sum;
                // i 顯示到 6 位，pi 顯示到小數點後 10 位
                System.out.printf("i = %6d, pi = %.10f%n", i, pi);
            }
            sign = -sign; // 符號翻轉
        }

        // Part (b): 找最小的 i 使 pi 四捨五入到小數點第 5 位為 3.14159
        sum = 0.0;
        sign = 1;
        int iMin = -1;
        // 從頭開始逐步累加，直到滿足條件
        for (int i = 1; ; i++) {
            sum += sign * (1.0 / (2 * i - 1));
            double pi = 4.0 * sum;
            double roundedTo5 = Math.round(pi * 100000.0) / 100000.0;
            if (roundedTo5 == 3.14159) {
                iMin = i;
                break;
            }
            // 安全機制，避免無限迴圈
            if (i > 2000000) {
                break;
            }
            sign = -sign;
        }

        System.out.println("i = " + iMin);
    }
}
```
