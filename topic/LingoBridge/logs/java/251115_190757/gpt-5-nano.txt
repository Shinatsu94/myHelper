import java.util.Scanner;

/*
檔案名稱: CSIE_41343236_1017_3.java
功能: 將使用者輸入之整數的位數反轉，並保留符號；例如 123 -> 321, 2500 -> 52, 8531 -> 1358
作者: 教師
日期: 2025-11-15
說明: 優化點: 不直接對整數取絕對值，改用逐位取模的絕對值以避免 Integer.MIN_VALUE 的溢位問題；使用 long 避免溢位風險；增加對輸入存在性檢查。
*/

public class CSIE_41343236_1017_3 {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        long num;
        if (scanner.hasNextLong()) {
            num = scanner.nextLong();
        } else {
            scanner.close();
            return;
        }

        long n = num;
        long reversed = 0;

        // 逐位反轉，對餘數取絕對值避免整數溢位
        while (n != 0) {
            int digit = (int) Math.abs(n % 10);
            reversed = reversed * 10 + digit;
            n /= 10;
        }

        if (num < 0) {
            reversed = -reversed;
        }

        System.out.println(reversed);
        scanner.close();
    }
}
