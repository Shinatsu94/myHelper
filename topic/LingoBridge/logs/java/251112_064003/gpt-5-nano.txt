/**
 * 程式名稱 : LeibnizPi
 * 
 * 功能說明：計算勒貝尼茲級數近似的 pi 值，輸出 i=10000,10000...100000 時的 pi，並找出使 pi >= 3.14159 的最小 i。
 *
 * 日期: 2025/11/11
 *
 * 作者: 教師助理（範例程式）
 *	
 * 學號: 00000000
 * 
 */
public class LeibnizPi {
    public static void main(String[] args) {
        // Part (a): 輸出 i = 10000, 20000, ..., 100000 對應的 pi 值，pi 顯示 10 位小數
        double sum = 0.0;
        for (int terms = 1; terms <= 100000; terms++) {
            // 第 terms 項的符號與分母
            double term = (terms % 2 == 1) ? 1.0 : -1.0;
            sum += term / (2.0 * terms - 1.0);

            // 每當 terms 為 10000 的倍數時輸出對應的 pi 值
            if (terms % 10000 == 0) {
                double pi = 4.0 * sum;
                System.out.println("i = " + String.format("%6d", terms) + ", pi = " + String.format("%.10f", pi));
            }
        }

        // Part (b): 找到最小的 i 使 pi >= 3.14159
        double sumForGoal = 0.0;
        int iForGoal = 0;
        for (int terms = 1; ; terms++) {
            double term = (terms % 2 == 1) ? 1.0 : -1.0;
            sumForGoal += term / (2.0 * terms - 1.0);
            double piEstimate = 4.0 * sumForGoal;
            if (piEstimate >= 3.14159) {
                iForGoal = terms;
                break;
            }
        }
        System.out.println("i = " + iForGoal);
    }
}
