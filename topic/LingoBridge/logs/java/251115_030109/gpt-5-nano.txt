修改後完整程式碼
/***************************************
 * File: LowerTriangularStars.java
 * 功能: 建立並列印下三角形的二維陣列，內容為 '*' 字元。
 * 作者: 教師
 * 日期: 2025-11-14
 * 說明: 讀取整數 n (1 <= n <= 20)，輸出 n 行的下三角星號模式。
 ***************************************/
import java.util.Scanner;

public class LowerTriangularStars {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        sc.close();

        // 基本輸入檢驗，遵守題意的範圍限制
        if (n <= 0 || n > 20) {
            return;
        }

        // 使用 2D char 陣列表示下三角形，只有下三角填 '*'，其餘位置保持未使用
        char[][] tri = new char[n][n];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= i; j++) {
                tri[i][j] = '*';
            }
        }

        // 輸出以上三角形內容，元素之間不輸出空白符號
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= i; j++) {
                System.out.print(tri[i][j]);
            }
            System.out.println();
        }
    }
}
