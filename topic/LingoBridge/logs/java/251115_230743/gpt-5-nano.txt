import java.util.Scanner;

public class CSIE_41343205_1003_1 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int a = sc.nextInt();
        int b = sc.nextInt();

        // 使用位元異或運算進行原地交換，且不使用額外變數
        a = a ^ b;
        b = a ^ b;
        a = a ^ b;

        System.out.println(a);
        System.out.println(b);

        sc.close();
    }
}
