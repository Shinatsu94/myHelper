/*
 * 檔案名稱: CSIE_41343215_1031_3.java
 * 功能: 讀取長度 n 與 n 個整數，將其排序後由小到大輸出，輸入支援負數
 * 作者: 學生
 * 日期: 2025-11-12
 * 備註: 不使用 EOF，執行一次程式即可結束
 */

import java.util.Scanner;
import java.util.Arrays;

public class CSIE_41343215_1031_3 {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        // 讀取陣列長度
        if (!scanner.hasNextInt()) {
            scanner.close();
            return;
        }
        int n = scanner.nextInt();

        // 讀取 n 個整數到陣列
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (scanner.hasNextInt()) {
                arr[i] = scanner.nextInt();
            } else {
                // 若輸入不足，以 0 作為預設值
                arr[i] = 0;
            }
        }
        scanner.close();

        // 進行排序（由小到大）
        Arrays.sort(arr);

        // 輸出排序後的陣列，元素之間以空格分隔，結尾換行
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());
    }
}

- 從原先印出星號的輸出，改為讀取長度與數值、執行排序再輸出，符合題目需求。
- 使用 Arrays.sort(arr) 進行排序，簡潔且穩定，降低實作錯誤風險，若未來想自訂排序演算法也較容易替換。
