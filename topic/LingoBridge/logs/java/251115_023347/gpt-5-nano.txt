/*
檔案名稱: LowerTriangularArray.java
功能: 依輸入的 n (0 < n <= 20)，建立並列印一個下三角形的二維字元陣列，其左下角為 '*'，每行列印至對應的列數。
作者: 你的名字
日期: 2025-11-14
說明: 使用二維陣列儲存下三角形，並只印出有效的三角區域，避免輸出多餘空白。
*/

import java.util.Scanner;

public class LowerTriangularArray {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        // 讀取輸入
        int n;
        if (!scanner.hasNextInt()) {
            scanner.close();
            return;
        }
        n = scanner.nextInt();

        // 輸入範圍檢查，符合題意 (0 < n <= 20)
        if (n <= 0 || n > 20) {
            scanner.close();
            return;
        }

        // 建立並填入下三角形的二維陣列
        char[][] arr = new char[n][n];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= i; j++) {
                arr[i][j] = '*';
            }
        }

        // 逐行列印，只輸出有效的下三角區域
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= i; j++) {
                System.out.print(arr[i][j]);
            }
            System.out.println();
        }

        scanner.close();
    }
}
