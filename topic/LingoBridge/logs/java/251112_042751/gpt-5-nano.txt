/*
 * File: Main.java
 * Description: 讀入兩個整數，使用不新增變數的方式交換兩數，並輸出交換後的值（各一行）。
 * Author: 你的名字

import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner scan = new Scanner(System.in);

        // 讀入兩個整數
        int a = scan.nextInt();
        int b = scan.nextInt();

        // 不使用額外變數的交換法（加減法）
        a = a + b;
        b = a - b;
        a = a - b;

        // 輸出交換後的結果
        System.out.println(a);
        System.out.println(b);

        scan.close();
    }
}
