/*
檔案名稱: csie_1031_optimized.java
功能: 讀入整數長度 n 與 n 個整數，輸出從小到大的排序結果
作者: 教師
日期: 2025-11-14
說明: 使用 Arrays.sort 進行排序，輸出中間不留多餘空格，最後換行
*/
import java.util.Scanner;
import java.util.Arrays;

public class csie_1031_optimized {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int n = sc.nextInt();
        int[] arr = new int[n];

        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                arr[i] = 0; // 若輸入不足，預設為 0，避免例外
            }
        }

        Arrays.sort(arr);

        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        sc.close();
    }
}
