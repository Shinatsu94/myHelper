/*
 * 檔案名稱: SelectionSort.java
 * 功能說明: 讀取使用者輸入的長度與對應的整數，將陣列內容由小到大排序（選擇排序），
 *         最後輸出排序結果，元素之間以單一空格分隔，並於末尾換行。
 * 作者: 資訊工程系教授
 * 日期: 2025-11-14
 * 備註: 具備負數支援，程式執行一次即結束，未使用 EOF。
 */

import java.util.Scanner;

public class SelectionSort {
    // 使用選擇排序將整數陣列由小到大排序
    public static void selectionSort(int[] arr) {
        int n = arr.length;

        for (int i = 0; i < n - 1; i++) {
            int minIndex = i;

            for (int j = i + 1; j < n; j++) {
                if (arr[j] < arr[minIndex]) {
                    minIndex = j;
                }
            }

            // 若找到的最小值不是目前的 i，才進行交換
            if (minIndex != i) {
                int temp = arr[i];
                arr[i] = arr[minIndex];
                arr[minIndex] = temp;
            }
        }
    }

    public static void main(String[] args) {
        Scanner s = new Scanner(System.in);
        int n = s.nextInt();

        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            arr[i] = s.nextInt();
        }

        selectionSort(arr);

        for (int i = 0; i < arr.length; i++) {
            if (i > 0) System.out.print(" ");
            System.out.print(arr[i]);
        }
        System.out.println(); // 輸出結尾換行

        s.close();
    }
}
