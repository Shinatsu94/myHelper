```java
/*
文件名稱: LeibnizPiPrinter.java
功能說明: 使用萊布尼茲公式計算 pi，輸出 i 為 10000, 20000, ..., 100000 時的 pi 值
作者: 論文/作業批改
日期: 2025-11-11
說明: pi 使用 10 位小數輸出，i 使用六位寬度對齊輸出
*/

import java.util.Locale;

public class LeibnizPiPrinter {
    public static void main(String[] args) {
        // 設定常用的十進位格式，以避免地區設定影響小數點
        Locale.setDefault(Locale.US);

        StringBuilder out = new StringBuilder();

        double sum = 0.0;
        int maxTerms = 100000;

        for (int k = 0; k < maxTerms; k++) {
            double term = 1.0 / (2.0 * k + 1.0);
            if (k % 2 == 0) {
                sum += term;
            } else {
                sum -= term;
            }

            int terms = k + 1;
            if (terms % 10000 == 0) {
                double pi = 4.0 * sum;
                out.append(String.format("i = %6d, pi = %.10f%n", terms, pi));
            }
        }

        System.out.print(out.toString());
    }
}
```
