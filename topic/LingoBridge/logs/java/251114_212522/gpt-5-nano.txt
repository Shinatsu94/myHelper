/*
檔案名稱: CSIE_41343119_1031_2.java
功能說明: 讀取一個整數 n，接著讀取 n 個整數，將陣列由小到大排序，最後輸出排序後的陣列。
作者: 金宗翰
學號: 41343119
日期: 2025-10-31
*/

import java.util.Scanner;
import java.util.Arrays;

public class CSIE_41343119_1031_2 {
    public static void main(String[] args) {
        Scanner scn = new Scanner(System.in);

        // 讀取陣列長度
        if (!scn.hasNextInt()) {
            scn.close();
            return;
        }
        int n = scn.nextInt();
        int[] A = new int[n];

        // 讀取 n 個整數
        for (int i = 0; i < n; i++) {
            if (scn.hasNextInt()) {
                A[i] = scn.nextInt();
            } else {
                // 若輸入不足，預設為 0（可依需求調整）
                A[i] = 0;
            }
        }

        // 由小到大排序
        Arrays.sort(A);

        // 輸出，數字之間以單一空格分隔，結尾換行
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(A[i]);
        }
        System.out.println(sb.toString());

        scn.close();
    }
}
