```java
/*
 * File: SortArrayAscending.java
 * Description: 讀取整數 n，接著讀入 n 個整數，將它們由小到大排序後輸出。
 * Author: 教師
 * Date: 2025-11-14
 */

import java.util.Scanner;
import java.util.Arrays;

public class SortArrayAscending {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取長度 n（若未提供，直接結束）
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();
        int[] arr = new int[Math.max(0, n)];

        // 依序讀入 n 個整數，可能分布在不同行
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                // 不足的輸入以 0 代替，避免例外
                arr[i] = 0;
            }
        }

        sc.close();

        // 排序
        Arrays.sort(arr);

        // 以空格分隔輸出，避免尾端空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < arr.length; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());
    }
}
```
