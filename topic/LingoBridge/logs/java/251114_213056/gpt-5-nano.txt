import java.util.Scanner;
import java.util.Arrays;

/*
檔案名稱: CSIE_41343205_1031_2.java
功能: 依輸入長度與元素，輸出由小到大排序的整數陣列
作者: 教師
日期: 2025-11-14
說明: 使用 Arrays.sort 進行排序，並以空格分隔輸出，結尾換行。
*/
public class CSIE_41343205_1031_2 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 先讀取陣列長度
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();

        // 讀取 n 個整數
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                // 若輸入不足，預設為 0，避免例外
                arr[i] = 0;
            }
        }

        // 排序（由小至大）
        Arrays.sort(arr);

        // 輸出，避免尾端多餘空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        sc.close();
    }
}
