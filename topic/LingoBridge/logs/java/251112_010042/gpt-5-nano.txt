import java.util.Scanner;

public class CSIE_41343212_1003_2 {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        int a = input.nextInt();
        int b = input.nextInt();

        // 不使用額外變數，透過算術交換兩個變數的值
        a = a + b;
        b = a - b;
        a = a - b;

        System.out.println(a);
        System.out.println(b);
    }
}
