修改後完整程式碼：
/*
檔案名稱: SortArray.java
功能: 讀取整數長度 n 與 n 個整數，將其升序排序並輸出排序後的結果
作者: 你的名字
日期: 2025-11-14
說明: 使用單一的一維陣列與 Arrays.sort 進行排序，支援負數，輸出為以空格分隔的一行結果。
*/
import java.util.Arrays;
import java.util.Scanner;

public class SortArray {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取陣列長度
        int n = sc.nextInt();
        int[] arr = new int[n];

        // 讀取 n 個整數
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                // 若輸入不足，預設為 0（實務上應有錯誤處理；此處保持容錯）
                arr[i] = 0;
            }
        }

        // 進行排序（升冪）
        Arrays.sort(arr);

        // 輸出排序後的結果，空格分隔，換行結尾
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        sc.close();
    }
}
