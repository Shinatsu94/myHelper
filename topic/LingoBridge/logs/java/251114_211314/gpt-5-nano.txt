/**
 * 檔案名稱: CSIE_41343236_1031_1.java
 * 功能: 讀取長度 n 與 n 個整數，排序後輸出由小到大的序列
 * 作者: (你的名字)
 * 日期: 2025-11-14
 */

import java.util.Scanner;
import java.util.Arrays;

public class CSIE_41343236_1031_1 {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        // 取得陣列長度
        if (!scanner.hasNextInt()) {
            System.out.println();
            scanner.close();
            return;
        }
        int n = scanner.nextInt();

        // 非正長度的情況，直接輸出換行並結束
        if (n <= 0) {
            System.out.println();
            scanner.close();
            return;
        }

        int[] arr = new int[n];

        // 讀取 n 個整數到陣列
        for (int i = 0; i < n; i++) {
            if (scanner.hasNextInt()) {
                arr[i] = scanner.nextInt();
            } else {
                // 若輸入不足，預設為 0，可依需求調整
                arr[i] = 0;
            }
        }

        // 由小到大排序
        Arrays.sort(arr);

        // 使用 StringBuilder 組裝輸出，避免每次輸出造成 IO 負擔
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }

        // 輸出結果並換行
        System.out.println(sb.toString());

        scanner.close();
    }
}
