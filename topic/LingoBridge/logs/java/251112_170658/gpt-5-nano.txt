/*
檔案名稱: CSIE_41343233_1031_2.java
功能: 讀取整數個數 n 與 n 個整數，輸出其升序排序結果
作者: 教授
日期: 2025-11-12
*/
import java.util.Scanner;
import java.util.Arrays;

public class CSIE_41343233_1031_2 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int n = sc.nextInt();
        int[] arr = new int[n];

        for (int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();
        }

        Arrays.sort(arr);

        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }

        System.out.print(sb.toString());

        sc.close();
    }
}

改進說明
- 使用 Arrays.sort 直接完成排序，避免自己實作排序邏輯容易出錯且效率較低的問題，同時天然支援負數與任意長度的陣列。
