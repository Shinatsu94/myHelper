/*
File: CSIE_41343203_1031_3b.java
描述: 讀入整數 n，接著讀入 n 個整數，輸出升序排序後的結果，以空格分隔。
作者: 您的教學助理
日期: 2025-11-12
*/
import java.util.Scanner;
import java.util.Arrays;

public class CSIE_41343203_1031_3b {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取長度 n
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();
        if (n <= 0) {
            sc.close();
            return;
        }

        // 讀取 n 個整數到一維陣列
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                // 若輸入不足，預設為 0
                arr[i] = 0;
            }
        }

        // 排序：由小到大
        Arrays.sort(arr);

        // 輸出排序結果，元素以空格分隔
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        sc.close();
    }
}

- 透過 Arrays.sort 提升排序可靠性與效能，避免自行實作排序演算法的潛在錯誤。
- 使用 StringBuilder 組裝輸出，減少多次 I/O 呼叫提升效能。
