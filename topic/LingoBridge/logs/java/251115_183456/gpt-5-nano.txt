import java.util.Scanner;

/*
 * File: CSIE_41343212_1017_2.java
 * Description: 判斷使用者輸入的數字各位數字是否嚴格遞增（從左到右）。
 * Author: 教師
 * Date: 2025-11-15
 */

public class CSIE_41343212_1017_2 {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        String num = "";
        if (input.hasNextLine()) {
            num = input.nextLine().trim();
        }

        boolean isWellOrdered = isWellOrdered(num);
        System.out.println(num + " is " + (isWellOrdered ? "well-ordered" : "not well-ordered") + " number.");
        input.close();
    }

    private static boolean isWellOrdered(String s) {
        if (s == null || s.length() == 0) {
            return false;
        }

        // 檢查是否全為數字
        for (int i = 0; i < s.length(); i++) {
            if (!Character.isDigit(s.charAt(i))) {
                return false;
            }
        }

        // 比較相鄰位數是否嚴格遞增
        for (int i = 0; i < s.length() - 1; i++) {
            int a = s.charAt(i) - '0';
            int b = s.charAt(i + 1) - '0';
            if (a >= b) {
                return false;
            }
        }
        return true;
    }
}
