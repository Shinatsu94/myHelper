import java.util.Scanner;

// 檔案名稱: CSIE_20251031_41343252_3.java
// 功能: 建立一個下三角形的二維陣列，並找出使用者指定數字的位置
// 作者: 依需求填寫
// 日期: 2025-11-14
// 說明: 先填充完整的下三角形陣列，之後再搜尋指定數字的位置，若未找到則輸出相應訊息

public class CSIE_20251031_41343252_3 {
    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);
        int height = in.nextInt();
        int target = in.nextInt();

        // 建立下三角形二維陣列
        int[][] tri = new int[height][];
        int count = 0;
        for (int i = 0; i < height; i++) {
            tri[i] = new int[i + 1];
            for (int j = 0; j <= i; j++) {
                count++;
                tri[i][j] = count;
            }
        }

        // 尋找 target 的位置
        int foundRow = -1;
        int foundCol = -1;
        boolean found = false;
        search:
        for (int i = 0; i < height; i++) {
            for (int j = 0; j <= i; j++) {
                if (tri[i][j] == target) {
                    foundRow = i;
                    foundCol = j;
                    found = true;
                    break search; // 同時跳出兩層迴圈
                }
            }
        }

        if (!found) {
            System.out.print("not find " + target + " in array.");
        } else {
            // 輸出格式: target[row, col]，行列皆從 1 開始
            System.out.print(target + "[" + (foundRow + 1) + ", " + (foundCol + 1) + "]");
        }

        in.close();
    }
}

修改部分說明與修改原因
- 將填充與搜尋分離為兩個階段：先完整填充陣列再進行搜尋，確保整個下三角形陣列都被填滿，符合題意「建立完整的下三角形二維陣列」。同時避免在填充過程中因找到目標而中斷造成資料不完整。
- 使用清晰的變數命名與結構：height、target、tri、count 等名稱比原本的 n、x、a、logi、logj 更具說明性。
- 加入程式標頭註解與註解：說明檔案用途、功能、日期，方便閱讀與維護。
- 使用標籤式 break（break search）來一次跳出多層迴圈，若找到目標就立即結束搜尋，提升效率。
- 針對輸出格式保持與原題一致，若找不到輸出 "not find X in array."，若找到輸出 "X[Row, Col]"，索引皆以 1 開始。
