// filename: SortArray.java
// function: 讀取長度 n，接著讀入 n 個整數，進行由小到大排序並輸出排序結果
// author: 你的教授名稱
// date: 2025/11/14
// version: 1.1

import java.util.Scanner;
import java.util.Arrays;

public class SortArray {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        // 讀取陣列長度
        if (!scanner.hasNextInt()) {
            scanner.close();
            return;
        }
        int n = scanner.nextInt();

        // 讀入 n 個整數，並放入陣列
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (scanner.hasNextInt()) {
                arr[i] = scanner.nextInt();
            } else {
                // 若輸入數量不足，補 0 以避免例外；實務上可選擇拋出例外或結束
                arr[i] = 0;
            }
        }
        scanner.close();

        // 由小到大排序
        Arrays.sort(arr);

        // 輸出排序結果，元素以空白分隔
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.print(sb.toString());
    }
}
