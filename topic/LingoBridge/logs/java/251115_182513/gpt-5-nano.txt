/*
 * File: CSIE_41343101_1017_2.java
 * 功能: 判斷輸入整數的各位數字是否從左到右嚴格遞增(well-ordered number)
 * 作者: 學生
 * 日期: 2025-11-15
 */

import java.util.Scanner;

public class CSIE_41343101_1017_2 {
  public static void main(String[] args) {
    Scanner sc = new Scanner(System.in);
    String str = sc.next();
    boolean isWellOrdered = true;

    for (int i = 0; i < str.length() - 1; i++) {
      if (str.charAt(i) >= str.charAt(i + 1)) {
        isWellOrdered = false;
        break;
      }
    }

    if (isWellOrdered) {
      System.out.println(str + " is well-ordered number.");
    } else {
      System.out.println(str + " is not well-ordered number.");
    }

    sc.close();
  }
}
