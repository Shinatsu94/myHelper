/*
File: SelectionSortExample.java
Description: 將任意長度的一維整數陣列進行升序排序，輸入長度與元素，輸出排序後結果
Author: 教師
Date: 2025-11-14
*/

import java.util.Scanner;

public class SelectionSortExample {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);

        // 讀取陣列長度
        if (!input.hasNextInt()) {
            input.close();
            return;
        }
        int n = input.nextInt();
        int[] arr = new int[n];

        // 逐一輸入陣列內容
        for (int i = 0; i < n; i++) {
            arr[i] = input.nextInt();
        }

        // 選擇排序：由小到大
        for (int i = 0; i < n - 1; i++) {
            int minIndex = i;
            for (int j = i + 1; j < n; j++) {
                if (arr[j] < arr[minIndex]) {
                    minIndex = j;
                }
            }
            // 交換
            int tmp = arr[i];
            arr[i] = arr[minIndex];
            arr[minIndex] = tmp;
        }

        // 輸出排序後的陣列，格式與題目要求相符
        if (n > 0) {
            System.out.print(arr[0]);
            for (int i = 1; i < n; i++) {
                System.out.print(" " + arr[i]);
            }
        }
        System.out.println();

        input.close();
    }
}
