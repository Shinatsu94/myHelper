/*
Filename: CSIE_41343205_1031_1.java
Description: 讀入一個整數 n，接著讀入 n 個整數，將其升序排序後輸出，輸出格式為空格分隔且末尾不帶空格。
Author: 程式設計講師
Date: 2025-11-14
*/

import java.util.Scanner;
import java.util.Arrays;

public class CSIE_41343205_1031_1 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int n = sc.nextInt();
        int[] arr = new int[n];

        for (int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();
        }

        // 使用內建排序以提升穩定性與效率
        Arrays.sort(arr);

        // 輸出不留尾端空格
        for (int i = 0; i < arr.length; i++) {
            if (i > 0) System.out.print(" ");
            System.out.print(arr[i]);
        }
        System.out.println();

        sc.close();
    }
}
