import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        // 讀入兩個整數 A, B
        int A = scanner.nextInt();
        int B = scanner.nextInt();

        // 先輸出原始值
        System.out.println(A);
        System.out.println(B);

        // 使用不額外變數的方式交換 A, B
        A = A + B;
        B = A - B;
        A = A - B;

        // 輸出交換後的值
        System.out.println(A);
        System.out.println(B);

        scanner.close();
    }
}
