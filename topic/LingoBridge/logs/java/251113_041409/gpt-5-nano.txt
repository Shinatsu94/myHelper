// 作者：YourName
// 日期：2025-11-12

public class CSIE_41343217_0926_2 {

    public static void main(String[] args) {
        printCodePoints();
    }

    // 依序輸出: 虎、科、大、林、武、杰的內碼
    public static void printCodePoints() {
        String[] firstPart = {"虎", "科", "大"};
        String[] secondPart = {"林", "武", "杰"};

        for (String ch : firstPart) {
            printBig5Code(ch);
        }
        for (String ch : secondPart) {
            printBig5Code(ch);
        }
    }

    // 將單一文字轉為 Big5 編碼的內碼（兩字元組成的整數值）並輸出
    private static void printBig5Code(String ch) {
        try {
            byte[] bytes = ch.getBytes("Big5");
            if (bytes.length == 2) {
                int code = ((bytes[0] & 0xFF) << 8) | (bytes[1] & 0xFF);
                System.out.println(ch + " " + code);
            } else {
                // 若遇到非雙字元組的字，顯示 N/A
                System.out.println(ch + " " + "N/A");
            }
        } catch (java.io.UnsupportedEncodingException e) {
            // 編碼不支援時的保護機制
            System.out.println(ch + " " + "N/A");
        }
    }
}
