/*
 * File: CSIE_20251031_41343252_2.java
 * 功能: 讀取長度與整數，排序後輸出，輸出以空格分隔，避免尾端空格，支援負數。
 * 作者: 請自行填寫
 * 日期: 2025-11-14
 */

import java.util.Arrays;
import java.util.Scanner;

public class CSIE_20251031_41343252_2 {
    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);

        int n = in.nextInt();
        int[] a = new int[n];
        for (int i = 0; i < n; i++) {
            a[i] = in.nextInt();
        }

        // 使用內建排序提升可讀性與效能
        Arrays.sort(a);

        // 輸出，避免尾端多餘的空格，並加上換行
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(a[i]);
        }
        System.out.println(sb.toString());

        in.close();
    }
}
