修改後完整程式碼（可直接存為 PiLeibniz.java）：
/*
  File: PiLeibniz.java
  功能: 使用萊布尼茲公式計算 pi，輸出 i=10000 到 100000，步階為 10000，
        i 顯示六位數，pi 顯示小數點後十位。
  作者: 你的名字
  日期: 2025-11-11
*/

public class PiLeibniz {
    public static void main(String[] args) {
        int maxI = 100000;
        int step = 10000;
        double sum = 0.0;

        // k 從 0 累加到 maxI-1，對應到 i = k+1
        for (int k = 0; k < maxI; k++) {
            // 第 k 項的符號與分母
            double term = ((k % 2 == 0) ? 1.0 : -1.0) / (2.0 * k + 1.0);
            sum += term;

            // 當累加到 i = k+1 等於目前的 step 值時輸出
            if ((k + 1) % step == 0) {
                int i = k + 1;
                double pi = 4.0 * sum;
                System.out.printf("i = %6d, pi = %.10f%n", i, pi);
            }
        }
    }
}
