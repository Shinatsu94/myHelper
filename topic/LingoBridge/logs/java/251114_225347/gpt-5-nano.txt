```java
/*
 * File: SortArray.java
 * 功能: 讀入陣列長度 n，接著讀入 n 個整數，將陣列以升序排序，並輸出排序結果（以空格分隔）。
 * 作者: 你的名字
 * 日期: 2025-11-14
 * 說明: 本程式符合題目要求的一次性輸入、排序與輸出流程。
 */

import java.util.Arrays;
import java.util.Scanner;

public class SortArray {
    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);

        // 讀取陣列長度
        int n = in.nextInt();
        int[] arr = new int[n];

        // 讀取 n 個整數
        for (int i = 0; i < n; i++) {
            arr[i] = in.nextInt();
        }

        // 升序排序
        Arrays.sort(arr);

        // 輸出排序後的陣列，元素之間以空格分隔，末尾不留空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.print(sb.toString());

        in.close();
    }
}
```
