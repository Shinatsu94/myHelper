檔案名稱: Main.java
功能: 讀取整數陣列長度與元素，並輸出升序排序結果
作者: 教師
日期: 2025-11-14

import java.util.*;

public class Main {
    public static void main(String[] args) {
        // 使用 Scanner 取得使用者輸入
        Scanner sc = new Scanner(System.in);

        // 讀取陣列長度 n
        int n = 0;
        if (sc.hasNextInt()) {
            n = sc.nextInt();
        }

        // 讀取 n 個整數到陣列
        int[] a = new int[n];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                a[i] = sc.nextInt();
            } else {
                a[i] = 0; // 若輸入不足，預設為 0，避免例外
            }
        }

        // 對整個陣列做升序排序
        Arrays.sort(a);

        // 輸出排序後的結果，元素以空格分隔
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(a[i]);
        }

        System.out.println(sb.toString());

        sc.close();
    }
}
