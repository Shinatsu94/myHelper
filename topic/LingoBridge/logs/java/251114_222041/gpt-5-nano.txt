import java.util.Arrays;
import java.util.Scanner;

/**
 * 文件名稱: CSIE_41343121_1031_3b.java
 * 功能: 讀取一個整數 n，接著讀取 n 個整數，將它們由小到大排序，並輸出排序後的結果。
 * 作者: 徐唯祐
 * 日期: 10/31
 * 註解: 程式內容包含必要的註解說明與適當的縮排。
 */
public class CSIE_41343121_1031_3b {
    public static void main(String[] args) {
        Scanner scn = new Scanner(System.in);

        // 讀取陣列長度
        int n = 0;
        if (scn.hasNextInt()) {
            n = scn.nextInt();
        }

        // 讀取 n 個整數，若不足以讀取則以 0 作為預設值
        int[] arr = new int[Math.max(0, n)];
        for (int i = 0; i < n; i++) {
            if (scn.hasNextInt()) {
                arr[i] = scn.nextInt();
            } else {
                arr[i] = 0;
            }
        }

        // 排序
        Arrays.sort(arr);

        // 輸出，元素以單一空格分隔，結尾換行
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < arr.length; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        scn.close();
    }
}
