import java.util.*;

public class LeibnizPiCalculator {
    /*
     檔案名稱 : LeibnizPiCalculator.java
     功能     : 使用勒貝格級數近似 π，輸出 i=10000,20000,...,100000 對應的 π 值（精度 10 位小數）
                 同時找出最小的 i，使 pi 的近似值四捨五入到小數點後五位為 3.14159，輸出該 i。
     作者     : 自動評分用範例
     日期     : 2025-11-12
    */
    public static void main(String[] args) {
        final int PRINT_UP_TO = 100000; // 需要輸出 pi 的最大 i
        StringBuilder sb = new StringBuilder();

        // 逐項累加勒貝格級數項，i = k+1
        double sum = 0.0;
        for (int k = 0; k < PRINT_UP_TO; k++) {
            double term = ((k % 2 == 0) ? 1.0 : -1.0) / (2.0 * k + 1.0);
            sum += term;
            int i = k + 1;
            if (i % 10000 == 0) {
                double pi = 4.0 * sum;
                sb.append(String.format("i = %6d, pi = %.10f%n", i, pi));
            }
        }

        // 找出最小的 i，使 pi 四捨五入到小數點後五位等於 3.14159
        sum = 0.0;
        int iMin = -1;
        for (int k = 0; k < 3000000; k++) { // 設一個上限以避免無限迴圈
            double term = ((k % 2 == 0) ? 1.0 : -1.0) / (2.0 * k + 1.0);
            sum += term;
            double pi = 4.0 * sum;
            // 四捨五入至五位小數後若等於 3.14159，則條件成立
            if (Math.round(pi * 100000.0) == 314159) {
                iMin = k + 1;
                break;
            }
        }

        sb.append("\n");
        if (iMin != -1) {
            sb.append(String.format("i = %d%n", iMin));
        } else {
            sb.append("未找到符合條件的 i").append(System.lineSeparator());
        }

        System.out.print(sb.toString());
    }
}
