/*
File: PiLeibniz.java
Description: 使用 Leibniz 公式計算 π，輸出指定的 i 與 π 值，以及找出使 π 四捨五入至小數點後第5位等於 3.14159 的最小 i。
Author: Anonymous
Date: 2025-11-11
*/

public class PiLeibniz {
    public static void main(String[] args) {
        // Part (a): 印出 i = 10000, 20000, ..., 100000 對應的 pi 值
        double sum = 0.0;
        int maxI = 100000;
        for (int i = 0; i < maxI; i++) {
            double term = (i % 2 == 0) ? (1.0 / (2 * i + 1)) : (-1.0 / (2 * i + 1));
            sum += term;
            int terms = i + 1;
            if (terms % 10000 == 0) {
                double pi = 4.0 * sum;
                System.out.printf("i = %6d, pi = %.10f%n", terms, pi);
            }
        }

        // Part (b): 找出使 pi 四捨五入至小數點後第5位等於 3.14159 的最小 i
        final double TARGET = 3.14159;
        sum = 0.0;
        int minI = -1;
        int upperBound = 2000000; // 適度上限，預期在此範圍內可找到
        for (int i = 0; i < upperBound; i++) {
            double term = (i % 2 == 0) ? (1.0 / (2 * i + 1)) : (-1.0 / (2 * i + 1));
            sum += term;
            double pi = 4.0 * sum;
            double rounded = Math.round(pi * 1e5) / 1e5; // 四捨五入到小數點後第5位
            if (rounded == TARGET) {
                minI = i + 1;
                break;
            }
        }

        if (minI != -1) {
            System.out.printf("i = %6d%n", minI);
        } else {
            System.out.println("i = not found within limit");
        }
    }
}
