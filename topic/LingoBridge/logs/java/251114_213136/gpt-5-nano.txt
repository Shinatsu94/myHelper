檔案名稱: CSIE_41343208_1031_2.java
功能說明: 讀入任意長度整數陣列，將元素由小到大排序，並輸出排序後的內容。
作者: (你的名字)
日期: (日期)
註解: 包含程式標頭與必要的註解，說明每段程式的用途。
*/

import java.util.Arrays;
import java.util.Scanner;

public class CSIE_41343208_1031_2 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取陣列長度
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();

        // 讀取陣列元素
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                // 若輸入不足，提前退出
                break;
            }
        }

        // 對陣列排序，處理負數也沒問題
        Arrays.sort(arr);

        // 輸出排序後的結果，元素間用空格分隔，末端換行
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        sc.close();
    }
}
