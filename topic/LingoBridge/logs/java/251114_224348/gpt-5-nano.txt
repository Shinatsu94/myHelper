/*
檔案名稱: SortArrayAsc.java
功能說明: 讀取整數長度 n，接著讀入 n 個整數，輸出其升序排序結果（由小到大）。
作者: 你的姓名
日期: 2025-11-14
說明: 使用 Arrays.sort 進行排序，支援負數。一次讀入，不使用 EOF。
*/
import java.util.Arrays;
import java.util.Scanner;

public class SortArrayAsc {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取長度 n
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();
        if (n <= 0) {
            sc.close();
            return;
        }

        // 讀入 n 個整數
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                // 若輸入不足，預設為 0（防止例外，可依需求調整）
                arr[i] = 0;
            }
        }
        sc.close();

        // 升序排序
        Arrays.sort(arr);

        // 輸出，元素以空格分隔，最後不留尾隨空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());
    }
}
