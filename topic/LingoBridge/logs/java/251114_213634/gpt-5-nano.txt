import java.util.Scanner;
import java.util.Arrays;

 /*
  File: CSIE_41343220_1031_2.java
  功能: 讀取一維整數陣列，將其內容以遞增排序輸出
  作者: 你的名字
  日期: 2025-11-14
 */

public class CSIE_41343220_1031_2 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取陣列長度
        if (!sc.hasNextInt()) {
            return;
        }
        int n = sc.nextInt();

        int[] data = new int[n];

        // 讀取陣列元素
        for (int i = 0; i < n; i++) {
            data[i] = sc.hasNextInt() ? sc.nextInt() : 0;
        }

        // 進行升序排序
        Arrays.sort(data);

        // 輸出排序結果，元素之間以空格分隔
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(data[i]);
        }
        System.out.print(sb.toString());
    }
}
