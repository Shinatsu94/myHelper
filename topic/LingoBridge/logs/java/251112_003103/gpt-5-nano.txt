/**
 * 檔名: Main.java
 * 功能: 讀取兩個整數，交換其值後輸出交換後的結果
 * 作者: 請自行填寫
 * 日期: 2025-11-11
 * 說明: 本程式採用位元運算的 XOR 方式交換 a 與 b，且不使用額外變數。
 */
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        // 使用 try-with-resources 以確保 Scanner 會被自動關閉
        try (Scanner sc = new Scanner(System.in)) {
            int a = sc.nextInt();
            int b = sc.nextInt();

            // 不使用其他變數，透過 XOR 交換 a 與 b
            a = a ^ b;
            b = a ^ b;
            a = a ^ b;

            System.out.println(a);
            System.out.println(b);
        }
    }
}
