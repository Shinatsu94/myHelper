// 文件名稱: CSIE_41343227_1031_3b.java
// 功能: 讀取長度 n，接著讀入 n 個整數，排序後輸出由小到大的序列
// 作者: 你的姓名
// 日期: 2025-11-12
// 備註: 不使用 EOF，執行一次程式即可結束

import java.util.Arrays;
import java.util.Scanner;

public class CSIE_41343227_1031_3b {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取長度 n
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();

        // 讀入 n 個整數
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                arr[i] = 0; // 若輸入不足，預設為 0（可依需求調整）
            }
        }

        // 排序
        Arrays.sort(arr);

        // 輸出，元素之間以空白分隔，結尾換行
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.print(sb.toString());

        sc.close();
    }
}

- 將原本不相關的 2D char 陣列與 '*' 圖樣移除，改為整數陣列並進行排序，直接符合題目要求的輸入/輸出格式。
- 使用 Arrays.sort 提供穩定且簡潔的排序解法，提升正確性與邏輯清晰度。
- 增加程式標頭與註解，提升可讀性與維護性。
