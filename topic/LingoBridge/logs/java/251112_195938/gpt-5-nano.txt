修改後完整程式碼
/*
 * File: CSIE_41343202_1031_3A.java
 * Description: 從輸入讀取一個長度 n，接著讀入 n 個整數，將它們升冪排序並輸出。
 * Author: Student
 * Date: 2025-11-12
 */

import java.util.Scanner;
import java.util.Arrays;

public class CSIE_41343202_1031_3A{
    public static void main(String[] args){
        Scanner scanner = new Scanner(System.in);

        // 讀取陣列長度
        if (!scanner.hasNextInt()) {
            scanner.close();
            return;
        }
        int n = scanner.nextInt();

        // 讀取 n 個整數，放入陣列
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (scanner.hasNextInt()) {
                arr[i] = scanner.nextInt();
            } else {
                // 若輸入不足，避免例外，預設為 0
                arr[i] = 0;
            }
        }

        // 排序
        Arrays.sort(arr);

        // 輸出，數字間以單一空格分隔，最後換行
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        scanner.close();
    }
}
