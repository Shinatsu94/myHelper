import java.util.Scanner;

/*
檔案名稱: CSIE_41343128_1031_3a_opt.java
功能: 建立並列印下三角形的二維結構，但實際輸出直接逐行產生星號，避免額外陣列儲存。
作者: 教師
日期: 2025-11-14
說明: 試題要求輸出由上到下、每行星號數量遞增的下三角形；此版本避免使用不必要的 2D 陣列，僅直接輸出每行字串。
*/

public class CSIE_41343128_1031_3a_opt {
    public static void main(String[] args) {
        Scanner scn = new Scanner(System.in);

        int n = 0;
        if (scn.hasNextInt()) {
            n = scn.nextInt();
        }

        // 檢查輸入是否在允許範圍內
        if (n <= 0 || n > 20) {
            scn.close();
            return; // 不輸出任何內容
        }

        // 逐行輸出下三角形星號，避免不必要的陣列儲存
        for (int i = 1; i <= n; i++) {
            StringBuilder line = new StringBuilder(i);
            for (int j = 0; j < i; j++) {
                line.append('*');
            }
            System.out.println(line.toString());
        }

        scn.close();
    }
}
