/*
 * 檔案名稱: CSIE_41343216_1031_1.java
 * 功能: 讀取長度與整數陣列，將其升序排序後輸出
 * 作者: 學生
 * 日期: 2025-11-14
 */

import java.util.Scanner;
import java.util.Arrays;

public class CSIE_41343216_1031_1 {
    public static void main(String[] args) {
        // 使用 try-with-resources 以確保 Scanner 會自動關閉
        try (Scanner input = new Scanner(System.in)) {
            int length = input.nextInt();      // 讀取陣列長度
            int[] numbers = new int[length];   // 建立陣列

            for (int i = 0; i < length; i++) {
                numbers[i] = input.nextInt();  // 逐一讀入整數
            }

            Arrays.sort(numbers);              // 由小到大排序

            // 一次性組裝輸出，避免多次輸出造成的效能問題
            StringBuilder sb = new StringBuilder();
            for (int i = 0; i < length; i++) {
                if (i > 0) sb.append(' ');
                sb.append(numbers[i]);
            }

            System.out.println(sb.toString()); // 輸出排序結果，並換行
        }
    }
}
