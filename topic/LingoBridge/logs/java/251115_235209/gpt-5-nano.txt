```java
/*
檔案名稱: CSIE_41343252_1003.java
功能說明: 讀入三個整數，使用兩種不同的方式（兩個三元運算子求最大與最小，並用總和減去最大與最小得到中位數）找出最大值、最小值與中位數，並以 min、mid、max 的順序輸出。
作者: 教授
日期: 2025-11-15
*/
import java.util.Scanner;

public class CSIE_41343252_1003 {
    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);
        int a = in.nextInt();
        int b = in.nextInt();
        int c = in.nextInt();

        // 使用三元運算子分別找出最大值與最小值
        int maxVal = (a > b ? (a > c ? a : c) : (b > c ? b : c));
        int minVal = (a < b ? (a < c ? a : c) : (b < c ? b : c));

        // 中位數透過 sum - max - min，為避免溢位，轉為 long 計算再轉回 int
        long sum = (long) a + b + c;
        int midVal = (int) (sum - maxVal - minVal);

        // 輸出順序依題意為 min, mid, max
        System.out.println(minVal);
        System.out.println(midVal);
        System.out.println(maxVal);

        in.close();
    }
}
```
