學生作答程式：
/* file name: CSIE_41343212_0926_1.java
   student: 李品劭
   ID: 41343212
   function: to print the code of Chinese word
 */
public class CSIE_41343212_0926_1 {
    public static void main(String[] args) {
        char firstName = '李';
        char middleName = '品';
        char lastName = '劭';
        char firstSchoolName = '虎';
        char middleSchoolName = '科';
        char lastSchoolName = '大';

        System.out.println("李的內碼 = " + (int) firstName);
        System.out.println("品的內碼 = " + (int) middleName);
        System.out.println("劭的內碼 = " + (int) lastName);

        System.out.println("虎的內碼 = " + (int) firstSchoolName);
        System.out.println("科的內碼 = " + (int) middleSchoolName);
        System.out.println("大的內碼 = " + (int) lastSchoolName);
    }
}
