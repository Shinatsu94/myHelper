/*
檔案名稱: CSIE_41343241_1031_1.java
功能: 讀入長度與 N 個整數，將數列由小到大排序，輸出排序結果
作者: 學生
日期: 2025-11-14
*/

import java.util.Scanner;
import java.util.Arrays;

public class CSIE_41343241_1031_1 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();
        }

        Arrays.sort(arr);

        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());
        sc.close();
    }
}
