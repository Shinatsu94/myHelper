修改後完整程式碼
```java
/*
檔案名稱: CSIE_51015114_1031_3.java
功能: 讀取長度與整數序列，升序排序並輸出
作者: 你的姓名
日期: 2025-11-14
*/

import java.util.Scanner;
import java.util.Arrays;

public class CSIE_51015114_1031_3 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取陣列長度
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();

        // 讀取 n 個整數
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                arr[i] = 0; // 若輸入不足，預設為 0
            }
        }
        sc.close();

        // 排序
        Arrays.sort(arr);

        // 輸出：單行、數字之間以空格分隔
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());
    }
}
```
