import java.util.Scanner;

public class SwapTwoNumbers {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int A = sc.nextInt();
        int B = sc.nextInt();

        // 輸出交換前的原始值
        System.out.println(A);
        System.out.println(B);

        // 不使用額外變數，利用加減法進行交換
        A = A + B;
        B = A - B;
        A = A - B;

        // 輸出交換後的值
        System.out.println(A);
        System.out.println(B);

        sc.close();
    }
}
