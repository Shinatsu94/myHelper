/* 
 * File: SortArrayAscending.java
 * 功能: 讀取整數個數 n，接著讀入 n 個整數，將它們升序排序，並印出排序後的結果 (以單行、以空格分隔)
 * 作者: 教學用範例
 * 日期: 2025-11-14
 */
import java.util.*;

public class SortArrayAscending {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        
        // 讀取陣列長度
        if (!sc.hasNextInt()) {
            return;
        }
        int n = sc.nextInt();
        if (n <= 0) {
            // 若長度不正，直接結束
            return;
        }

        // 讀入 n 個整數
        int[] numbers = new int[n];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                numbers[i] = sc.nextInt();
            } else {
                // 若輸入不足，預設為 0
                numbers[i] = 0;
            }
        }

        // 排序
        Arrays.sort(numbers);

        // 輸出成單行，元素之間以空格分隔，避免尾端空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(numbers[i]);
        }
        System.out.print(sb.toString());
    }
}
