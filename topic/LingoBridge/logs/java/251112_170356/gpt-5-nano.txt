檔案名稱: CSIE_41343227_1031_1.java
功能: 讀取整數 n 與 n 個整數，對陣列進行升序排序，輸出排序結果
作者: 
日期: 2025-11-12
*/
import java.util.*;

public class CSIE_41343227_1031_1 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();
        }

        // 使用 Arrays.sort 提升效率與穩定性
        Arrays.sort(arr);

        // 輸出排序結果，元素以空格分隔
        for (int i = 0; i < arr.length; i++) {
            if (i > 0) System.out.print(' ');
            System.out.print(arr[i]);
        }
        System.out.println();
        sc.close();
    }
}
