/**
 * File: SortArrayAsc.java
 * 功能: 讀取使用者輸入的長度 n，接著讀取 n 個整數，將它們升序排序，並輸出排序結果（以空白分隔）。
 * 作者: 教師
 * 日期: 2025-11-12
 */

import java.util.Scanner;
import java.util.Arrays;

public class SortArrayAsc {
    public static void main(String[] args) {
        // 讀取使用者輸入
        Scanner sc = new Scanner(System.in);

        // 讀取陣列長度
        int n = sc.nextInt();

        // 讀取 n 個整數放入陣列
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();
        }

        // 將陣列排序（升序）
        Arrays.sort(arr);

        // 輸出排序後的內容，以空白分隔
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        // 關閉資源
        sc.close();
    }
}

- 改用清晰的變數命名與結構：讀取長度 n、讀取 n 個整數到陣列、排序、再輸出，提升可讀性與可維護性。
