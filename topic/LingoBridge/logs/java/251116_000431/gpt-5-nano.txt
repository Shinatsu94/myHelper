```java
/*
 * File: CSIE_41343203_1003_3.java
 * Description: 找出 1~100000 中，數字各位數字總和等於輸入值 A 的數字個數
 * Author: 學生
 * Date: 2025-11-15
 * Notes: 已修正命名遮蔽、迴圈範圍與註解，提升可讀性與正確性。
 */

import java.util.Scanner;

public class CSIE_41343203_1003_3 {

    // 計算整數 n 的各位數字總和，例如 123 -> 1+2+3 = 6
    public static int sumDigits(int n) {
        int sum = 0;
        while (n > 0) {
            sum += n % 10;
            n /= 10;
        }
        return sum;
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 目標和：輸入值 A
        int target = sc.nextInt();

        int count = 0;
        // 遍歷 1 ~ 100000，符合條件則計數
        for (int i = 1; i <= 100000; i++) {
            if (sumDigits(i) == target) {
                count++;
            }
        }

        System.out.println(count);
        sc.close();
    }
}
```

- 修改 1：將輸入變數名稱從 a 改為 target，並將計算函式改名為 sumDigits，避免與變數同名造成遮蔽與編譯錯誤。
- 修改 2：將計算函式分離為獨立靜態方法 sumDigits，增加可重用性與可讀性。
- 修改 3：迴圈範圍改為 for (i = 1; i <= 100000; i++)，符合題目要求與避免包含 0。
- 修改 4：增加程式標頭註解與內部註解，提升可維護性與可讀性。
