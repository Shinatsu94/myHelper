/*
 * 文件名: CSIE_41343208_1031_3b_sorted.java
 * 功能: 讀取長度 n，接著讀取 n 個整數，輸出升冪排序結果
 * 作者: 
 * 日期: 2025-11-14
 */

import java.util.Arrays;
import java.util.Scanner;

public class CSIE_41343208_1031_3b_sorted {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }

        int n = sc.nextInt();
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                // 若輸入不足，預設為 0
                arr[i] = 0;
            }
        }

        // 對整數陣列進行升冪排序
        Arrays.sort(arr);

        // 輸出排序後的結果，數字以空格分隔
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        sc.close();
    }
}
