/*
 * 檔案名稱: CSIE_41343112_1017_3.java
 * 功能: 將使用者輸入的整數之各位數字反轉，正負號保持原本符號；輸出不保留前導零。
 * 作者: 自動評測系統
 * 日期: 2025-11-15
 */

import java.util.Scanner;

public class CSIE_41343112_1017_3 {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        long n;
        if (scanner.hasNextLong()) {
            n = scanner.nextLong();
        } else {
            scanner.close();
            return;
        }
        scanner.close();

        // 處理正負號：先對絕對值反轉，最終再乘回符號
        long sign = (n < 0) ? -1 : 1;
        long m = Math.abs(n);
        long rev = 0;

        while (m > 0) {
            rev = rev * 10 + (m % 10);
            m /= 10;
        }

        rev *= sign;

        System.out.println(rev);
    }
}
