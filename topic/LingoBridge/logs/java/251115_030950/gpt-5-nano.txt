/**
 * File: LowerTriangularMatrixSearch.java
 * 功能: 建立下三角形二維陣列，輸入高度 n，從 1 開始填入數值，尋找使用者指定的數字與其位置，未在陣列中時輸出提示。
 * 作者: 學生
 * 日期: 2025-11-14
 */

import java.util.Scanner;

public class LowerTriangularMatrixSearch {
    public static void main(String[] args) {
        Scanner scn = new Scanner(System.in);
        int n = scn.nextInt();
        int target = scn.nextInt();

        int[][] list = new int[n][];
        for (int i = 0; i < n; i++) {
            list[i] = new int[i + 1];
        }

        int val = 1;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= i; j++) {
                list[i][j] = val++;
            }
        }

        boolean found = false;
        int foundRow = -1;
        int foundCol = -1;

        outer:
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= i; j++) {
                if (list[i][j] == target) {
                    found = true;
                    foundRow = i;
                    foundCol = j;
                    break outer;
                }
            }
        }

        if (found) {
            System.out.print(target + "[" + (foundRow + 1) + ", " + (foundCol + 1) + "]");
        } else {
            System.out.print("not find " + target + " in array.");
        }

        System.out.println();
        scn.close();
    }
}
