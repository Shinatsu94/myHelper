Author: 你的名字

public class SwapTwoNumbers {
    public static void main(String[] args) {
        try (Scanner scanner = new Scanner(System.in)) {
            int A = scanner.nextInt();
            int B = scanner.nextInt();

            // 不使用額外變數，利用加法交換
            A = A + B;
            B = A - B;
            A = A - B;

            System.out.println(A);

        }
    }
}
