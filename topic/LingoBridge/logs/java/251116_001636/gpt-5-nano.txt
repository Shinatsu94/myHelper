- 變數命名改為 targetSum 與 count，以提升語意清晰度。

修改後的完整程式碼如下：

/*
 * 文件名稱: CSIE_41343240_1003_3.java
 * 功能: 輸入整數 A，統計在 1..100000 之間，數字各位數字總和等於 A 的數字個數
 * 作者: 黃榮裕 (41343240)
 * 日期: 2025-10-03
 * 備註: 包含對輸入的基本檢查與效能優化
 */

import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.io.IOException;

public class CSIE_41343240_1003_3 {
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        String line = br.readLine();
        if (line == null) {
            System.out.println(0);
            return;
        }
        line = line.trim();
        if (line.isEmpty()) {
            System.out.println(0);
            return;
        }
        int targetSum;
        try {
            targetSum = Integer.parseInt(line);
        } catch (NumberFormatException e) {
            System.out.println(0);
            return;
        }

        // 若 targetSum 超過 45，因 1..99999 最大位數之和為 45；100000 的位數和為 1
        if (targetSum <= 0 || targetSum > 45) {
            System.out.println(0);
            return;
        }

        int count = 0;
        for (int i = 1; i <= 100000; i++) {
            if (digitSum(i) == targetSum) {
                count++;
            }
        }
        System.out.println(count);
    }

    private static int digitSum(int n) {
        int sum = 0;
        while (n > 0) {
            sum += n % 10;
            n /= 10;
        }
        return sum;
    }
}
