/*
 * 檔案名稱: CSIE_41343216_1031_1.java
 * 功能說明: 讀入整數陣列長度 n，接著讀入 n 個整數，將陣列由小到大排序，並輸出結果（以空白分隔）。
 * 作者: 你的姓名
 * 日期: 2025-11-12
 *
 * 使用說明: 一次性執行程序，輸出排序結果，元素之間以單一空白分隔。
 */

import java.util.Scanner;
import java.util.Arrays;

public class CSIE_41343216_1031_1 {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);

        int n = input.nextInt();
        int[] arr = new int[n];

        for (int i = 0; i < n; i++) {
            arr[i] = input.nextInt();
        }

        Arrays.sort(arr);

        if (n > 0) {
            StringBuilder sb = new StringBuilder();
            for (int i = 0; i < n; i++) {
                if (i > 0) sb.append(' ');
                sb.append(arr[i]);
            }
            System.out.println(sb.toString());
        }

        input.close();
    }
}
