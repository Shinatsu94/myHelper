修改後完整程式碼
/* 
 * File: CSIE_41343209_1031_2.java
 * 功能: 讀入長度 n 與 n 個整數，將陣列內容由小到大排序，並印出排序後結果。
 * 作者: 教師
 * 日期: 2025-11-12
 */
import java.util.*;

public class CSIE_41343209_1031_2 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }

        int n = sc.nextInt();
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                arr[i] = 0; // 若輸入不足，預設為 0
            }
        }

        // 使用內建排序，從小到大
        Arrays.sort(arr);

        // 輸出，確保不出現尾端多餘空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        sc.close();
    }
}
