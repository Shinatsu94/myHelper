```java
/*
 * 檔案名稱：CSIE_41343101_1031_3a.java
 * 功能說明：讀入一個整數 n，接著讀入 n 個整數，將它們升序排序並輸出。
 * 作者：教授
 * 日期：2025-11-14
 * 備註：不使用 EOF，程式執行一次後結束。
 */

import java.util.Arrays;
import java.util.Scanner;

public class CSIE_41343101_1031_3a {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取長度
        int n = sc.nextInt();
        int[] arr = new int[Math.max(0, n)]; // 保證不為負長度

        // 讀入元素
        for (int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();
        }

        // 排序
        Arrays.sort(arr);

        // 輸出排序結果，數字以單一行、空格分隔
        for (int i = 0; i < arr.length; i++) {
            if (i > 0) System.out.print(" ");
            System.out.print(arr[i]);
        }
        System.out.println();

        sc.close();
    }
}
```
