public class SwapTwoNumbers {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀入兩個整數，分別存放於 A 與 B
        int A = sc.nextInt();
        int B = sc.nextInt();

        // 使用 XOR 交換法，避免使用額外變數
        A = A ^ B;
        B = A ^ B;
        A = A ^ B;

        // 輸出交換後的結果，每行一個
        System.out.println(A);
        System.out.println(B);

        sc.close();
    }
}
