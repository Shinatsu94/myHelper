```java
/*
 * File: LowerTriangularStars.java
 * Description: 建立並列印下三角形的二維不等長陣列（jagged array），
 *              每行包含自下而上的星號數量：第 i 行有 i+1 個 '*'
 * Author: 教師資訊
 * Date: 2025-11-14
 * Notes: 0 < n <= 20
 */

import java.util.Scanner;

public class LowerTriangularStars {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt(); // 讀入高度
        sc.close();

        // 建立 jagged array：第 i 行有 i+1 個元素
        char[][] arr = new char[n][];
        for (int i = 0; i < n; i++) {
            arr[i] = new char[i + 1];
            for (int j = 0; j <= i; j++) {
                arr[i][j] = '*';
            }
        }

        // 輸出下三角形陣列
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= i; j++) {
                sb.append(arr[i][j]);
            }
            if (i < n - 1) sb.append('\n');
        }
        System.out.print(sb.toString());
    }
}
```
