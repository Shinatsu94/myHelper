二、修改後完整程式碼
/*
 filename: ArraySortAscending.java
 功能: 讀入整數 n 代表長度，接著讀入 n 個整數，將它們升冪排序並輸出，使用單行輸出，以空格分隔。
 作者: 你的名字
 日期: 2025-11-12
*/
import java.util.Arrays;
import java.util.Scanner;

public class ArraySortAscending {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取長度 n
        if (!sc.hasNextInt()) {
            return;
        }
        int n = sc.nextInt();
        if (n <= 0) {
            // 對於非正長度，直接結束
            return;
        }

        // 讀取 n 個整數並放入陣列
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                // 若輸入不足，補 0 以避免例外；實務上應該提示錯誤或終止
                arr[i] = 0;
            }
        }

        // 排序
        Arrays.sort(arr);

        // 輸出：單行、元素以空格分隔
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.print(sb.toString());
    }
}
