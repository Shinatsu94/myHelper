/*
檔案名稱: SortArrayAsc.java
功能: 讀入長度 n 與 n 個整數，對陣列做升冪排序，輸出排序後的陣列內容（以空格分隔，單行輸出）
作者: 金宗翰
日期: 2025-11-14
*/

import java.util.Scanner;
import java.util.Arrays;

public class SortArrayAsc {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取陣列長度
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();
        int[] arr = new int[n];

        // 讀取 n 個整數
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                // 若輸入不足，預設為 0，避免 ArrayIndexOutOfBounds
                arr[i] = 0;
            }
        }

        // 排序
        Arrays.sort(arr);

        // 輸出，單行、空格分隔，無尾端空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.println(sb.toString());

        sc.close();
    }
}
