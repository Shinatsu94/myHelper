public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int A = sc.nextInt();
        int B = sc.nextInt();

        // 使用不新增變數的交換方式：位元異或交換
        // A = A ^ B; B = A ^ B; A = A ^ B;
        // 若 A、B 為整數，仍能正常完成交換

        B = A ^ B;
        A = A ^ B;

        System.out.println(A);
        System.out.println(B);

        sc.close();
    }
}
