// 檔案名稱: CSIE_41343136_1031_3a.java
// 功能: 依題意讀取整數長度與數值，進行升序排序，輸出排序結果
// 作者: 您的教師姓名
// 日期: 2025-11-14
import java.util.Arrays;
import java.util.Scanner;

public class CSIE_41343136_1031_3a {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取陣列長度
        if (!sc.hasNextInt()) {
            // 無可讀取的整數，結束程序
            return;
        }
        int n = sc.nextInt();

        // 輕量防呆：若 n <= 0，直接輸出空行或不輸出內容
        if (n <= 0) {
            System.out.println();
            sc.close();
            return;
        }

        int[] arr = new int[n];

        // 依序讀入 n 個整數
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                // 不足的輸入，以 0 代替；也可選擇其他預設策略
                arr[i] = 0;
            }
        }

        // 升序排序
        Arrays.sort(arr);

        // 輸出排序結果，空格分隔且不在末尾多餘空格
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.print(sb.toString());

        sc.close();
    }
}
