public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        int A = scanner.nextInt();
        int B = scanner.nextInt();

        // 先印出原值
        System.out.println(A);
        System.out.println(B);

        // 不使用額外變數，直接在 A, B 上交換值
        A = A ^ B;
        B = A ^ B;
        A = A ^ B;

        // 印出交換後的值
        System.out.println(A);
        System.out.println(B);

        scanner.close();
    }
}
