/*
 * 檔案名稱: Main.java
 * 功能說明: 讀取使用者輸入的長度與長度個整數，將陣列內容由小到大排序，並輸出排序結果。
 * 作者: ChatGPT
 * 日期: 2025-11-14
 * 註解: 程式中保留基本註解以說明流程與邏輯，維持可讀性。
 */

import java.util.Arrays;
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取陣列長度
        if (!sc.hasNextInt()) {
            // 沒有輸入時，輸出空行並結束
            System.out.println();
            return;
        }
        int len = sc.nextInt();

        if (len <= 0) {
            // 長度為0或負數時，輸出空行並結束
            System.out.println();
            return;
        }

        int[] nums = new int[len];

        // 讀取陣列元素
        for (int i = 0; i < len; i++) {
            if (sc.hasNextInt()) {
                nums[i] = sc.nextInt();
            } else {
                // 不足的情況，補上 0 以穩健處理
                nums[i] = 0;
            }
        }

        // 對陣列進行排序
        Arrays.sort(nums);

        // 一次性輸出排序結果，避免多次 I/O
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < len; i++) {
            if (i > 0) sb.append(' ');
            sb.append(nums[i]);
        }
        System.out.println(sb.toString());

        // 不強制關閉 System.in，以避免某些測資環境之影響
        // sc.close();
    }
}
