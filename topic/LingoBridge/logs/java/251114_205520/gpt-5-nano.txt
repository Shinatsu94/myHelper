/*
檔案名稱: SortArrayAscending.java
功能說明: 讀取任意長度整數陣列，將內容升序排序，輸出排序結果
作者: 你選擇的名字
日期: 2025-11-14
*/
import java.util.Scanner;

public class SortArrayAscending {
    public static void main(String[] args) {
        // 使用 try-with-resources 以確保 Scanner 正確關閉
        try (Scanner sc = new Scanner(System.in)) {
            if (!sc.hasNextInt()) {
                // 無有效輸入時結束
                return;
            }

            int n = sc.nextInt();
            int[] arr = new int[n];

            for (int i = 0; i < n; i++) {
                arr[i] = sc.nextInt();
            }

            // 選擇排序：由小到大
            for (int i = 0; i < arr.length - 1; i++) {
                int minIdx = i;
                for (int j = i + 1; j < arr.length; j++) {
                    if (arr[j] < arr[minIdx]) {
                        minIdx = j;
                    }
                }
                // 交換 arr[i] 與 arr[minIdx]
                int tmp = arr[i];
                arr[i] = arr[minIdx];
                arr[minIdx] = tmp;
            }

            // 輸出：使用單一字串拼接，避免尾端多餘空格
            StringBuilder sb = new StringBuilder();
            for (int i = 0; i < arr.length; i++) {
                if (i > 0) sb.append(' ');
                sb.append(arr[i]);
            }
            System.out.println(sb.toString());
        }
    }
}
