/*
File: CSIE_41343226_1031_3B.java
功能: 建立下三角二維陣列，填入 1..n(n+1)/2，尋找指定數字的座標
作者: 作者姓名
日期: 2025-11-14
*/
import java.util.Scanner;

public class CSIE_41343226_1031_3B {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int height = sc.nextInt();  // 陣列高度 n
        int target = sc.nextInt();  // 要尋找的數字

        int[][] tri = new int[height][height];

        // Fill lower-triangular part with sequential numbers
        int val = 1;
        for (int r = 0; r < height; r++) {
            for (int c = 0; c <= r; c++) {
                tri[r][c] = val++;
            }
        }

        boolean found = false;
        int foundRow = -1;
        int foundCol = -1;

        // Search for target
        for (int r = 0; r < height; r++) {
            for (int c = 0; c <= r; c++) {
                if (tri[r][c] == target) {
                    found = true;
                    foundRow = r + 1; // 1-based
                    foundCol = c + 1; // 1-based
                    break;
                }
            }
            if (found) break;
        }

        if (found) {
            System.out.println(target + "[" + foundRow + ", " + foundCol + "]");
        } else {
            System.out.println("not find " + target + " in array.");
        }

        sc.close();
    }
}
