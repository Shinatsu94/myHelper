File: WellOrderedChecker.java
Description: Determine if input number's digits are strictly increasing from left to right.

import java.util.Scanner;

public class WellOrderedChecker {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        String input = sc.nextLine().trim();
        sc.close();

        boolean isWellOrdered = isWellOrdered(input);
        if (isWellOrdered) {
            System.out.println(input + " is well-ordered number.");
        } else {
            System.out.println(input + " is not well-ordered number.");
        }
    }

    private static boolean isWellOrdered(String s) {
        if (s == null || s.isEmpty()) return false;
        // 這裡假設只接受正整數；若遇到負號直接判定為非遞增數
        if (s.charAt(0) == '-') return false;

        // 確認全為數字
        for (int i = 0; i < s.length(); i++) {
            char c = s.charAt(i);
            if (c < '0' || c > '9') return false;
        }

        // 檢查從左到右是否嚴格遞增
        for (int i = 0; i < s.length() - 1; i++) {
            if (s.charAt(i) >= s.charAt(i + 1)) {
                return false;
            }
        }
        return true;
    }
}
