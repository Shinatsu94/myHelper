/*
檔名: LeibnizPi.java
功能: 使用勒布尼茲級數近似 pi，輸出 i = 10000, 20000, ..., 100000 對應的 pi 值
      pi 的格式為小數點後 10 位，i 顯示為 6 位寬度右對齊。
      同時找出使 pi 四捨五入為 3.14159 的最小 i，並輸出該 i。
作者: [你的名字]
日期: 2025-11-11
說明: 本程式直接使用迴圈累加勒布尼茲級數的項，逐步計算並輸出要求的格式。
*/
public class LeibnizPi {
    public static void main(String[] args) {
        int maxPrintI = 100000;      // 要輸出到的最大 i（包含 100000）
        int printStep = 10000;         // 每 10000 顯示一次結果
        double sum = 0.0;              // 累加和，為 4 倍的近似 pi
        int firstFiveDecimalI = -1;    // 找到的最小 i，使 pi 四捨五入為 3.14159
        int limit = 200000;            // 計算上限，足以找到符合條件的 i（例如 136121）

        // 從 i = 1 開始累加項
        for (int i = 1; i <= limit; i++) {
            // 第 i 項：(+1)/(2i-1) 或 (-1)/(2i-1)
            double term = (i % 2 == 1) ? 1.0 / (2.0 * i - 1.0)
                                       : -1.0 / (2.0 * i - 1.0);
            sum += term;

            // 輸出要求：i 為 10000, 20000, ..., 100000，pi 四捨五入到小數點後 10 位
            if (i % printStep == 0 && i <= maxPrintI) {
                double pi = 4.0 * sum;
                System.out.println("i = " + String.format("%6d", i) +
                                   ", pi = " + String.format("%0.10f", pi));
            }

            // 找出使 pi 四捨五入為 3.14159 的最小 i
            long roundedToFive = Math.round(4.0 * sum * 100000); // pi * 100000 的四捨五入值
            if (firstFiveDecimalI == -1 && roundedToFive == 314159) {
                firstFiveDecimalI = i;
                // 找到後繼續執行，等到最後再輸出
            }

            // 一旦找到了最小 i，且已超過最後的輸出區段，即可結束計算
            if (firstFiveDecimalI != -1 && i > maxPrintI) {
                break;
            }
        }

        // 輸出最小的 i（不輸出 pi 值，只輸出 i 本身）
        System.out.println("i = " + firstFiveDecimalI);
    }
}
