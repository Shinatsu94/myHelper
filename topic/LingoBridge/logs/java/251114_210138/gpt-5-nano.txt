/*
File: Main.java
功能: 讀入整數 n 及 n 個整數，對陣列進行升序排序，輸出排序結果。
作者: 你的姓名
日期: 2025-11-14
說明: 使用 Arrays.sort 取代手動排序以提升可讀性與穩定性。
*/

import java.util.Scanner;
import java.util.Arrays;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 讀取長度 n
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }
        int n = sc.nextInt();

        // 若 n <= 0，直接輸出換行並結束
        if (n <= 0) {
            System.out.println();
            sc.close();
            return;
        }

        int[] arr = new int[n];
        // 讀取 n 個整數
        for (int i = 0; i < n; i++) {
            if (sc.hasNextInt()) {
                arr[i] = sc.nextInt();
            } else {
                // 若輸入不足，預設為 0
                arr[i] = 0;
            }
        }

        // 升序排序
        Arrays.sort(arr);

        // 輸出排序後的陣列，元素間以單一空格分隔
        for (int i = 0; i < n; i++) {
            if (i > 0) System.out.print(" ");
            System.out.print(arr[i]);
        }
        System.out.println();

        sc.close();
    }
}
