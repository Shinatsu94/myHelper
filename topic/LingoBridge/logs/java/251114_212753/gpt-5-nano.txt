```java
/*
 * File: CSIE_41343136_1031_2.java
 * 功能: 讀入長度 n 與 n 個整數，將陣列由小至大排序後輸出
 * 作者: 教師
 * 日期: 2025-11-14
 * 版本: 1.1
 * 備註: 不使用 EOF，單次執行並輸出排序結果
 */

import java.util.Scanner;
import java.util.Arrays;

public class CSIE_41343136_1031_2 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        if (!sc.hasNextInt()) {
            return;
        }

        int n = sc.nextInt();
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();
        }

        Arrays.sort(arr);

        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }

        System.out.print(sb.toString());
        sc.close();
    }
}
```
