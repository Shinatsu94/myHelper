import java.util.Scanner;

/**
 * File: CSIE_51015108_1003_3.java
 * Title: 數字位數總和等於指定值的計數
 * Description: 統計 1~100000 之間，有多少數字的各位數字之和等於輸入值 A
 * Author: 作者
 * Date: 日期
 */
public class CSIE_51015108_1003_3 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int targetSum = sc.nextInt(); // 輸入的位數總和要求
        int count = 0;

        // 遍歷 1 到 100000，統計位數總和等於 targetSum 的數字
        for (int num = 1; num <= 100000; num++) {
            if (digitSum(num) == targetSum) {
                count++;
            }
        }

        System.out.println(count);
        sc.close();
    }

    // 通用的位數總和函式，對任意正整數皆有效
    private static int digitSum(int n) {
        int sum = 0;
        while (n > 0) {
            sum += n % 10;
            n /= 10;
        }
        return sum;
    }
}
