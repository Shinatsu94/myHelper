/*
File: CSIE_41343236_1017_2.java
功能: 判斷輸入數字的各位數是否嚴格遞增（left-to-right）。
作者: 學生
日期: 2025-11-15
說明: 程式僅執行一次，輸入為單一數字字串，輸出是否為 well-ordered number。
*/

import java.util.Scanner;

public class CSIE_41343236_1017_2 {

    // 判斷數字字串的各位是否嚴格遞增
    public static boolean isWellOrdered(String num) {
        if (num == null || num.length() <= 1) {
            return true;
        }

        int n = num.length();
        char prev = num.charAt(0);

        for (int i = 1; i < n; i++) {
            char cur = num.charAt(i);
            // 若前一位數字不小於後一位，則不是嚴格遞增
            if (prev >= cur) {
                return false;
            }
            prev = cur;
        }
        return true;
    }

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        String input = scanner.nextLine().trim();

        if (isWellOrdered(input)) {
            System.out.println(input + " is well-ordered number.");
        } else {
            System.out.println(input + " is not well-ordered number.");
        }

        scanner.close();
    }
}
