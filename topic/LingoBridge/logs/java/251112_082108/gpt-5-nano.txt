```java
/*
 * File: LeibnizPiPrinter.java
 * 功能: 使用莱布尼茲公式計算 pi，輸出 i 為 10000,20000,...,100000 時的近似值，
 *       以及找出使 pi >= 3.14159 的最小 i。
 * 作者: 你自己
 * 日期: 2025/10/17
 * 程式說明: 以遞增累加的方式計算 4 * sum_{k=0}^{i-1} (-1)^k / (2k+1)
 */
public class LeibnizPiPrinter {
    public static void main(String[] args) {
        // 避免地區設定影響小數點顯示
        // 輸出格式：i = 6位數對齊, pi = 小數點後10位
        int[] steps = {10000, 20000, 30000, 40000, 50000, 60000, 70000, 80000, 90000, 100000};

        double sum = 0.0;
        int currentK = 0;

        // 逐步增加 i，利用累積避免重複計算
        for (int i : steps) {
            for (; currentK < i; currentK++) {
                double term = 1.0 / (2 * currentK + 1);
                if ((currentK & 1) == 0) { // k 偶數，正號
                    sum += term;
                } else {                   // k 奇數，負號
                    sum -= term;
                }
            }
            double pi = 4.0 * sum;
            System.out.printf("i = %6d, pi = %.10f%n", i, pi);
        }

        // 找出 pi >= 3.14159 的最小 i
        int k = 0;
        double s = 0.0;
        int iMin = 0;
        while (true) {
            double term = 1.0 / (2 * k + 1);
            if ((k & 1) == 0) s += term;
            else s -= term;
            double piValue = 4.0 * s;
            if (piValue >= 3.14159) {
                iMin = k + 1; // i 為前 k+1 項
                break;
            }
            k++;
        }
        System.out.printf("i = %6d%n", iMin);
    }
}
```

- 輸出格式符合題意：使用 System.out.printf，將 i 以 %6d 對齊，π 值以 %.10f 顯示小數點後 10 位，符合指定的顯示格式。
