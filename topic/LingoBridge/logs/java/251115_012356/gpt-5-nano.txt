功能: 將輸入的一維整數陣列由小到大排序，輸出排序結果

public class CSIE_41343121_1031_1 {
    public static void main(String[] args) {
        // 讀取整數輸入，第一個為陣列長度 n，接著 n 個整數為陣列元素
        java.util.Scanner scn = new java.util.Scanner(System.in);

        if (!scn.hasNextInt()) {
            scn.close();
            return;
        }

        int n = scn.nextInt();
        if (n <= 0) {
            scn.close();
            return;
        }

        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            if (scn.hasNextInt()) {
                arr[i] = scn.nextInt();
            } else {
                // 若輸入不足，提前結束
                break;
            }
        }

        // 冒泡排序法：由小到大
        for (int i = 0; i < n - 1; i++) {
            boolean swapped = false;
            for (int j = 0; j < n - 1 - i; j++) {
                if (arr[j] > arr[j + 1]) {
                    int tmp = arr[j];
                    arr[j] = arr[j + 1];
                    arr[j + 1] = tmp;
                    swapped = true;
                }
            }
            if (!swapped) break; // 若整趟沒交換，表示已排序完成
        }

        // 輸出排序結果，元素以空白分隔，結尾不多餘空白
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.print(sb.toString());

        scn.close();
    }
}
