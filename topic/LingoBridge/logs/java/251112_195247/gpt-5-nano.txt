/*
File: SortArrayAscending.java
功能: 讀入長度 n，接著讀入 n 個整數，輸出由小到大排序的結果。
作者: 請自行填寫
日期: 2025-11-12
*/
import java.util.Arrays;
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        if (!sc.hasNextInt()) {
            sc.close();
            return;
        }

        int n = sc.nextInt();
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();
        }

        Arrays.sort(arr);

        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }

        System.out.println(sb.toString());
        sc.close();
    }
}

- 使用 StringBuilder 的連接輸出，避免多次 System.out.print 導致效率下降。
