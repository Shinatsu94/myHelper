/*
 * 檔案名稱: SortArray.java
 * 功能說明: 讀取使用者輸入的長度 n，接著讀入 n 個整數，輸出由小到大排序後的內容。
 * 作者: 教師
 * 日期: 2025-11-14
 * 版本: 1.0
 */
import java.util.Arrays;
import java.util.Scanner;

public class SortArray {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();
        }
        sc.close();

        Arrays.sort(arr);

        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(arr[i]);
        }
        System.out.print(sb.toString());
    }
}
