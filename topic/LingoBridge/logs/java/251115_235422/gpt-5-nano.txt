/**
 * 檔案名稱: CSIE_51015120_1003_2.java
 * 功能描述: 讀入三個整數，輸出最小值、中間值、最大值，並使用兩種不同方式找值。
 * 作者: 資訊工程系教授
 * 日期: 2025-11-15
 * 備註: 使用三元運算子求最大、最小；中間值以 sum - max - min 計算；使用 try-with-resources 自動關閉 Scanner。
 */

import java.util.Scanner;

public class CSIE_51015120_1003_2 {
    public static void main(String[] args) {
        try (Scanner sc = new Scanner(System.in)) {
            int a = sc.nextInt();
            int b = sc.nextInt();
            int c = sc.nextInt();

            int max = (a > b) ? ((a > c) ? a : c) : ((b > c) ? b : c);
            int min = (a < b) ? ((a < c) ? a : c) : ((b < c) ? b : c);

            long sum = (long) a + b + c; // 避免溢位
            int mid = (int) (sum - max - min);

            System.out.println(min);
            System.out.println(mid);
            System.out.println(max);
        }
    }
}
