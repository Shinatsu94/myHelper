/*
 * File: LowerTriangularSearch.java
 * 功能: 建立一個下三角形二維陣列，填入從1開始的連續數字，找出指定數字在陣列中的位置（列、行，皆以1開頭）並輸出。
 * 作者: 你的姓名
 * 日期: 2025-11-14
 */

import java.util.Scanner;

public class LowerTriangularSearch {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        int n = scanner.nextInt(); // 陣列高度（層數）
        int num = scanner.nextInt(); // 要尋找的數字

        int maxNum = n * (n + 1) / 2; // 下三角形總元素數

        if (num < 1 || num > maxNum) {
            System.out.println("not find " + num + " in array.");
        } else {
            // 建立實際的下三角形二維陣列並填值
            int[][] tri = new int[n][];
            int val = 1;
            for (int i = 0; i < n; i++) {
                tri[i] = new int[i + 1];
                for (int j = 0; j <= i; j++) {
                    tri[i][j] = val++;
                }
            }

            // 使用公式直接計算 num 所在的列與欄
            // row 為 1-based
            int row = (int) Math.ceil((Math.sqrt(8.0 * num + 1) - 1) / 2.0);
            int startOfRow = row * (row - 1) / 2; // 這一列前面的數字總數
            int col = num - startOfRow; // 1-based 的欄位

            System.out.println(num + "[" + row + ", " + col + "]");
        }

        scanner.close();
    }
}
