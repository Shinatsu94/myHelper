/*
檔案名稱: LeibnizPiCalculator.java
功能: 計算 pi 的近似值（Leibniz 公式）：
      pi = 4 * (1 - 1/3 + 1/5 - 1/7 + ... + (-1)^(n+1) / (2n-1))
      (a) 輸出 i = 10000, 20000, ..., 100000 對應的 pi 值，格式符合題目要求
      (b) 找到最小的 i 使 pi（四倍和）到小數點後 5 位等於 3.14159
作者: 教授
日期: 2025-11-11
*/
public class LeibnizPiCalculator {
    public static void main(String[] args) {
        // (a) 逐步列出 i 為 10000 的倍數時的 pi 值
        for (int i = 10000; i <= 100000; i += 10000) {
            double sum = 0.0;
            for (int j = 1; j <= i; j++) {
                double term = 1.0 / (2 * j - 1);
                sum += (j % 2 == 1) ? term : -term;
            }
            double pi = sum * 4.0;
            System.out.println("i = " + String.format("%6d", i) + ", pi = " + String.format("%.10f", pi));
        }

        System.out.println();

        // (b) 找出使 pi 的近似值到小數第 5 位為 3.14159 的最小 i
        int terms = 0;
        double sum = 0.0;
        // sum 為當前前 terms 項的和；檢查的條件是 floor(sum*4*100000)/100000 是否等於 3.14159
        while (Math.floor(sum * 4 * 100000.0) / 100000.0 != 3.14159) {
            terms++;
            double term = 1.0 / (2 * terms - 1);
            sum += (terms % 2 == 1) ? term : -term;
        }
        System.out.println("i = " + String.format("%6d", terms));
    }
}
