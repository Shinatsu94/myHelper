```java
/*
 * File: LowerTriangle.java
 * Description: 建立並列印高度為 n 的下三角形二維陣列
 * Author: 你的名字
 * Date: 2025-11-14
 * Notes: 依題意輸入 0 < n <= 20，輸出下三角形的 '*' 圖形
 */

import java.util.Scanner;

public class LowerTriangle {
    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);
        int n = in.nextInt();

        // 輸入範圍檢查
        if (n <= 0 || n > 20) {
            in.close();
            return;
        }

        // 建立並填入下三角星號的二維陣列
        char[][] triangle = new char[n][n];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= i; j++) {
                triangle[i][j] = '*';
            }
        }

        // 組裝輸出，使用 StringBuilder 提升效能
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= i; j++) {
                sb.append(triangle[i][j]);
            }
            sb.append('\n'); // 每行結尾換行，符合輸出格式
        }

        System.out.print(sb.toString());

        in.close();
    }
}
```
