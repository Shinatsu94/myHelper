/*
 * 文件名稱: PiLeibnizCalculator.java
 * 功能說明: 使用 Leibniz 無窮級數近似圓周率，輸出 i=10000,20000,...,100000 對應的 pi 值，
 *           以及找出最小 i 使 pi 在區間 3.14159 <= pi < 3.14160。
 * 作者: 你的姓名
 * 日期: 2025-11-15
 * 版本: 1.0
 */

public class PiLeibnizCalculator {
    public static void main(String[] args) {
        final int STEP = 10000;

        // 計算並輸出 pi 在 i = 10000, 20000, ..., 100000 的值
        double sum = 0.0;
        int sign = 1; // 符號，初始為正
        for (int k = 1; k <= 100000; k++) {
            sum += 4.0 * sign / (2 * k - 1);
            sign = -sign;
            if (k % STEP == 0) {
                System.out.printf("i = %6d, pi = %.10f%n", k, sum);
            }
        }

        // 找出最小的 i 使得 pi 落在 [3.14159, 3.14160) 的區間
        double s = 0.0;
        int a = 1;
        int i = 0;
        while (s <= 3.14159 || s >= 3.14160) {
            i++;
            s += 4.0 * a / (2 * i - 1);
            a = -a;
        }
        System.out.printf("%ni = %d", i);
    }
}
