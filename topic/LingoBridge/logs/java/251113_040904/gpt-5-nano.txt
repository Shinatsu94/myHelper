public class CSIE_41343150_0926_2_Optimized {
    public static void main(String[] args) {
        printWithCode("虎科大");
        printWithCode("林武杰");
    }

    // 將字串中的每個字元與其Unicode碼點輸出
    private static void printWithCode(String text) {
        for (int i = 0; i < text.length(); i++) {
            char ch = text.charAt(i);
            int codePoint = ch; // BMP 字元的Unicode碼點
            System.out.println(ch + " " + codePoint);
        }
    }
}
