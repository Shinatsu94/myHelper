/**
 * 檔案名稱: SwapTwoNumbers.java
 * 功能: 讀入兩個整數 A, B，印出交換後的兩個數字（僅使用 A、B 兩個變數進行交換）
 * 作者: 逐步提供
 * 日期: 2025-11-11
 */
import java.util.Scanner;

public class SwapTwoNumbers {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int A = sc.nextInt();
        int B = sc.nextInt();

        // 不可使用額外變數 → 以加減法交換 A 與 B 的值
        A = A + B;
        B = A - B;
        A = A - B;

        System.out.println(A);
        System.out.println(B);

        sc.close();
    }
}
