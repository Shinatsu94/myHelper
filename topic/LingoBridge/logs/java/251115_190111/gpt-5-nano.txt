/*
 * File: ReverseDigits.java
 * Description: 讀取正整數（可包含前後空白），輸出其位元反轉後的整數表示，會移除前導零；若全為零則輸出 0。
 * Author: 你的名字
 * Date: 2025-11-15
 */
import java.util.Scanner;

public class ReverseDigits {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        String input = sc.nextLine().trim();
        sc.close();

        boolean negative = false;
        String digits = input;

        // 處理負號（若題目未要求可忽略此部份）
        if (!digits.isEmpty() && digits.charAt(0) == '-') {
            negative = true;
            digits = digits.substring(1);
        }

        // 反轉字串表示的數字
        StringBuilder reversed = new StringBuilder(digits).reverse();

        // 移除反轉後字串的前導零
        int idx = 0;
        while (idx < reversed.length() && reversed.charAt(idx) == '0') idx++;
        String result = (idx == reversed.length()) ? "0" : reversed.substring(idx);

        // 輸出結果
        System.out.print(negative ? "-" + result : result);
    }
}
