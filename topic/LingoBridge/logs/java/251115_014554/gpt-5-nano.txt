```java
/*
檔案名稱: CSIE_51015108_1031_1.java
功能: 讀取整數長度 n 與 n 個整數，對陣列內容由小到大排序，並輸出排序結果
作者: 教師
日期: 2025-11-14
*/
import java.util.Scanner;
import java.util.Arrays;

public class CSIE_51015108_1031_1 {
    public static void main(String[] args) {
        Scanner s = new Scanner(System.in);
        int n = s.nextInt();
        int[] array = new int[n];
        for (int i = 0; i < n; i++) {
            array[i] = s.nextInt();
        }
        s.close();

        if (n > 0) {
            Arrays.sort(array);
        }

        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (i > 0) sb.append(' ');
            sb.append(array[i]);
        }
        System.out.println(sb.toString());
    }
}
```
