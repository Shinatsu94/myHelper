修改後完整程式碼：

/*
 * File: CSIE_41343112_1003_3.java
 * 功能: 統計 1 至 100000 中，數字各位數字加總等於輸入 A 的數字個數
 * 作者: 教授
 * 日期: 2025-11-15
 * 版本: 1.0
 * 說明: 使用獨立的 digitSum 函式與效率較高的 I/O 設計，並加入程式標頭註解與內部註解
 */

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;

public class CSIE_41343112_1003_3 {
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        String line = br.readLine();
        int A = Integer.parseInt(line != null ? line.trim() : "0");

        int count = 0;
        // 依題意範圍計數，排除 0
        for (int i = 1; i <= 100000; i++) {
            if (digitSum(i) == A) {
                count++;
            }
        }
        System.out.println(count);
    }

    // 計算整數 x 的數字之和
    private static int digitSum(int x) {
        int sum = 0;
        while (x > 0) {
            sum += x % 10;
            x /= 10;
        }
        return sum;
    }
}
